import numpy as np

def binary_crossentropy(target, output, from_logits=False):
    if not from_logits:
        output = np.clip(output, 1e-7, 1 - 1e-7)
        output = np.log(output / (1 - output))
    return (target * -np.log(1/(1 + np.exp(-output))) +
            (1 - target) * -np.log(1 - 1/(1 + np.exp(-output))))
