import numpy as np

def standardize_weights(y,
                        sample_weight=None,
                        class_weight=None,
                        sample_weight_mode=None):
    """Performs sample weight validation and standardization.

    Everything gets normalized to a single sample-wise (or timestep-wise)
    weight array.

    # Arguments
        y: Numpy array of model targets to be weighted.
        sample_weight: User-provided `sample_weight` argument.
        class_weight: User-provided `class_weight` argument.
        sample_weight_mode: One of `None` or `"temporal"`.
            `"temporal"` indicated that we expect 2D weight data
            that will be applied to the last 2 dimensions of
            the targets (i.e. we are weighting timesteps, not samples).

    # Returns
        A Numpy array of target weights, one entry per sample to weight.

    # Raises
        ValueError: In case of invalid user-provided arguments.
    """
    if sample_weight_mode is not None:
        if sample_weight_mode != 'temporal' and sample_weight_mode != None:
            raise ValueError('sample_weight_mode should be None or "temporal". Found: ' + str(sample_weight_mode))
        if len(y.shape) < 2 and sample_weight_mode == 'temporal':
            raise ValueError('Found a sample_weight array for an input with shape ' + str(y.shape) + '. Timestep-wise sample weighting (use of sample_weight_mode="temporal") is restricted to outputs that are at least 2D, i.e. that have a time dimension.')
        if sample_weight is not None and len(sample_weight.shape) != 2 and sample_weight_mode == 'temporal':
            raise ValueError('Found a sample_weight array with shape ' + str(sample_weight.shape) + '. In order to use timestep-wise sample weighting, you should pass a 2D sample_weight array.')
    else:
        if sample_weight is not None and sample_weight_mode != "temporal" and len(sample_weight.shape) != 1:
            raise ValueError('Found a sample_weight array with shape ' + str(sample_weight.shape) + '. In order to use timestep-wise sample weights, you should specify sample_weight_mode="temporal" in compile(). If you just mean to use sample-wise weights, make sure your sample_weight array is 1D.')

    if sample_weight is not None and class_weight is not None:
        warnings.warn('Found both `sample_weight` and `class_weight`: `class_weight` argument will be ignored.')
    
    weights_flag = False

    if sample_weight is not None:
        if len(sample_weight.shape) > 1:
            raise ValueError('Found a sample_weight with shape' + str(sample_weight.shape) + '. Expected sample_weight with rank less than or equal to 1.')
        if sample_weight.ndim == 1 and len(sample_weight) != len(y):
            raise ValueError('The length of the sample_weight array ' + str(len(sample_weight)) + ' should be equal to the number of samples ' + str(len(y)) + '.')
        if sample_weight.ndim == 1 and len(sample_weight) == len(y):
            weights_flag = True
            weights = sample_weight
        if sample_weight.ndim == 2 and len(sample_weight) != len(y[0]):
            raise ValueError('The length of the sample_weight array ' + str(len(sample_weight)) + ' should be equal to the number of timesteps ' + str(len(y[0])) + '.')
        if sample_weight.ndim == 2 and len(sample_weight) == len(y[0]):
            weights_flag = True
            weights = sample_weight
        if not weights_flag:
            raise ValueError('Invalid shape of sample_weight: ' + str(sample_weight.shape) + ' for input with shape: ' + str(y.shape) )
        return weights
        
    elif isinstance(class_weight, dict):
        if len(y.shape) > 1:
            raise ValueError('`class_weight` not supported for 2+ dimensional targets.')
        
        weights = np.asarray([class_weight[cls] for cls in y])
        return weights
    else:
        if sample_weight_mode is None:
            return np.ones((y.shape[0],), dtype=np.float32)
        else:
            return np.ones((y.shape[0], y.shape[1]), dtype=np.float32)
