def join_path(self, *args):
    parts = []
    for arg in args:
        arg = self._unquote(arg).replace('/', '\\')
        if '\\' in arg:
            parts.extend([a for a in arg.split('\\') if a])
        else:
            parts.append(arg)
    path = '\\'.join(parts)
    if path.startswith('~'):
        return path
    return path
