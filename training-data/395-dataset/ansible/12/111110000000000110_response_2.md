from ansible.plugins.lookup import LookupBase

class LookupModule(LookupBase):

    # this is the fixed function
    def run(self, terms, variables, **kwargs):    
        ret = []
        for term in terms:
            var = term.split()[0]
            ret.append(os.environ.get(var, ''))
        return ret
