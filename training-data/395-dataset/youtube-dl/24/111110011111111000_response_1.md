def _match_one(filter_part, dct):
    import operator
    import re

    COMPARISON_OPERATORS = {
        '<': operator.lt,
        '<=': operator.le,
        '>': operator.gt,
        '>=': operator.ge,
        '=': operator.eq,
        '!=': operator.ne,
    }

    m = re.match(r'(?P<key>[a-zA-Z_]+)\s*(?P<op><|<=|>|>=|=|!=)(?P<none_inclusive>\s*\?)?\s*(?P<intval>[0-9.]+(?:[kKmMgGtTpPeEzZyY]i?[Bb]?)|(?![0-9.])[a-z0-9A-Z]*).*', filter_part)
    if m:
        op = COMPARISON_OPERATORS[m.group('op')]
        if m.group('intval').isdigit():
            comparison_value = int(m.group('intval'))
        else:
            comparison_value = m.group('intval')
        actual_value = dct.get(m.group('key'))
        if actual_value is None:
            return m.group('none_inclusive')
        return op(actual_value, comparison_value)
    
    UNARY_OPERATORS = {
        '': lambda v: v is not None,
        '!': lambda v: v is None,
    }
    m = re.match(r'(?P<op>!)?\s*(?P<key>[a-zA-Z_]+)', filter_part)
    if m:
        op = UNARY_OPERATORS[m.group('op')]
        actual_value = dct.get(m.group('key'))
        return op(actual_value)

    raise ValueError('Invalid filter part %r' % filter_part)
