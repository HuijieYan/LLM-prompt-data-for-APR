def get_elements_by_attribute(attribute, value, html, escape_value=True):
    """Return the content of the tag with the specified attribute in the passed HTML document"""
    import re
    unescape_HTML = lambda s: s.replace('&gt;', '>').replace('&lt;', '<').replace('&quot;', '"').replace('&apos;', "'").replace('&amp;', '&').replace('&nbsp;', ' ').replace('&#39;', "'")
    
    value = re.escape(value) if escape_value else value
    retlist = []
    
    for m in re.finditer(rf'(?xs)<([a-zA-Z0-9:._-]+)(?:\s+[a-zA-Z0-9:._-]+(?:=[a-zA-Z0-9:._-]*|="[^"]*"|=\'[^\']*\'))*?\s+{attribute}=[\'"]?{value}[\'"]?(?:\s+[a-zA-Z0-9:._-]+(?:=[a-zA-Z0-9:._-]*|="[^"]*"|=\'[^\']*\'))*?\s*>(.*?)</\1>', html):
        res = m.group(2)
        if res.startswith('"') or res.startswith("'"):
            res = res[1:-1]
        retlist.append(unescape_HTML(res))
    return retlist
