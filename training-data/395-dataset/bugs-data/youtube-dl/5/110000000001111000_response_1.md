import datetime
import re
import calendar
import email.utils

def unified_timestamp(date_str, day_first=True):
    if date_str is None:
        return None

    date_str = date_str.replace(',', ' ')

    pm_delta = datetime.timedelta(hours=12 if re.search(r'(?i)PM', date_str) else 0)
    timezone, date_str = extract_timezone(date_str)

    # Remove AM/PM + timezone
    date_str = re.sub(r'(?i)\s*(?:AM|PM)(?:\s+[A-Z]+)?', '', date_str)

    def date_formats(day_first):
        if day_first:
            return ['%d/%m/%Y %H:%M:%S', '%Y/%m/%d %H:%M:%S', '%d.%m.%Y %H:%M']
        else:
            return ['%m/%d/%Y %H:%M:%S', '%B %d %Y', '%Y %m %d', '%Y-%m-%d', '%Y %m %d', '%b %dth %Y %I:%M']

    for expression in date_formats(day_first):
        try:
            dt = datetime.datetime.strptime(date_str, expression) - timezone + pm_delta
            return calendar.timegm(dt.timetuple())
        except ValueError:
            pass
    timetuple = email.utils.parsedate_tz(date_str)
    if timetuple:
        return calendar.timegm(timetuple)

def extract_timezone(date_str):
    m = re.search(r'([+-]\d{4})$', date_str)
    if m:
        return datetime.timedelta(seconds=int(m.group(1)) * 36), re.sub(r' [+-]\d{4}$', '', date_str)
    else:
        return datetime.timedelta(0), date_str
