```python
import re

def urljoin(base, path):
    if isinstance(base, bytes):
        base = base.decode('utf-8')

    if re.match(r'^(?:https?:)?//', path):
        return path

    if not isinstance(path, str) or not path:
        return None

    if not isinstance(base, str) or not re.match(
            r'^(?:https?:)?//', base):
        return None

    return base + path
```
