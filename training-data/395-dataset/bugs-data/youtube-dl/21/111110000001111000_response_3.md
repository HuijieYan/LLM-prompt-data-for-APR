```python
import re
from six import urlparse

def urljoin(base, path):
    if not isinstance(path, str) or not path:
        return None
    if re.match(r'^(?:https?:)?//', path):
        return path
    if not isinstance(base, str) or not re.match(r'^(?:https?:)?//', base):
        return None
    return urlparse.urljoin(base, path)
```
