def sudo_support(match_func: Callable[['Command'], bool]) -> Callable[['Command'], bool]:
    def wrapper(command: Command) -> bool:
        if command.sudo:
            return False
        else:
            return match_func(command)
    return wrapper


@sudo_support
def match_func(command: 'Command') -> bool:
    return (command.script_parts
            and 'not found' in command.stderr
            and bool(get_close_matches(command.script_parts[0],
                                      get_all_executables())))
