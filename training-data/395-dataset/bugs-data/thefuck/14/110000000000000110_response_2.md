import os

def _get_overridden_aliases(self):
    overridden_aliases = os.environ.get('TF_OVERRIDDEN_ALIASES', '').strip()
    if overridden_aliases:
        return [alias.strip() for alias in overridden_aliases.split(',')]
    else:
        return ['cd', 'grep', 'ls', 'man', 'open']
