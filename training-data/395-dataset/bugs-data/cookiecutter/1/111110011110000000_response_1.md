```python
import os
import json
from collections import OrderedDict

class ContextDecodingException(Exception):
    pass

def apply_overwrites_to_context(original_context, new_overwrites):
    for k, v in new_overwrites.items():
        if isinstance(v, dict):
            # Recursion to drill into nested context
            apply_overwrites_to_context(original_context[k], v)
        else:
            original_context[k] = v

def generate_context(context_file='cookiecutter.json', default_context=None, extra_context=None):
    context = OrderedDict()

    try:
        with open(context_file) as file_handle:
            obj = json.load(file_handle, object_pairs_hook=OrderedDict)
    except FileNotFoundError:
        raise FileNotFoundError(f"{context_file} not found")
    except ValueError as e:
        full_fpath = os.path.abspath(context_file)
        json_exc_message = str(e)
        our_exc_message = (
            f'JSON decoding error while loading "{full_fpath}".  Decoding'
            f' error details: "{json_exc_message}"'
        )
        raise ContextDecodingException(our_exc_message)

    file_name = os.path.split(context_file)[1]
    file_stem = file_name.split('.')[0]
    context[file_stem] = obj

    if default_context:
        apply_overwrites_to_context(obj, default_context)
    if extra_context:
        apply_overwrites_to_context(obj, extra_context)

    return context
```
