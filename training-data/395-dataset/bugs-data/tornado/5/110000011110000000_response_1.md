import math

def _update_next(self, current_time):
    if self._next_timeout <= current_time:
        callback_time_sec = self.callback_time / 1000.0
        num_callbacks = math.floor((current_time - self._next_timeout) / callback_time_sec) + 1
        self._next_timeout += num_callbacks * callback_time_sec
