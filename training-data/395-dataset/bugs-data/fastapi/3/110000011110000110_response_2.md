async def serialize_response(
    *,
    field: pydantic.fields.ModelField = None,
    response_content: Any,
    include: typing.Union[Set[int], Dict[str, typing.Any]] = None,
    exclude: typing.Union[Set[int], Dict[str, typing.Any]] = set(),
    by_alias: bool = True,
    exclude_unset: bool = False,
    is_coroutine: bool = True,
) -> Any:
    if field:
        errors = []
        if exclude_unset and isinstance(response_content, pydantic.main.BaseModel):
            if PYDANTIC_1:
                response_content = response_content.dict(exclude_unset=exclude_unset, by_alias=by_alias)
            else:
                response_content = response_content.dict(
                    exclude_unset=exclude_unset, skip_defaults=exclude_unset, by_alias=by_alias
                )  # pragma: nocover
        if is_coroutine:
            value, errors_ = field.validate(response_content, {}, loc=("response",))
        else:
            value, errors_ = await run_in_threadpool(
                field.validate, response_content, {}, loc=("response",)
            )
        if isinstance(errors_, pydantic.error_wrappers.ErrorWrapper):
            errors.append(errors_)
        elif isinstance(errors_, list):
            errors.extend(errors_)
        if errors:
            raise pydantic.error_wrappers.ValidationError(errors, field.type_)
        return fastapi.encoders.jsonable_encoder(
            value,
            include=include,
            exclude=exclude,
            by_alias=by_alias,
            exclude_unset=exclude_unset,
        )
    else:
        return fastapi.encoders.jsonable_encoder(response_content)
