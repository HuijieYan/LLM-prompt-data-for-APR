async def serialize_response(
    *,
    field: ModelField = None,
    response_content: Any,
    include: Optional[Union[SetIntStr, DictIntStrAny]] = None,
    exclude: Optional[Union[SetIntStr, DictIntStrAny]] = None,
    by_alias: bool = True,
    exclude_unset: bool = False,
    is_coroutine: bool = True,
) -> Any:
    if field:
        errors = []
        if exclude_unset and isinstance(response_content, BaseModel):
            response_dict_params = {}
            if PYDANTIC_1:
                response_dict_params['exclude_unset'] = exclude_unset
            else:
                response_dict_params['skip_defaults'] = exclude_unset
            response_content = response_content.dict(**response_dict_params)  # pragma: nocover
        if is_coroutine:
            value, errors_ = await field.validate(response_content, {}, loc=("response",))
        else:
            value, errors_ = await run_in_threadpool(
                field.validate, response_content, {}, loc=("response",)
            )
        if isinstance(errors_, ErrorWrapper):
            errors.append(errors_)
        elif isinstance(errors_, list):
            errors.extend(errors_)
        if errors:
            raise ValidationError(errors, field.type_)
        return jsonable_encoder(
            value,
            include=include,
            exclude=exclude,
            by_alias=by_alias,
            exclude_unset=exclude_unset,
        )
    else:
        return jsonable_encoder(response_content)
