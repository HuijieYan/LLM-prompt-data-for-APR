from fastapi.exceptions import RequestValidationError
from fastapi import HTTP_422_UNPROCESSABLE_ENTITY, Request, JSONResponse

from decimal import Decimal
from typing import List

# this is the fixed function
async def request_validation_exception_handler(
    request: Request, exc: RequestValidationError
) -> JSONResponse:

    errors = []
    for err in exc.errors():
        formatted_err = {
            "loc": err.get("loc"),
            "msg": err.get("msg"),
            "type": err.get("type"),
            "ctx": {
                "limit_value": (err.get("ctx").get("limit_value") if err.get("ctx").get("limit_value") else None)
            }
        }
        errors.append(formatted_err)
    return JSONResponse(
        status_code=HTTP_422_UNPROCESSABLE_ENTITY, content={"detail": errors}
    )
