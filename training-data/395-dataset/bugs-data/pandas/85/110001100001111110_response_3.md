                level_values = Categorical(np.asarray(level_values), categories=level_index.categories, ordered=level_index.ordered)
            grouper = level_values.map(mapper)

            ok_codes, uniques = algos.factorize(indexer[mask], sort=True)

            codes[mask] = ok_codes

            # Remove unobserved levels from level_index
            level_index = level_index[uniques]
