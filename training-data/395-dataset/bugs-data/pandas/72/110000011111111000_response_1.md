def setitem(self, indexer, value):
    """
    Set the value inplace, returning a maybe different typed block.

    Parameters
    ----------
    indexer : tuple, list-like, array-like, slice
        The subset of self.values to set
    value : object
        The value being set

    Returns
    -------
    Block

    Notes
    -----
    `indexer` is a direct slice/positional indexer. `value` must
    be a compatible shape.
    """
    transpose = self.ndim > 1

    if value is None:
        if self.is_numeric:
            value = np.nan

    if self._can_hold_element(value):
        if lib.is_scalar(value):
            value = convert_scalar(values, value)
    else:
        find_dtype = False

        if hasattr(value, "dtype"):
            dtype = value.dtype
            find_dtype = True
        elif lib.is_scalar(value) and not isna(value):
            dtype, _ = infer_dtype_from_scalar(value, pandas_dtype=True)
            find_dtype = True

        if find_dtype:
            dtype = find_common_type([self.values.dtype, dtype])
            if not is_dtype_equal(self.dtype, dtype):
                b = self.astype(dtype)
                return b.setitem(indexer, value)

    if is_extension_array_dtype(getattr(value, "dtype", None)):
        arr_value = value
    else:
        arr_value = np.array(value)

    if not self._can_hold_element(value):
        dtype, _ = maybe_promote(arr_value.dtype)
        self.values = self.values.astype(dtype)

    if transpose:
        self.values = self.values.T

    check_setitem_lengths(indexer, value, self.values)

    if is_empty_indexer(indexer, arr_value):
        pass
    elif is_scalar_indexer(indexer, arr_value):
        self.values[indexer] = value
    elif (
        len(arr_value.shape)
        and arr_value.shape[0] == self.values.shape[0]
        and arr_value.size == self.values.size
    ):
        self.values[indexer] = value
        try:
            self.values = self.values.astype(arr_value.dtype)
        except ValueError:
            pass
    else:
        self.values[indexer] = value

    if transpose:
        self.values = self.values.T
    block = self.make_block(self.values)
    return block
