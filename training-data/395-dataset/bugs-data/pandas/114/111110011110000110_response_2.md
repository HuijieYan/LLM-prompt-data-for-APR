    def get_value(self, series, key, getter=None) -> Any:
        """
        Parameters
        ----------
        series : Series
        key : label
        getter : a bound label-oriented scalar accessory such as .loc (optional)
            to support releasing the GIL in the C code paths

        Returns
        -------
        value : Label
        """
        if getter is not None:
            key = getter(key)

        s = getattr(series, "_values", series)
        try:
            return self._engine.get_value(s, key, tz=getattr(series.dtype, "tz", None))
        except KeyError as e1:
            if len(self) > 0 and (self.holds_integer() or self.is_boolean()):
                raise

            try:
                return libindex.get_value_at(s, key)
            except IndexError:
                raise
            except TypeError:
                # generator/iterator-like
                if is_iterator(key):
                    raise InvalidIndexError(key)
                else:
                    raise e1
            except Exception:
                raise e1
        except TypeError:
            # e.g. "[False] is an invalid key"
            if is_scalar(key):
                raise IndexError(key)
            raise InvalidIndexError(key)
