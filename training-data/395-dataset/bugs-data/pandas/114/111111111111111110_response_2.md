    def get_value(self, series, key):
        if not isinstance(series, (pd.Series, pd.DataFrame)):
            raise ValueError("The input must be a Series or DataFrame")

        if isinstance(series, pd.DataFrame):
            data = {col: series[col]._values for col in series.columns}
            df = pd.DataFrame(data)
        else:
            df = pd.DataFrame({series.name: series._values})

        if not (isinstance(key, str) or isinstance(key, int)):
            raise ValueError("The key must be a string or an integer")

        if isinstance(key, str):
            if key not in df.columns:
                raise KeyError(f"KeyError: '{key}' not in index")
            else:
                return df[key].loc[key]
        else:
            if key < 0 or key >= len(df):
                raise KeyError(f"Positional indexing is out-of-bounds")
            else:
                return df[key].loc[key]
