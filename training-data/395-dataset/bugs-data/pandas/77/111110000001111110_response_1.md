import numpy as np

def na_logical_op(x: np.ndarray, y: np.ndarray, op: np.ufunc):
    try:
        result = op(x, y)
    except TypeError:
        if isinstance(y, np.ndarray):
            x = np.asarray(x, dtype=object)
            y = np.asarray(y, dtype=object)
            result = np.vectorize(op)(x, y)
        else:
            if not np.isnan(y):
                y = bool(y)
            try:
                result = op(x, y)
            except (
                TypeError,
                ValueError,
                AttributeError,
                OverflowError,
                NotImplementedError,
            ):
                typ = type(y).__name__
                raise TypeError(
                    f"Cannot perform '{op.__name__}' with a dtyped [{x.dtype}] array "
                    f"and scalar of type [{typ}]"
                )
    return result
