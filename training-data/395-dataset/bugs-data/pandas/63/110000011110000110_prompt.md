    """
    Require they keys to be the same type as the index. (so we don't
    fallback)
    """
    # allow arbitrary setting
    if is_setter:
        return list(key)

    for ax, i in zip(self.obj.axes, key):
        if ax.is_integer():
            if not is_integer(i):

                )
        else:
            if is_integer(i) and not (ax.holds_integer() or ax.is_floating()):
                raise ValueError(
