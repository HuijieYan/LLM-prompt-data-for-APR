    def searchsorted(self, value, side="left", sorter=None):
        if isinstance(value, (np.ndarray, Index)):
            value = np.array(value, dtype='timedelta64[ns]', copy=False)
        elif isinstance(value, int):
            value = np.timedelta64(value, 'ns')
        elif isinstance(value, float):
            value = np.timedelta64(int(value), 'ns')
        elif isinstance(value, (str, datetime.timedelta)):
            value = pd.Timedelta(value)
        else:
            raise TypeError("Value must be Timedelta, string, integer, float, timedelta or convertible")
    
        return self.values.searchsorted(value, side=side, sorter=sorter)
