def astype(self, dtype, copy=True):
    dtype = pandas.api.types.pandas_dtype(dtype)
    if isinstance(dtype, pandas.StringDtype):
        if copy:
            return self.copy()
        return self
    if dtype == "Int64":
        return self.to_numeric(errors="coerce").astype(dtype)  
    return super().astype(dtype, copy)
