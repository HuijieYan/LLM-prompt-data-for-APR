    def astype(self, dtype, copy=True):
        dtype = pandas_dtype(dtype)
        if isinstance(dtype, StringDtype):
            if copy:
                return self.copy()
            return self
        if isinstance(dtype, Int64Dtype):
            try:
                arr = self._ndarray.astype(dtype.numpy_dtype, copy=copy)
                mask = self.isna()
                if mask.any():
                    values = arr[~mask]
                    values = values.astype(dtype.numpy_dtype)
                    arr = values.copy()
                return arr
            except TypeError:
                raise TypeError('data type not understood')
        return super().astype(dtype, copy)
