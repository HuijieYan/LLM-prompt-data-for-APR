import numpy as np

def diff(arr, n: int, axis: int = 0):
    """
    difference of n between self,
    analogous to s-s.shift(n)

    Parameters
    ----------
    arr : ndarray
    n : int
        number of periods
    axis : int
        axis to shift on

    Returns
    -------
    shifted
    """
    
    na = np.nan
    dtype = arr.dtype

    is_timedelta = False
    if np.issubdtype(arr.dtype, np.datetime64):
        dtype = np.float64
        arr = arr.view("i8")
        na = np.datetime64('nat')
        is_timedelta = True

    out_arr = np.empty(arr.shape, dtype=dtype)

    na_indexer = [slice(None)] * arr.ndim
    na_indexer[axis] = slice(None, n) if n >= 0 else slice(n, None)
    out_arr[tuple(na_indexer)] = na

    if arr.ndim == 1:
        res = arr[n:] if n >= 0 else arr[:n]
        lag = arr[:-n] if n > 0 else arr[-n:]
        result = np.where(res == na, na, res - lag)
        out_arr[n:] if n >= 0 else out_arr[:n] = result
    else:
        res_indexer = [slice(None)] * arr.ndim
        res_indexer[axis] = slice(n, None) if n >= 0 else slice(None, n)
        
        lag_indexer = [slice(None)] * arr.ndim
        lag_indexer[axis] = slice(None, -n) if n > 0 else slice(-n, None)
        
        res = arr[tuple(res_indexer)]
        lag = arr[tuple(lag_indexer)]
        
        mask = (arr[tuple(res_indexer)] == na) | (arr[tuple(lag_indexer)] == na)
        result = res - lag
        result[mask] = na
        out_arr[tuple(res_indexer)] = result

    if is_timedelta:
        out_arr = out_arr.astype("int64").view("timedelta64[ns]")

    return out_arr
