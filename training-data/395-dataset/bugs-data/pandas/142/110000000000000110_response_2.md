import numpy as np

def diff(arr, n: int, axis: int = 0):
    """
    difference of n between self,
    analogous to s-s.shift(n)

    Parameters
    ----------
    arr : ndarray
    n : int
        number of periods
    axis : int
        axis to shift on

    Returns
    -------
    shifted
    """

    dtype = arr.dtype

    out_arr = np.diff(arr, n, axis=axis)

    return out_arr
