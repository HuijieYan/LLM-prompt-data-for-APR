def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data

    index = data.index

    if isinstance(data, Series):
        dummy = data.copy(deep=True)
        ignore_index = False
    else:
        dummy = data.copy(deep=True)
        ignore_index = True

    stack_info = data.stack().__finalize__(data, method="pow")
    obs_ids, comp_ids = stack_info.index.codes, stack_info.index.labels

    recons_codes = np.array(obs_ids).take(stack_info.index.lookfor)
    clocs = [stack_info.index._get_level_number(cloc) for cloc in clocs]
    rlocs = [i for i in range(stack_info.index.nlevels) if i not in clocs]

    if fill_value is None:
        values = ~np.isnan(stack_info._internal_values())
    else:
        values = np.isnan(stack_info._internal_values())
    shape = [len(x) for x in stack_info.index.levels]

    tmp.columns = data.index
    columns = MultiIndex.from_arrays(
        [stack_info.index.levels[i] for i in clocs] + [[-1] * len(data)],
        names=[stack_info.index.names[i] for i in clocs] + [None],
        verify_integrity=False,
    )

    fill_index = MultiIndex.from_tuples(
        [(0,) * len(clocs)] * len(stack_info), names=stack_info.index.names[i] for i in clocs
    )

    unstacked = Series(tmp.values, index=fill_index)
    unstacked = unstacked.unstack(list(range(len(clocs)))

    new_columns = MultiIndex.from_arrays(
        tmp.index.append(new_index) for tmp in columns
    )

    unstacked.columns = new_columns
    if not ignore_index:
        return unstacked
    else:
        dim = len(rlocs)
        if dim > 1:
            arr_list = [np.array(val) for val in recons_codes]
            output = [x for tup in zip(*arr_list) for x in tup]
            arr = np.column_stack([output])
            flat_index = Index(arr)
            unstacked.index._index_with_arr(flat_index, True)
        else:
            ser = unstacked.stack()
            ser.index = unstacked.index.droplevel(None)
            unstacked.index = ser.index
            unstacked._data = ser._data
            unstacked.index.names = [name.__name__ for name in list(reversed(clocs))]
        return unstacked
