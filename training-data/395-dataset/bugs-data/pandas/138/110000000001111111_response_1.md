```python
import numpy as np
import pandas as pd

def _coerce_to_type(x):
    """
    if the passed data is of datetime/timedelta type,
    this method converts it to numeric so that qcut method can
    handle it
    """
    dtype = None

    if isinstance(x, pd.Series) and x.dtype == 'bool':
        x = x.astype(int)
        dtype = np.dtype("int64")
    
    return x, dtype
```
