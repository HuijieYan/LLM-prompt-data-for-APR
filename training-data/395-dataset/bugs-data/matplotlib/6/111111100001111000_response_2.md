def invalid_shape_exception(csize, xsize):
    return ValueError(
        f"'c' argument has {csize} elements, which is inconsistent "
        f"with 'x' and 'y' with size {xsize}.")

def _parse_scatter_color_args(c, edgecolors, kwargs, xsize, get_next_color_func):
    facecolors = kwargs.pop('facecolors', None)
    facecolors = kwargs.pop('facecolor', facecolors)
    edgecolors = kwargs.pop('edgecolor', edgecolors)
    kwcolor = kwargs.pop('color', None)

    if kwcolor is not None and c is not None:
        raise ValueError("Supply a 'c' argument or a 'color'"
                         " kwarg but not both; they differ but"
                         " their functionalities overlap.")

    if kwcolor is not None:
        try:
            mcolors.to_rgba_array(kwcolor)
        except ValueError as err:
            raise ValueError(
                "'color' kwarg must be an color or sequence of color "
                "specs.  For a sequence of values to be color-mapped, use "
                "the 'c' argument instead.") from err
        if edgecolors is None:
            edgecolors = kwcolor
        if facecolors is None:
            facecolors = kwcolor

    if edgecolors is None and not rcParams['_internal.classic_mode']:
        edgecolors = rcParams['scatter.edgecolors']

    c_was_none = c is None
    if c is None:
        c = (facecolors if facecolors is not None
             else "b" if rcParams['_internal.classic_mode']
             else get_next_color_func())
    c_is_string_or_strings = (
        isinstance(c, str)
        or (np.iterable(c) and len(c) > 0
            and isinstance(cbook.safe_first_element(c), str)))

    c_is_mapped = False if c_was_none else True
    valid_shape = True

    if not c_was_none and kwcolor is None and not c_is_string_or_strings:
        try:
            c = np.asanyarray(c, dtype=float)
            c_is_mapped = True
            if c.size != xsize:
                c_is_mapped = False
        except ValueError:
            pass
    if not c_is_mapped and c.size in ((3,), (4,)):
        raise ValueError(
            "'c' argument looks like a single numeric RGB or "
            "RGBA sequence, which should be avoided as value-"
            "mapping will have precedence in case its length "
            "matches with 'x' & 'y'.  Please use a 2-D array "
            "with a single row if you really want to specify "
            "the same RGB or RGBA value for all points.")

    try:
        colors = mcolors.to_rgba_array(c) if c_is_mapped else mcolors.to_rgba_array(c)
    except (TypeError, ValueError) as err:
        if "RGBA values should be within 0-1 range" in str(err):
            raise
        else:
            raise ValueError(
                f"'c' argument must be a color, a sequence of colors, "
                f"or a sequence of numbers, not {c}") from err
    else:
        if len(colors) not in (0, 1, xsize):
            raise ValueError(
                f"'c' argument has {c.size} elements, which is inconsistent "
                f"with 'x' and 'y' with size {xsize}.")

    return c, colors, edgecolors
