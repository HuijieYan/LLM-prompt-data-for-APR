import urllib.parse

def _safe_ParseResult(parts, encoding='utf8', path_encoding='utf8'):
    return (
        urllib.parse.to_native_str(parts.scheme),
        urllib.parse.to_native_str(parts.netloc.encode('idna')),

        # default encoding for path component SHOULD be UTF-8
        urllib.parse.quote_to_bytes(parts.path, path_encoding), urllib.parse.quote_to_bytes(parts.params, path_encoding),

        # encoding of query and fragment follows page encoding
        # or form-charset (if known and passed)
        urllib.parse.quote_to_bytes(parts.query, encoding), urllib.parse.quote_to_bytes(parts.fragment, encoding)
    )
