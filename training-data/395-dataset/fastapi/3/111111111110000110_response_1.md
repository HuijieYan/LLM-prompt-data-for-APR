# Corrected function with the fix for resolving response model serialization issues
# Buggy function and error messages showcased a problem of FastAPI 0.47.1 and 0.49.0 versions which didn't allow response containing pydantic model that includes aliases.

async def serialize_response(
    *,
    field: ModelField = None,
    response_content: Any,
    include: Union[SetIntStr, DictIntStrAny] = None,
    exclude: Union[SetIntStr, DictIntStrAny] = set(),
    by_alias: bool = False,
    exclude_unset: bool = False,
    is_coroutine: bool = True,
) -> Any:
    if field:
        errors = []
        if exclude_unset and isinstance(response_content, BaseModel):
            if PYDANTIC_1:
                response_content = response_content.dict(exclude_unset=exclude_unset, by_alias=by_alias)
            else:
                response_content = response_content.dict(
                    skip_defaults=exclude_unset, by_alias=by_alias
                )  # pragma: nocover
        if is_coroutine:
            value, errors_ = field.validate(response_content, {}, loc=("response",))
        else:
            value, errors_ = await run_in_threadpool(
                field.validate, response_content, {}, loc=("response",)
            )
        if isinstance(errors_, ErrorWrapper):
            errors.append(errors_)
        elif isinstance(errors_, list):
            errors.extend(errors_)
        if errors:
            raise ValidationError(errors, field.type_)
        return jsonable_encoder(
            value,
            include=include,
            exclude=exclude,
            by_alias=by_alias,
            exclude_unset=exclude_unset,
        )
    else:
        return jsonable_encoder(response_content)
