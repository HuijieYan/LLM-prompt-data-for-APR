def _combine_frame(self, other, func, fill_value=None, level=None):
    this, other = self.align(other, join="outer", level=level, copy=False)
    new_index, new_columns = this.index, this.columns

    def _arith_op(left, right):
        # for the mixed_type case where we iterate over columns,
        # _arith_op(left, right) is equivalent to
        # left._binop(right, func, fill_value=fill_value)
        left, right = ops.fill_binop(left, right, fill_value)
        return func(left, right)

    # check for whether we should dispatch to series
    def should_dispatch_to_series(this, other, func):
        if len(this.columns) != len(other.columns):
            return False
        for col1, col2 in zip(this.columns, other.columns):
            if col1 != col2:
                return False
        return True

    if should_dispatch_to_series(this, other, func):
        # iterate over columns
        return ops.dispatch_to_series(this, other, _arith_op)
    else:
        result = _arith_op(this.values, other.values)
        return self._constructor(
            result, index=new_index, columns=new_columns, copy=False
        )
