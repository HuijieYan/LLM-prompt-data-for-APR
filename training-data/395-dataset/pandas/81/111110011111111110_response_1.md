def astype(self, dtype, copy=True, na_value=pd.NA):
    """
    Cast to a NumPy array or IntegerArray with 'dtype'.

    Parameters
    ----------
    dtype : str or dtype
        Typecode or data-type to which the array is cast. Must be a valid numeric dtype.
    copy : bool, default True
        Whether to copy the data, even if not necessary. If False,
        a copy is made only if the old dtype does not match the
        new dtype.
    na_value : scalar, optional
        Scalar missing value indicator to use in numpy array. Defaults
        to the native missing value indicator of this array (pd.NA).

    Returns
    -------
    array : ndarray or IntegerArray
        NumPy ndarray or IntegerArray with 'dtype' for its dtype.

    Raises
    ------
    TypeError
        if incompatible type with an integer dtype and equivalent of same_kind
        casting
    """

    # if we are astyping to an existing IntegerDtype we can fastpath
    if isinstance(dtype, _IntegerDtype):
        result = self._data.astype(dtype.numpy_dtype, copy=False)
        return type(self)(result, mask=self._mask, copy=False)

    # coerce
    if is_float_dtype(dtype):
        # In astype, we consider dtype=float to also mean na_value=np.nan
        kwargs = dict(na_value=np.nan)
    else:
        kwargs = {}

    data = self.to_numpy(dtype=dtype, copy=False, na_value=na_value)
    return astype_nansafe(data, dtype, copy=False)
