from typing import Any, Dict, List, Optional, Tuple, Union
from pydantic.fields import ModelField
from pydantic.error_wrappers import ErrorWrapper
from pydantic.file import UploadFile
from fastapi.params import Form
from copy import deepcopy
import asyncio

BaseConfig = None
PYDANTIC_1 = None
sequence_types = None
sequence_shapes = {"sequence_shapes"}  # assuming this is a set

def get_field_info(field):
    return field

def lenient_issubclass(field, bytes):
    return field

def sequence_shape_to_type(field_shape):
    return field_shape

async def request_body_to_args(
    required_params: List[ModelField],
    received_body: Optional[Union[Dict[str, Any], UploadFile]],
) -> Tuple[Dict[str, Any], List[ErrorWrapper]]:
    values = {}
    errors = []
    if required_params:
        field = required_params[0]
        field_info = get_field_info(field)
        embed = getattr(field_info, "embed", None)
        if len(required_params) == 1 and not embed:
            received_body = {field.alias: received_body}
        for field in required_params:
            value: Any = None
            if received_body is not None:
                if field.shape in sequence_shapes and isinstance(
                    received_body, UploadFile  # Changed FormData to UploadFile
                ):
                    value = received_body.file.read()  # Use file property to read UploadFile contents
                else:
                    value = received_body.get(field.alias)
            if (
                value is None
                or (isinstance(field_info, Form) and value == "")  # Changed params.Form to Form
                or (
                    isinstance(field_info, Form)
                    and field.shape in sequence_shapes
                    and len(value) == 0
                )
            ):
                if field.required:
                    if PYDANTIC_1:
                        errors.append(
                            ErrorWrapper(MissingError(), loc=("body", field.alias))
                        )
                    else:  # pragma: nocover
                        errors.append(
                            ErrorWrapper(  # type: ignore
                                MissingError(),
                                loc=("body", field.alias),
                                config=BaseConfig,
                            )
                        )
                else:
                    values[field.name] = deepcopy(field.default)
                continue
            if (
                isinstance(field_info, UploadFile)  # Changed params.File to UploadFile
                and lenient_issubclass(field.type_, bytes)
                and isinstance(value, UploadFile)
            ):
                value = await value.read()
            elif (
                field.shape in sequence_shapes
                and isinstance(field_info, UploadFile)  # Changed params.File to UploadFile
                and lenient_issubclass(field.type_, bytes)
                and isinstance(value, sequence_types)  # Assuming sequence_types is a sequence type
            ):
                awaitables = [sub_value.read() for sub_value in value]
                contents = await asyncio.gather(*awaitables)
                value = sequence_shape_to_type[field.shape](contents)
            v_, errors_ = field.validate(value, values, loc=("body", field.alias))
            if isinstance(errors_, ErrorWrapper):
                errors.append(errors_)
            elif isinstance(errors_, list):
                errors.extend(errors_)
            else:
                values[field.name] = v_
    return values, errors
