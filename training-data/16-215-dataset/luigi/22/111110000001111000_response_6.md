import time

class Worker(object):
    """
    Structure for tracking worker activity and keeping their references.
    """

    def __init__(self, worker_id, last_active=None):
        self.id = worker_id
        self.reference = None  
        self.last_active = last_active  
        self.started = time.time()
        self.tasks = set()  
        self.info = {}
