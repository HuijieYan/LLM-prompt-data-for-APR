def _dict_arg(self, name, value):
    command = []
    if value and isinstance(value, dict):
        for prop, val in value.items():  # Changed variable name from value to val
            command += [name]  # Changed quotes to double quotes
            if '"' not in str(val):
                command += ['{0}={1}'.format(prop, val)]  # Changed double quotes to single quotes
            else:
                command += ['{0}="{1}"'.format(prop, val)]  # Changed double quotes to single quotes
    return command
