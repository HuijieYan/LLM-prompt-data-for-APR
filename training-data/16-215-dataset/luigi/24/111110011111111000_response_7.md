def _dict_arg(self, name, value):
    command = []
    if value and isinstance(value, dict):
        for prop, val in value.items():  # Changed the variable name from "value" to "val" to avoid conflict with the parameter name
            command += [name, '"{0}={1}"'.format(prop, val)]  # Changed "value" to "val"
    return command
