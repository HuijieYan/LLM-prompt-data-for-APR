def get_new_command(command):
    # If --set-upstream or -u are passed, remove it and its argument. This is
    # because the remaining arguments are concatenated onto the command suggested
    # by git, which includes --set-upstream and its argument
    git_push_index = command.script_parts.index('git')
    push_index = command.script_parts.index('push')
    end_index = len(command.script_parts)
    for index, arg in enumerate(command.script_parts):
        if arg in ['-u', '--set-upstream']:
            del command.script_parts[index]
            del command.script_parts[index]
            if index < end_index:
                end_index -= 2
    
    push_upstream = command.stderr.split('\n')[-3].strip().partition('git ')[2]
    return ' '.join(command.script_parts[:push_index+1]) + ' ' + push_upstream
