import re

def replace_argument(command_string, flag, new_value):
    return re.sub(r'{}(\s+\w+)?'.format(flag), new_value, command_string)

def git_support(fn):
    def wrapper(command):
        new_command = Command(command.script, command.stdout, command.stderr)
        new_command.script_parts = command.script_parts.copy()
        return fn(new_command)
    return wrapper

@git_support
def get_new_command(command):
    # If --set-upstream or -u are passed, remove it and its argument. This is
    # because the remaining arguments are concatenated onto the command suggested
    # by git, which includes --set-upstream and its argument
    upstream_flag = '--set-upstream'
    upstream_short_flag = '-u'
    if upstream_flag in command.script:
        command.script_parts.remove(upstream_flag)
    elif upstream_short_flag in command.script:
        upstream_index = command.script_parts.index(upstream_short_flag)
        command.script_parts.pop(upstream_index)
        command.script_parts.pop(upstream_index)
    push_upstream = command.stderr.split('\n')[-3].strip().partition('git ')[2]
    return replace_argument(" ".join(command.script_parts), 'push', push_upstream)
