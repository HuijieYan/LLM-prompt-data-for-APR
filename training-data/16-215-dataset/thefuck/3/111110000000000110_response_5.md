class Fish(Generic):
    def info(self):
        """Returns the name and version of the current shell"""
        proc = Popen(['fish', '-c', 'echo $FISH_VERSION'],
                     stdout=PIPE, stderr=DEVNULL, shell=True) # Added shell=True
        version = proc.communicate()[0].decode('utf-8').strip() # Used proc.communicate() instead of proc.stdout.read()
        return u'Fish Shell {}'.format(version)
