# The declaration of the class containing the fixed function
class Fish(Generic):

    # the fixed function
    def info(self):
        """Returns the name and version of the current shell"""
        proc = Popen(['fish', '-c', 'echo $FISH_VERSION'],
                     stdout=PIPE, stderr=DEVNULL, shell=True)  # added shell=True
        version = proc.communicate()[0].decode('utf-8').strip()  # used communicate() to collect output
        return u'Fish, version {}'.format(version)  # changed the format of the version string
