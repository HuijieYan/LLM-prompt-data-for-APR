import re

@git_support
def match(command):
    script = command.script.split()
    stderr = command.stderr

    stash_pattern = re.compile(r'stash')
    usage_pattern = re.compile(r'usage:')

    if any(re.search(stash_pattern, word) for word in script) and re.search(usage_pattern, stderr):
        return True
    else:
        return False
