from PIL import Image
from io import BytesIO
from scrapy.pipelines.images import FilesPipeline

# The declaration of the class containing the fixed function
class ImagesPipeline(FilesPipeline):
    """
    Abstract pipeline that implements the image thumbnail generation logic
    """

    # The fixed function
    def convert_image(self, image, size=None):
        # Fix for PNG images with RGBA mode
        if image.format == 'PNG' and image.mode == 'RGBA':
            background = Image.new('RGB', image.size, (255, 255, 255))
            background.paste(image, (0, 0), image)
            image = background

        # Ensure the image mode is RGB
        if image.mode != 'RGB':
            image = image.convert('RGB')

        # Resize the image if a size is specified
        if size:
            image = image.copy()
            image.thumbnail(size, Image.ANTIALIAS)

        # Save the image to a buffer
        buf = BytesIO()
        image.save(buf, 'JPEG')
        return image, buf
