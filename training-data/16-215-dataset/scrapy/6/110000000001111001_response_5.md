To fix the bug, the condition should only check for the `RGBA` mode, and the image should be converted to `RGB` if it is in `RGBA` mode.

from PIL import Image
from io import BytesIO

def convert_image(self, image, size=None):
    if image.mode == 'RGBA':
        background = Image.new('RGB', image.size, (255, 255, 255))
        background.paste(image, image)
        image = background
    elif image.mode != 'RGB':
        image = image.convert('RGB')

    if size:
        image = image.copy()
        image.thumbnail(size, Image.ANTIALIAS)

    buf = BytesIO()
    image.save(buf, 'JPEG')
    return image, buf
