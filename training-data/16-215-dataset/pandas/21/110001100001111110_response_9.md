def _get_with(self, key):
    # other: fancy integer or otherwise
    if isinstance(key, (slice, ABCDataFrame)):
        raise TypeError(
            "Indexing a Series with DataFrame is not "
            "supported, use the appropriate DataFrame column"
        )
    elif isinstance(key, tuple):
        return self._get_values_tuple(key)
    elif not is_list_like(key):
        # e.g. scalars that aren't recognized by lib.is_scalar, GH#32684
        return self.loc[key]

    if isinstance(key, (list, np.ndarray, Index)):
        key_type = "string"  # Set key_type to 'string' for all array-like indexers
    else:
        key_type = key.inferred_type

    if key_type in ("integer", "boolean", "string"):
        # We treat all types of array-like indexers the same way
        if not self.index._should_fallback_to_positional():
            return self.loc[key]
        else:
            return self.iloc[key]

    return self.reindex(key)
