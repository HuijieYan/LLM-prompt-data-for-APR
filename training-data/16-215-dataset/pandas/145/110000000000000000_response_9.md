import pandas as pd
import numpy as np
from pandas.core.arrays import Categorical

def dispatch_to_series(left, right, func, str_rep=None, axis=None):
    """
    Evaluate the frame operation func(left, right) by evaluating
    column-by-column, dispatching to the Series implementation.

    Parameters
    ----------
    left : DataFrame
    right : scalar or DataFrame
    func : arithmetic or comparison operator
    str_rep : str or None, default None
    axis : {None, 0, 1, "index", "columns"}

    Returns
    -------
    DataFrame
    """
    # Note: we use iloc to access columns for compat with cases
    #       with non-unique columns.
    import pandas.core.computation.expressions as expressions
    import pandas.core.common as com
    from pandas import DataFrame, Series
    from pandas.core.generic import NDFrame, ABCDataFrame, ABCSeries
    from pandas.core import algorithms
    from pandas.core.dtypes.generic import ABCSeries

    right = com.apply_if_callable(right, left)
    assert isinstance(right, (NDFrame, ABCSeries, np.ndarray, Categorical))

    if axis is not None:
        if axis == 1:
            pass  # currently only have column axis operations handled
        elif axis == 0:
            assert (isinstance(right, Series) or isinstance(right, np.ndarray) or
                    com.is_scalar(right))
        else:
            raise ValueError("Not implemented")

    new_data = algorithms.dispatch_to_series(left, right, func, axis)
    return new_data
