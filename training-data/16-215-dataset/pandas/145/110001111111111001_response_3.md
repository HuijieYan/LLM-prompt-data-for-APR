    Parameters

    right : scalar or DataFrame

def masked_arith_op(x, y, op):

    mask = (~(xrav.astype('O') == pd.NaT)) & (~(yrav.astype('O') == pd.NaT))

    if mask.any():
        with np.errstate(all="ignore"):
            result = np.empty(x.size, dtype=x.dtype)
            result[mask] = op(xrav[mask], yrav[mask])
