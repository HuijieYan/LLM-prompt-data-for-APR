import numpy as np
import pandas.core.computation.expressions as expressions
from pandas.core.indexing import _LocIndexer
from pandas.core.construction import extract_array
from pandas.core.frame import DataFrame, _shared_docs
from pandas.core.generic import NDFrame, ABCDataFrame, ABCSeries
import pandas.core.algorithms as algorithms
from pandas.core.dtypes.common import is_list_like, is_scalar

def dispatch_to_series(left, right, func, str_rep=None, axis=None):
    """
    Evaluate the frame operation func(left, right) by evaluating
    column-by-column, dispatching to the Series implementation.

    Parameters
    ----------
    left : DataFrame
    right : scalar or DataFrame
    func : arithmetic or comparison operator
    str_rep : str or None, default None
    axis : {None, 0, 1, "index", "columns"}

    Returns
    -------
    DataFrame
    """
    if axis is not None:
        raise NotImplementedError("axis argument is not supported yet")

    if not isinstance(func, str):
        raise ValueError("`func` must be a string representing the operation (e.g., '+', '-', '*', '/', '==', '<', '>', etc.)")

    if not isinstance(left, NDFrame) or not isinstance(right, (ABCDataFrame, ABCSeries)) and not is_scalar(right):
        raise TypeError("Unsupported types: Not a DataFrame, Series, or scalar")

    right = extract_array(right, extract_numpy=True)

    lf = left
    if isinstance(right, ABCDataFrame):
        assert right._indexed_same(lf), "Index of DataFrame does not match"
    elif isinstance(right, ABCSeries):
        assert right.index.equals(lf.index), "Index of Series does not match"
        if axis == 'columns':
            assert right.index.equals(lf.columns), "Index of Series does not match columns of DataFrame"
            lf = lf.T

    new_data = algorithms.dispatch_to_series(frame=lf, right=right, func=func)
    return new_data
