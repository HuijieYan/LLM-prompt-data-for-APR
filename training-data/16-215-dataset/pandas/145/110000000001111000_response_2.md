import pandas as pd
import numpy as np
def dispatch_to_series(left, right, func, str_rep=None, axis=None):

    assert axis == "columns" and isinstance(right, pd.Series) and right.index.equals(left.index)

    a = left
    b = right

    expressions = __import__("pandas.core.computation.expressions", fromlist=[""])

    new_data = expressions.evaluate(lambda a, b: pd.DataFrame({i: func(a[i], b) for i in a}), str_rep, a, b)
    return new_data
