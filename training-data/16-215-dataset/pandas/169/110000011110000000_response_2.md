import pandas as pd
import numpy as np

def quantile(self, q=0.5, axis=0, numeric_only=True, interpolation="linear"):
    self._check_percentile(q)

    data = self._get_numeric_data() if numeric_only else self
    axis = pd._libs.lib.infer_dtype_from_object(axis)
    
    result = np.nanpercentile(data, q, axis=axis, interpolation=interpolation)

    if isinstance(q, (list, np.ndarray)):
        if result.ndim == 2:
            result = pd.DataFrame(result, index=q, columns=self.columns)
        else:
            result = pd.Series(result, index=self.columns)
    else:
        result = pd.Series(result, index=self.columns)

    return result

# Test
def test_quantile_empty_no_columns(self):
    df = pd.DataFrame(pd.date_range("1/1/18", periods=5))
    df.columns.name = "captain tightpants"
    result = df.quantile(0.5)
    expected = pd.Series([], index=[], name=0.5)
    expected.index.name = "captain tightpants"
    pd.testing.assert_series_equal(result, expected)

    result = df.quantile([0.5])
    expected = pd.DataFrame([], index=[0.5], columns=[])
    expected.columns.name = "captain tightpants"
    pd.testing.assert_frame_equal(result, expected)
