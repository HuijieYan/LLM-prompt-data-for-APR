def quantile(self, q=0.5, axis=0, numeric_only=True, interpolation="linear"):
    """
    Return values at the given quantile over requested axis.

    ... (rest of the docstring remains the same) ...

    """
    self._check_percentile(q)

    data = self._get_numeric_data() if numeric_only else self
    axis = self._get_axis_number(axis)
    is_transposed = axis == 1

    if is_transposed:
        data = data.T

    # Use iloc to access the actual data for quantile calculation
    result = data._data.quantile(
        qs=q, axis=0 if axis == 1 else 1, interpolation=interpolation, transposed=is_transposed
    )

    # If result is not a dataframe or series, create Series from it
    if not isinstance(result, (pd.DataFrame, pd.Series)):
        result = pd.Series(result, name=q)

    if is_transposed:
        result = result.T

    return result
