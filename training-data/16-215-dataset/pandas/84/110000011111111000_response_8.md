def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data

    index = data.index

    cnames = [name for name in index.names if name in clocs]
    rnames = [name for name in index.names if name not in clocs]

    cindex = index.droplevel(rnames)

    if isinstance(data, pd.Series):
        unstacked = cindex.unstack(fill_value=fill_value)
        unstacked.index = pd.MultiIndex.from_tuples([unstacked.index], names=cnames)
    else:
        unstacked = cindex.unstack(fill_value=fill_value)
        unstacked.columns = pd.MultiIndex.from_tuples([unstacked.columns], names=cnames)

    return unstacked
