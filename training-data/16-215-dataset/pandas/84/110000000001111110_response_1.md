import pandas as pd

def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data

    index = data.index
    clocs = [index._get_level_number(i) for i in clocs]

    def _unstack_single(data, clocs, fill_value=fill_value):
        return data.unstack(clocs, fill_value=fill_value)

    if len(clocs) == 1:
        return _unstack_single(data, clocs, fill_value=fill_value)
    
    unstacked = data
    level_data = list(index.levels[i] for i in clocs)
    code_data = list(index.codes[i] for i in clocs)
    name_data = list(index.names[i] for i in clocs)
    dummy_index = pd.MultiIndex.from_arrays(code_data + [len(index.codes[0])*[0]], names=name_data+["__placeholder__"])
    unstacked.index = dummy_index
    unstacked = _unstack_single(unstacked, len(dummy_index)-1, fill_value)
    levels = level_data + [index.levels[i] for i in range(len(index.levels)) if i not in clocs]
    codes = code_data + [index.codes[i] for i in range(len(index.levels)) if i not in clocs]
    names = name_data + [index.names[i] for i in range(len(index.levels)) if i not in clocs]
    
    return unstacked.set_index(pd.MultiIndex(levels=levels, codes=codes, names=names))

# Test Cases
idx = pd.MultiIndex.from_product([['a', 'b', 'c'], [1, 2, 3]], names=[('A', 'a'), ('B', 'b')])

s = pd.Series(1, index=idx)
result1 = s.unstack(("A", "a"))


testing_loc = [idx.names[0] + (i,) for i in [idx.levels[0][0], idx.levels[0][-1]]]
implicit = _unstack_multiple(s, testing_loc)

result1.equals(implicit)
