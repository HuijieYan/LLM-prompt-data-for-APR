import pandas as pd

def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data

    levels = data.index.names

    if isinstance(clocs, tuple):
        clocs = [clocs]

    clocs_index = [levels.index(c) for c in clocs]
    rlocs_index = [i for i in range(len(levels)) if i not in clocs_index]

    clevels = [data.index.levels[i] for i in clocs_index]
    ccodes = [data.index.codes[i] for i in clocs_index]
    cnames = [data.index.names[i] for i in clocs_index]
    rlevels = [data.index.levels[i] for i in rlocs_index]
    rcodes = [data.index.codes[i] for i in rlocs_index]
    rnames = [data.index.names[i] for i in rlocs_index]

    shape = [len(x) for x in clevels]

    group_index = pd.MultiIndex.from_product(clevels, names=cnames).codes

    obs_ids, comp_ids = pd.factorize(group_index)

    obs_ids = obs_ids % shape[0]
    comp_ids = comp_ids % shape[0]

    recons_codes = [comp_ids[obs_ids], obs_ids]

    new_columns = pd.MultiIndex.from_product(clevels, names=cnames)
    new_index = pd.MultiIndex.from_product(rlevels, names=rnames)

    dummy = data.copy()
    dummy.index = pd.MultiIndex.from_arrays(rcodes + recons_codes, names=rnames + cnames)

    unstacked = dummy.unstack(clocs_index[0])

    if isinstance(unstacked, pd.Series):
        unstacked = unstacked.unstack(fill_value=fill_value)
    else:
        unstacked = unstacked.unstack(fill_value=fill_value).unstack(fill_value=fill_value)

    if not isinstance(data, pd.Series):
        unstacked.columns = new_columns

    unstacked.index = new_index

    return unstacked
