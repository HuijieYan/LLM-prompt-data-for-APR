def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data

    index = data.index

    if isinstance(index, pd.MultiIndex):
        levels = index.names
        clevels = [levels[i] for i in clocs]
    else:
        levels = [None for _ in range(index.nlevels)]
        clevels = [index.name]

    rlocs = [i for i in range(index.nlevels) if i not in clocs]

    group_index = index._compliance_codes(for_unstacking=True)

    comp_ids, obs_ids = group_index.compress(group_index, sort=False)
    recons_codes = group_index.decons_obs_group_ids(comp_ids, obs_ids, False)

    if rlocs == []:
        # Everything is in clocs, so the dummy df has a regular index
        dummy_index = pd.Index(obs_ids, name="__placeholder__")
    else:
        dummy_index = pd.MultiIndex.from_arrays(
            index = pd.concat([index.levels[i] for i in rlocs] + [pd.Index(obs_ids)], ignore_index=True),
            codes = pd.concat([index.codes[i] for i in rlocs] + [pd.Index(comp_ids)], ignore_index=True),
            names = [index.names[i] for i in rlocs] + ["__placeholder__"]
        )

    if isinstance(data, pd.Series):
        dummy = data.copy()
        dummy.index = dummy_index
        dummy.name = data.name
        unstacked = dummy.unstack("__placeholder__", fill_value)
        new_levels = clevels
        new_names = [d.name for d in data.index.levels[i] for i in clocs]
        new_codes = recons_codes
    else:
        if isinstance(data.columns, pd.MultiIndex):
            result = data
            for i in range(len(clocs)):
                val = clocs[i]
                result = result.unstack(i, fill_value)
            return result

        dummy = data.copy()
        dummy.index = dummy_index

        unstacked = dummy.unstack("__placeholder__", fill_value)
        if isinstance(unstacked, pd.Series):
            name = unstacked.index.name
        else:
            name = unstacked.columns.name
        new_levels = [d.name for d in unstacked.index.levels[0]] + clevels
        new_names = [name] + [index.names[i] for i in clocs]

        new_codes = [unstacked.index.codes[0]]
        for rec in recons_codes:
            new_codes.append(rec.take(unstacked.index.codes[-1]))

    new_columns = pd.MultiIndex(levels=new_levels, codes=new_codes, names=new_names)

    if isinstance(unstacked, pd.Series):
        unstacked.index = new_columns
    else:
        unstacked.columns = new_columns

    return unstacked
