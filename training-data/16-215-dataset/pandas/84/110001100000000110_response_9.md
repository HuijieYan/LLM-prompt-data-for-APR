def _unstack_multiple(data, level, fill_value=None):
    if len(level) == 0:
        return data

    index = data.index

    level_list = [index._get_level_number(i) for i in level]

    remaining_levels = [i for i in range(index.nlevels) if i not in level_list]

    level_data = [index.levels[i] for i in level_list]
    level_codes = [index.codes[i] for i in level_list]
    level_names = [index.names[i] for i in level_list]
    remaining_data = [index.levels[i] for i in remaining_levels]
    remaining_codes = [index.codes[i] for i in remaining_levels]
    remaining_names = [index.names[i] for i in remaining_levels]

    shape = [len(x) for x in level_data]
    group_index = get_group_index(level_codes, shape, sort=False, xnull=False)

    comp_ids, obs_ids = compress_group_index(group_index, sort=False)
    recons_codes = decons_obs_group_ids(comp_ids, obs_ids, shape, level_codes, xnull=False)

    if remaining_levels == []:
        dummy_index = Index(obs_ids, name="__placeholder__")
    else:
        dummy_index = MultiIndex(
            levels=remaining_data + [obs_ids],
            codes=remaining_codes + [comp_ids],
            names=remaining_names + ["__placeholder__"],
            verify_integrity=False,
        )

    if isinstance(data, Series):
        dummy = data.copy()
        dummy.index = dummy_index

        unstacked = dummy.unstack("__placeholder__", fill_value=fill_value)
        new_levels = level_data
        new_names = level_names
        new_codes = recons_codes
    else:
        if isinstance(data.columns, MultiIndex):
            result = data
            for i in range(len(level_list)):
                val = level_list[i]
                result = result.unstack(val, fill_value=fill_value)
                level_list = [v if i > v else v - 1 for v in level_list]

            return result

        dummy = data.copy()
        dummy.index = dummy_index

        unstacked = dummy.unstack("__placeholder__", fill_value=fill_value)
        if isinstance(unstacked, Series):
            unstcols = unstacked.index
        else:
            unstcols = unstacked.columns
        new_levels = [unstcols.levels[0]] + level_data
        new_names = [data.columns.name] + level_names

        new_codes = [unstcols.codes[0]]
        for rec in recons_codes:
            new_codes.append(rec.take(unstcols.codes[-1]))

    new_columns = MultiIndex(
        levels=new_levels, codes=new_codes, names=new_names, verify_integrity=False
    )

    if isinstance(unstacked, Series):
        unstacked.index = new_columns
    else:
        unstacked.columns = new_columns

    return unstacked
