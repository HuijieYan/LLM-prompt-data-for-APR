def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data

    # NOTE: This function only deals with MultiIndexes
    # Check if the index is a MultiIndex
    if not isinstance(data.index, pd.MultiIndex):
        raise ValueError("The index of the input data should be a MultiIndex")

    index = data.index

    clevels = [index.levels[index.names.index(i)] for i in clocs]
    cloc_codes = [index.codes[index.names.index(i)] for i in clocs]

    shape = [len(x) for x in clevels]
    
    group_index = get_group_index(cloc_codes, shape, sort=False, xnull=False)
    
    comp_ids, obs_ids = compress_group_index(group_index, sort=False)
    recons_codes = decons_obs_group_ids(comp_ids, obs_ids, shape, cloc_codes, xnull=False)
    
    if isinstance(data, pd.Series):
        # Unstack for Series
        new_columns = pd.MultiIndex.from_product(clevels, names=clocs)
        unstacked = pd.Series(data.values[obs_ids], index=new_columns)
    else:
        # Unstack for DataFrame
        new_columns = pd.MultiIndex.from_product(clevels, names=clocs)
        new_data = data.to_records(index=True, convert_datetime64=True)
        unstacked = pd.DataFrame(new_data[obs_ids], columns=new_columns)

    return unstacked
