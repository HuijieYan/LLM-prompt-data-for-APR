```python
def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data

    index = data.index

    # Converting tuple of tuples (if passed) to list of tuples instead
    if isinstance(clocs, tuple):
        clocs = [clocs]

    cimals = []

    for cloctup in clocs:
        cimla = []
        for cloc in cloctup:
            clocnum = index._get_level_number(cloc)
            cimla.append(clocnum)
        cimals.append(cimla)

    # Processing for case when hierarchical index is involved
    if len(cimals) == 1:
        result = data.unstack(level=cimals[0], fill_value=fill_value)
    else:
        result = data
        for i, cimal in enumerate(cimals):
            fill_value = fill_value if i == len(cimals) - 1 else None
            level = cimal.pop(0)  # Extract the first value from the list
            result = result.unstack(level=level, fill_value=fill_value)

    return result
```
