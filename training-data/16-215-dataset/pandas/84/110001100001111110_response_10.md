def _unstack_multiple(data, clocs, fill_value=None):
    if not isinstance(clocs, (list, tuple)):
        clocs = [clocs]

    if len(clocs) == 0:
        return data

    if isinstance(data, DataFrame) and isinstance(data.columns, MultiIndex):
        result = data.copy()
        level = clocs[0]
        for i in range(len(data.columns.levels)):
            if level in data.columns.levels[i]:
                result = result.unstack(level, fill_value=fill_value)
                break
        return result
    else:
        index = data.index
        names = index.names
        if isinstance(names[0], (tuple, list)):
            names = [n[0] for n in names]
        clocs = [names.index(i) for i in clocs]
        
        if {**data.index._get_attributes_dict(), **{'level': clocs}} not in data._attributes:
            result = data.unstack(level=clocs, fill_value=fill_value)
            return result
        else:
            raise ValueError("Not implemented error")
