def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data

    index = data.index

    if len(clocs) == 1:
        loc = clocs[0]
        level = index._get_level_number(loc)
        unstacked = data.unstack(level, fill_value=fill_value)
        return unstacked
    elif len(clocs) == 2:  
        loc1, loc2 = clocs
        level1 = index._get_level_number(loc1)
        level2 = index._get_level_number(loc2)
        unstacked = data.unstack(level2, fill_value=fill_value)
        new_levels = index.levels[level1].append(unstacked.columns.levels[0])
        new_columns = pd.MultiIndex.from_product([new_levels, unstacked.columns.levels[1]])
        unstacked.columns = new_columns
        return unstacked
    else:
        raise NotImplementedError
