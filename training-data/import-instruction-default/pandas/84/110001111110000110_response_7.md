def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data
    
    # NOTE: This doesn't deal with hierarchical columns yet

    index = data.index

    clocs = [i if isinstance(i, tuple) else (i,) for i in clocs]
    
    cnames = index.names
    cidx = [index._get_level_number(i) for i in clocs]
    
    ridx = [i for i in range(len(index.names)) if i not in cidx]
    rnames = [cnames[i] for i in ridx]
    ridx = [index._get_level_number(i) for i in rnames]

    shape = [len(index.levels[i]) for i in cidx]
    group_index = get_group_index(index.codes, shape, sort=False, xnull=False)

    compressed, obs_ids = compress_group_index(group_index, sort=False)
    recons_codes = decons_obs_group_ids(compressed, obs_ids, shape, index.codes, xnull=False)

    if not ridx:
        dummy_index = Index(obs_ids, name="__placeholder__")
    else:
        rlevels = [index.levels[i] for i in ridx]
        rcodes = [index.codes[i] for i in ridx]
        dummy_index = MultiIndex(levels=rlevels + [obs_ids],
                                 codes=rcodes + [compressed],
                                 names=rnames + ["__placeholder__"],
                                 verify_integrity=False)

    if isinstance(data, Series):

        dummy = data.copy()
        dummy.index = dummy_index
        unstacked = dummy.unstack("__placeholder__", fill_value=fill_value)
        new_levels = [index.levels[i] for i in cidx]
        new_names = [cnames[i] for i in cidx]
        new_codes = recons_codes
    else:
        dummy = data.copy()
        dummy.index = dummy_index
        unstacked = dummy.unstack("__placeholder__", fill_value=fill_value)
        if isinstance(unstacked, Series):
            unstcols = unstacked.index
        else:
            unstcols = unstacked.columns
        new_levels = [unstcols.levels[-1]] + [index.levels[i] for i in cidx]
        new_names = [unstacked.columns.names[-1]] + [cnames[i] for i in cidx]
        new_codes = [unstcols.codes[-1]]
        for rec in recons_codes:
            new_codes.append(rec.take(unstcols.codes[-1]))

    new_columns = MultiIndex(levels=new_levels, codes=new_codes, names=new_names, verify_integrity=False)

    if isinstance(unstacked, Series):
        unstacked.index = new_columns
    else:
        unstacked.columns = new_columns

    return unstacked
