def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data

    # NOTE: This doesn't deal with hierarchical columns yet

    index = data.index
    
    if isinstance(index, MultiIndex):
        ccodes = [index.get_level_codes(i) for i in clocs]
        cnames = [index.names[i] for i in clocs]
        rlocs = [i for i in range(index.nlevels) if i not in clocs]
        obs_ids = Index(range(ccodes[0].max() + 1), name="__placeholder__")
        dummy_index = MultiIndex(
            levels=[index.levels[i] for i in rlocs] + [obs_ids],
            codes=[index.codes[i] for i in rlocs] + [ccodes[0]],
            names=[index.names[i] for i in rlocs] + ["__placeholder__"],
            verify_integrity=False,
        )
    else:
        cnames = [None]
        ccodes = [None]
        dummy_index = Index(obs_ids, name="__placeholder__")

    shape = [len(x) for x in index.levels]
    group_index = get_group_index(index.codes, shape, sort=False, xnull=False)

    comp_ids, obs_ids = compress_group_index(group_index, sort=False)
    recons_codes = decons_obs_group_ids(comp_ids, obs_ids, shape, index.codes, xnull=False)

    if isinstance(data, Series):
        dummy = data.copy()
        dummy.index = dummy_index

        unstacked = dummy.unstack("__placeholder__", fill_value=fill_value)
        new_levels = list(index.levels[i] for i in clocs)
        new_names = list(cnames)
        new_codes = list(recons_codes)

    else:
        dummy = data.copy()
        dummy.index = dummy_index

        unstacked = dummy.unstack("__placeholder__", fill_value=fill_value)
        new_names = [data.columns.name] + cnames
        new_levels = [index.levels[i] for i in rlocs] + [obs_ids]
        new_codes = [index.codes[i] for i in rlocs] + [comp_ids]

    new_columns = MultiIndex(
        levels=new_levels, codes=new_codes, names=new_names, verify_integrity=False
    )

    if isinstance(unstacked, Series):
        unstacked.index = new_columns
    else:
        unstacked.columns = new_columns

    return unstacked
