This results in a `ValueError` and a subsequent `KeyError`.

        for col_loc in clocs[::-1]:  # Iterate over clocs in reverse order
            result = result.unstack(col_loc, fill_value=fill_value)

    levels = [index.levels[index._get_level_number(col_loc)] for col_loc in clocs]
    codes = [index.codes[index._get_level_number(col_loc)] for col_loc in clocs]
    names = [index.names[index._get_level_number(col_loc)] for col_loc in clocs]

    shape = [len(x) for x in levels]
    group_index = get_group_index(codes, shape, sort=False, xnull=False)

    dummy_index_levels = index.levels[:]
    for idx, level in zip(clocs, levels):
        dummy_index_levels[index._get_level_number(idx)] = obs_ids

    dummy_index = MultiIndex(
        levels=dummy_index_levels,
        codes=index.codes[:len(dummy_index_levels)],
        names=index.names[:len(dummy_index_levels)],
        verify_integrity=False,
    )

    # Prepare dummy DataFrame
    dummy = data.copy()
    dummy.index = dummy_index

    new_columns = MultiIndex(

    )

    if isinstance(unstacked, Series):
        unstacked.index = new_columns
    else:
        unstacked.columns = new_columns
