@Substitution("\ndata : DataFrame, index : Optional[Hashable], columns : Optional[Hashable], values : Optional[Hashable]")
@Appender(_shared_docs["pivot"], indents=1)
def pivot(data: "DataFrame", index=None, columns=None, values=None) -> "DataFrame":
    """
    Pivot a DataFrame from long to wide format.

    Parameters
    ----------
    data : DataFrame
    index : hashable, optional
        Column to use to make new frame's index. If None, uses existing index.
    columns : hashable
        Column to use to make new frame's columns.
    values : hashable, optional
        Column(s) to use for populating new frame's values.

    Returns
    -------
    DataFrame

    See also
    --------
    DataFrame.pivot_table : Create a spreadsheet-style pivot table as a DataFrame.
    DataFrame.melt : "Unpivots" a DataFrame from wide format to long format, optionally leaving identifier variables set.

    Examples
    --------
    >>> df = pd.DataFrame({"foo": ['one', 'one', 'two', 'two'],"bar": ['A', 'A', 'B', 'C'],"baz": [1, 2, 3, 4]})
    >>> df.pivot(columns='foo', values='baz')
       foo  one  two
    0    A  1.0  NaN
    1    A  2.0  NaN
    2    B  NaN  3.0
    3    C  NaN  4.0
    """    
    if columns is None:
        raise ValueError("columns is not optional. Please provide a value for 'columns' parameter.")
    if values is None:
        cols = [columns] if index is None else [index, columns]
        append = index is None
        indexed = data.set_index(cols, append=append)
    else:
        if index is None:
            index = data.index
        else:
            index = data[index]
        index = MultiIndex.from_arrays([index, data[columns]])

        if is_list_like(values) and not isinstance(values, tuple):
            # Exclude tuple because it is seen as a single column name
            indexed = data._constructor(
                data[values].values, index=index, columns=values
            )
        else:
            indexed = data._constructor_sliced(data[values].values, index=index)
    return indexed.unstack(columns)
