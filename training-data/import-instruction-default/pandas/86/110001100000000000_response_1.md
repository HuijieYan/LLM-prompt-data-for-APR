# The corrected version of the function
@Substitution("\nframe : DataFrame")
@Appender(_shared_docs["pivot"], indents=1)
def pivot(frame: "DataFrame", index=None, columns=None, values=None) -> "DataFrame":
    if values is None:
        cols = [columns] if index is None else [index, columns]
        append = index is None
        indexed = frame.set_index(cols, append=append)
    else:
        if index is None:
            index = frame.index
        else:
            index = frame[index]
        index = MultiIndex.from_arrays([index, frame[columns]])

        if is_list_like(values) and not isinstance(values, tuple):
            # Exclude tuple because it is seen as a single column name
            indexed = frame._constructor(
                frame[values].values, index=index, columns=values
            )
        else:
            indexed = frame._constructor_sliced(frame[values].values, index=index)
    return indexed.unstack(columns)
