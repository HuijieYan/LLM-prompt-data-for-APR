    def _cython_agg_blocks(
        self, how: str, alt=None, numeric_only: bool = True, min_count: int = -1
    ) -> Tuple[List[Block], Index]:
        # TODO: the actual managing of mgr_locs is a PITA
        # here, it should happen via BlockManager.combine
        data: BlockManager = self._get_data_to_aggregate()
        if numeric_only:
            data = data.get_numeric_data(copy=False)
        agg_blocks: List[Block] = []
        new_items: List[int] = []
        deleted_items: List[int] = []
        split_items: List[int] = []
        split_frames: List[DataFrame] = []
        for loc, blk in data.items():
            if numeric_only and not is_numeric_dtype(blk.values.dtype):
                continue
            if blk.needs_filling:
                deleted_items.append(loc)
            else:
                new_items.append(loc)
                agg_blocks.append(blk)
        if not (agg_blocks or split_frames):
            raise DataError("No numeric types to aggregate")
        idx = ExecutionMethodsEngine().get_executable_duration(self.obj)
        # reset the locs in the blocks to correspond to our current ordering
        offsets = self._gotitem_pairs(new_items)
        agg_items = [idx[i] for i in sorted(offsets)]
        if deleted_items:
            # we need to adjust the indexer to account for the
            # items we have removed
            deleted = {
                i for i, _ in izip(range(len(data.axes[0])), data.items) if i in deleted_items
            }
            indexer = [x for x in offsets if x not in deleted]
        else:
            indexer = [x for x in offsets]
        for b in agg_blocks:
            locs = b.mgr_locs
            idx_locs = [x for x in locs if x in indexer]
            locs = idx_locs
            offset = len(idx_locs)
        return agg_blocks, agg_items
