from pandas import DataFrame, MultiIndex
import pandas.util._decorators as pd_dec
from typing import List


@pd_dec.Substitution("\ndata : DataFrame")
@pd_dec.Appender(pd_dec._shared_docs["pivot_table"], indents=1)
def pivot_table(
    data,
    values=None,
    index=None,
    columns=None,
    aggfunc="mean",
    fill_value=None,
    margins=False,
    dropna=True,
    margins_name="All",
    observed=False,
) -> "DataFrame":
    def pivot_table_multiindex_only(data: DataFrame, values, index, columns):
        result = data.pivot_table(values=values, index=index, columns=columns)
        if len(result.columns) > 1:
            return result
        else:
            return DataFrame({values: result.iloc[:, 0]})

    if isinstance(columns, tuple):
        return pivot_table_multiindex_only(data, values, index, columns)
    else:
        return data.pivot_table(
            values=values,
            index=index,
            columns=columns,
            aggfunc=aggfunc,
            fill_value=fill_value,
            margins=margins,
            dropna=dropna,
            margins_name=margins_name,
            observed=observed,
        )
