class TimeGrouper(Grouper):
    """
    Custom groupby class for time-interval grouping.
    
    Parameters
    ----------
    freq : pandas date offset or offset alias for identifying bin edges
    closed : closed end of interval; 'left' or 'right'
    label : interval boundary to use for labeling; 'left' or 'right'
    convention : {'start', 'end', 'e', 's'}
        If axis is PeriodIndex
    """

    def _adjust_bin_edges(self, binner, ax_values):
        # Please ignore the body of this function

    def _get_time_bins(self, ax):
        if not isinstance(ax, DatetimeIndex):
            raise TypeError(
                "axis must be a DatetimeIndex, but got "
                f"an instance of {type(ax).__name__}"
            )
    
        if len(ax) == 0:
            binner = labels = DatetimeIndex(data=[], freq=self.freq, name=ax.name)
            return binner, [], labels
    
        first, last = self._get_date_range_edges(ax.min(), ax.max())
        binner, labels = self._generate_bins_labels(self.freq, first, last, ax.tz, ax.name)
        bin_edges = self._adjust_bin_edges(binner, ax.asi8)
        bins = lib.generate_bins_dt64(ax.asi8, bin_edges, self.closed, hasnans=ax.hasnans)
    
        if self.closed == "right":
            labels = binner if self.label != "right" else labels[1:]
        elif self.label == "right":
            labels = labels[1:]
    
        binner, labels = self._handle_nans(binner, labels, ax.hasnans)
    
        if len(bins) < len(labels):
            labels = labels[:len(bins)]
    
        return binner, bins, labels

    def _get_date_range_edges(self, start, end):
        return start.floor(self.freq), end.floor(self.freq) + self.freq

    def _generate_bins_labels(self, freq, start, end, tz, name):
        binner = labels = date_range(freq=freq, start=start, end=end, tz=tz, name=name, ambiguous="infer", nonexistent="shift_forward")
        return binner, labels

    def _handle_nans(self, binner, labels, has_nans):
        if has_nans:
            binner = binner.insert(0, NaT)
            labels = labels.insert(0, NaT)
        return binner, labels
