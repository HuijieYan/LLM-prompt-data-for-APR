    def _get_time_bins(self, ax):

            )

        start = ax.min().floor('D')
        end = ax.max().ceil('D') + Period('1D') 
        # Generate the range of dates based on frequency, start, and end
        binner = labels = date_range(start=start, end=end, freq=self.freq, tz=ax.tz, name=ax.name, closed=self.closed)
