def _convert_key(self, key, is_setter: bool = False):
    """
    Require they keys to be the same type as the index. (so we don't
    fallback)
    """
    # allow arbitrary setting
    if is_setter:
        if not is_list_like(key):
            key = [key]
        return key

    for ax, i in zip(self.obj.axes, key):
        if is_numeric_dtype(ax):
            if not is_numeric_dtype(i):
                raise ValueError(
                    f"At based indexing on an {ax.dtype} index "
                    f"can only have {ax.dtype} indexers"
                )
        else:
            if is_numeric_dtype(i) and not is_list_like(ax):
                raise ValueError(
                    f"At based indexing on a non-{ax.dtype} "
                    f"index can only have non-{ax.dtype} "
                    "indexers"
                )
    return key
