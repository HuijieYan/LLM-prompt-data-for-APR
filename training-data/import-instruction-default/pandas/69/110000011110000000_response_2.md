def _convert_key(self, key, is_setter: bool = False):
    """
    Require they keys to be the same type as the index. (so we don't
    fallback)

    Parameters
    ----------
    key : object
        The key to convert.
    is_setter : bool, default False
        Indicates whether the key is for setting.

    Returns
    -------
    converted_key : object
        The converted key.

    Raises
    ------
    ValueError
        If based indexing on an integer index can only have integer indexers or
        based indexing on a non-integer index can only have non-integer indexers.
    """
    # allow arbitrary setting
    if is_setter:
        return list(key)

    for ax, i in zip(self.obj.axes, key):
        if is_numeric_dtype(ax):
            if is_integer(i) and not is_integer(ax):
                raise ValueError(
                    "At based indexing on an integer index can only have integer indexers"
                )
            elif is_float(i) and not is_float(ax):
                raise ValueError(
                    "At based indexing on a non-integer index can only have non-integer indexers"
                )

    return key
