def _convert_key(self, key, is_setter: bool = False):
    """
    Require they keys to be the same type as the index. (so we don't
    fallback)
    """
    from pandas.core.dtypes.common import is_integer, is_sequence
    # allow arbitrary setting
    if is_setter:
        return list(key)

    for ax, i in zip(self.obj.axes, key):
        if is_numeric_dtype(ax.dtype) and not is_sequence(i) and not is_list_like(i):
            raise ValueError(
                "At based indexing on an numeric index can only have numeric or sequence indexers"
            )
        elif is_numeric_dtype(ax.dtype) and is_sequence(i) and not all(is_numeric_dtype(t) or is_iterator(t) for t in i):
            raise ValueError(
                "At based indexing on an numeric index can only have numeric or sequence indexers"
            ) 
        elif not is_numeric_dtype(ax.dtype) and not is_integer(i):
            raise ValueError(
                "At based indexing on a non-numeric index can only have non-numeric indexers"
            )
    return key
