def _convert_key(self, key, is_setter: bool = False):
    """
    Require they keys to be the same type as the index. (so we don't
    fallback)
    """
    from pandas.core.dtypes.common import is_float, is_integer, is_iterator, is_list_like, is_numeric_dtype, is_scalar, is_sequence

    # allow arbitrary setting
    if is_setter:
        return list(key)

    for ax, i in zip(self.obj.axes, key):
        if is_numeric_dtype(ax):
            if not is_numeric_dtype(i):
                raise ValueError(
                    f"At based indexing on {ax} index can only have {ax} indexers"
                )
        else:
            if not is_numeric_dtype(i) and not is_list_like(ax):
                raise ValueError(
                    f"At based indexing on {ax} index can only have non-{ax} indexers"
                )
    return key
