from pandas.core.dtypes.common import is_float, is_integer, is_iterator, is_list_like, is_numeric_dtype, is_scalar, is_sequence

def _convert_key(self, key, is_setter: bool = False):
    """
    Require they keys to be the same type as the index. (so we don't
    fallback)
    """
    # allow arbitrary setting
    if is_setter:
        return list(key)

    for ax, i in zip(self.obj.axes, key):
        if is_numeric_dtype(ax.dtype):
            if is_integer(i) and not is_integer(ax.dtype.type):
                raise ValueError(
                    "At based indexing on an integer index "
                    "can only have integer indexers"
                    )
            elif is_float(i) and not is_float(ax.dtype.type):
                raise ValueError(
                    "At based indexing on an non-integer "
                    "index can only have non-integer "
                    "indexers"
                    )
        else:
            if is_sequence(i):
                for ind in i:
                    _ = ax.isin([ind])
            else:
                _ = ax.isin([i])
    return key
