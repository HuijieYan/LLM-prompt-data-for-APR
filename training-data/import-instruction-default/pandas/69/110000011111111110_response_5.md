def _convert_key(self, key, is_setter: bool = False):
    """
    Require they keys to be the same type as the index. (so we don't
    fallback)
    """
    if is_setter:
        return list(key)

    if any([self.obj.axes[0].is_integer(), self.obj.axes[1].is_integer()]):
        if not is_integer(key[0]):
            raise ValueError(
                "At based indexing on an integer index "
                "can only have integer indexers"
            )
    else:
        if is_integer(key[0]):
            if not self.obj.axes[0].holds_integer() and not self.obj.axes[1].holds_integer():
                raise ValueError(
                    "At based indexing on an non-integer "
                    "index can only have non-integer "
                    "indexers"
                )

    return key
