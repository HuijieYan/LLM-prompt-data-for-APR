def _convert_key(self, key, is_setter: bool = False):
    """
    Require they keys to be the same type as the index. (so we don't
    fallback)
    """
    # allow arbitrary setting
    if is_setter:
        return list(key)

    for ax, i in zip(self.obj.axes, key):
        # Check if the index is of numeric type
        if is_numeric_dtype(ax):
            # Check if the index is integer type
            if is_integer(ax):
                # Check if the key is not an integer
                if not is_integer(i):
                    raise ValueError("At based indexing on an integer index can only have integer indexers")
            else:
                # Check if the key is integer and index is non-integer
                if is_integer(i) and not is_integer(ax):
                    raise ValueError("At based indexing on an non-integer index can only have non-integer indexers")
    return key
