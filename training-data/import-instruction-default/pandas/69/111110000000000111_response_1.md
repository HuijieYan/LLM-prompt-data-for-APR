@Appender(IndexingMixin.at.__doc__)
class _AtIndexer(_ScalarAccessIndexer):

    def _convert_key(self, key, is_setter: bool = False):
        """
        Require they keys to be the same type as the index. (so we don't
        fallback)
        """
        if is_setter:
            if is_iterator(key) or is_list_like(key):
                key = list(key)
            else:
                key = [key]
    
        for ax, i in zip(self.obj.axes, key):
            if is_numeric_dtype(ax):
                if not is_numeric_dtype(i):
                    raise ValueError(
                        f"At based indexing on a {ax.dtype} index "
                        f"can only have {ax.dtype} indexers"
                    )
            else:
                if is_numeric_dtype(i) and not is_numeric_dtype(ax):
                    raise ValueError(
                        f"At based indexing on a {ax.dtype} "
                        f"index can only have {ax.dtype} "
                        "indexers"
                    )
        return key
