def _convert_key(self, key, is_setter: bool = False):
    """
    Require they keys to be the same type as the index. (so we don't
    fallback)
    """
    # allow arbitrary setting
    if is_setter:
        return list(key)

    for ax, i in zip(self.obj.axes, key):
        if isinstance(ax, pd.tseries.index.DatetimeIndex):
            if not is_float(i):
                raise ValueError(
                    "At based indexing on an integer index "
                    "can only have integer indexers"
                )
        elif isinstance(ax, pd.tseries.index.TimedeltaIndex):
            if not is_numeric_dtype(ax.dtype) or not is_float(i):
                raise ValueError(
                    "At based indexing on an non-integer "
                    "index can only have non-integer "
                    "indexers"
                )
        else:
            if not is_list_like(ax) or not is_iterator(ax) or not is_numeric_dtype(ax.dtype) or not is_sequence(ax) or not is_scalar(ax) or not is_float(ax) or not is_integer(ax):
                raise ValueError(
                    "At based indexing on an non-integer "
                    "index can only have non-integer "
                    "indexers"
                )
    return key
