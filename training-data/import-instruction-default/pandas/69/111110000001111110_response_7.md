@Appender(IndexingMixin.at.__doc__)
class _AtIndexer(_ScalarAccessIndexer):
    def _convert_key(self, key, is_setter: bool = False):
        """
        Require they keys to be the same type as the index. (so we don't
        fallback)
        """
        # allow arbitrary setting
        if is_setter:
            return list(key)
    
        for ax, i in zip(self.obj.axes, key):
            if is_numeric_dtype(ax):
                if is_float(i) and not is_float(ax):
                    raise ValueError(
                        "At based indexing on an non-float index "
                        "can only have non-float indexers"
                    )
                elif is_integer(i) and not is_integer(ax):
                    raise ValueError(
                        "At based indexing on an integer index "
                        "can only have integer indexers"
                    )
            else:
                if is_sequence(i) and not is_list_like(ax):
                    raise ValueError(
                        "At based indexing on an non-list-like "
                        "index can only have non-list-like "
                        "indexers"
                    )
                elif is_scalar(i) and not is_scalar(ax):
                    raise ValueError(
                        "At based indexing on an non-scalar "
                        "index can only have non-scalar "
                        "indexers"
                    )
        return key
