    def _convert_key(self, key, is_setter: bool = False):
        """
        Require they keys to be the same type as the index. (so we don't
        fallback)
        """
        # allow arbitrary setting
        if is_setter:
            return list(key)
    
        for ax, i in zip(self.obj.axes, key):
            if is_numeric_dtype(ax):
                i_is_float = is_float(i)
                if i_is_float and not is_float(ax):
                    raise ValueError(
                        "At based indexing on a non-float index "
                        "can only have non-float indexers"
                    )
                if not i_is_float and is_float(ax):
                    raise ValueError(
                        "At based indexing on a float index "
                        "can only have float indexers"
                    )
            elif is_list_like(ax):
                raise ValueError(
                    "At based indexing on a list-like index "
                    "is not supported"
                )
            elif is_scalar(ax):
                raise ValueError(
                    "At based indexing on a scalar index "
                    "is not supported"
                )
            else:
                if is_integer(i):
                    raise ValueError(
                        "At based indexing on an non-integer "
                        "index can only have non-integer "
                        "indexers"
                    )
        return key
