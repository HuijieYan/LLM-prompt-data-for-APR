# The corrected version of the function
def _convert_key(self, key, is_setter: bool = False):
    """
    Require they keys to be the same type as the index. (so we don't
        fallback)
    """
    # allow arbitrary setting
    if is_setter:
        return list(key)

    for ax, i in zip(self.obj.axes, key):
        if pd.api.types.is_float(ax[0]):
            if not pd.api.types.is_float(i):
                raise ValueError(
                    "At based indexing on an float index "
                    "can only have float indexers"
                )
        else:
            if pd.api.types.is_float(i) and not pd.api.types.is_numeric_dtype(ax):
                raise ValueError(
                    "At based indexing on an non-float "
                    "index can only have non-float "
                    "indexers"
                )
    return key
