def _convert_key(self, key, is_setter: bool = False):
    """
    Require they keys to be the same type as the index. (so we don't
    fallback)
    """
    # allow arbitrary setting
    if is_setter:
        return list(key)

    for ax, i in zip(self.obj.axes, key):
        if not is_sequence(ax) and not is_iterator(ax):
            ax = [ax]

        if is_numeric_dtype(ax[0]) and not is_numeric_dtype(i):
            raise ValueError(
                f"At based indexing on an {ax[0].kind} index can only have {ax[0].kind} indexers"
            )
        elif not is_numeric_dtype(ax[0]) and is_numeric_dtype(i):
            raise ValueError(
                f"At based indexing on a non-{ax[0].kind} index can only have non-{ax[0].kind} indexers"
            )

    return key
