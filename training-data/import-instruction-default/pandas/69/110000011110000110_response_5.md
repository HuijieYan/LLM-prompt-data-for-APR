from pandas.core.dtypes.common import is_float, is_integer, is_iterator, is_list_like, is_numeric_dtype, is_scalar, is_sequence

def _convert_key(self, key, is_setter: bool = False):
    """
    Require they keys to be the same type as the index. (so we don't
    fallback)
    """
    # allow arbitrary setting
    if is_setter:
        return list(key)

    for ax, i in zip(self.obj.axes, key):
        if is_numeric_dtype(ax.dtype):
            ax_is_integer = is_integer(ax.dtype)
            ax_is_float = is_float(ax.dtype)
            if ax_is_integer:
                if not is_integer(i):
                    raise ValueError(
                        f"At based indexing on an integer index "
                        f"can only have integer indexers"
                    )
            elif ax_is_float:
                if not is_scalar(i):
                    raise ValueError(
                        f"At based indexing on a non-integer "
                        f"index can only have non-integer "
                        f"indexers"
                    )
            elif is_sequence(i):
                if is_list_like(i) or is_iterator(i):
                    raise ValueError(
                        f"At based indexing on a non-integer index "
                        f"can only have non-integer indexers"
                    )
    return key
