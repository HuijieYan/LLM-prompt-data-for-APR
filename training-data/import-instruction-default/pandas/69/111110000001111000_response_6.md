# The corrected version of the buggy function
@Appender(IndexingMixin.at.__doc__)
class _AtIndexer(_ScalarAccessIndexer):


    def _convert_key(self, key, is_setter: bool = False):
        """
        Require they keys to be the same type as the index. (so we don't
        fallback)
        """
        # allow arbitrary setting
        if is_setter:
            return list(key)
        
        for ax, i in zip(self.obj.axes, key):
            if ax.is_floating() and not is_float(i):
                raise ValueError(
                    "At based indexing on an floating index "
                    "can only have floating indexers"
                )
            if ax.is_integer() and not is_integer(i):
                raise ValueError(
                    "At based indexing on an integer index "
                    "can only have integer indexers"
                )
            if ax.is_numeric() and not is_numeric_dtype(i):
                raise ValueError(
                    "At based indexing on a numeric index "
                    "can only have numeric indexers"
                )
            if ax.is_categorical() and is_list_like(i):
                raise ValueError(
                    "At based indexing on a categorical index "
                    "can only have single label indexers"
                )
            if ax.is_sequence() and not is_iterator(i):
                raise ValueError(
                    "At based indexing on a sequence index "
                    "can only have sequence indexers"
                )
        return key
