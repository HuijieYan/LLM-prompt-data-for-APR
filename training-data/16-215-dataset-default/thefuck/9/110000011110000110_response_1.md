def get_new_command(command):
    # If --set-upstream or -u are passed, remove it and its argument. This is
    # because the remaining arguments are concatenated onto the command suggested
    # by git, which includes --set-upstream and its argument
    script_parts = command.script.split()
    if '--set-upstream' in script_parts:
        upstream_option_index = script_parts.index('--set-upstream')
        script_parts.pop(upstream_option_index)
        script_parts.pop(upstream_option_index)
    elif '-u' in script_parts:
        upstream_option_index = script_parts.index('-u')
        script_parts.pop(upstream_option_index)
        script_parts.pop(upstream_option_index)
    
    push_upstream = command.stderr.split('\n')[-3].strip().partition('git ')[2]
    result = " ".join(script_parts)
    if 'push' in result:
        result = result.replace('push', f'push --set-upstream {push_upstream}')
    else:
        result = f'git push --set-upstream {push_upstream} {result}'

    return result
