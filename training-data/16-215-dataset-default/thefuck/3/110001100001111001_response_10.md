from subprocess import Popen, PIPE, DEVNULL

def info(self):
    """Returns the name and version of the current shell"""
    proc = Popen(['fish', '-c', 'echo "Fish Shell $FISH_VERSION"'],
                 stdout=PIPE, stderr=DEVNULL)

    return version
