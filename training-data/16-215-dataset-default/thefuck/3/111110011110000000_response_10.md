from subprocess import Popen, PIPE, DEVNULL
from typing import Generic

class Fish(Generic):

    def info(self):
        """Returns the name and version of the current shell"""
        proc = Popen(['fish', '--version'], stdout=PIPE, stderr=DEVNULL)  # fixed command to get fish version
        version = proc.stdout.read().decode('utf-8').strip().split(', ')[1]  # extract version from output
        return f'Fish Shell {version}'  # use f-string for formatting

# Fixed test function
def test_info(self, shell, Popen):
    Popen.return_value.stdout.read.return_value = b'fish, version 3.5.9\n'  # fixed return value
    assert shell.info() == 'Fish Shell 3.5.9'
    assert Popen.call_args[0][0] == ['fish', '--version']  # ensure the correct command is called
