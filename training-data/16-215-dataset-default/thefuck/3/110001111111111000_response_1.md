def info(self):
    """Returns the name and version of the current shell"""
    proc = Popen(['fish', '-c', 'echo $FISH_VERSION'],
                 stdout=PIPE, stderr=DEVNULL)
    version = proc.stdout.readline().decode('utf-8').strip()  # Use readline instead of read
    return u'Fish Shell {}'.format(version.replace('fish, version ', ''))  # Remove "fish, version" from the version string
