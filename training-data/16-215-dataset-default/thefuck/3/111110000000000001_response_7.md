from subprocess import PIPE, DEVNULL, Popen
from typing import Generic

class Fish(Generic):

    def info(self):
        """Returns the name and version of the current shell"""
        proc = subprocess.Popen(['fish', '-c', 'echo $FISH_VERSION'],
