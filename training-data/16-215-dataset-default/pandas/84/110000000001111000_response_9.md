```python
def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data

    # NOTE: This doesn't deal with hierarchical columns yet

    index = data.index

    clocs = [index._get_level_number(i) for i in clocs]

    rlocs = [i for i in range(index.nlevels) if i not in clocs]

    clevels = [index.levels[i] for i in clocs]
    ccodes = [index.codes[i] for i in clocs]
    cnames = [index.names[i] for i in clocs]
    rlevels = [index.levels[i] for i in rlocs]
    rcodes = [index.codes[i] for i in rlocs]
    rnames = [index.names[i] for i in rlocs]

    shape = [len(x) for x in clevels]
    group_index = get_group_index(ccodes, shape, sort=False, xnull=False)

    comp_ids, obs_ids = compress_group_index(group_index, sort=False)
    recons_codes = decons_obs_group_ids(comp_ids, obs_ids, shape, ccodes, xnull=False)
    
    new_columns = [None] * len(ccodes)
    for i in range(len(ccodes)):
        new_columns[i] = make_new_index(ccodes[i], clevels[i], obs_ids)
    
    new_index = reconstruct_index(comp_ids, rlocs, rlevels, fill_value, obs_ids)
    new_index = new_index.set_names(rnames, inplace=True)

    unstacked = unstack_data(data, fill_value, new_columns, cnames, recons_codes)
    unstacked.index = new_index

    return unstacked
```
The make_new_index function constructs the new level index based on the compressed data. The reconstruct_index function constructs a new hierarchical index based on the compressed data, filtered from the existing index using rlocs, or reconstructed based on the shape parameter. The unstack_data function then takes the preprocessed data, columns, and levels to create new multiindex data.
