import pandas as pd
from pandas.core.reshape.reshape import get_dummies
from pandas.core.dtypes.common import notna
from pandas.core.groupby.grouper import _get_data_to_unstack
from pandas.core.dtypes.common import (
    is_bool_indexer,
    is_extension_array_dtype,
    is_list_like,
    is_scalar,
    is_dtype_equal,
)

def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data
    
    clocs = list(clocs)
    index = data.index

    rlocs = [i for i in range(index.nlevels) if i not in clocs]

    reshaped = data.unstack(clocs, fill_value=fill_value)

    dummy = pd.DataFrame(index=reshaped.index)
    dummy[reshaped.columns] = reshaped
    reshaped = dummy

    unstacked = reshaped.stack(fill_value=fill_value)

    return unstacked
