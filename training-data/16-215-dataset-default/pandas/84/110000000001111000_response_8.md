def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data

    # extract the first level index name
    first_level_name = clocs[0][0]

    # unstack the data
    unstacked = data.unstack(level=first_level_name, fill_value=fill_value)

    return unstacked
