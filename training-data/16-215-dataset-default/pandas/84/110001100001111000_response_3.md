def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data

    index = data.index

    clocs = [index._get_level_number(i) for i in clocs]

    rlocs = [i for i in range(index.nlevels) if i not in clocs]

    clevels = [index.levels[i] for i in clocs]
    rlevels = [index.levels[i] for i in rlocs]

    shape = [len(x) for x in clevels]
    group_index = get_group_index(index.codes, shape, sort=False, xnull=False)

    comp_ids, obs_ids = compress_group_index(group_index, sort=False)
    recons_codes = decons_obs_group_ids(comp_ids, obs_ids, shape, index.codes, xnull=False)

    dummy_index = MultiIndex.from_product([rlevels, [obs_ids]], names=rlevels.names + ['__placeholder__'])

    if isinstance(data, Series):
        dummy = data.copy()
        dummy.index = dummy_index
    else:
        dummy = data.copy()
        dummy.index = dummy_index

    unstacked = dummy.unstack(fill_value=fill_value)
    
    if isinstance(data, DataFrame) and isinstance(unstacked, Series):
        unstacked = unstacked.unstack(fill_value=fill_value)

    new_columns = MultiIndex.from_product([unstacked.columns.levels[0], clevels[0]], names=[data.columns.name, clevels[0].name])
    
    if isinstance(unstacked, Series):
        unstacked = unstacked.to_frame()

    unstacked.columns = new_columns

    return unstacked
