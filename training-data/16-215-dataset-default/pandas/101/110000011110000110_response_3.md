import numpy as np
import pandas as pd
import pytest


def astype_nansafe(arr, dtype, copy=True, skipna=False):
    """
    Cast the elements of an array to a given dtype in a NaN-safe manner.

    Parameters
    ----------
    arr : ndarray
    dtype : np.dtype
    copy : bool, default True
        If False, a view will be attempted but may fail, if
        e.g. the item sizes don't align.
    skipna: bool, default False
        Whether or not we should skip NaN when casting as a string-type.

    Raises
    ------
    ValueError
        The dtype was a datetime64/timedelta64 dtype, but it had no unit.
    """

    if isinstance(dtype, pd.CategoricalDtype):
        return pd.Series(arr).astype(dtype)

    if not isinstance(dtype, np.dtype):
        dtype = np.dtype(dtype)

    if dtype.kind in ["U", "S"]:
        return arr.astype(dtype)

    elif np.issubdtype(dtype, np.datetime64):
        if np.issubdtype(arr.dtype, np.datetime64):
            if np.isnat(arr).any():
                return arr.astype(dtype)
            else:
                raise ValueError("Cannot convert NaT values to integer")
        else:
            raise TypeError(
                f"cannot astype a datetimelike from [{arr.dtype}] to [{dtype}]"
            )

    elif np.issubdtype(dtype, np.timedelta64):
        if np.issubdtype(arr.dtype, np.timedelta64):
            if np.isnat(arr).any():
                return arr.astype(dtype)
            else:
                raise ValueError("Cannot convert NaT values to integer")
        else:
            raise TypeError(
                f"cannot astype a datetimelike from [{arr.dtype}] to [{dtype}]"
            )

    elif np.issubdtype(arr.dtype, np.floating) and np.issubdtype(dtype, np.integer):
        if np.isnan(arr).any():
            raise ValueError("Cannot convert NaN values to integer")
        else:
            return arr.astype(dtype)

    elif np.issubdtype(arr.dtype, np.object_):  # noqa
        return arr.astype(dtype)

    else:
        return arr.astype(dtype, copy=copy)


@pytest.mark.parametrize("val", [np.datetime64("NaT"), np.timedelta64("NaT")])
@pytest.mark.parametrize("typ", [np.int64])
def test_astype_nansafe(val, typ):
    arr = np.array([val])

    msg = "Cannot convert NaT values to integer"
    with pytest.raises(ValueError, match=msg):
        astype_nansafe(arr, dtype=typ)
