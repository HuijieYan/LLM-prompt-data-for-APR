import pandas as pd
import numpy as np
import pandas.core.computation.expressions as expressions
from pandas.core.indexing import maybe_convert_ix

def dispatch_to_series(left, right, func, str_rep=None, axis=None):
    """
    Evaluate the frame operation func(left, right) by evaluating
    column-by-column, dispatching to the Series implementation.

    Parameters
    ----------
    left : DataFrame
    right : scalar or DataFrame
    func : arithmetic or comparison operator
    str_rep : str or None, default None
    axis : {None, 0, 1, "index", "columns"}

    Returns
    -------
    DataFrame
    """
    right = maybe_convert_ix(right, left)
    if np.isscalar(right) or isinstance(right, pd.core.generic.NDFrame) and right.shape[1] == left.shape[1]:
        def column_op(a, b):
            return a.apply(lambda x: func(x, b))

    elif isinstance(right, pd.core.generic.NDFrame) and right.shape == left.shape:
        def column_op(a, b):
            return a.apply(lambda x: func(x, b.loc[a.index]))

    else:
        raise NotImplementedError(right)

    new_data = left.apply(column_op)
    return new_data
