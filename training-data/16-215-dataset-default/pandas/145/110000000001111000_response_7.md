```python
import numpy as np
import pandas as pd
import pandas.core.computation.expressions as expressions
from pandas.api.types import is_scalar

def dispatch_to_series(left, right, func, str_rep=None, axis=None):
    """
    Evaluate the frame operation func(left, right) by evaluating
    column-by-column, dispatching to the Series implementation.

    Parameters
    ----------
    left : DataFrame
    right : scalar or DataFrame
    func : arithmetic or comparison operator
    str_rep : str or None, default None
    axis : {None, 0, 1, "index", "columns"}

    Returns
    -------
    DataFrame
    """
    
    # 처리하기 쉽도록 right 값을 zero dimension에서 꺼냄
    right = np.squeeze(right)
    
    # right가 scalar이거나 zero dimension Array 일 때 해당 값을 고정된 값을 리턴하도록 조정하기
    if is_scalar(right):
        right_dict = {i: right for i in range(len(left.columns))}
        new_data = expressions.evaluate(right_dict, str_rep, left, right)
        
    elif isinstance(right, pd.DataFrame):
        right_columns_matched = right.columns.equals(left.columns)
        right_index_matched = right.index.equals(left.index)

        if right_columns_matched and right_index_matched:
            new_data = left.apply(lambda a: func(a, right))

        elif right_columns_matched and axis == "columns":
            if right.index.equals(left.index):
                new_data = left.apply(lambda a: a.apply(func, b=right, axis=1))
            else:
                raise ValueError("Index of the right DataFrame should match the index of the left DataFrame")

        else:
            raise ValueError("Columns of the right DataFrame should match the columns of the left DataFrame")
    
    elif isinstance(right, pd.Series):
        if right.index.equals(left.index):
            new_data = left.apply(lambda a: a.apply(func, b=right))
        elif right.index.equals(left.columns):
            new_data = left.apply(lambda a: func(a, right))
        else:
            raise ValueError("Indexes of right Series should either match with index or columns of the left DataFrame")
    
    return new_data
```
