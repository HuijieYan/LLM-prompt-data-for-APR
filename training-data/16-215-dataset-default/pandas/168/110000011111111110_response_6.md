def _get_grouper(
    obj,
    key=None,
    axis=0,
    level=None,
    sort=True,
    observed=False,
    mutated=False,
    validate=True,
):
    group_axis = obj._get_axis(axis)

    if level is not None:
        if isinstance(group_axis, pd.MultiIndex):
            if isinstance(level, list):
                level = level[0]

            if key is None and pd.api.types.is_scalar(level):
                key = group_axis.get_level_values(level)
                level = None

        else:
            if pd.api.types.is_list_like(level):
                nlevels = len(level)
                if nlevels == 1:
                    level = level[0]
                elif nlevels == 0:
                    raise ValueError("No group keys passed!")
                else:
                    raise ValueError("multiple levels only valid with MultiIndex")

            if isinstance(level, str):
                if obj.index.name != level:
                    raise ValueError(f"level name {level} is not the name of the index")
            elif level > 0 or level < -1:
                raise ValueError("level > 0 or level < -1 only valid with MultiIndex")

            level = None
            key = group_axis

    if isinstance(key, pd.Grouper):
        if key.key is None:
            binner, grouper, obj = key._get_grouper(obj, validate=False)
            return grouper, [], obj
        else:
            return key, {key.key}, obj
    elif isinstance(key, pd.BaseGrouper):
        return key, [], obj
    else:
        is_tuple = isinstance(key, tuple)
        all_hashable = is_tuple and all(pd.api.types.is_hashable(k) for k in key)

        if is_tuple:
            if (all_hashable and key not in obj and set(key).issubset(obj)) or not all_hashable:
                msg = "Interpreting tuple 'by' as a list of keys, rather than a single key. Use 'by=[...]' instead of 'by=(...)'. In the future, a tuple will always mean a single key."
                warnings.warn(msg, FutureWarning, stacklevel=5)
                key = list(key)

        if not pd.api.types.is_list_like(key):
            keys = [key]
            match_axis_length = False
        else:
            keys = key
            match_axis_length = len(keys) == len(group_axis)

        any_callable = any(callable(g) or isinstance(g, dict) for g in keys)
        any_groupers = any(isinstance(g, pd.Grouper) for g in keys)
        any_arraylike = any(
            isinstance(g, (list, tuple, pd.Series, pd.Index, np.ndarray)) for g in keys
        )

            not any_callable
            and not any_arraylike

            group_axis, groupings, sort=sort, mutated=mutated
        )
