def _get_grouper(
    obj,
    key=None,
    axis=0,
    level=None,
    sort=True,
    observed=False,
    mutated=False,
    validate=True,
):
    group_axis = obj._get_axis(axis)

    def is_in_axis(key):
        if isinstance(obj.columns, pd.MultiIndex) and key in obj.columns:
            return True
        elif isinstance(key, (int, str)) and key in obj.index.names:
            return True
        else:
            return False

    def is_in_obj(gpr):
        try:
            return id(gpr) == id(obj[gpr.name])
        except Exception:
            return False
        
    if level is not None:
        levels = level if isinstance(level, (tuple, list)) else [level]
        
        if key is None and is_in_axis(level):
            key = group_axis.get_level_values(level)

        for level in levels:
            if level not in obj.index.names:
                raise ValueError("level name {} is not the name of the index".format(level))
        
        level = None

    if isinstance(key, (str, int)):
        key = [key]

    match_axis_length = len(key) == len(group_axis)

    if not (any(callable(g) or isinstance(g, dict) for g in key) or 
            any(isinstance(g, Grouper) for g in key) or 
            any(isinstance(g, (list, tuple, pd.Series, pd.Index, np.ndarray)) for g in key)):
        if isinstance(obj, pd.DataFrame):
            all_in_columns_index = all(g in obj.columns or g in obj.index.names for g in key)
        elif isinstance(obj, pd.Series):
            all_in_columns_index = all(g in obj.index.names for g in key)
            
        if not all_in_columns_index:
            key = [com.asarray_tuplesafe(key)]

    for i, gpr in enumerate(key):
        if is_in_obj(gpr):
            in_axis, name = True, gpr.name
            exclusions.append(name)
        else:
            raise KeyError(gpr)

    groupings = [Grouping(group_axis, g, obj=obj, name=name, level=None, sort=sort, observed=observed, in_axis=in_axis) for g, name in zip(key, exclusions)]
    
    if len(groupings) == 0 and len(obj):
        raise ValueError("No group keys passed!")

    grouper = BaseGrouper(group_axis, groupings, sort=sort, mutated=mutated)
    
    return grouper, exclusions, obj
