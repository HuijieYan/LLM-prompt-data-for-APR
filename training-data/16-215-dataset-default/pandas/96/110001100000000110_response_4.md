@apply_wraps
def apply(self, other):
    if isinstance(other, pd.Timestamp):
        n = self.n
        if n >= 0:
            if self.is_on_offset(other):
                offset = other
                while n > 0:
                    offset += self
                    if self.is_on_offset(offset):
                        n -= 1
                return offset
            else:
                return self._next_opening_time(other)
        else:
            if self.is_on_offset(other):
                offset = other
                while n < 0:
                    offset += self
                    if self.is_on_offset(offset):
                        n += 1
                return offset
            else:
                return self._prev_opening_time(other)

    else:
        raise TypeError("Only know how to combine business hour with datetime")
