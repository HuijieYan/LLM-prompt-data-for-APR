# this is the fixed function
def apply(self, other):
    if isinstance(other, pd.Timestamp):
        original_other = other
        if isinstance(other, pd.DatetimeIndex):
            other = other[0]

        other_infer = other
        if self.start > self.end:
            if other.time() in self.end:
                # adjustment to move to previous business day
                other = other - pd.Timedelta(seconds=1)

        if other.time() in self.start:
            other = original_other + pd.Timedelta(minutes=1)
            if other_infer in self.index:
                pass
            else:
                other = original_other
            while other.time() not in self.start:
                other = other + pd.Timedelta(minutes=1)
            return other

        # calendar days
        if self.day.offset == 0:
            return other

        orig_day = other.day
        while True:
            other = other + self.day
            if other.day == orig_day:
                break

        while self.holiday.get(key=other.date()):
            other = other + self.day

        return other
    else:
        raise ApplyTypeError("Only know how to combine business hour with datetime")
