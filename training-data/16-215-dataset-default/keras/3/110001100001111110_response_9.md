def _clone_functional_model(model, input_layers=None):
    """Clone a functional `Model` instance.

    Model cloning is similar to calling a model on new inputs,
    except that it creates new layers (and thus new weights) instead
    of sharing the weights of the existing layers.

    # Arguments
        model: Instance of `Model`.
        input_layers: optional list of input layers
            to build the model upon. If not provided,
            placeholders will be created.

    # Returns
        An instance of `Model` reproducing the behavior
        of the original model, on top of new input layers,
        using newly instantiated weights.

    # Raises
        ValueError: in case of invalid `model` argument value.
    """

    # Check if the model is an instance of Model or Sequential
    if not isinstance(model, Model):
        raise ValueError('Expected `model` argument to be a `Model` instance, got {}'.format(model))

    # Check if the model is an instance of Sequential
    if isinstance(model, Sequential):
        raise ValueError('Expected `model` argument to be a functional `Model` instance, got a `Sequential` instance instead: {}'.format(model))

    layer_map = {}  # Cache for created layers.
    tensor_map = {}  # Map {reference_tensor: (corresponding_tensor, mask)}

    # Create placeholders to build the model on top of if no input layers are provided
    if input_layers is None:
        input_layers = [Input(batch_shape=layer.input_shape, dtype=layer.dtype, sparse=layer.sparse, name=layer.name) for layer in model._input_layers]

        for original, cloned in zip(model._input_layers, input_layers):
            layer_map[original] = cloned

    for layer in model.layers:
        if layer not in layer_map:
            # Clone layer
            new_layer = layer.__class__.from_config(layer.get_config())
            layer_map[layer] = new_layer

    for original, cloned in zip(model._input_layers, input_layers):
        layer_map[original] = cloned

    # Iterate over every node in the reference model, in depth order.
    depth_keys = list(model._nodes_by_depth.keys())
    depth_keys.sort(reverse=True)
    for depth in depth_keys:
        nodes = model._nodes_by_depth[depth]
        for node in nodes:
            layer = node.outbound_layer
            if layer not in layer_map:
                # Clone layer
                new_layer = layer.__class__.from_config(layer.get_config())
                layer_map[layer] = new_layer

            reference_input_layers = node.inbound_layers

            computed_data = []  # List of tuples (input, mask).
            for x in reference_input_layers:
                if x in tensor_map:
                    computed_data.append(tensor_map[x])

            if len(computed_data) == len(reference_input_layers):
                input_data = [x[0] for x in computed_data]
                kwargs = {} if not node.arguments else node.arguments

                output_tensor = layer(input_data, **kwargs)
                tensor_map[layer] = output_tensor

    # Check that we did compute the model outputs, then instantiate a new model from inputs and outputs.
    output_tensors = [tensor_map[x] for x in model.outputs]
    return Model(input_layers, output_tensors, name=model.name)
