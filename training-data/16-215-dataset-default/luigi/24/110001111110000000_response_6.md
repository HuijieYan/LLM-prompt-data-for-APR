def _dict_arg(self, name, value):
    command = []
    if value and isinstance(value, dict):
        for prop, value_item in value.items():  # Changed the variable name from 'value' to 'value_item' to avoid conflict
            command += [name, '"{0}={1}"'.format(prop, value_item)]  # Changed the variable name from 'value' to 'value_item' to avoid conflict
    return command
