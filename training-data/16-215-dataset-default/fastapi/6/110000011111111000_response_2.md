from fastapi import Form
from starlette.datastructures import FormData

def request_body_to_args(
    required_params: List[ModelField],
    received_body: Optional[Union[Dict[str, Any], FormData]],
) -> Tuple[Dict[str, Any], List[ErrorWrapper]]:
    values = {}
    errors = []
    for field in required_params:
        field_info = get_field_info(field)
        if isinstance(received_body, FormData) and field.alias in received_body.keys():
            value = received_body.getlist(field.alias) if field.shape in sequence_shapes else received_body.get(field.alias)
        else:
            value = None
        if value is None and field.required:
            errors.append(
                ErrorWrapper(MissingError(), loc=("body", field.alias))
            )
        else:
            values[field.alias] = value
    return values, errors
