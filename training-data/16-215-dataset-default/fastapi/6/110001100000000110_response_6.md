from typing import List, Optional, Union, Dict, Any, Tuple
from fastapi.params import File, Form, ModelField
from fastapi.datastructures import UploadFile
from pydantic.error_wrappers import ErrorWrapper
from pydantic.fields import SHAPE_SEQUENCE
from pydantic.main import BaseConfig
from pydantic.schema import get_field_info
from copy import deepcopy


async def request_body_to_args(
        required_params: List[ModelField],
        received_body: Optional[Union[Dict[str, Any], FormData]],
) -> Tuple[Dict[str, Any], List[ErrorWrapper]]:
    values = {}
    errors = []
    if required_params:
        for field in required_params:
            value: Any = None
            field_info = get_field_info(field)
            shape = field_info.shape
            if received_body is not None:
                if shape in SHAPE_SEQUENCE and isinstance(received_body, FormData):
                    value = received_body.getlist(field.alias)
                else:
                    value = received_body.get(field.alias)
            if value is None or (
                    isinstance(field_info, Form) and value == ""
            ) or (
                    isinstance(field_info, Form) and shape in SHAPE_SEQUENCE and len(value) == 0
            ):
                if field.required:
                    errors.append(
                        ErrorWrapper(MissingError(), loc=("body", field.alias))
                    )
                else:
                    values[field.name] = deepcopy(field.default)
                continue
            if (
                    isinstance(field_info, File)
                    and lenient_issubclass(field.type_, bytes)
                    and isinstance(value, UploadFile)
            ):
                value = await value.read()
            elif (
                    shape in SHAPE_SEQUENCE
                    and isinstance(field_info, File)
                    and lenient_issubclass(field.type_, bytes)
                    and isinstance(value, sequence_types)
            ):
                awaitables = [sub_value.read() for sub_value in value]
                contents = await asyncio.gather(*awaitables)
                value = sequence_shape_to_type[shape](contents)
            v_, errors_ = field.validate(value, values, loc=("body", field.alias))
            if isinstance(errors_, ErrorWrapper):
                errors.append(errors_)
            elif isinstance(errors_, list):
                errors.extend(errors_)
            else:
                values[field.name] = v_
    return values, errors
