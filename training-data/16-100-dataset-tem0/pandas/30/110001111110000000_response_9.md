import pandas as pd
import numpy as np
from pandas.core.dtypes.common import is_datetime64_any_dtype as is_datetime
from pandas.core.dtypes.missing import isna
from pandas.core.dtypes.common import iNaT
from pandas.core.tools.datetimes import to_datetime

class JSONReader:
    def __init__(self, data):
        self.data = data
        self.min_stamp = pd.Timestamp.min.value
        self.date_unit = None
        self._STAMP_UNITS = ['ns']

    def read(self):
        return self._get_object_parser(self.data)

    def _get_object_parser(self, data):
        return SeriesParser(data).parse()

class SeriesParser:
    def __init__(self, data):
        self.data = data

    def parse(self):
        self._try_convert_types()
        return self.data

    def _try_convert_types(self):
        self._try_convert_data()

    def _try_convert_data(self):
        new_data, result = self._try_convert_to_date(self.data)
        return new_data, result

    def _try_convert_to_date(self, data):
        if not len(data):
            return data, False

        new_data = data
        if new_data.dtype == "object":
            try:
                new_data = data.astype("int64")
            except (TypeError, ValueError, OverflowError):
                pass

        if issubclass(new_data.dtype.type, np.number):
            in_range = (
                isna(new_data._values)
                | (new_data > self.min_stamp)
                | (new_data._values == iNaT)
            )
            if not in_range.all():
                return data, False

        date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
        for date_unit in date_units:
            try:
                new_data = to_datetime(new_data, errors="raise", unit=date_unit)
            except (ValueError, OverflowError):
                continue
            return new_data, True
        return data, False

def read_json(data, typ):
    json_reader = JSONReader(data)
    result = json_reader.read()
    return result

def test_readjson_bool_series():
    # GH31464
    result = read_json("[true, true, false]", typ="series")
    expected = pd.Series([True, True, False])
    assert result.equals(expected)
