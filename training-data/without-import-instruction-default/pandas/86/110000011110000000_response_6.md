import pandas as pd
import pytest
from pandas.core.indexes.api import _shared_docs
from pandas._libs import tslib

def Substitution(arg):
    def dec(obj):
        obj.__doc__ = obj.__doc__.replace("\ndata : DataFrame", arg)
        return obj
    return dec

def Appender(data, indents):
    def dec(obj):
        obj.__doc__ += "\n" + data['pivot'].replace("\n", "\n" + " " * indents)
        return obj
    return dec

def pivot(data: "DataFrame", index=None, columns=None, values=None):
    """
    pivot(data, index=None, columns=None, values=None) -> DataFrame

    Return reshaped DataFrame organized by given index / column values.

    Parameters
    ----------
    data : DataFrame
    index : int or str, optional
        Column to use to make the new frame's index. If none, uses existing index.
    columns : int or str, optional
        Column to use to make the new frame's columns.
    values : int or str, optional
        Column(s) in the DataFrame to pivot to the column level.
    """
    if values is None:
        cols = [columns] if index is None else [index, columns]
        append = index is None
        indexed = data.set_index(cols, append=append)
    else:
        if index is None:
            index = data.index
        else:
            index = data[values]  # Fix assignment
        index = tslib.MultiIndex.from_arrays([data[index], data[columns]])  # Update MultiIndex.from_arrays call

        if isinstance(values, str):  # Check if values is a string
            indexed = pd.DataFrame(data[values].values, index=index, columns=[values])  # Reconstruct DataFrame
        elif isinstance(values, list):
            indexed = pd.DataFrame(data[values].values, index=index, columns=values)  # Reconstruct DataFrame
    return indexed.unstack(columns)
