import pandas as pd
import pytest

def pivot(data: "DataFrame", index=None, columns=None, values=None) -> "DataFrame":
    if columns is None:
        raise ValueError("The 'columns' argument is required in df.pivot")
    if values is None:
        cols = [columns] if index is None else [index, columns]
        append = index is None
        indexed = data.set_index(cols, append=append)
    else:
        if index is None:
            index = data.index
        else:
            index = data[index]
        index = pd.MultiIndex.from_arrays([index, data[columns]])
        
        if isinstance(values, str):
            indexed = data._constructor(data[values].values, index=index, columns=[values])
        else:
            indexed = data._constructor_sliced(data[values].values, index=index)
    return indexed.unstack(columns)

def test_pivot_columns_none_raise_error():
    # GH 30924
    df = pd.DataFrame(
        {"col1": ["a", "b", "c"], "col2": [1, 2, 3], "col3": [1, 2, 3]}
    )
    with pytest.raises(ValueError, match="The 'columns' argument is required in df.pivot"):
        pivot(df, index="col1", values="col3")
