import pandas as pd
import pytest
from pandas.core.reshape.pivot import Substitution, Appender

@Substitution("\ndata : DataFrame")
@Appender(_shared_docs["pivot"], indents=1)
def pivot(data: "DataFrame", index=None, columns=None, values=None) -> "DataFrame":
    if columns is None:
        raise TypeError("pivot() missing 1 required argument: 'columns'")        
    if values is None:
        cols = [columns] if index is None else [index, columns]
        append = index is None
        indexed = data.set_index(cols, append=append)
    else:
        if index is None:
            index = data.index
        else:
            index = data[index]
        index = pd.MultiIndex.from_arrays([index, data[columns]])
        if isinstance(values, str) or not hasattr(values, '__iter__'):
            indexed = pd.DataFrame(data[values].values, index=index, columns=[values])
        else:
            indexed = pd.DataFrame(data[values].values, index=index)
    return indexed.unstack(columns)

def test_pivot_columns_none_raise_error(self):
    df = pd.DataFrame({"col1": ["a", "b", "c"], "col2": [1, 2, 3], "col3": [1, 2, 3]})
    msg = r"pivot\(\) missing 1 required argument: 'columns'"
    with pytest.raises(TypeError, match=msg):
        pivot(data=df, index="col1", values="col3")
