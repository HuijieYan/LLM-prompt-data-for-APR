from pandas.core.reshape import pivot
from pandas.core.frame import DataFrame

def pivot_data(data: DataFrame, index=None, columns=None, values=None) -> DataFrame:
    if values is None:
        return pivot.pivot(data, index, columns)  # call the pivot function from pandas
    else:
        return pivot.pivot_table(data, index=index, columns=columns, values=values)  # call the pivot_table function from pandas
