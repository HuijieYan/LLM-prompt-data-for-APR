import numpy as np
import pytest

def astype_nansafe(arr, dtype, copy: bool = True, skipna: bool = False):
    import pandas as pd

    if not isinstance(dtype, np.dtype):
        dtype = pd.api.types.pandas_dtype(dtype)

    if np.issubdtype(dtype, np.datetime64):
        if np.issubdtype(arr.dtype, np.datetime64):
            if dtype == arr.dtype:
                return arr
            else:
                try:
                    return arr.astype(dtype)
                except ValueError:
                    if skipna and np.any(pd.isna(arr)):
                        return arr
                    raise

        if np.issubdtype(arr.dtype, np.timedelta64):
            try:
                return arr.view('i8').astype(dtype)
            except ValueError:
                if skipna and np.any(pd.isna(arr)):
                    return arr
                raise

    if np.issubdtype(dtype, np.timedelta64):
        if np.issubdtype(arr.dtype, np.timedelta64) and dtype == arr.dtype:
            return arr
        return arr.view('i8').astype(dtype)

    if np.issubdtype(dtype, np.floating) and np.issubdtype(arr.dtype, np.integer):
        if not np.isfinite(arr).all():
            raise ValueError("Cannot convert non-finite values (NA or inf) to integer")

    if dtype == np.dtype("datetime64") or dtype == np.dtype("timedelta64"):
        msg = (
            f"The '{dtype.name}' dtype has no unit. Please pass in "
            f"'{dtype.name}[ns]' instead."
        )
        raise ValueError(msg)
    return arr.astype(dtype, copy=copy)

@pytest.mark.parametrize("val", [np.datetime64("NaT"), np.timedelta64("NaT")])
@pytest.mark.parametrize("typ", [np.int64])
def test_astype_nansafe(val, typ):
    arr = np.array([val])

    msg = "Cannot convert NaT values to integer"
    with pytest.raises(ValueError, match=msg):
        astype_nansafe(arr, dtype=typ)
