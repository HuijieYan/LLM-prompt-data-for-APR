import numpy as np

_INT64_DTYPE = np.dtype('int64')
_TD_DTYPE = np.dtype('timedelta64[ns]')

def astype_nansafe(arr, dtype, copy: bool = True, skipna: bool = False):
    """
    Cast the elements of an array to a given dtype in a nan-safe manner.

    Parameters
    ----------
    arr : ndarray
    dtype : np.dtype
    copy : bool, default True
        If False, a view will be attempted but may fail, if
        e.g. the item sizes don't align.
    skipna: bool, default False
        Whether or not we should skip NaN when casting as a string-type.

    Raises
    ------
    ValueError
        The dtype was a datetime64/timedelta64 dtype, but it had no unit.
    """

    if np.issubdtype(dtype, np.integer):
        if skipna and np.issubdtype(arr.dtype, np.datetime64):
            return arr

        if np.issubdtype(arr.dtype, np.datetime64) and not skipna:
            return arr

        if np.issubdtype(arr.dtype, np.timedelta64):
            if np.issubdtype(dtype, np.datetime64):
                raise TypeError(f'Cannot convert from [{arr.dtype}] to [{dtype}]')
            if not np.isnat(arr).all():
                return arr.astype(dtype)

            if _INT64_DTYPE == dtype:
                if copy:
                    arr = arr.copy()
                arr[~arr.isnat] = arr[~arr.isnat].astype(dtype)
                return arr

            if _TD_DTYPE == dtype:
                return arr.astype(_TD_DTYPE)

            result = arr.astype(np.float64).astype(dtype)
            result[arr.isnat] = np.nan
            return result
    # Similar logic for other dtype conversions

    return arr.astype(dtype, copy=copy)
