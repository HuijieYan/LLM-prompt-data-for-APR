import numpy as np
import pandas as pd
import pytest


def astype_nansafe(arr, dtype, copy: bool = True, skipna: bool = False):
    """
    Cast the elements of an array to a given dtype in a nan-safe manner.

    Parameters
    ----------
    arr : ndarray
    dtype : np.dtype or Python type
        The data type to which the elements of the array should be cast.
    copy : bool, default True
        If False, a view will be attempted but may fail if the item sizes don't align.
    skipna : bool, default False
        Whether or not to skip NaN values when casting as a string-type.

    Raises
    ------
    ValueError
        The dtype was a datetime64/timedelta64 dtype, but it had no unit.

    Returns
    -------
    np.ndarray
        The array with elements cast to the specified dtype.
    """

    if isinstance(dtype, type(pd.CategoricalDtype())) and np.nan in arr:
        raise ValueError("Cannot convert CategoricalDtype containing NaNs to integer")

    if np.issubdtype(dtype, np.datetime64) or np.issubdtype(dtype, np.timedelta64):
        if dtype.kind == "M" or dtype.kind == "m":
            return arr.astype(dtype)
        else:
            dtype = f"{dtype.name}[ns]"
            return arr.astype(dtype)

    if np.issubdtype(arr.dtype, np.floating) and np.issubdtype(dtype, np.integer):
        if not np.isfinite(arr).all():
            raise ValueError("Cannot convert non-finite values (NA or inf) to integer")

    if copy or isinstance(arr.dtype, pd.StringDtype) or isinstance(dtype, pd.StringDtype):
        return arr.astype(dtype, copy=True)

    return arr.view(dtype)


# A passing test for the corrected function
@pytest.mark.parametrize("val", [np.datetime64("NaT"), np.timedelta64("NaT")])
@pytest.mark.parametrize("typ", [np.int64])
def test_astype_nansafe(val, typ):
    arr = np.array([val])

    msg = "Cannot convert CategoricalDtype containing NaNs to integer"
    with pytest.raises(ValueError, match=msg):
        astype_nansafe(arr, dtype=typ)
