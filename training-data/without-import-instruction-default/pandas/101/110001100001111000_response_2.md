import numpy as np
import pandas as pd

_INT64_DTYPE = np.dtype('int64')
_TD_DTYPE = np.dtype('timedelta64')
_M_DTYPE = np.dtype('datetime64[ns]')

def astype_nansafe(arr, dtype, copy: bool=True, skipna: bool=False):
    """
    Cast the elements of an array to a given dtype in a nan-safe manner.

    Parameters
    ----------
    arr : ndarray
    dtype : np.dtype or str
    copy : bool, default True
        If False, a view will be attempted but may fail, if
        e.g. the item sizes don't align.
    skipna: bool, default False
        Whether or not we should skip NaN when casting as a string-type.

    Raises
    ------
    ValueError
        The dtype was a datetime64/timedelta64 dtype, but it had no unit.
    """

    if isinstance(dtype, str):
        dtype = np.dtype(dtype)

    # dispatch on extension dtype if needed
    if pd.api.types.is_extension_array_dtype(dtype):
        return dtype.construct_array_type()._from_sequence(arr, dtype=dtype, copy=copy)

    if np.issubdtype(dtype, np.floating) and not np.issubdtype(arr.dtype, np.floating):
        arr = arr.astype(np.float64)

    if is_timedelta64_dtype(arr) or is_datetime64_dtype(arr):
        return _astype_datearr(arr, dtype, copy=copy)

    if dtype.name in ("datetime64", "timedelta64"):
        msg = (
            f"The '{dtype.name}' dtype has no unit. Please pass in "
            f"'{dtype.name}[ns]' instead."
        )
        raise ValueError(msg)

    if copy or pd.api.types.is_object_dtype(arr) or pd.api.types.is_object_dtype(dtype):
        # Explicit copy, or required since NumPy can't view from / to object.
        return arr.astype(dtype, copy=True)

    return arr.view(dtype)

def is_datetime64_dtype(arr):
    return arr.dtype.type is np.datetime64

def is_timedelta64_dtype(arr):
    return arr.dtype.type is np.timedelta64

def _astype_datearr(arr, dtype, copy):
    if is_timedelta64_dtype(arr):
        return _astype_timedelta(arr, dtype, copy=copy)
    elif is_datetime64_dtype(arr):
        return _astype_datetime(arr, dtype, copy=copy)

def _astype_timedelta(arr, dtype, copy):
    if pd.api.types.is_object_dtype(dtype):
        return pd.Series(arr).astype(dtype).values
    elif dtype == np.int64:
        return arr.view(dtype)
    elif dtype == _TD_DTYPE:
        return arr.astype(_TD_DTYPE, copy=copy)

    if dtype.kind == 'm':
        result = arr.astype(dtype)
        result[arr.isna()] = np.nan
        return result.astype(np.float64)

    raise TypeError(f"cannot astype a timedelta from [{arr.dtype}] to [{dtype}]")

def _astype_datetime(arr, dtype, copy):
    if pd.api.types.is_object_dtype(dtype):
        return pd.Series(arr).astype(dtype).values
    elif dtype == np.int64:
        return arr.view(dtype)

    if dtype.kind == 'M':
        return arr.astype(dtype)
    
    raise TypeError(f"cannot astype a datetimelike from [{arr.dtype}] to [{dtype}]")
