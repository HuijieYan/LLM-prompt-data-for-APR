import numpy as np
from numpy import datetime64, timedelta64

def astype_nansafe(arr, dtype, copy: bool = True, skipna: bool = False):
    """
    Cast the elements of an array to a given dtype a nan-safe manner.

    Parameters
    ----------
    arr : ndarray
    dtype : np.dtype
    copy : bool, default True
        If False, a view will be attempted but may fail, if
        e.g. the item sizes don't align.
    skipna: bool, default False
        Whether or not we should skip NaN when casting as a string-type.

    Raises
    ------
    ValueError
        The dtype was a datetime64/timedelta64 dtype, but it had no unit.
    """

    if np.issubdtype(arr.dtype, np.datetime64) or np.issubdtype(arr.dtype, np.timedelta64):

        if np.issubdtype(arr.dtype, np.datetime64):
            if not np.issubdtype(dtype, np.integer):
                raise TypeError(f"cannot astype a datetimelike from [{arr.dtype}] to [{dtype}]")
            elif dtype.kind != 'i' or dtype.itemsize != 8:
                raise ValueError("The dtype was a datetime64 dtype, but it had no unit.")

        elif np.issubdtype(arr.dtype, np.timedelta64):
            if not np.issubdtype(dtype, np.integer):
                raise TypeError(f"cannot astype a timedelta from [{arr.dtype}] to [{dtype}]")
            elif dtype.kind != 'i' or dtype.itemsize != 8:
                raise ValueError("The dtype was a timedelta64 dtype, but it had no unit.")
        
        if copy:
            return arr.astype(dtype, copy=True)
        else:
            return arr.view(dtype)
    
    else:
        if not isinstance(dtype, np.dtype):
            dtype = np.dtype(dtype)

        if np.issubdtype(dtype, np.floating) and np.issubdtype(arr.dtype, np.floating):
            if np.isnan(arr).any():
                arr = arr.astype(dtype)
                arr[np.isnan(arr)] = np.nan
                return arr
            else:
                return arr.astype(dtype)
        
        elif np.issubdtype(dtype, np.integer) and np.issubdtype(arr.dtype, np.floating):
            if np.isfinite(arr).all():
                return arr.astype(dtype)
            else:
                raise ValueError("Cannot convert non-finite values (NA or inf) to integer")
        
        else:
            if copy:
                return arr.astype(dtype, copy=True)
            else:
                return arr.view(dtype)
