from pandas import date_range, DatetimeIndex, NaT, Timestamp, lib
from pandas.tseries.frequencies import to_offset
from pandas.tseries.frequencies import Day

def _get_time_bins(self, ax):
    if not isinstance(ax, DatetimeIndex):
        raise TypeError(
            "axis must be a DatetimeIndex, but got "
            f"an instance of {type(ax).__name__}"
        )

    if len(ax) == 0:
        binner = labels = DatetimeIndex(data=[], freq=self.freq, name=ax.name)
        return binner, [], labels

    first = ax.min()
    last = ax.max()
    offset = to_offset(self.freq)
    first = offset.rollback(first)
    last = offset.rollback(last)
    bins = date_range(start=first, end=last, freq=self.freq, tz=ax.tz, name=ax.name, closed=self.closed, ambiguous='infer')
  
    bins = bins.append(bins[-1] + offset)
    bins = bins.insert(0, bins[0] - offset)

    if self.closed == "right":
        labels = bins[1:]
        if self.label == "left":
            labels = labels[:-1]
    else:
        labels = bins[:-1]
        if self.label == "right":
            labels = labels[1:]
    
    binner, bin_edges = self._adjust_bin_edges(bins, ax.asi8)

    if ax.hasnans:
        binner = binner.insert(0, NaT)
        labels = labels.insert(0, NaT)

    if len(bins) < len(labels):
        labels = labels[:len(bins)]

    return binner, lib.generate_bins_dt64(ax.asi8, bin_edges, self.closed, hasnans=ax.hasnans), labels
