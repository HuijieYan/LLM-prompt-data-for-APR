```python
from typing import List, Tuple
import numpy as np
import pandas as pd
from pandas.core.internals import Block

def _cython_agg_blocks(
    self, how: str, alt=None, numeric_only: bool = True, min_count: int = -1
) -> Tuple[List[Block], pd.Index]:
    # TODO: the actual managing of mgr_locs is a PITA
    # here, it should happen via BlockManager.combine

    data = self._get_data_to_aggregate()

    if numeric_only:
        data = data._get_numeric_data(copy=False)

    agg_blocks = []
    new_items = []
    deleted_items = []
    split_items = []
    split_frames = []

    no_result = object()
    for block in data.blocks:
        result = no_result
        locs = block.mgr_locs.as_array
        try:
            result, _ = self.grouper.aggregate(
                block.values, how, axis=1, min_count=min_count
            )
        except NotImplementedError:
            if alt is None:
                assert how == "ohlc"
                deleted_items.append(locs)
                continue
            obj = self.obj[data.items[locs]]
            if obj.shape[1] == 1:
                obj = obj.iloc[:, 0]
            s = data._get_grouper(obj, self.grouper)
            try:
                result = s.aggregate(lambda x: alt(x, axis=self.axis))
                result = result._validate(allow_object=True)
                blocks = result._data.blocks
                if not result._data.get_bool_indexer(result.columns, null_val=True).any():
                    result = blocks[0].get_values()
            except TypeError:
                deleted_items.append(locs)
                continue
            except ValueError:
                split_items.append(locs)
                split_frames.append(result._data.blocks[0])
                continue
            except Exception as e:
                raise e
            else:
                pass

        if not isinstance(result, (pd.DataFrame, pd.Series, np.ndarray, list, tuple)):
            if result is no_result:
                pass
            else:
                raise TypeError(
                    f"unsupported type: {type(result).__name__}"
                )

        if block._can_hold_element(result):
            result = pd._try_cast(result, block.values.dtype, copy=False)
            agg_block = block.make_block(result, placement=block.mgr_locs)
        else:
            raise TypeError(
                "invalid apply result at {0!r}".format(block)
            )

        new_items.append(np.array(list(block.mgr_locs)))

        agg_blocks.append(agg_block)

    if not (agg_blocks or split_frames):
        raise ValueError("No numeric types to aggregate")

    if len(split_items) > 0:
        for locs, result in zip(split_items, split_frames):
            result_block = result._data.blocks[0]
            assert len(locs) == result_block.shape[1]
            blklocs = result.from_element(locs)
            new_items.append(np.array(list(blklocs)))
            agg_blocks.append(result_block)

    indexer = np.concatenate(new_items)
    agg_items = data.items.take(np.sort(indexer))

    if len(deleted_items) > 0:
        di = np.concatenate(deleted_items)
        ai = np.arange(len(data.items))
        m = np.zeros(len(data.items))
        m[di] = 1
        indexer = (ai - m.cumsum())[indexer]

    offset = 0
    for block in agg_blocks:
        loc = len(block.mgr_locs)
        block.mgr_locs = indexer[offset:offset + loc]
        offset += loc

    return agg_blocks, agg_items
```
