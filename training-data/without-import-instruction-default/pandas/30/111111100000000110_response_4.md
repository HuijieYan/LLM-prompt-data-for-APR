import numpy as np
import pandas as pd
from pandas._libs.tslibs.nattype import iNaT
from pandas.api.types import isna
from pandas._libs.tslibs.timestamps import Timestamp, NaTType
from pandas.errors import OutOfBoundsDatetime

class Parser():

    def _try_convert_to_date(self, data):
        if not len(data):
            return data, False
        
        new_data = data
        if new_data.dtype == "object":
            try:
                new_data = data.astype("int64")
            except (TypeError, ValueError, OverflowError):
                pass
        
        if issubclass(np.min_scalar_type(new_data.dtype), np.number):
            in_range = (
                isna(new_data) | (new_data > self.min_stamp) | (new_data == iNaT)
            )
            if not in_range.all():
                return data, False
        
        date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
        for date_unit in date_units:
            try:
                new_data = pd.to_datetime(new_data, errors="raise", unit=date_unit, utc=self.utc)
            except (ValueError, OverflowError, OutOfBoundsDatetime):
                continue
            return new_data, True
        return data, False
