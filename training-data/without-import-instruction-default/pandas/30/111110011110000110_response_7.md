import numpy as np
import pandas as pd
from pandas.api.types import is_integer_dtype
from pandas.api.extensions import ExtensionDtype

class Parser():

    def _try_convert_to_date(self, data):
        if not len(data):
            return data, False

        new_data = data
        if new_data.dtype == "object":
            try:
                new_data = pd.to_datetime(new_data.astype("int64"), errors='coerce')
            except (TypeError, ValueError, OverflowError):
                pass

        if is_integer_dtype(new_data):
            in_range = (
                pd.isna(new_data)
                | (new_data > self.min_stamp)
            )
            if not in_range.all():
                return data, False

        if isinstance(new_data.dtype, ExtensionDtype):
            date_unit = self.date_unit if self.date_unit else 'ns'
            try:
                new_data = pd.to_datetime(new_data, errors='raise', unit=date_unit)
                return new_data, True
            except (ValueError, OverflowError):
                return data, False
        else:
            return data, False

parser = Parser()
print(parser._try_convert_to_date(pd.Series([True, True, False])))  # Output will depend on self.min_stamp and self.date_unit
