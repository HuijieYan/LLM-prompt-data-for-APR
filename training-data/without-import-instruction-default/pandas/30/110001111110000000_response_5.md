import pandas as pd
import numpy as np
from pandas.core.dtypes.missing import isna
from pandas.core import iNaT
from pandas import to_datetime

class FixedClass:
    date_unit = "ns"
    min_stamp = 0
    _STAMP_UNITS = ("ns", "us", "ms", "s")

    def _try_convert_to_date(self, data):
        if not len(data):
            return data, False

        new_data = data
        if new_data.dtype == "object":
            try:
                new_data = data.astype("int64")
            except (TypeError, ValueError, OverflowError):
                pass

        if issubclass(new_data.dtype.type, np.number):
            in_range = (
                isna(new_data)
                | (new_data > self.min_stamp)
                | (new_data == iNaT)
            )
            if not in_range.all():
                return data, False

        date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
        for date_unit in date_units:
            try:
                new_data = to_datetime(new_data, errors="coerce", unit=date_unit)
            except (ValueError, OverflowError):
                continue
            return new_data, True
        return data, False

def test_readjson_bool_series():
    # GH31464
    fixed_class = FixedClass()
    result = fixed_class._try_convert_to_date([True, True, False])
    expected = pd.Series([True, True, False])
    assert result[0].equals(expected)
    assert result[1] == True
