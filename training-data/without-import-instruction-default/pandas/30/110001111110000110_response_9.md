import numpy as np
import pandas as pd
from pandas.io.json import read_json
from pandas.testing import assert_series_equal

class JSONReader:
    def __init__(self, data, typ):
        self.data = data
        self.typ = typ
        self._STAMP_UNITS = ["s", "ms", "us", "ns"]
        self.date_unit = "ns"
        self.min_stamp = pd.Timestamp.min.value

    def read(self):
        obj = self._get_object_parser(self.data)
        return obj

    def _get_object_parser(self, json):
        return SeriesParser(json, typ=self.typ).parse()

class SeriesParser:
    def __init__(self, json, typ):
        self.json = json
        self.typ = typ

    def parse(self):
        obj, result = self._try_convert_types()
        return obj

    def _try_convert_types(self):
        return self._try_convert_data(self.json)

    def _try_convert_data(self, data):
        return self._try_convert_to_date(data)

    def _try_convert_to_date(self, data):
        if not len(data):
            return data, False

        new_data = data
        if new_data.dtype == "object":
            try:
                new_data = pd.to_datetime(new_data.astype("int64"), errors="raise", unit=self.date_unit)
            except (TypeError, ValueError, OverflowError):
                pass
            return new_data, True
        else:
            return data, False
