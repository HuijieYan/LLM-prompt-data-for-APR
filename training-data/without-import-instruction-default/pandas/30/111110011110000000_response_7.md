import numpy as np
import pandas as pd

class Parser():
    
    _STAMP_UNITS = ("ns",)
    
    def __init__(self, date_unit=None, min_stamp=0):
        self.date_unit = date_unit
        self.min_stamp = min_stamp
        
    def isna(self, values):
        return pd.isna(values)
    
    def iNaT(self):
        return pd.NaT
    
    def to_datetime(self, data, errors, unit):
        return pd.to_datetime(data, errors=errors, unit=unit)
    
    def _try_convert_to_date(self, data):
        if len(data) == 0:
            return data, False
        
        new_data = data
        if new_data.dtype == "object":
            try:
                new_data = data.astype("int64")
            except (TypeError, ValueError, OverflowError):
                pass
        
        if issubclass(new_data.dtype.type, np.number):
            in_range = (
                self.isna(new_data._values)
                | (new_data > self.min_stamp)
                | (new_data._values == self.iNaT())
            )
            if not in_range.all():
                return data, False
        
        date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
        for date_unit in date_units:
            try:
                new_data = self.to_datetime(new_data, errors="raise", unit=date_unit)
            except (ValueError, OverflowError):
                continue
            return new_data, True
        return data, False
