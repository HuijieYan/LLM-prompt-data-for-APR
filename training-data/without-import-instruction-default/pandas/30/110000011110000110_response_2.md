import pandas as pd
from pandas.io.json import json_normalize

def read_json(data, typ):
    """
    Read a JSON string or file and return a Pandas object.

    Parameters:
    data : str or file
        The JSON string or file to be read.
    typ : {'series', 'frame'}
        Determines the type of Pandas object to return.

    Returns:
    result : Series or DataFrame
        The Pandas object containing the parsed data.

    Raises:
    TypeError: When the JSON data is not convertible to the specified type.
    """
    result = json_normalize(data)
    if typ == "series":
        return result.squeeze()
    elif typ == "frame":
        return result
    else:
        raise ValueError("Invalid type. Must be 'series' or 'frame'.")

def _try_convert_to_date(self, data):
    """
    Try to parse a ndarray-like input into a date column.

    Try to coerce objects in epoch/ISO formats and integers/floats in epoch
    formats. Return a boolean indicating if parsing was successful.
    """
    # no conversion on empty
    if not len(data):
        return data, False

    new_data = data
    if new_data.dtype == "object":
        try:
            new_data = new_data.astype("int64")
        except (TypeError, ValueError, OverflowError):
            pass

    # Convert to datetime
    try:
        new_data = pd.to_datetime(new_data, errors="coerce")
        return new_data, True
    except ValueError:
        return data, False
