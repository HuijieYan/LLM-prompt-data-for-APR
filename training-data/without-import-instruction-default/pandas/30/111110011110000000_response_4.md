class Parser():

    def _try_convert_to_date(self, data, date_unit, min_stamp):
        if not len(data):
            return data, False

        new_data = data
        if new_data.dtype == "object":
            try:
                new_data = data.astype("int64")
            except (TypeError, ValueError, OverflowError):
                pass

        if issubclass(new_data.dtype.type, np.number):
            in_range = (
                isna(new_data)
                | (new_data > min_stamp)
                | (new_data == iNaT)
            )
            if not in_range.all():
                return data, False

        date_units = (date_unit,) if date_unit else self._STAMP_UNITS
        for date_unit in date_units:
            try:
                new_data = to_datetime(new_data, errors="coerce", unit=date_unit)
            except (ValueError, OverflowError):
                continue
            return new_data, True
        return data, False
