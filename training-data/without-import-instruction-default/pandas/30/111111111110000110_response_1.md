    def _try_convert_to_date(self, data, errors="raise"):

        Try to parse a ndarray like into a date column.

        formats. Return a boolean if parsing was successful.
        """
        # no conversion on empty
        if not len(data):
            return data, False

        if not is_numeric_dtype(data) and not isinstance(data, str) and not isinstance(data, (np.ndarray, pd.Series)):
            if errors == "coerce":
                return data, False
            if errors == "raise":
                raise ParserError(f"{errors} is not supported for parsing bool data to dates")

        new_data = data

        if not is_numeric_dtype(new_data):
            if errors == "coerce":
                from pandas.api.types import is_bool_dtype
                coerce_result = new_data.copy()
                coerce_result[is_bool_dtype(new_data)] = new_data[is_bool_dtype(new_data)]
                new_data = coerce_result
            else:
                if errors != "ignore":
                    raise ParserError(
                        f"An incorrect type of values was passed in, "
                        f"{type(data)} cannot be used for date conversion."
                    )

        if is_numeric_dtype(new_data) and not pd.api.types.is_datetime64_any_dtype(new_data):
            date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
            for date_unit in date_units:
                try:
                    new_data = pd.to_datetime(new_data, errors=errors, unit=date_unit)
                except (ValueError, OverflowError):
                    continue
                return new_data, True

        return data, False
