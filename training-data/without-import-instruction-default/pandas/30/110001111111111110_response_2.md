```python
import numpy as np
import pandas as pd
from pandas._libs.tslibs.nattype import iNaT
from pandas.core.dtypes.common import is_datetime64_any_dtype, is_datetime64tz_dtype, is_object_dtype
from pandas.tseries.tools import to_datetime


def _try_convert_to_date(data, min_stamp, date_unit):
    if is_datetime64_any_dtype(data) or is_datetime64tz_dtype(data):
        return data, True
    if is_object_dtype(data):
        try:
            new_data = data.astype("int64")
        except (TypeError, ValueError, OverflowError):
            pass
    if issubclass(data.dtype.type, np.number):
        in_range = (
            data.isnull()
            | (data > min_stamp)
            | (data == iNaT)
        )
        if not in_range.all():
            return pd.Series(data), False

    date_units = (date_unit,) if date_unit else ('s', 'ms', 'us', 'ns')
    for date_unit in date_units:
        try:
            data = to_datetime(data, errors="raise", unit=date_unit)
        except (ValueError, OverflowError):
            continue
        return data, True
    return pd.Series(data), False
```
