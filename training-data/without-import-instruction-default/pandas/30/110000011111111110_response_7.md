import numpy as np
import pandas as pd
from pandas import Series
from pandas.util._decorators import Appender
from pandas.errors import iNaT, isna
from pandas.core.arrays.datetimes import DatetimeArray
from pandas.core.algorithms import isin
from pandas.types.dtypes import CategoricalDtype
from pandas.core.dtypes.common import (
    is_categorical,
    is_extension_array_dtype,
)
from pandas.api.types import is_float, is_integer_dtype
from pandas.core.dtypes.generic import ABCDatetimeArray, ABCIndexClass

def _try_convert_to_date(self, data):
    """
    Try to parse a ndarray like into a date column.

    Try to coerce object in epoch/iso formats and integer/float in epoch
    formats. Return a boolean if parsing was successful.
    """
    # no conversion on empty
    if not len(data):
        return data, False

    new_data = data
    if is_integer_dtype(new_data) or is_float(new_data):
        date_range_max = np.iinfo("int64").max
        if new_data.max() > date_range_max:
            return data, False

    new_data_dtype = new_data.dtype
    if (
        new_data_dtype == "object"
        and isinstance(new_data[0], str)
        and isinstance(new_data[0], bytes)
    ):
        try:
            new_data = new_data.astype("int64")
        except (TypeError, ValueError, OverflowError):
            pass

    elif is_extension_array_dtype(new_data):
        if isinstance(new_data, ABCDatetimeArray):
            result = new_data._try_coerce_to_datetimelike(errors="raise")
            if not result[1]:
                return data, False

    else:
        categorical = is_categorical(new_data)
        if not categorical and isinstance(new_data, ABCIndexClass):
            categorical = (
                is_categorical(new_data.array)
                if isinstance(new_data.array, ABCIndexClass)
                else is_categorical(new_data.array)
            )
        if categorical:
            new_data = new_data.astype(str)

        if (is_integer_dtype(new_data_dtype) or is_float(new_data) and not categorical):
            mask = isna(new_data) | (new_data > self.min_stamp)

            if iNaT is not None:
                mask |= new_data._values == iNaT
            if not mask.all():
                return data, False

        elif not isinstance(new_data_dtype, DatetimeArray):
            # ignore numbers that are out of range
            if issubclass(new_data_dtype.type, np.number):
                in_range = isin(
                    new_data, [-np.inf, self.min_stamp, np.inf], assume_unique=False
                )
                if not in_range.all():
                    return data, False

        conversions = list(self._get_date_formats())

        for _, date_unit in conversions:
            try:
                new_data = pd.to_datetime(new_data, errors="raise", unit=date_unit)
                return new_data, True
            except Exception:
                continue

    return data, False
