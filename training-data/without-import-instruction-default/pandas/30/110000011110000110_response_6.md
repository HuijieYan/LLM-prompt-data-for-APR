import pandas as pd
import numpy as np
from pandas.core.dtypes.missing import isna, iNaT
from pandas.core.dtypes.common import is_datetime64_any_dtype
from pandas.core.arrays.datetimes import DatetimeArray
from pandas.errors import OutOfBoundsDatetime

def _try_convert_to_date(data, date_unit, min_stamp):
    """
    Try to parse a ndarray like into a date column.

    Try to coerce object in epoch/iso formats and integer/float in epoch
    formats. Return a boolean if parsing was successful.
    """
    # no conversion on empty
    if not len(data):
        return data, False

    new_data = pd.Series(data)
    if new_data.dtype == "object":
        try:
            new_data = data.astype("int64")
        except (TypeError, ValueError, OverflowError):
            pass

    # ignore numbers that are out of range
    if isinstance(new_data, np.ndarray) and getattr(new_data.dtype, "kind", None) in ["i", "u", "f"]:
        in_range = (
            isna(new_data)
            | (new_data > min_stamp)
            | (new_data == iNaT)
        )
        if not in_range.all():
            return data, False

    date_units = (date_unit,) if date_unit else DatetimeArray._STAMP_UNITS
    for date_unit in date_units:
        try:
            new_data = pd.to_datetime(new_data, errors="raise", unit=date_unit)
        except (ValueError, OverflowError, OutOfBoundsDatetime):
            continue
        return new_data, True
    return data, False

_min_stamp = np.datetime64('NaT').astype('datetime64[ns]')
_date_unit = 's'
