def _convert_key(self, key, is_setter: bool = False):
    """
    Require they keys to be the same type as the index. (so we don't
    fallback)
    """
    if is_setter:
        return list(key)

    index_type = type(key[0])

    for i in key:
        if type(i) != index_type:
            raise ValueError(
                "At based indexing can only have indexers of the same type as the index"
            )

    return key
