def _convert_key(self, key, is_setter: bool = False):
    """
    Require they keys to be the same type as the index. (so we don't
    fallback)
    """
    # allow arbitrary setting
    if is_setter:
        return list(key)

    for ax, i in zip(self.obj.index, key):  # Use self.obj.index instead of self.obj.axes
        if self.obj.index.is_integer():  # Use self.obj.index.is_integer() instead of ax.is_integer()
            if not is_integer(i):  # Use self.obj.index.is_integer() instead of is_integer(i)
                raise ValueError(
                    "At based indexing on an integer index "
                    "can only have integer indexers"
                )
        else:
            if is_integer(i) and not self.obj.index.holds_integer():  # Use self.obj.index.holds_integer() instead of ax.holds_integer()
                raise ValueError(
                    "At based indexing on an non-integer "
                    "index can only have non-integer "
                    "indexers"
                )
    return key
