def _convert_key(self, key, is_setter: bool = False):
    """
    Require the keys to be the same type as the index. (so we don't
    fallback)
    """
    # allow arbitrary setting
    if is_setter:
        return list(key)

    for ax, i in zip(self.obj.axes, key):
        if issubclass(ax.dtype.type, np.floating) and not issubclass(type(i), np.floating):
            raise ValueError(
                "At based indexing on an float index "
                "can only have float indexers"
            )
        elif issubclass(ax.dtype.type, np.integer) and not issubclass(type(i), np.integer):
            raise ValueError(
                "At based indexing on an integer index "
                "can only have integer indexers"
            )
    return key
