def _convert_key(self, key, is_setter: bool = False):
    """
    Require they keys to be the same type as the index. (so we don't
    fallback)
    """
    # allow arbitrary setting
    if is_setter:
        return list(key)

    for ax, i in zip(self.obj.axes, key):
        if (isinstance(i, int) and isinstance(self.obj.index[0], int)) or (isinstance(i, float) and isinstance(self.obj.index[0], float)):
            if not type(self.obj.index[0])(i) == i:  # cast to the index type to ensure type consistency
                raise ValueError(
                    f"At based indexing on an {type(self.obj.index[0]).__name__} index "
                    f"can only have {type(self.obj.index[0]).__name__} indexers"
                )
        else:
            if (isinstance(i, int) and not isinstance(self.obj.index[0], int)) or (isinstance(i, float) and not isinstance(self.obj.index[0], float)):
                raise ValueError(
                    f"At based indexing on an non-{type(self.obj.index[0]).__name__} "
                    f"index can only have non-{type(self.obj.index[0]).__name__} "
                    "indexers"
                )
    return key
