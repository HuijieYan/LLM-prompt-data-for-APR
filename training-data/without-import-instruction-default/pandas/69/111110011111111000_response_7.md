def _convert_key(self, key, is_setter: bool = False):
    """
    Require they keys to be the same type as the index. (so we don't
    fallback)
    """
    # allow arbitrary setting
    if is_setter:
        return list(key)

    for ax, i in zip(self.obj.axes, key):
        # Changing is_integer() to isinstance(i, int) to properly identify integer types
        if isinstance(i, int) and isinstance(ax, pd.Int64Index):
            if not isinstance(ax, pd.Int64Index):
                raise ValueError(
                    "At based indexing on an integer index "
                    "can only have integer indexers"
                )
        else:
            if isinstance(i, float) and not isinstance(ax, pd.Float64Index):
                raise ValueError(
                    "At based indexing on an non-integer "
                    "index can only have non-integer "
                    "indexers"
                )
    return key
