# The declaration of the class containing the buggy function
@Appender(IndexingMixin.at.__doc__)
class _AtIndexer(_ScalarAccessIndexer):

    def _convert_key(self, key, is_setter: bool = False):
        """
        Require the keys to be the same type as the index. (so we don't
        fallback)
        """
        # allow arbitrary setting
        if is_setter:
            return list(key)
    
        for ax, i in zip(self.obj.axes, key):
            ax_type = self.obj.index.get_level_type(0)
            if ax_type == 'floating':
                i = float(i)
            elif ax_type == 'integer':
                i = int(i)
            elif ax_type == "mixed-integer":
                if not is_integer(i):
                    raise ValueError(
                        "At based indexing on a mixed-integer index "
                        "can only have integer indexers"
                    )
            elif ax_type == "mixed-integer-float":
                if not is_integer(i) and not is_float(i):
                    raise ValueError(
                        "At based indexing on a mixed-integer-float "
                        "index can only have integer and float indexers"
                    )
            else:
                raise ValueError("Unknown index type")
        return key
