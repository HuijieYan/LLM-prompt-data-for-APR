def _convert_key(self, key, is_setter: bool = False):
    """
    Require the keys to be the same type as the index (so we don't
    fallback)
    """
    # allow arbitrary setting
    if is_setter:
        return list(key)

    for ax, i in zip(self.obj.axes, key):
        if isinstance(self.obj, pd.Series):
            if not isinstance(i, type(ax[0])):
                raise ValueError(
                    "At based indexing on an {} index can only have {} indexers".format(
                        "integer" if np.issubdtype(ax.dtype, np.integer) else "non-integer",
                        "integer" if np.issubdtype(ax.dtype, np.integer) else "non-integer"
                    )
                )
        else:  # for DataFrame
            i = next(iter(key))  # assuming all index types are the same in a DataFrame
            if not isinstance(i, type(ax[0])):
                raise ValueError(
                    "At based indexing on an {} index can only have {} indexers".format(
                        "integer" if np.issubdtype(ax.dtype, np.integer) else "non-integer",
                        "integer" if np.issubdtype(ax.dtype, np.integer) else "non-integer"
                    )
                )
    return key
