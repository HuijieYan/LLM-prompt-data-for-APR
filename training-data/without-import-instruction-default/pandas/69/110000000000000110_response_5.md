# this is the fixed function
def _convert_key(self, key, is_setter: bool = False):
    """
    Require the keys to be the same type as the index. (so we don't
    fallback)
    """
    # allow arbitrary setting
    if is_setter:
        return key  # removed list conversion

    for ax, i in zip(self.obj.axes, key):
        if ax.is_integer():
            if not isinstance(i, int):  # changed is_integer to isinstance
                raise ValueError(
                    "Integer index can only have integer indexers"
                )
        else:
            if isinstance(i, int) and not ax.holds_integer():  # changed is_integer to isinstance
                raise ValueError(
                    "Non-integer index can only have non-integer indexers"
                )
    return key
