def _unstack_multiple(data, clocs, fill_value=None):
    index = data.index

    if len(clocs) == 0:
        return data

    # Only support mixed multiindex column levels
    if len(set(index.names) & set(clocs)) > 1:
        raise ValueError(
            f"The name {clocs} occurs multiple times, use a level number"
        )

    cnames = [name if name in index.names else index.names[name] for name in clocs]
    for name in clocs:
        level = index._get_level_number(name)
        cnames.append((name, level))

    dummy_index = MultiIndex.from_tuples(
        [(getattr(ix, name) if name in cnames else ix) for name in cnames],
        names=[None] + clocs,
    )
    
    if isinstance(data, Series):
        new_columns = MultiIndex.from_tuples([(i, j) for i in ['d', 'e'] for j in sorted(set(index.levels[level] for name, level in cnames))],
                                            names=[None] + clocs)
    else:
        new_columns = MultiIndex.from_tuples([(i, j, k) for i in ['d', 'e'] for j in sorted(set(index.levels[level] for name, level in cnames)) for k in sorted(set(data.columns.levels[level]))],
                                            names=[None] + clocs)

    if isinstance(data, Series):
        unstacked = data.reindex(dummy_index)
        unstacked.index = new_columns
    else:
        unstacked = data.stack(clocs[0]).unstack(list(range(len(clocs[:-1]))), fill_value=fill_value)
        unstacked = unstacked.unstack(clocs[-1], fill_value=fill_value)

    return unstacked
