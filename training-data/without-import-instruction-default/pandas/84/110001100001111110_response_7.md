# The corrected and bug-fixed function

def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data

    # NOTE: This doesn't deal with hierarchical columns yet

    index = data.index

    # Convert tuple of tuple(s) to list of tuple(s)
    if isinstance(clocs, tuple):
        clocs = [clocs]

    # Convert the row index to a DataFrame and reset the index
    df = data.reset_index()

    # Unstack using the clocs
    unstacked = df.unstack(clocs, fill_value=fill_value)

    return unstacked
