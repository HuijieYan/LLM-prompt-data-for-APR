def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data

    index = data.index
    if isinstance(clocs, tuple):
        clocs = [clocs]

    clevels = [index.get_level_values(i[1]) for i in clocs]
    rlevels = [index.get_level_values(i) for i in range(index.nlevels) if i not in [i[0] for i in clocs]]

    # Unstacking for the case when hierarchical columns are present
    if isinstance(data.columns, pd.MultiIndex):
        result = data
        for cloc in sorted([i[0] for i in clocs]):
            result = result.unstack(cloc, fill_value=fill_value)
            clocs = [(i[0], i[1]-1) if i[0] >= cloc else i for i in clocs]
        return result
    else:
        shape = [len(x) for x in clevels]
        group_index = get_group_index([index.get_level_values(i[1]) for i in clocs], shape, sort=False, xnull=False)
      
        comp_ids, obs_ids = compress_group_index(group_index, sort=False)
        recons_codes = decons_obs_group_ids(comp_ids, obs_ids, shape, [index.get_level_values(i[1]) for i in clocs], xnull=False)

        if rlevels == []:
            # Everything is in clocs, so the dummy df has a regular index
            dummy_index = pd.Index(obs_ids, name="__placeholder__")
        else:
            dummy_index = pd.MultiIndex(
                levels=rlevels + [obs_ids],
                codes=recons_codes + [comp_ids],
                names=[index.names[i] if i < len(index.names) else "__placeholder__" for i in [i[0] for i in clocs]]+[None],
                verify_integrity=False
            )

        if isinstance(data, pd.Series):
            dummy = data.copy()
            dummy.index = dummy_index

            unstacked = dummy.unstack("__placeholder__", fill_value=fill_value)
            new_levels = clevels
            new_names = [i[1] for i in clocs]
            new_codes = [index.get_level_values(i[1]).codes for i in clocs]
        else:
            dummy = data.copy()
            dummy.index = dummy_index

            unstacked = dummy.unstack("__placeholder__", fill_value=fill_value)
            if isinstance(unstacked, pd.Series):
                unstcols = unstacked.index
            else:
                unstcols = unstacked.columns
            new_levels = [index.get_level_values(i[1]) for i in clocs] + clevels
            new_names = [index.names[i[1]] for i in clocs] + [data.columns.name]
            new_codes = [unstcols.get_level_values(i).codes for i in range(len(unstcols.levels))]
    
        new_columns = pd.MultiIndex(
            levels=new_levels, codes=new_codes, names=new_names, verify_integrity=False
        )

        if isinstance(unstacked, pd.Series):
            unstacked.index = new_columns
        else:
            unstacked.columns = new_columns

        return unstacked
