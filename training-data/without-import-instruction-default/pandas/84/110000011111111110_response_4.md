def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data

    if isinstance(data, Series):
        # Unstack when clocs are passed
        if isinstance(data.index, MultiIndex):
            result = data.unstack(clocs)
            return result

    # Handle when data is a DataFrame
    index = data.index

    if isinstance(clocs, tuple):  # Unstack if clocs is a tuple
        clocs = list(clocs)

    clocs = [index._get_level_number(i) for i in clocs]

    if isinstance(data.columns, MultiIndex):
        result = data
        for i in range(len(clocs)):
            result = result.unstack(clocs[i], fill_value=fill_value)
        return result

    return data.unstack(clocs)
