from pandas import Index, MultiIndex, Series
from pandas.core.reshape.reshape import _unstack_multiple as original_unstack_multiple

def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data

    index = data.index
    clocs = [c if isinstance(c, int) else index._get_level_number(c) for c in clocs]

    return original_unstack_multiple(data, clocs, fill_value)

# The assertions to validate the correction.
idx = MultiIndex.from_product([['a', 'b', 'c'], [1, 2, 3]], names=[('A', 'a'), ('B', 'b')])
s = Series(1, index=idx)
s.unstack(clocs=(('A', 'a'))) # expects to return the unstacked DataFrame from the Series s
