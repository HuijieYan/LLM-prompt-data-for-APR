@apply_wraps
def apply(self, other):
    if isinstance(other, pd.Timestamp):
        n = self.n
        business_hours = self.business_hours
        start_time = self.start_time

        if n >= 0:
            # check if the time is in after the specified end time or not on the current offset
            if other.time() in self.end_time or not self._is_on_offset(other):
                other = self._next_opening_time(other)
        else:
            # check if the time is on the specified start time
            if other.time() in self.start_time:
                other = other - pd.Timedelta(seconds=1)
            # check of the time is not on the current offset
            if not self._is_on_offset(other):
                other = self._next_opening_time(other)
                other = self._get_closing_time(other)

        # Calculate the total business hours in a day
        total_business_hours = sum(
            self._get_business_hours_by_sec(st, en)
            for st, en in zip(self.start_time, self.end_time)
        )

        # Calculate the business days and remaining hours
        business_days, remaining_seconds = divmod(abs(n * 60), total_business_hours // 60)
        if n < 0:
            business_days, remaining_seconds = -business_days, -remaining_seconds

        # Adjust the dates for business days
        if business_days != 0:
            skip_bd = BusinessDay(n=business_days)
            if self._next_bday.is_on_offset(other):
                other = other + skip_bd
            else:
                prev_open = self._prev_opening_time(other)
                remain = other - prev_open
                other = prev_open + skip_bd + remain

        # Remaining business hours adjustment
        remaining_hours = pd.Timedelta(minutes=remaining_seconds)

        if n >= 0:
            while remaining_hours != pd.Timedelta(0):
                business_hour = (
                    self._get_closing_time(self._prev_opening_time(other)) - other
                )
                if remaining_hours < business_hour:
                    other += remaining_hours
                    remaining_hours = pd.Timedelta(0)
                else:
                    remaining_hours -= business_hour
                    other = self._next_opening_time(other + business_hour)
        else:
            while remaining_hours != pd.Timedelta(0):
                business_hour = self._next_opening_time(other) - other
                if remaining_hours > business_hour or remaining_hours == business_hour:
                    other += remaining_hours
                    remaining_hours = pd.Timedelta(0)
                else:
                    remaining_hours -= business_hour
                    other = self._get_closing_time(
                        self._next_opening_time(
                            other + business_hour - pd.Timedelta(seconds=1)
                        )
                    )

        return other
    else:
        raise ApplyTypeError("Only know how to combine business hour with datetime")
