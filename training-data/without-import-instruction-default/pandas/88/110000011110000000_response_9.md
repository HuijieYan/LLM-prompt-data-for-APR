from pandas import DataFrame, MultiIndex
import pandas.core.reshape.pivot as pivot

def fix_pivot_table(input_data, values=None, index=None, columns=None, aggfunc="mean", fill_value=None, margins=False, dropna=True, margins_name="All", observed=False) -> DataFrame:
    """
    Fixed pivot_table method
    """

    result = pivot.pivot_table(input_data, values=values, index=index, columns=columns, aggfunc=aggfunc, fill_value=fill_value, margins=margins, dropna=dropna, margins_name=margins_name, observed=observed)
    
    if isinstance(result, DataFrame) and result.columns.nlevels > 1:
        result = result.droplevel(0, axis=1)

    return result
