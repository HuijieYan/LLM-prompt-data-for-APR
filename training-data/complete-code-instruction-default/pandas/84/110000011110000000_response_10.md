# The bug-fixed function
def _unstack_multiple(data, clocs, fill_value=None):
    import pandas as pd

    if len(clocs) == 0:
        return data

    # NOTE: This doesn't deal with hierarchical columns yet

    index = data.index

    if isinstance(index, pd.MultiIndex):
        names = list(index.names)
    elif isinstance(index, pd.Index):
        names = [index.name]

    clocs = [names.index(i) for i in clocs]

    rlocs = [i for i in range(len(names)) if i not in clocs]

    cnames = [names[i] for i in clocs]
    rnames = [names[i] for i in rlocs]

    shape = [len(idx) for i, idx in enumerate(index.levels) if i in clocs]
    group_index = index.get_level_values(clocs[0]).values

    for i in range(1, len(clocs)):
        group_index *= shape[i]
        group_index += index.get_level_values(clocs[i]).values

    unstacked = data.unstack(cnames) if not isinstance(index, pd.Index) else data
    unstacked.index = unstacked.index.droplevel(clocs)

    new_columns = pd.MultiIndex.from_tuples(
        unstacked.columns, names=rnames+cnames
    )

    unstacked.columns = new_columns

    return unstacked
