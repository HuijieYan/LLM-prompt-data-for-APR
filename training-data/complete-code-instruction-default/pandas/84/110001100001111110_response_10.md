def _unstack_multiple(data, clocs, fill_value=None):
    import pandas as pd

    # Ensure clocs is a list
    if not isinstance(clocs, list):
        clocs = [clocs]

    if len(clocs) == 0:
        return data

    index = data.index

    # Use _get_level_number for each level name in clocs
    clocs = [index._get_level_number(i) if isinstance(i, tuple) else i for i in clocs]

    rlocs = [i for i in range(index.nlevels) if i not in clocs]

    clevels = [index.levels[i] for i in clocs]
    ccodes = [index.codes[i] for i in clocs]
    cnames = [index.names[i] for i in clocs]
    rlevels = [index.levels[i] for i in rlocs]
    rcodes = [index.codes[i] for i in rlocs]
    rnames = [index.names[i] for i in rlocs]

    shape = [len(clevels[0])]
    group_index = pd.factorize(index.to_frame(index=False).to_records(index=False).tolist())

    comp_ids, obs_ids = pd.Series(group_index[0]), group_index[1].unique()
    recons_codes = [pd.Index(ccodes[0]).get_indexer(comp_ids)]

    if rlocs == []:
        # Everything is in clocs, so the dummy df has a regular index
        dummy_index = pd.Index(obs_ids, name="__placeholder__")
    else:
        dummy_index = pd.MultiIndex.from_arrays([obs_ids, pd.Index(comp_ids)], names=[rnames[0], "__placeholder__"])

    if isinstance(data, pd.Series):
        dummy = data.copy()
        dummy.index = dummy_index

        unstacked = dummy.unstack("__placeholder__", fill_value=fill_value)
        new_levels = clevels
        new_names = cnames
        new_codes = recons_codes
    else:
        if isinstance(data.columns, pd.MultiIndex):
            result = data
            for i in range(len(clocs)):
                val = clocs[i]
                result = result.unstack(val, fill_value=fill_value)
                clocs = [v if i > v else v - 1 for v in clocs]

            return result

        dummy = data.copy()
        dummy.index = dummy_index

        unstacked = dummy.unstack("__placeholder__", fill_value=fill_value)
        if isinstance(unstacked, pd.Series):
            unstcols = unstacked.index
        else:
            unstcols = unstacked.columns
        new_levels = [unstcols.levels[0]] + clevels
        new_names = [data.columns.name] + cnames

        new_codes = [unstcols.codes[0]]
        for rec in recons_codes:
            new_codes.append(rec.take(unstcols.codes[-1]))

    new_columns = pd.MultiIndex.from_arrays(new_codes, names=new_names)

    if isinstance(unstacked, pd.Series):
        unstacked.index = new_columns
    else:
        unstacked.columns = new_columns

    return unstacked
