# Fixed function
def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data

    # NOTE: This doesn't deal with hierarchical columns yet

    index = data.index

    cloc_idx = tuple(index._get_level_number(i) if isinstance(i, tuple) else i for i in clocs)

    if isinstance(data, Series):
        unstacked = data.unstack(cloc_idx, fill_value=fill_value)
    else:
        unstacked = data.unstack(cloc_idx, fill_value=fill_value)
        if isinstance(unstacked, Series):
            unstacked.index = unstacked.index.swaplevel(i=-1, j=0)
        else:
            unstacked.index = unstacked.columns.swaplevel(i=-1, j=0)

    return unstacked
