def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data

    index = data.index

    levels = [index.get_level_values(i) for i in clocs]
    names = [index.names[i] for i in clocs]

    temp_df = data.reset_index()
    values = [temp_df.pop(lvl) for lvl in levels]
    dummy_index = temp_df.__placeholder__
    
    unstacked = temp_df.set_index(dummy_index, append=True).unstack(fill_value=fill_value)

    new_levels = [unstacked.columns] + levels
    new_names = [data.columns.name] + names
    new_columns = pd.MultiIndex.from_tuples(list(zip(*new_levels)), names=new_names)

    if isinstance(unstacked, pd.Series):
        unstacked.index = new_columns
    else:
        unstacked.columns = new_columns

    return unstacked
