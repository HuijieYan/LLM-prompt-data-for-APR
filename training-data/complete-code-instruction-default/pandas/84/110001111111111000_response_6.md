def _unstack_multiple(data, clocs, fill_value=None):
    index = data.index

    if isinstance(clocs, tuple):
        clocs = [clocs]

    rlocs = [i for i in range(index.nlevels) if i not in [index._get_level_number(i) for i in clocs]]

    levels = np.split(index.levels, [index._get_level_number(i) for i in clocs])
    codes = np.split(index.codes, [index._get_level_number(i) for i in clocs])

    new_levels = levels[0].tolist() + [index.levels[i] for i in rlocs]
    new_codes = codes[0].tolist() + [index.codes[i] for i in rlocs]

    new_index = MultiIndex(levels=new_levels, codes=new_codes, names=[index.names[i] for i in list(range(len(new_levels)))])

    dummy = data.reindex(new_index)
    unstacked = dummy.unstack(clocs, fill_value=fill_value)

    return unstacked
