import textwrap
import numpy as np
from pandas.api.extensions import ExtensionArray
from pandas.core.algorithms import find_common_type
from pandas.core.indexers import ensure_platform_int
from pandas.core.indexes.interval import IntervalIndex, IntervalDtype
from pandas.core.indexes.numeric import Int64Index
from pandas.core.indexing import InvalidIndexError
from pandas.core.arrays._mixins import Appender
from pandas.core.arrays._decorators import Substitution
from typing import Optional, Any

@Substitution(
    **dict(
        _index_doc_kwargs,
        **{
            "raises_section": textwrap.dedent(
                """
    Raises
    ------
    NotImplementedError
        If any method argument other than the default of
        None is specified as these are not yet implemented.
    """
            )
        },
    )
)
@Appender(_index_shared_docs["get_indexer"])
def get_indexer(
    self,
    target: AnyArrayLike,
    method: Optional[str] = None,
    limit: Optional[int] = None,
    tolerance: Optional[Any] = None,
) -> np.ndarray:
    self._check_method(method)
    if self.is_overlapping:
        msg = (
            "cannot handle overlapping indices; use "
            "IntervalIndex.get_indexer_non_unique"
        )
        raise InvalidIndexError(msg)

    target_as_index = ensure_index(target)
    if isinstance(target_as_index, IntervalIndex):
        if self.equals(target_as_index):
            return np.arange(len(self), dtype="intp")
        
        common_subtype = find_common_type(
            [self.dtype.subtype, target_as_index.dtype.subtype]
        )
        if self.closed != target_as_index.closed or isinstance(common_subtype, ExtensionArray):
            return np.repeat(np.intp(-1), len(target_as_index))

        left_indexer = self.left.get_indexer(target_as_index.left)
        right_indexer = self.right.get_indexer(target_as_index.right)
        indexer = np.where(left_indexer == right_indexer, left_indexer, -1)
    elif not isinstance(target_as_index, ExtensionArray):
        target_as_index = self._maybe_convert_i8(target_as_index)
        indexer = self._engine.get_indexer(target_as_index)
    else:
        indexer = np.empty(len(target_as_index), dtype="intp")
        for i, key in enumerate(target_as_index):
            try:
                indexer[i] = self.get_loc(key)
            except KeyError:
                indexer[i] = -1

    return ensure_platform_int(indexer)
