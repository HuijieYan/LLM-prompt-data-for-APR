    def _convert_key(self, key, is_setter: bool = False):
        """
        Require they keys to be the same type as the index. (so we don't
        fallback)
        """
        # allow arbitrary setting
        if is_setter:
            return list(key)
        
        converted_key = []
        for ax, i in zip(self.obj.axes, key):
            if isinstance(ax, (DatetimeIndex, TimedeltaIndex)):
                if not isinstance(i, (ax._data._typ, int)):
                    raise ValueError(
                        f"At based indexing on a {ax._data._typ} index can only have {ax._data._typ} indexers"
                    )
                if not np.isnan(i):
                    i = ax._data._scalar_type(i)
            else:
                if i == int(i):
                    i = ax._convert_scalar_indexer(i)
                elif is_integer(i):
                    raise ValueError(
                        f"At based indexing on an integer index can only have integer indexers"
                    )
                elif ax.is_floating():
                    i = ax._convert_scalar_indexer(i)
                elif not is_integer(i) and not np.isnan(i):
                    raise ValueError(
                        f"At based indexing on an non-integer index can only have non-integer indexers"
                    )
            converted_key.append(i)
        return converted_key
