def _convert_key(self, key, is_setter: bool = False):
    """
    Require they keys to be the same type as the index. (so we don't
    fallback)
    """
    # allow arbitrary setting
    if is_setter:
        return list(key)

    key_type = type(key[0])  # Get the type of the first element in the key

    if isinstance(self.obj, pd.DatetimeIndex) and key_type != pd.Timestamp:
        key = pd.to_datetime(key)
    elif isinstance(self.obj, pd.TimedeltaIndex) and key_type != pd.Timedelta:
        key = pd.to_timedelta(key)

    return key
