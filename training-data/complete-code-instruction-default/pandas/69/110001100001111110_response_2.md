# The relative path of the fixed file: pandas/core/indexing.py

# fixed function
def _convert_key(self, key, is_setter: bool = False):
    """
    Require they keys to be the same type as the index. (so we don't
    fallback)
    """
    # allow arbitrary setting
    if is_setter:
        return list(key)

    for ax, i in zip(self.obj.axes, key):
        if isinstance(ax, pd.core.indexes.numeric.Float64Index) and not isinstance(i, (int, float)):
            raise ValueError(
                "At based indexing on an integer index "
                "can only have integer indexers"
            )
        elif isinstance(ax, (pd.core.indexes.numeric.Int64Index, pd.core.indexes.numeric.Float64Index)) and not isinstance(i, (int, np.int64, np.int, np.float64, np.float)):
            raise ValueError(
                "At based indexing on an non-integer "
                "index can only have non-integer "
                "indexers"
            )

    return key
