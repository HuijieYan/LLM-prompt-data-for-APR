@Appender(IndexingMixin.at.__doc__)
class _AtIndexer(_ScalarAccessIndexer):

    def _convert_key(self, key, is_setter: bool = False):
        """
        Require they keys to be the same type as the index. (so we don't
        fallback)
        """
        # allow arbitrary setting
        if is_setter:
            return list(key)
    
        for ax, i in zip(self.obj.axes, key):
            if isinstance(ax, (DatetimeIndex, TimedeltaIndex)):
                if is_integer(i):
                    raise ValueError(
                        f"At based indexing on a {ax.dtype} index can only have non-integer indexers"
                    )
                elif not ax.is_type_compatible(i):
                    raise ValueError(
                        f"At based indexing on a {ax.dtype} index can only have non-{ax.dtype} indexers"
                    )
            else:
                if not is_integer(i):
                    raise ValueError(
                        "At based indexing on an integer index "
                        "can only have integer indexers"
                    )
        return key
