def _convert_key(self, key, is_setter: bool = False):
    """
    Require they keys to be the same type as the index. (so we don't
    fallback)
    """
    if is_setter:
        return list(key)

    if isinstance(self.obj, pd.Series):
        if isinstance(key, (tuple, list)):
            key = key[0]

        if self.obj.index.is_integer():
            if not isinstance(key, (int, np.integer)):
                raise ValueError(
                    "At based indexing on an integer index "
                    "can only have integer indexers"
                )
        else:
            if isinstance(key, (int, np.integer)) and not self.obj.index.is_integer():
                raise ValueError(
                    "At based indexing on a non-integer index "
                    "can only have non-integer indexers"
                )
    else:
        if isinstance(key, (tuple, list)):
            key = key[0]

        if pd.api.types.is_integer(self.obj):
            if not isinstance(key, (int, np.integer)):
                raise ValueError(
                    "At based indexing on an integer index "
                    "can only have integer indexers"
                )
        else:
            if isinstance(key, (int, np.integer)) and not pd.api.types.is_integer(self.obj):
                raise ValueError(
                    "At based indexing on a non-integer index "
                    "can only have non-integer indexers"
                )

    return key
