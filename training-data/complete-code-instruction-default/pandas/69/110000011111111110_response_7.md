# this is the fixed function
def _convert_key(self, key, is_setter: bool = False):
    """
    Require the keys to be the same type as the index. (so we don't fallback)
    """
    # allow arbitrary setting
    if is_setter:
        return list(key)

    for ax, i in zip(self.obj.axes, key):
        if isinstance(self.obj, pd.TimedeltaIndex):
            if isinstance(i, int) and not pd.api.types.is_timedelta64_any_dtype(self.obj.dtype):
                raise ValueError(
                    "At based indexing on a non-timedelta index "
                    "can only have non-integer indexers"
                )
            elif not pd.api.types.is_timedelta64_any_dtype(self.obj.dtype):
                if not isinstance(i, int):
                    raise ValueError(
                        "At based indexing on a non-timedelta "
                        "index can only have non-integer "
                        "indexers"
                    )
        elif isinstance(self.obj, pd.DatetimeIndex):
            if isinstance(i, int) and not pd.api.types.is_datetime64_any_dtype(self.obj.dtype):
                raise ValueError(
                    "At based indexing on a non-datetime index "
                    "can only have non-integer indexers"
                )
            elif not pd.api.types.is_datetime64_any_dtype(self.obj.dtype):
                if not isinstance(i, int):
                    raise ValueError(
                        "At based indexing on a non-datetime "
                        "index can only have non-integer "
                        "indexers"
                    )

    return key
