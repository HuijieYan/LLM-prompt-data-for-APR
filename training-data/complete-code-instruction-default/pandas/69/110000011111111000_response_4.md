def _convert_key(self, key, is_setter: bool = False):
    """
    Require the keys to be the same type as the index. (so we don't
    fallback)
    """
    # allow arbitrary setting
    if is_setter:
        return list(key)

    for ax, i in zip(self.axes, key):
        if issubclass(ax.dtype.type, np.datetime64) or issubclass(ax.dtype.type, np.timedelta64):
            if not issubclass(type(i), (np.datetime64, np.timedelta64)):
                raise ValueError(
                    "At based indexing on an {} index can only have {} indexers".format(
                        "integer" if issubclass(ax.dtype.type, np.datetime64) else "non-integer",
                        "integer" if issubclass(type(i), (int, np.int64)) else "non-integer"
                    )
                )
        elif not issubclass(type(i), ax.dtype.type.type):
            raise ValueError(
                "At based indexing on a {} index can only have {} indexers".format(
                    "floating" if np.issubdtype(ax.dtype, np.floating) else "non-float",
                    "floating" if isinstance(i, float) else "non-float"
                )
            )
    return key
