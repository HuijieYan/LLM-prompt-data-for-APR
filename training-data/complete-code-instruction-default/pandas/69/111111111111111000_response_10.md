def _convert_key(self, key, is_setter: bool = False):
        """
        Require the keys to be the same type as the index (so we don't
        fallback).
        """
        # allow arbitrary setting
        if is_setter:
            return list(key)
        
        key_type = type(key)
        is_integer = key_type is int or key_type is np.int64 or key_type is np.int32
        if is_integer:
            if self.obj.axes[0].is_integer():
                return key_type(key)
            else:
                ax_type = type(self.obj.axes[0][0])
                return ax_type(key)
        else:
            if not self.obj.axes[0].is_integer():
                return key_type(key)
            else:
                raise ValueError(
                    f"At based indexing on a non-integer index "
                    f"can only have non-integer indexers"
                )
