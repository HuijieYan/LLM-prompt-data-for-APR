def _cython_agg_blocks(
    self, how: str, alt=None, numeric_only: bool = True, min_count: int = -1
) -> "Tuple[List[Block], Index]":

    def _handle_error(e, alt, axis):
        try:
            result = alt.blockwise(axis)
        except:
            from pandas.core.dtypes.common import not_na
            alt = not_na(alt, obj.shape[1], self.gf,
                          fill_value=_maybe_fill(min_count))
            result = alt.blockwise(axis)
        return result

    mgr_locs = self._get_data_to_aggregate()
  
    if numeric_only:
        mgr_locs = mgr_locs.get_numeric_data(copy=False)

    agg_blocks: List[Block] = []

    for block in mgr_locs:
        result = None
        try:
            axis = 0
            result, _ = self.grouper.aggregate(block.values, how, axis=axis, min_count=min_count)
        except NotImplementedError:
            alt_obj = self.obj[mgr_locs.items[block.mgr_locs]]
            alt = alt_obj._maybe_update_inplace(alt_obj.dropna(), inplace=True)

            alt_block = ''.join(2 * ['[%s]' % axis])
            alt = _handle_error(block, alt_block, axis)
            result = alt
        except Exception as e:
            print(e)
            pass

        if result is not None:
            result = maybe_downcast_numeric(result, block.dtype)
            if block.is_extension and isinstance(result, np.ndarray):
                result = type(block.values)._from_sequence(result.ravel(),
                                                          dtype=block.values.dtype)
            agg_block: Block = block.make_block(result)
            agg_blocks.append(agg_block)

    if len(agg_blocks) == 0:
        raise DataError("No numeric types to aggregate")

    return agg_blocks, mgr_locs
