def _cython_agg_blocks(
        self, how: str, alt=None, numeric_only: bool = True, min_count: int = -1
    ) -> "Tuple[List[Block], Index]":
        # TODO: the actual managing of mgr_locs is a PITA
        # here, it should happen via BlockManager.combine
        
        data=self._get_data_to_aggregate()
        
        if numeric_only:
            data = data.get_numeric_data(copy=False)
        result=[]
        new_items=[]
        deleted_items=[]
        split_items=[]
        split_frames=[]
        no_res = object()
        for block in data.blocks:
            new_res = no_res
            locs = block.mgr_locs.as_array
            try:
                new_res, _ = self.grouper.aggregate(
                    block.values, how, axis=1, min_count=min_count
                )
            except NotImplementedError:
                if alt is None:
                    assert how == "ohlc"
                    deleted_items.append(locs)
                    continue
                
                obj = self.obj[data.items[locs]]
                if obj.shape[1] == 1:
                    obj = obj.iloc[:, 0]    
                
                s = get_groupby(obj, self.grouper)
                try:
                    new_res = s.aggregate(lambda x: alt(x, axis=self.axis))
                except TypeError:
                    deleted_items.append(locs)
                    continue
                else:
                    new_res = cast(DataFrame, new_res)
                    if len(new_res._data.blocks) != 1:
                        split_items.append(locs)
                        split_frames.append(new_res)
                        continue
        
                    assert len(new_res._data.blocks) == 1
                    new_res = new_res._data.blocks[0].values
                    if isinstance(new_res, np.ndarray) and new_res.ndim == 1:
                        new_res = new_res.reshape(1, -1)
                    
            assert not isinstance(new_res, DataFrame)
            
            if new_res is not no_res:
                new_res = maybe_downcast_numeric(new_res, block.dtype)

                if block.is_extension and isinstance(new_res, np.ndarray):
                    assert new_res.ndim == 1 or new_res.shape[0] == 1
                    try:
                        new_res = type(block.values)._from_sequence(
                            new_res.ravel(), dtype=block.values.dtype
                        )
                    except ValueError:
                        new_res = new_res.reshape(1, -1)

                new_block = block.make_block(new_res)
            
            new_items.append(locs)
            result.append(new_block)

        if not (result or split_frames):
            raise DataError("No numeric types to aggregate")

        if split_items:
            for locs, new_res in zip(split_items, split_frames):
                assert len(locs) == new_res.shape[1]
                for i, loc in enumerate(locs):
                    new_items.append(np.array([loc], dtype=locs.dtype))
                    result.append(new_res.iloc[:, [i]]._data.blocks[0])
        
        indexer = np.concatenate(new_items)
        agg_items = data.items.take(np.sort(indexer))

        if deleted_items:
            deleted = np.concatenate(deleted_items)
            ai = np.arange(len(data))
            mask = np.zeros(len(data))
            mask[deleted] = 1
            indexer = (ai - mask.cumsum())[indexer]
        
        offset = 0
        for blk in result:
            loc = len(blk.mgr_locs)
            blk.mgr_locs = indexer[offset : (offset + loc)]
            offset += loc
        
        return result, agg_items
