def _cython_agg_blocks(
    self, how: str, alt=None, numeric_only: bool = True, min_count: int = -1
) -> "Tuple[List[Block], Index]":
    # TODO: the actual managing of mgr_locs is a PITA
    # here, it should happen via BlockManager.combine

    data: BlockManager = self._get_data_to_aggregate()

    if numeric_only:
        data = data.get_numeric_data(copy=False)


    # Change conditional to check dtype of the data
    if data._data.dtypes.any() != "category":
        agg_blocks = data._data.agg(how, min_count=min_count)
        if isinstance(agg_blocks, DataFrame):
            to_concat = []

            for block in agg_blocks.blocks:
                to_concat.append(block.internal_values())
            
            result = np.concatenate(to_concat, axis=self.axis)
            new_blocks = np.array_split(result, len(to_concat))

            new_items = []
            offset = 0
            for blk in new_blocks:
                new_block = Block(blk, self.grouper.indices, self.grouper.group_info, self._block_shape)
                new_items.append(blk)
                new_blocks.append(new_block)

            agg_blocks = ColumnZip(new_blocks, new_items, index=agg_blocks._index)

            return agg_blocks
        elif isinstance(agg_blocks, DataFrame) and "Series" in agg_blocks.dtypes:
            s = agg_blocks.iloc[:, 0]
            index = s.index
            return s
        else:
            # Groupby methods with 'how' argument calling
            if isinstance(agg_blocks, Series):
                # Return the same type as original if 'how' call is getting new groups which doesn't have 'value'
                return agg_blocks
            else:
                return
    else:
        # probably you may raise warning about non-numeric types on which mean is being calculated
        raise TypeError("Non-numeric types are not allowed for mean aggregation in groupby")
