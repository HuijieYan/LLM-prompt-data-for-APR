# The corrected version of the function
import numpy as np
import pandas as pd
from pandas.api.types import is_numeric_dtype, is_datetime64_dtype


def _try_convert_to_date(self, data):
    """
    Try to parse a ndarray like into a date column.

    Try to coerce object in epoch/iso formats and integer/float in epoch
    formats. Return a boolean if parsing was successful.
    """
    if not len(data):
        return data, False

    new_data = data
    if new_data.dtype == "object":
        if len(data) > 0:
            elements_are_numeric = all(pd.api.types.is_numeric(x) for x in data)
            if elements_are_numeric:
                new_data = pd.to_datetime(data, unit="s", origin="unix")
            else:
                # in the case of non-numeric data, keep it as is
                pass

    date_units = (self.date_unit,) if self.date_unit else ("s", "ms", "us", "ns")
    for date_unit in date_units:
        try:
            new_data = pd.to_datetime(new_data, errors="raise", format=None, unit=date_unit, infer_datetime_format=False)
        except (ValueError, OverflowError):
            continue

        all_elements_are_datetime = all(pd.api.types.is_datetime64_any_dtype(x) for x in new_data)
        if all_elements_are_datetime:
            return new_data, True

    return data, False
