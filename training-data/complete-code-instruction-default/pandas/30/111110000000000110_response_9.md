class Parser():

    def _try_convert_to_date(self, data):
        """
        Try to parse a ndarray like into a date column.

        Try to coerce object in epoch/iso formats and integer/float in epoch
        formats. Return a boolean if parsing was successful.
        """
        import numpy as np
        from pandas.core.dtypes.dtypes import DatetimeTZDtype
        from pandas.core.dtypes.common import is_datetime64_any_dtype, is_datetimetz, is_extension_array_dtype, is_object_dtype, is_integer, is_float
        from pandas.core.missing import isna
        from pandas.core.tools.datetimes import to_datetime
        from pandas.compat.numpy import datetime64, iNaT, long

        # no conversion on empty
        if not len(data):
            return data, False

        # coerce numeric to datetimes
        if is_object_dtype(data) and not is_datetimetz(data) and not isinstance(data.dtype, DatetimeTZDtype):
            # If data is object and has strings (e.g. formatted dates), try to avoid a cast to
            # strings and only parse the data, if it's not a timestamp/iso format
            data = to_datetime(data, errors="ignore")
            if is_datetime64_any_dtype(data.dtype):
                return data, True

            if not is_object_dtype(data):
                return data, False

            data = data.astype('str')

        new_data = data
        unit = self.date_unit
        if is_object_dtype(new_data):
            # Option 1: allow coercion to integer of objects
            # e.g. epoch timestamps
            if is_float(new_data).all() or is_integer(new_data).all():
                new_data = new_data.astype('int64', copy=False)

            # ignore numbers that are out of range
            if is_numeric_exact_dtype(new_data.dtype):
                in_range = (
                    isna(new_data._values)
                    | (new_data > self.min_stamp)
                    | (new_data._values == iNaT)
                )
                if not in_range.all():
                    return data, False

            if is_object_dtype(new_data):
                # option 3: strings and object dtype
                # since the date_unit is not specified, try all units to coerce
                date_units = (unit,) if unit else self._STAMP_UNITS
                for u in date_units:
                    try:
                        new_data = to_datetime(new_data, errors="ignore", unit=u)
                        return new_data, True
                    except (ValueError, OverflowError):
                        pass
                return data, False
