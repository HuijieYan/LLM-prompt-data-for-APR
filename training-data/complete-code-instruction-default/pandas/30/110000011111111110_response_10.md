```python
import numpy as np
import pandas as pd
from pandas.core.dtypes.missing import isna
from pandas.core import ops
from pandas._libs.tslibs import iNaT
from pandas.core.arrays.datetimes import DatetimeArray
from pandas.core.dtypes.common import infer_dtype_from_object

def _try_convert_to_date(self, data):
    """
    Try to parse a ndarray like into a date column.

    Try to coerce object in epoch/iso formats and integer/float in epoch
    formats. Return a boolean if parsing was successful.
    """
    # no conversion on empty
    if not len(data):
        return data, False
    
    new_data = data
    if new_data.dtype == "bool":
        return new_data, False
    
    if new_data.dtype == "object":
        inferred_type = infer_dtype_from_object(new_data)
        if inferred_type in ['string', 'datetime64']:
            new_data = DatetimeArray(new_data)
        elif inferred_type == 'integer':
            try:
                new_data = data.astype("int64")
            except (TypeError, ValueError, OverflowError):
                pass

    # ignore numbers that are out of range
    if ops.is_number(new_data):
        in_range = (
            isna(np.asarray(new_data._values, dtype=bool))
            | (new_data > self.min_stamp)
            | (new_data._values == iNaT)
        )
        if not in_range.all():
            return data, False

    date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
    for date_unit in date_units:
        new_data = DatetimeArray(new_data, tz='UTC').to_numpy(dtype='datetime64', errors='ignore', unit=date_unit)
        if new_data is not None:
            return new_data, True
    return data, False
```
