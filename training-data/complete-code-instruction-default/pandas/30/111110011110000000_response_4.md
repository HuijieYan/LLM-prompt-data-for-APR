import numpy as np
import pandas as pd
from pandas.api.types import is_numeric_dtype
from pandas.api.extensions import is_extension_array_dtype
from pandas.core.dtypes.cast import (
    maybe_cast_to_datetime,
    _delegate_to_extension_array,
)

class Parser():

    def _try_convert_to_date(self, data):
        if not len(data):
            return data, False
    
        new_data = data
        if is_extension_array_dtype(new_data):
            new_data = maybe_cast_to_datetime(new_data, copy=True)

        elif is_numeric_dtype(new_data):
            tzone = getattr(self, "tz", None)
            new_data = maybe_cast_to_datetime(new_data, copy=True, unit=self.date_unit, errors="raise", tz=tzone)

        elif new_data.dtype == "object":
            try:
                new_data = new_data.astype("int64")
            except (TypeError, ValueError, OverflowError):
                pass

        return new_data, True
