import numpy as np
import pandas as pd
from pandas.core.dtypes.common import is_datetime64_any_dtype as is_datetime

def _try_convert_to_date(self, data):
    """
    Try to parse a ndarray-like data into a date column.

    Try to coerce object in epoch/iso formats and integer/float in epoch formats. 
    Return a boolean if parsing was successful.
    """
    if not len(data):
        return data, False

    new_data = data
    if new_data.dtype == "object":
        try:
            new_data = data.astype("int64")
        except (TypeError, ValueError, OverflowError):
            pass

    if is_datetime(new_data):
        return new_data, True
    else:
        return data, False
