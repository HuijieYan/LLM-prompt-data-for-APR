@apply_wraps
def apply(self, other):
    if isinstance(other, pd.Timestamp):
        n = self.n

        # Clone timestamp
        other = pd.Timestamp(other)

        # Adjust other to the beginning of the next business hour
        if n >= 0:
            if not self.is_on_offset(other) or other.time() not in self._offset_open:
                other = self._next_opening_time(other)
        else:
            if other.time() in self._offset_open:
                # Move to the previous business day
                other = other - pd.Timedelta("1s")
            if not self.is_on_offset(other):
                other = self._next_opening_time(other)
                other = self._get_closing_time(other)

        # Get the total business hours within a day
        businesshours = sum(
            self._get_business_hours_by_sec(st, en)
            for st, en in zip(self._offset_open, self._offset_close)
        )

        bd, r = divmod(abs(n * 60), businesshours // 60)
        if n < 0:
            bd, r = -bd, -r

        # Adjust by business days first
        if bd != 0:
            skip_bd = BizDay(n=bd)
            if not self.is_on_offset(other):
                prev_open = self._prev_opening_time(other)
                remain = other - prev_open
                other = prev_open + skip_bd + remain
            else:
                other = other + skip_bd

        # Remaining business hours to adjust
        bhour_remain = pd.Timedelta(minutes=r)

        if n >= 0:
            while bhour_remain != pd.Timedelta(0):
                bhour = (self._get_closing_time(self._prev_opening_time(other)) - other)
                if bhour_remain < bhour:
                    other += bhour_remain
                    bhour_remain = pd.Timedelta(0)
                else:
                    bhour_remain -= bhour
                    other = self._next_opening_time(other + bhour)
        else:
            while bhour_remain != pd.Timedelta(0):
                bhour = self._next_opening_time(other) - other
                if bhour_remain > bhour or (bhour_remain == bhour and other.nanosecond != 0):
                    other += bhour_remain
                    bhour_remain = pd.Timedelta(0)
                else:
                    bhour_remain -= bhour
                    other = self._get_closing_time(self._next_opening_time(other + bhour - pd.Timedelta("1s")))

        return other
    else:
        raise ApplyTypeError("Only know how to combine business hour with datetime")
