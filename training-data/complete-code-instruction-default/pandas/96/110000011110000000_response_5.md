@apply_wraps
def apply(self, other):
    if isinstance(other, pd.Timestamp):
        other_time = other.time()
        n = self.n
        
        if not pd.isna(other) and other_time >= self.end:
            business_hours = self._get_business_hours_by_sec(self.start[0], self.end[0])
            shift_days, remaining_hours = divmod(abs(n), business_hours / 60)
            if n < 0:
                shift_days, remaining_hours = -shift_days, -remaining_hours
            
            adjustment = BDay(n=shift_days, normalize=True)
            adjusted_date = pd.Timestamp(other) + adjustment
            
            return adjusted_date
        
    else:
        raise ApplyTypeError("Only know how to combine business hour with pandas Timestamp")
