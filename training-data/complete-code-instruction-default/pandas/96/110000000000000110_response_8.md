# The corrected version of the buggy function
from datetime import datetime, timedelta
from pandas.tseries.offsets import CustomBusinessHour
from pandas.tseries.offsets import CustomBusinessDay

def apply(self, other):
    if isinstance(other, datetime):
        # used for detecting edge condition
        nanosecond = getattr(other, "nanosecond", 0)
        # reset timezone and nanosecond
        # other may be a Timestamp, thus not use replace
        other = datetime(
            other.year,
            other.month,
            other.day,
            other.hour,
            other.minute,
            other.second,
            other.microsecond,
        )
        n = self.n
        self.start = [CustomBusinessHour._round(CustomBusinessHour._first, self.offsets[0])]
        self.end = [CustomBusinessHour._round(CustomBusinessHour._last, self.offsets[-1])]
        
        # adjust other to reduce number of cases to handle
        if n >= 0:
            if other.time() in self.end or not CustomBusinessHour._is_on_offset(other, self.start, self.end):
                other = CustomBusinessHour._next_opening_time(other, self.start, self.end)
        else:
            if other.time() in self.start:
                # adjustment to move to previous business day
                other = other - timedelta(seconds=1)
            if not CustomBusinessHour._is_on_offset(other, self.start, self.end):
                other = CustomBusinessHour._next_opening_time(other, self.start, self.end)
                other = CustomBusinessHour._get_closing_time(other, self.start, self.end)

        # get total business hours by sec in one business day
        businesshours = sum(
            CustomBusinessHour._get_business_hours_by_sec(st, en)
            for st, en in zip(self.start, self.end)
        )

        bd, r = divmod(abs(n * 60), businesshours // 60)
        if n < 0:
            bd, r = -bd, -r

        # adjust by business days first
        if bd != 0:
            skip_bd = CustomBusinessDay(n=bd)
            # midnight business hour may not on BusinessDay
            if not skip_bd.on_offset(other):
                prev_open = CustomBusinessHour._prev_opening_time(other, self.start, self.end)
                remain = other - prev_open
                other = prev_open + skip_bd + remain
            else:
                other = other + skip_bd

        # remaining business hours to adjust
        bhour_remain = timedelta(minutes=r)

        if n >= 0:
            while bhour_remain != timedelta(0):
                # business hour left in this business time interval
                bhour = (
                    CustomBusinessHour._get_closing_time(CustomBusinessHour._prev_opening_time(other, self.start, self.end))
                    - other
                )
                if bhour_remain < bhour:
                    # finish adjusting if possible
                    other += bhour_remain
                    bhour_remain = timedelta(0)
                else:
                    # go to next business time interval
                    bhour_remain -= bhour
                    other = CustomBusinessHour._next_opening_time(other + bhour, self.start, self.end)
        else:
            while bhour_remain != timedelta(0):
                # business hour left in this business time interval
                bhour = CustomBusinessHour._next_opening_time(other, self.start, self.end) - other
                if (
                    bhour_remain > bhour
                    or bhour_remain == bhour
                    and nanosecond != 0
                ):
                    # finish adjusting if possible
                    other += bhour_remain
                    bhour_remain = timedelta(0)
                else:
                    # go to next business time interval
                    bhour_remain -= bhour
                    other = CustomBusinessHour._get_closing_time(
                        CustomBusinessHour._next_opening_time(
                            other + bhour - timedelta(seconds=1), self.start, self.end
                        ), self.start, self.end
                    )

        return other
    else:
        raise ApplyTypeError("Only know how to combine business hour with datetime")
