from datetime import datetime, timedelta
from pandas.tseries.offsets import CustomBusinessHour
import pandas as pd

def apply(self, other):
    if isinstance(other, datetime):
        # used for detecting edge condition
        nanosecond = getattr(other, "nanosecond", 0)
        # reset timezone and nanosecond
        # other may be a Timestamp, thus not use replace
        other = datetime(
            other.year,
            other.month,
            other.day,
            other.hour,
            other.minute,
            other.second,
            other.microsecond,
        )
        n = self.n
        offset = self.offset

        # adjust other to reduce number of cases to handle
        if n >= 0:
            if other.time() in self.end or not self.on_offset(other):
                other = self.next_opening_time(other)
        else:
            if other.time() in self.start:
                # adjustment to move to previous business day
                other = other - timedelta(seconds=1)
            if not self.on_offset(other):
                other = self.next_opening_time(other)
                other = self.get_closing_time(other)

        # get total business hours by sec in one business day
        business_hours = sum(
            self.get_business_hours_by_seconds(start, end)
            for start, end in zip(self.start, self.end)
        )

        business_days, remainder = divmod(abs(n * 60), business_hours // 60)
        if n < 0:
            business_days, remainder = -business_days, -remainder

        # adjust by business days first
        if business_days != 0:
            skip_business_days = CustomBusinessHour(n=business_days, start=self.start, end=self.end, holidays=self.holidays)
            if not self.next_bday.on_offset(other):
                prev_open = self.prev_opening_time(other)
                remain = other - prev_open
                other = prev_open + skip_business_days + remain
            else:
                other = other + skip_business_days

        # remaining business hours to adjust
        business_hours_remainder = timedelta(minutes=remainder)

        if n >= 0:
            while business_hours_remainder != timedelta(0):
                # business hour left in this business time interval
                business_hour = (
                    self.get_closing_time(self.prev_opening_time(other)) - other
                )
                if business_hours_remainder < business_hour:
                    # finish adjusting if possible
                    other += business_hours_remainder
                    business_hours_remainder = timedelta(0)
                else:
                    # go to next business time interval
                    business_hours_remainder -= business_hour
                    other = self.next_opening_time(other + business_hour)
        else:
            while business_hours_remainder != timedelta(0):
                # business hour left in this business time interval
                business_hour = self.next_opening_time(other) - other
                if (
                    business_hours_remainder > business_hour
                    or business_hours_remainder == business_hour
                    and nanosecond != 0
                ):
                    # finish adjusting if possible
                    other += business_hours_remainder
                    business_hours_remainder = timedelta(0)
                else:
                    # go to next business time interval
                    business_hours_remainder -= business_hour
                    other = self.get_closing_time(
                        self.next_opening_time(
                            other + business_hour - timedelta(seconds=1)
                        )
                    )

        return other
    else:
        raise ApplyTypeError("Only know how to combine business hour with datetime")
