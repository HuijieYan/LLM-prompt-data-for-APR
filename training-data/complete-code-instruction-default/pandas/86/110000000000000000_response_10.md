@Substitution("\ndata : DataFrame")
@Appender(_shared_docs["pivot"], indents=1)
def pivot(data: "DataFrame", index=None, columns=None, values=None) -> "DataFrame":
    if values is None:
        if index is None:
            indexed = data.pivot(index, columns)
        else:
            indexed = data.pivot(index, columns, data)
    else:
        indexed = data.pivot(index, columns, values)
    return indexed
