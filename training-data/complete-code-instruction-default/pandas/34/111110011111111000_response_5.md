    """
    Custom groupby class for time-interval grouping.

    ----------
    freq : pandas date offset or offset alias for identifying bin edges
    closed : closed end of interval; 'left' or 'right'

    def _get_time_bins(self, ax):
        if not isinstance(ax, DatetimeIndex):
            raise TypeError(
                "axis must be a DatetimeIndex, but got "
                f"an instance of {type(ax).__name__}"
            )

        if len(ax) == 0:
            binner = labels = ax.copy()
            return binner, [], labels

        first = ax[0].normalize()
        last = (ax[-1] + Timedelta(days=1)).normalize()

        if last.tzinfo != first.tzinfo:
            last = last.tz_localize(first.tzinfo)

        bin_edges = date_range(start=first, end=last, freq=self.freq, tz=ax.tz)
        binner = cut(ax, bin_edges, labels=False, right=self.closed)
        labels = cut(ax, bin_edges, labels=False, right=self.label)

        return binner, bin_edges, labels
