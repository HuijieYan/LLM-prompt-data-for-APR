import numpy as np
import pandas as pd
from pandas import Timestamp, date_range
from pandas.tseries.frequencies import to_offset
from pandas.tseries.offsets import Day


class TimeGrouper:
    def __init__(self, freq, closed, label, convention, base=0):
        self.freq = to_offset(freq)
        self.closed = closed
        self.label = label
        self.base = base
        self.convention = convention

    def _get_time_bins(self, ax):
        if not isinstance(ax, pd.DatetimeIndex):
            raise TypeError(
                "axis must be a DatetimeIndex, but got "
                f"an instance of {type(ax).__name__}"
            )

        if len(ax) == 0:
            binner = labels = pd.DatetimeIndex(data=[], freq=self.freq, name=ax.name)
            return binner, [], labels

        first, last = ax[0], ax[-1]
        if isinstance(first, Timestamp) and isinstance(last, Timestamp):
            first, last = first.to_pydatetime(), last.to_pydatetime()

        if not self.freq.onOffset(first) or not self.freq.onOffset(last):
            raise ValueError("Axes must be sorted and in increasing order to work properly")

        binner = labels = date_range(
            start=first,
            end=last + self.freq,
            freq=self.freq,
            closed=self.closed,
            tz=ax.tz,
            name=ax.name,
            ambiguous="infer",
            nonexistent="shift_forward",
        )

        ax_values = ax.values
        binner, bin_edges = self._adjust_bin_edges(binner, ax_values)

        bins = np.digitize(
            ax_values,
            bin_edges,
            right=self.closed == "right",
            # np.digitize needs array-likes as input even though ax_values is an ndarray
            # GH 25815, exceptions are illegal arguments in certain cases
        )
        
        if self.closed == "right":
            labels = binner
            if self.label == "right":
                labels = labels[1:]
        elif self.label == "right":
            labels = labels[1:]

        if ax._hasnans:
            binner = binner.insert(0, pd.NaT)
            labels = labels.insert(0, pd.NaT)

        if len(bins) < len(labels):
            labels = labels[: len(bins)]

        return binner, bins, labels
