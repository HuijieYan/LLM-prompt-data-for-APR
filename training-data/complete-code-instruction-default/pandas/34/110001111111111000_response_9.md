from pandas.tseries.frequencies import to_offset
from pandas.core.resample import DatetimeIndex, date_range
from pandas import NaT, DatetimeIndex, date_range, lib


def _get_time_bins(self, ax):
    if not isinstance(ax, DatetimeIndex):
        raise TypeError(
            "axis must be a DatetimeIndex, but got "
            f"an instance of {type(ax).__name__}"
        )

    if len(ax) == 0:
        binner = labels = DatetimeIndex(data=[], freq=self.freq, name=ax.name)
        return binner, [], labels

    first = ax.min()
    last = ax.max()
    first_diff = first.round("1D") - first
    last_diff = last.round("1D") - last
    last_offset = to_offset(self.freq).rollback(last)
    if last_diff.total_seconds() > 0:
        last = last_offset + to_offset(self.freq)

    binner = labels = date_range(
        freq=self.freq,
        start=first.round("1D"),
        end=last,
        tz=ax.tz,
        name=ax.name,
        ambiguous="infer",
        nonexistent="shift_forward",
    )

    ax_values = ax.asi8
    binner, bin_edges = _adjust_bin_edges(self, binner, ax_values)

    # general version, knowing nothing about relative frequencies
    bins = lib.generate_bins_dt64(
        ax_values, bin_edges, self.closed, hasnans=ax.hasnans
    )

    if self.closed == "right":
        labels = binner
        if self.label == "right":
            labels = labels[1:]
    elif self.label == "right":
        labels = labels[1:]

    if ax.hasnans:
        binner = binner.insert(0, NaT)
        labels = labels.insert(0, NaT)

    # if we end up with more labels than bins
    # adjust the labels
    # GH4076
    if len(bins) < len(labels):
        labels = labels[: len(bins)]

    return binner, bins, labels
