def _get_time_bins(self, ax):
    if not isinstance(ax, pd.DatetimeIndex):
        raise TypeError(
            "axis must be a DatetimeIndex, but got "
            f"an instance of {type(ax).__name__}"
        )

    if len(ax) == 0:
        binner = labels = pd.DatetimeIndex(data=[], freq=self._freq, name=ax.name)
        return binner, [], labels

    first = ax.min()
    last = ax.max() + to_offset(self.freq)
    bin_edges = np.concatenate(([first], ax.values))
    binner = last
    labels = bin_edges

    if self.closed == "left":
        bins = np.digitize(ax, bin_edges, right=True)
    elif self.closed == "right":
        bins = np.digitize(ax, bin_edges, right=False)

    return binner, bins, labels
