def astype_nansafe(arr, dtype, copy: bool = True):
    """
    Cast the elements of an array to a given dtype a nan-safe manner.

    Parameters
    ----------
    arr : ndarray
    dtype : np.dtype
    copy : bool, default True
        If False, a view will be attempted but may fail, if
        e.g. the item sizes don't align.

    Raises
    ------
    ValueError
        The dtype was a datetime64/timedelta64 dtype, but it had no unit.
    """
    if is_datetime64_dtype(dtype) or is_timedelta64_dtype(dtype):
        if dtype == 'datetime64':
            dtype = 'datetime64[ns]'
        elif dtype == 'timedelta64':

        if c_dtype.kind == "M":
            return arr.astype(c_dtype)

        raise TypeError(f"cannot astype a datetimelike from [{arr.dtype}] to [{c_dtype}]")

            return tslibs.ints_to_pytimedelta(arr.view(np.int64))

        elif c_dtype == _TD_DTYPE:
            return arr.astype(_TD_DTYPE, copy=copy)

        raise TypeError(f"cannot astype a timedelta from [{arr.dtype}] to [{c_dtype}]")

        # if we have a datetime/timedelta array of objects
        # then coerce to a proper dtype and recall astype_nansafe

        elif is_datetime64_dtype(c_dtype):
            from pandas import to_datetime

            return astype_nansafe(to_datetime(arr).values, c_dtype, copy=copy)
        elif is_timedelta64_dtype(c_dtype):
            from pandas import to_timedelta

            return astype_nansafe(to_timedelta(arr).values, c_dtype, copy=copy)

    if str(c_dtype) in ("datetime64", "timedelta64"):
        msg = (
            f"The '{c_dtype.name}' dtype has no unit. Please pass in "
            f"'{c_dtype.name}[ns]' instead."
        )
        raise ValueError(msg)

    if copy or is_object_dtype(arr) or is_object_dtype(c_dtype):
        # Explicit copy, or required since NumPy can't view from / to object.
        return arr.astype(c_dtype, copy=True)

    return arr.view(c_dtype)
