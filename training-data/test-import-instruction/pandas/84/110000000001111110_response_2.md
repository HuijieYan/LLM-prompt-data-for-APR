def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data
    
    # NOTE: This doesn't deal with hierarchical columns yet
    
    index = data.index
    
    cvalues = _get_level_values(data, clocs)
    
    rvalues = [index.get_level_values(i) for i in range(index.nlevels) if i not in cvalues]
    
    if isinstance(data, pd.Series):
        unstacked = data.unstack(clocs, fill_value=fill_value)
    else:
        unstacked = data.unstack(clocs, fill_value=fill_value)
        unstacked.index = pd.MultiIndex.from_product(rvalues, names=index.names[~index.names.isin(clocs)])
        
    return unstacked

# helper function to get level values
def _get_level_values(data, clocs):
    index = data.index
    return [index._get_level_number(i) for i in clocs]
