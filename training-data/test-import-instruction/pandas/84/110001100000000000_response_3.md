def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data

    index = data.index
    rlocs = [i for i in range(index.nlevels) if i not in clocs]

    clevels = [index.levels[i] for i in clocs]
    ccodes = [index.codes[i] for i in clocs]
    cnames = [index.names[i] for i in clocs]
    rlevels = [index.levels[i] for i in rlocs]
    rcodes = [index.codes[i] for i in rlocs]
    rnames = [index.names[i] for i in rlocs]

    comp_ids, obs_ids = _group_index.get_group_index(ccodes, sort=False, xnull=False)
    recons_codes = _group_index.decons_obs_group_ids(comp_ids, obs_ids, len(clevels), ccodes, xnull=False)

    dummy_index = _get_dummy_index(obs_ids, comp_ids, rlocs, rlevels, rcodes, rnames)

    if isinstance(data, Series):
        unstacked, new_columns = _unstack_series(data, clevels, cnames, recons_codes, fill_value, dummy_index)
    else:
        unstacked, new_columns = _unstack_df(data, clocs, clevels, cnames, recons_codes, fill_value, dummy_index)

    return unstacked


def _get_dummy_index(obs_ids, comp_ids, rlocs, rlevels, rcodes, rnames):
    if rlocs == []:
        # Everything is in clocs, so the dummy df has a regular index
        dummy_index = pd.Index(obs_ids, name="__placeholder__")
    else:
        dummy_index = pd.MultiIndex(
            levels=rlevels + [obs_ids],
            codes=rcodes + [comp_ids],
            names=rnames + ["__placeholder__"],
            verify_integrity=False,
        )
    return dummy_index


def _unstack_series(data, clevels, cnames, recons_codes, fill_value, dummy_index):
    dummy = data.copy()
    dummy.index = dummy_index

    unstacked = dummy.unstack("__placeholder__", fill_value=fill_value)
    new_levels = clevels
    new_names = cnames
    new_codes = recons_codes
    new_columns = pd.MultiIndex(
        levels=new_levels, codes=new_codes, names=new_names, verify_integrity=False
    )
    unstacked.index = new_columns
    return unstacked, new_columns


def _unstack_df(data, clocs, clevels, cnames, recons_codes, fill_value, dummy_index):
    if isinstance(data.columns, pd.MultiIndex):
        result = data
        for i in range(len(clocs)):
            val = clocs[i]
            result = result.unstack(val, fill_value=fill_value)
            clocs = [v if i > v else v - 1 for v in clocs]
        return result

    dummy = data.copy()
    dummy.index = dummy_index

    unstacked = dummy.unstack("__placeholder__", fill_value=fill_value)
    unstcols = unstacked.index if isinstance(unstacked, Series) else unstacked.columns
    new_levels = [unstcols.levels[0]] + clevels
    new_names = [data.columns.names[0]] + cnames

    new_codes = [unstcols.codes[0]]
    for rec in recons_codes:
        new_codes.append(rec.take(unstcols.codes[-1]))

    new_columns = pd.MultiIndex(
        levels=new_levels, codes=new_codes, names=new_names, verify_integrity=False
    )

    if isinstance(unstacked, Series):
        unstacked.index = new_columns
    else:
        unstacked.columns = new_columns

    return unstacked, new_columns
