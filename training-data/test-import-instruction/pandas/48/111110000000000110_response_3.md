def _cython_agg_blocks(
    self, how: str, alt=None, numeric_only: bool = True, min_count: int = -1
) -> "Tuple[List[Block], Index]":
    data: BlockManager = self._get_data_to_aggregate()
    
    if numeric_only:
        data = data.convert(numeric_only)
    
    new_blocks = []
    for blk in data.blocks:
        result, _ = self.grouper.aggregate(blk.values, how, axis=1, min_count=min_count)
        new_block = blk.convert(result, copy=False)
        new_blocks.append(new_block)
    
    # reset the index to correspond to the current ordering
    indexer = np.concatenate([blk.mgr_locs for blk in new_blocks])
    aggregate_items = data.items.take(np.sort(indexer))
    offset = 0
    for blk in new_blocks:
        length = len(blk.mgr_locs)
        blk.mgr_locs = indexer[offset : offset + length]
        offset += length
    
    return new_blocks, aggregate_items
