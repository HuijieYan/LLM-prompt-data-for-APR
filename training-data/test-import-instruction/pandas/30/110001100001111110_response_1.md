def _try_convert_to_date(self, data):
    """
    Try to parse an ndarray-like object into a date column.

    Try to coerce objects in epoch/iso formats and integers/floats in epoch
    formats. Return a boolean if parsing was successful.
    """
    # no conversion on empty
    if not len(data):
        return data, False

    if hasattr(data, 'dtype'):
        new_data = data
    else:
        new_data = pd.Series(data)

    if pd.api.types.is_object_dtype(new_data):
        try:
            new_data = pd.to_datetime(new_data.astype("int64"), errors="raise", unit=self.date_unit)
        except (TypeError, ValueError, OverflowError):
            pass

    # ignore numbers that are out of range
    if pd.api.types.is_numeric_dtype(new_data):
        in_range = (
            pd.isna(new_data)
            | (new_data > self.min_stamp)
            | (pd.isna(new_data))    # added pd.isna call
        )
        if not in_range.all():
            return data, False

    date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
    for date_unit in date_units:
        try:
            new_data = pd.to_datetime(new_data, errors="raise", unit=date_unit)
        except (ValueError, OverflowError):
            continue
        return new_data, True
    return data, False
