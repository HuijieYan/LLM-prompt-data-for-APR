    def _try_convert_to_date(self, data):
        """
        Try to parse a ndarray like into a date column.

        Try to coerce object in epoch/iso formats and integer/float in epoch
        formats. Return a boolean if parsing was successful.
        """
        # Early return on empty data
        if len(data) == 0:
            return data, False

        new_data = data
        if is_object_dtype(new_data):
            try:
                new_data = new_data.astype("int64")
            except (TypeError, ValueError, OverflowError):
                pass

        # Convert integer-like types to datetime
        if is_scalar(self.min_stamp):
            invalid_int64 = np.iinfo(np.int64)
            if np.isnan(self.min_stamp) or self.min_stamp <= invalid_int64.min or self.min_stamp >= invalid_int64.max:
                raise OutOfBoundsDatetime("Out of bounds nanosecond timestamp: {self.min_stamp}")

        if is_integer_dtype(new_data):
            invalid_int64 = np.iinfo(np.int64)
            if new_data.min() < invalid_int64.min or new_data.max() > invalid_int64.max:
                return data, False

            # Convert to datetime
            if self.min_stamp != 0:
                new_data += self.min_stamp
            result = DatetimeArray(new_data, dtype="datetime64[ns]", freq=None, tz=None, copy=True)
            return result, True

        # Return false for unprocessable data
        return data, False
