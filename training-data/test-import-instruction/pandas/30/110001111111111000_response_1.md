```python
import numpy as np
from pandas.core.dtypes.common import isna, classes
from pandas.core.reshape.concat import concat
from pandas.core.arrays.datetimes import DatetimeArray, integrally_scaled, should_sanitize
from pandas._libs.tslibs import iNaT

def _try_convert_to_date(self, data):
    """
    Try to parse a ndarray like into a date column.

    Try to coerce object in epoch/iso formats and integer/float in epoch
    formats. Return a boolean if parsing was successful.
    """
    if isna(data).all():
        return data, False

    new_data = data
    if new_data.dtype == "object":
        try:
            new_data = new_data.astype("int64")
        except (TypeError, ValueError, OverflowError):
            pass

    # ignore numbers that are out of range or non-convertible
    numeric_dtype = np.result_type(new_data.dtype)
    if issubclass(numeric_dtype.type, np.number):
        in_range = (
            isna(new_data) | (new_data >= self.min_stamp) | (new_data == iNaT)
        )
        if not in_range.all():
            return data, False

    date_units = [self.date_unit] if self.date_unit else self._STAMP_UNITS
    timestamp_array = None # initialize a timestamp array
    
    for date_unit in date_units:
        if should_sanitize(new_data):
            new_data = new_data.astype("datetime64")
        
        try:
            # Create the timestamp array and parse the data
            arr = integrally_scaled(new_data, date_unit, conversion_data=self.unit_data)
            arr = DatetimeArray._simple_new(arr, freq=date_unit)
            arr._data = arr._data.astype('datetime64')

            if timestamp_array is None:
                timestamp_array = arr
            else:
                timestamp_array = concat([timestamp_array, arr])
        
        except (ValueError, OverflowError):
            continue
        
        # If there were no errors, return the new_data and True
        return new_data, True

    # If all cases failed, return the original data and False
    return data, False
```
