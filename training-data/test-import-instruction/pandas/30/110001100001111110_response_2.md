import pandas as pd
import numpy as np
from pandas.api.types import is_numeric_dtype, is_object_dtype

class BuggyClass:
    def __init__(self, date_unit=None, min_stamp=31536000):
        self.date_unit = date_unit
        self.min_stamp = np.timedelta64(min_stamp, "s").astype("datetime64[ns]")
        self._STAMP_UNITS = ("s", "ms", "us", "ns")

    def _try_convert_to_date(self, data):
        if is_numeric_dtype(data):
            in_range = isna(data) | (data >= self.min_stamp)
            return data.astype("datetime64[ns]"), in_range.all()
          
        if is_object_dtype(data):
            try:
                new_data = data.astype("int64")
                in_range = isna(new_data) | (new_data >= self.min_stamp.astype("int64"))
                return new_data.astype("datetime64[ns]"), in_range.all()
            except (TypeError, ValueError, OverflowError):
                pass

        date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
        for date_unit in date_units:
            try:
                new_data = pd.to_datetime(data, errors="raise", unit=date_unit)
                return new_data, True
            except (ValueError, OverflowError):
                continue

        return data, False
