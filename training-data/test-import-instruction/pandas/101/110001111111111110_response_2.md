import numpy as np
from numpy import nan as NaN
import pandas._libs.lib as lib


def astype_nansafe(arr, dtype, copy: bool = True, skipna: bool = False):
    """
    Cast the elements of an array to a given dtype in a nan-safe manner.

    Parameters
    ----------
    arr : ndarray
    dtype : np.dtype
    copy : bool, default True
        If False, a view will be attempted but may fail, if
        e.g. the item sizes don't align.
    skipna: bool, default False
        Whether or not we should skip NaN when casting as a string-type.

    Raises
    ------
    ValueError
        The dtype was a datetime64/timedelta64 dtype, but it had no unit.
    """

    # dispatch on extension dtype if needed
    if np.issubdtype(dtype, np.datetime64):
        raise ValueError(f"The '{dtype.name}' dtype has no unit. Please pass in '{dtype.name}[ns]' instead.")
    elif np.issubdtype(dtype, np.timedelta64):
        raise ValueError(f"The '{dtype.name}' dtype has no unit. Please pass in '{dtype.name}[ns]' instead.")

    if copy or np.issubdtype(arr.dtype, np.object) or np.issubdtype(dtype, np.object):
        # Explicit copy, or required since NumPy can't view from / to object.
        return arr.astype(dtype, copy=True)

    if np.issubdtype(dtype, np.integer) and np.issubdtype(arr.dtype, np.floating):
        if not np.isfinite(arr).all():
            raise ValueError("Cannot convert non-finite values (NA or inf) to integer")

    return _astype_nansafe(arr, dtype, skipna)


def _astype_nansafe(arr, dtype, skipna):
    if assert_is_compatible_with_dtype(arr.dtype, dtype):
        return arr.astype(dtype)

    if arr.dtype.kind == "M":
        if skipna:
            arr = arr[~lib.isna(arr)]
        return arr.view(dtype)
    elif arr.dtype.kind == "m":
        if skipna:
            mask = lib.isna(arr)
            result = arr.astype(dtype).astype(np.float64)
            result[mask] = NaN
            return result
        return arr.view(dtype)

    if arr.dtype.kind == "O":
        if np.issubdtype(dtype, np.integer):
            result = arr.astype(dtype, copy=True)
            # check aenciaos for the floating precision
            if np.issubdtype(arr.dtype, np.floating) and not np.isfinite(result).all():
                raise ValueError("Cannot convert non-finite values (NA or inf) to integer")
            return result
        else:
            return arr.astype(dtype, copy=True)

    raise TypeError(f"cannot astype a datetimelike from [{arr.dtype}] to [{dtype}]")


def assert_is_compatible_with_dtype(arr_dtype, target_dtype):
    # Type conversion
    conversions = [
        ("M", "m"),
        (np.float32, np.float64),
        (np.int32, np.int64),
    ]
    if arr_dtype.kind == "O" and type(target_dtype) is not str and np.issubdtype(target_dtype, np.floating):
        arr_dtype = np.find_common_type([arr_dtype], []).dtype
    if arr_dtype != target_dtype:
        return False

    return True
