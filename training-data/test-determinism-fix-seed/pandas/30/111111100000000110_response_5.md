import numpy as np
from pandas.errors import iNaT, isna
from pandas._libs.tslibs.timestamps import Timestamp
from pandas.core.arrays import DatetimeArray
from pandas.core.dtypes.common import is_datetime64_any_dtype
from pandas.tseries.frequencies import date_range
from pandas.errors import OutOfBoundsDatetime
import pandas as pd
from pandas.errors import ParserError


class Parser():
    def _try_convert_to_date(self, data):
        """
        Try to parse a ndarray like into a date column.

        Try to coerce object in epoch/iso formats and integer/float in epoch
        formats. Return a boolean if parsing was successful.
        """
        if len(data) == 0:
            dt_ = DatetimeArray._simple_new(np.array([], dtype="datetime64[ns]"))
            return dt_, False, False

        # no conversion on empty
        if not len(data):
            return data, False

        new_data = data
        if new_data.dtype == "object":
            if len(data) > 0:
                
                timestamp = np.asarray(data, dtype='datetime64[ns]')
                if not (np.isnat(timestamp) | np.isnan(timestamp)).all():
                    data = timestamp
                else:
                    try:
                        new_data = data.astype("int64")
                    except TypeError:
                        pass

        if is_datetime64_any_dtype(new_data):
            return new_data, False, False

        if issubclass(new_data.dtype.type, np.number):
            if isna(new_data).any():
                return data, False, False

        try:
            new_data = pd.to_datetime(new_data, errors="raise")
        except (ValueError, OverflowError, pd.errors.OutOfBoundsDatetime):
            return data, False, False

        return new_data, True, False
