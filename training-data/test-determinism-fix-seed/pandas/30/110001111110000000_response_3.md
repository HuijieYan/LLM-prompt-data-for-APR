import pandas as pd
import numpy as np
from pandas.api.types import is_scalar, is_list_like
from pandas.core.dtypes.common import is_datetime64_any_dtype, is_string_dtype, is_bool_dtype
from pandas.core.dtypes.missing import isna, notna
from pandas._libs.tslibs import Timestamp, NaT
from pandas.errors import AbstractMethodError, ParserError, OutOfBoundsDatetime

class ReadJson:
    def __init__(self, data, typ=None):
        self.data = data
        self.typ = typ
        self.min_stamp = Timestamp.min.value
        self.date_unit = None

    def read(self):
        obj = self._get_object_parser(self.data)
        return obj

    def _get_object_parser(self, data):
        if self.typ == "series":
            return SeriesParser(data).parse()
        raise AbstractMethodError()

class SeriesParser:
    def __init__(self, data):
        self.data = data
        self.name = None
        self.index = None

    def parse(self):
        names = None
        return self._parse(names)

    def _parse(self, names=None):
        if is_list_like(self.data):
            values = self.data
        elif is_scalar(self.data):
            values = [self.data]
        else:
            raise ParserError()

        if is_string_dtype(values) or is_bool_dtype(values):
            result = self._try_convert_to_date(pd.Series(values, name=self.name, index=self.index))
        else:
            raise NotImplementedError()
        return result

    def _try_convert_to_date(self, data):
        if not len(data):
            return data, False

        new_data = data
        if new_data.dtype == "object":
            try:
                new_data = new_data.astype("int64")
            except (TypeError, ValueError, OverflowError):
                pass

        if issubclass(new_data.dtype.type, np.number):
            in_range = (
                isna(new_data._values)
                | (new_data > self.min_stamp)
                | (new_data._values == NaT)
            )
            if not in_range.all():
                return data, False

        date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
        for date_unit in date_units:
            try:
                new_data = pd.to_datetime(new_data, errors="coerce", unit=date_unit)
            except (ValueError, OverflowError, OutOfBoundsDatetime):
                continue
            return new_data, True
        return data, False

read_json = ReadJson

def test_readjson_bool_series():
    result = read_json("[true, true, false]", typ="series")
    expected = pd.Series([True, True, False])
    pd.testing.assert_series_equal(result, expected)
