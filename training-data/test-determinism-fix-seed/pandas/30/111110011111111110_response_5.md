import numpy as np
import pandas as pd
from pandas._libs.tslibs.nattype import NaTType
from pandas.api.types import isna
from pandas.core.dtypes.dtypes import RangeDtype
from pandas.errors import OutOfBoundsDatetime

iNaT = NaTType()

# The declaration of the class containing the buggy function
class Parser():
    def __init__(self):
        self.min_stamp = 31536000
        self.date_unit = None
        self._STAMP_UNITS = ('s', 'ms', 'us', 'ns')

    # this is the buggy function you need to fix
    def _try_convert_to_date(self, data):
        """
        Try to parse a ndarray like into a date column.

        Try to coerce object in epoch/iso formats and integer/float in epoch
        formats. Return a boolean if parsing was successful.
        """
        # no conversion on empty
        if not len(data):
            return data, False

        new_data = data.copy()
        if isinstance(new_data.dtype, RangeDtype):
            new_data = new_data.astype("int64")

        # ignore numbers that are out of range
        if issubclass(new_data.dtype.type, np.number):
            in_range = (
                isna(new_data._values)
                | (new_data > self.min_stamp)
                | (new_data._values == iNaT)
            )
            if not in_range.all():
                return data, False

        date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
        for date_unit in date_units:
            try:
                new_data = pd.to_datetime(new_data, errors="coerce", unit=date_unit)
            except (ValueError, OverflowError, OutOfBoundsDatetime):
                continue
            if new_data.isna().all():
                return data, False
            return new_data, True
        return data, False
