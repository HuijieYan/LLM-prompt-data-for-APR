```python
import numpy as np
from pandas import isna, to_datetime, iNaT

class Parser():

    def _try_convert_to_date(self, data):
        if not len(data):
            return data, False

        new_data = data
        if new_data.dtype == "object":
            try:
                new_data = data.astype("int64")
            except (TypeError, ValueError, OverflowError):
                pass

        if issubclass(new_data.dtype.type, np.number):
            in_range = (
                isna(new_data._values)
                | (new_data > self.min_stamp)
                | (new_data._values == iNaT)
            )
            if not in_range.all():
                return data, False

        date_units = (self.date_unit,) if self.date_unit else ('s', 'ms', 'us', 'ns')
        for date_unit in date_units:
            try:
                new_data = to_datetime(new_data, errors="raise", unit=date_unit)
            except (ValueError, OverflowError):
                continue
            return new_data, True
        return data, False
```
I made the following corrections:

- Imported numpy as np to resolve the reference to np.number
- Removed reference to self._STAMP_UNITS when assigning date_units, as it was not defined in the function
- Changed self._STAMP_UNITS to ('s', 'ms', 'us', 'ns') in the case where date_unit is not available.
