        if not len(data):
            return data, False

        if new_data.dtype == "object":

                new_data = new_data.astype("int64")

            )
            if not in_range.all():
                return data, False

        date_units = (None,) if self.date_unit is None else [self.date_unit]
        for date_unit in date_units:
