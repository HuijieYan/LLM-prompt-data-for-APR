        """

        from pandas.io.json._json import TimestampParser
        from pandas._libs.tslibs import parsing
    
        # no conversion on empty
        if not len(data):
            return data, False
    
        new_data = data
        if new_data.dtype == "object":
            try:
                new_data = data.astype("int64")
            except (TypeError, ValueError, OverflowError):
                pass
    
        # throw error if we are unable to parse basic dtypes
        if new_data.dtype == "int64":
            new_data = parsing.try_parse_dates(new_data, 
                                              format="{:%Y-%m-%d %H:%M:%S}")
        elif new_data.dtype == "datetime64[ns]":
            new_data = parsing.try_parse_dates(new_data,
                                              format="{:%Y-%m-%d %H:%M:%S}")

        # ignore numbers that are out of range
        if issubclass(new_data.dtype.type, np.number):
            in_range = (
                pd.isna(new_data) |
                (new_data > self.min_stamp) |
                (new_data == TimestampParser.iNaT)
            )
            if not in_range.all():
                return data, False
    
        date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
        for date_unit in date_units:
            try:
                new_data = pd.to_datetime(new_data, errors="raise", unit=date_unit)
            except (ValueError, OverflowError):

            return new_data, True
