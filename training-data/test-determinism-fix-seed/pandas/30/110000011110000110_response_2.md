import pandas as pd
import numpy as np
from pandas.core.dtypes.missing import isna, iNaT
from pandas._libs.tslibs.nattype import NaTType
from pandas.core.tools.datetimes import to_datetime

def _try_convert_to_date(self, data):
    if not len(data):
        return data, False

    new_data = data
    if new_data.dtype == "object":
        try:
            new_data = pd.to_numeric(new_data, errors='ignore')
        except (TypeError, ValueError, OverflowError):
            pass

    if issubclass(new_data.dtype.type, np.number) and not isna(new_data).all():
        min_stamp = pd.Timestamp.min.to_datetime64()
        in_range = isna(new_data._values) | (new_data > min_stamp) | (new_data._values == iNaT)
        if not in_range.all():
            return data, False

    date_unit = (self.date_unit,) if self.date_unit else getattr(self, "_STAMP_UNITS", None)

        try:
            new_data = to_datetime(new_data, errors="raise", unit=date_unit)
        except (ValueError, OverflowError):
            continue
        return new_data, True
    return data, False
