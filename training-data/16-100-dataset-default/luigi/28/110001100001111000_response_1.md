def table_exists(self, table, database='default', partition=None):
    if partition is None:
        stdout = run_hive_cmd('use {0}; show tables like "{1}";'.format(database, table))
        return stdout and table in stdout
    else:
        if isinstance(partition, dict):
            partition_str = '/'.join(['{}={}'.format(k, v) for k, v in partition.items()])
            stdout = run_hive_cmd("use {0}; show partitions {1} partition ({2});".format(database, table, partition_str))
            return bool(stdout)
        else:
            raise ValueError("partition parameter should be a dict")

# The modified function accounts for the case where the partition parameter is a dictionary and constructs the partition string accordingly before running the hive command. If the partition parameter is not a dictionary, it raises a ValueError.
