import pandas as pd
from pandas.core.indexes.multi import MultiIndex
from pandas.core.reshape.pivot import pivot
import pytest
from pandas._testing import assert_frame_equal
from pandas.tests.reshape.test_pivot import TestPivotTable as parent
from pandas.util import Substitution, Appender
from pandas.errors import TypeError

_shared_docs = {
    "pivot": """
Pivot a DataFrame from long to wide format, indicating specified columns and values.
""",
}

# The source code of the corrected function
@Substitution("\ndata : DataFrame")
@Appender(_shared_docs["pivot"], indents=1)
def pivot(data: "DataFrame", index=None, columns=None, values=None) -> "DataFrame":
    if values is None:
        cols = [columns] if index is None else [index, columns]
        append = index is None
        indexed = data.set_index(cols, append=append)
    else:
        if index is None:
            index = data.index
            index = data[index]
        index = MultiIndex.from_arrays([index, data[columns]])

        if isinstance(values, list) and not isinstance(values, tuple):
            # Exclude tuple because it is seen as a single column name
            indexed = pd.DataFrame(data[values].values, index=index, columns=values)
        else:
            indexed = pd.DataFrame(data[values].values, index=index)
    result = indexed.unstack(columns)
    return result

# A fixed test function for the corrected function
class TestPivotTable(parent):
    def test_pivot_columns_none_raise_error(self):
        # GH 30924
        df = pd.DataFrame(
            {"col1": ["a", "b", "c"], "col2": [1, 2, 3], "col3": [1, 2, 3]}
        )
        msg = r"pivot\(\) missing 1 required argument: 'columns'"
        with pytest.raises(TypeError, match=msg):
            pivot(df, index="col1", values="col3")
