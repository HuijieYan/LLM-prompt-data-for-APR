def _cython_agg_blocks(
    self, how: str, alt=None, numeric_only: bool = True, min_count: int = -1
) -> "Tuple[List[Block], Index]":
    # TODO: the actual managing of mgr_locs is a PITA
    # here, it should happen via BlockManager.combine

    data: BlockManager = self._get_data_to_aggregate()

    if numeric_only:
        data = data.convert(numeric=True, copy=False)

    agg_blocks: List[Block] = []
    new_items: List[np.ndarray] = []
    deleted_items: List[np.ndarray] = []
    split_items: List[np.ndarray] = []

    no_result = object()
    for block_id, block in data.items():
        result = no_result
        locs = np.array(block.mgr_locs, copy=False, dtype=block.mgr_locs.dtype)
        try:
            result, _ = self.grouper.aggregate(
                block.values, how, axis=1, min_count=min_count
            )
        except NotImplementedError:
            if alt is None:
                assert how == "ohlc"
                deleted_items.append(locs)
                continue
            obj = self.obj[locs]
            if isinstance(obj, DataFrame):
                if len(obj.columns) == 1:
                    obj = obj.iloc[:, 0]
            else:
                obj = obj.squeeze()
            result = obj.groupby(self.grouper, sort=False).agg(alt)

        if not isinstance(result, DataFrame):
            if result is no_result:
                deleted_items.append(locs)
                continue
            if len(result._data.blocks) != 1:
                split_items.append(result._data.blocks[0].mgr_locs)
                split_frames.append(result)
                continue
            result = result._data.blocks[0].values
            if isinstance(result, np.ndarray) and result.ndim == 1:
                result = result.reshape(1, -1)

        if isinstance(result, DataFrame):
            raise ValueError("aggregate returned a DataFrame")

        if result is not no_result:
            result = maybe_downcast_numeric(result, block.dtype)
            if is_extension_array_dtype(block.dtype) and isinstance(result, np.ndarray):
                result = type(block.values)._from_sequence(
                    result, dtype=block.dtype, copy=False
                )
                result, _ = make_block(result, locs, ndim=2)
            else:
                result, _ = make_block(result, locs, ndim=2)

        new_items.append(locs)
        agg_blocks.append(result)
    
    if not (agg_blocks or split_frames):
        raise DataError("No numeric types to aggregate")

    if split_items:
        for split_locs, split_result in zip(split_items, split_frames):
            new_items.append(split_locs)
            split_result, _ = make_block(split_result, split_locs, ndim=2)
            agg_blocks.append(split_result)

        deleted = np.concatenate(deleted_items)
        indexer = sorted(set(range(len(data))).difference(deleted))

        indexer = sorted(np.unique(new_items, return_index=True)[1])

    if indexer.shape[0] < len(data.items):
        data = data.reindex(item=new_items[indexer])
        agg_blocks = data.blocks

    return agg_blocks, data.items.take(indexer)
