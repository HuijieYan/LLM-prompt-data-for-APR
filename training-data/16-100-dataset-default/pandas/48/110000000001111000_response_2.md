def _cython_agg_blocks(
    self, how: str, alt=None, numeric_only: bool = True, min_count: int = -1
) -> "Tuple[List[Block], Index]":
    data: BlockManager = self._get_data_to_aggregate()
    
    if numeric_only:
        data = data.select_dtypes(include=["number"], exclude=["boolean"])

    if data.empty:
        raise ValueError("All numeric types to aggregate can't be empty")

    agg_blocks: List[Block] = []

    num_blocks = data._get_numeric_data()
    for block in num_blocks.blocks:
        result = pd.DataFrame(block.mgr_locs)
        agg_blocks.append(result)

    return agg_blocks
