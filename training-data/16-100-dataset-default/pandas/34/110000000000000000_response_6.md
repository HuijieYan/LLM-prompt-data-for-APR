def _get_time_bins(self, ax: DatetimeIndex):
    if not isinstance(ax, DatetimeIndex):

        )

    if len(ax) == 0:
        binner = pd.DatetimeIndex(data=[], freq=self.freq, name=ax.name)
        return binner, [], binner

    first, last = ax.min(), ax.max()
    binner = pd.date_range(
        freq=self.freq,
        start=first,
        end=last,
        tz=ax.tz,
        name=ax.name,
        closed=self.closed,
        base=self.base,
        ambiguous="infer",
        nonexistent="shift_forward"
    )

    ax_values = ax.asi8
    bin_edges = self._adjust_bin_edges(binner, ax_values)

    bins = pd.arrays._generate_bins_dt64(
        ax_values, bin_edges, self.closed, has_nans=ax.hasnans
    )

    if self.closed == "right":
        labels = binner[1:] if self.label == "right" else binner
    elif self.label == "right":
        labels = binner[1:]

    if ax.hasnans:
        binner = binner.insert(0, pd.NaT)
        labels = labels.insert(0, pd.NaT)

    if len(bins) < len(labels):
        labels = labels[: len(bins)]

    return binner, bins, labels
