def _get_time_bins(self, ax):
    if not isinstance(ax, pd.DatetimeIndex):  # Fixed: Changed DatetimeIndex to pd.DatetimeIndex
        raise TypeError(
            "axis must be a DatetimeIndex, but got "
            f"an instance of {type(ax).__name__}"
        )

    if len(ax) == 0:
        binner = labels = pd.DatetimeIndex(data=[], freq=self.freq, name=ax.name)  # Fixed: Changed DatetimeIndex to pd.DatetimeIndex
        return binner, [], labels

    binner, last = _get_timestamp_range_edges(  # Changed first to binner, last to last
        ax.min(), ax.max(), self.freq, closed=self.closed, base=self.base
    )
    # GH #12037
    # use first/last directly instead of call replace() on them
    # because replace() will swallow the nanosecond part
    # thus last bin maybe slightly before the end if the end contains
    # nanosecond part and lead to `Values falls after last bin` error
    labels = pd.date_range(  # Fixed: Changed date_range to pd.date_range
        freq=self.freq,
        start=binner,  # Changed first to binner
        end=last,
        tz=ax.tz,
        name=ax.name,
        ambiguous="infer",
        nonexistent="shift_forward",
    )

    ax_values = ax.asi8
    binner, bin_edges = self._adjust_bin_edges(binner, ax_values)  # Fixed: Changed first to binner

    # general version, knowing nothing about relative frequencies
    bins = lib.generate_bins_dt64(
        ax_values, bin_edges, self.closed, hasnans=ax.hasnans
    )

    if self.closed == "right":
        labels = binner
        if self.label == "right":
            labels = labels[1:]
    elif self.label == "right":
        labels = labels[1:]

    if ax.hasnans:
        binner = binner.insert(0, pd.NaT)  # Fixed: Changed NaT to pd.NaT
        labels = labels.insert(0, pd.NaT)  # Fixed: Changed NaT to pd.NaT

    # if we end up with more labels than bins
    # adjust the labels
    # GH4076
    if len(bins) < len(labels):
        labels = labels[: len(bins)]

    return binner, bins, labels
