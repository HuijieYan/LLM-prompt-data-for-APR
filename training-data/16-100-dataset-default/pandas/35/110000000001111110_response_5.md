import weakref

def cache_readonly(fn):
    cache = {}
    def wrapper(self):
        if fn.__name__ not in cache:
            cache[fn.__name__] = weakref.ref(self)
        cached = cache[fn.__name__]
        return fn(cached(), len(self))
    return wrapper

@cache_readonly
def _engine(self, length):
    # To avoid a reference cycle, pass a weakref of self to _engine_type.
    period = weakref.ref(self)
    return self._engine_type(period, length)
