import weakref

def cache_readonly(func):
    cache = {}
    def inner(self):
        if self not in cache:
            cache[self] = func(self)
        return cache[self]
    return inner

@cache_readonly
def _engine(self):
    # To avoid a reference cycle, pass a weakref of self to _engine_type.
    period = weakref.ref(self)
