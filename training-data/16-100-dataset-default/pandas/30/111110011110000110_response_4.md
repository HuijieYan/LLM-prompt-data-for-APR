import pandas as pd
import numpy as np
from pandas.errors import iNaT
from pandas.api.types import isna
from pandas._libs.tslibs.timestamps import Timestamp
from pandas.core.arrays import DatetimeArray

class Parser():
    def __init__(self):
        self.date_unit = ''
        self.min_stamp = Timestamp('1970-01-01', tz=None)
        self._STAMP_UNITS = ['ns', 'datetime64[ns]']

    def _try_convert_to_date(self, data):
        new_data = data
        if new_data.dtype == "object":
            try:
                new_data = new_data.astype("int64")
            except (TypeError, ValueError, OverflowError):
                pass

        if issubclass(new_data.dtype.type, np.number):
            in_range = (
                isna(new_data._values)
                | (new_data > self.min_stamp.value)
                | (new_data._values == iNaT)
            )
            if not in_range.all():
                return data, False

        date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
        for date_unit in date_units:
            try:
                new_data = DatetimeArray(new_data, dtype='datetime64', tz=None)
            except (ValueError, OverflowError):
                continue
            return new_data, True
        return data, False


def read_json(json_data, typ):
    data = pd.Series(eval(json_data))
    parser = Parser()
    if typ == "series":
        return parser._try_convert_to_date(data)
    else:
        return data
