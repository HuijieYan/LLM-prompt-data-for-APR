import pandas as pd

class Parser():

    def _try_convert_to_date(self, data):
        if not len(data):
            return data, False

        new_data = data
        if new_data.dtype == "object":
            try:
                new_data = pd.to_datetime(data, errors="coerce")
            except (TypeError, ValueError, OverflowError):
                pass

        return new_data, True
