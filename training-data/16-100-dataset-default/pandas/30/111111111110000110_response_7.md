import numpy as np
import pandas as pd
from pandas.core.dtypes.common import is_integer
from pandas.core.dtypes.common import is_float
from pandas.core.dtypes.common import is_datetime64_any_dtype as is_datetime
from pandas.core.dtypes.common import is_bool_dtype
from pandas.core.dtypes.common import is_scalar

class Parser():

    def _try_convert_to_date(self, data):
        if not len(data):
            return data, True  # Changed to True to indicate successful parsing
    
        new_data = data
        if is_bool_dtype(new_data):
            return new_data, False  # Return original data and False
    
        if is_integer(new_data) or is_float(new_data):
            if is_scalar(data):
                new_data = [new_data]
            return new_data, False  # Return original data and False
    
        if is_datetime(new_data):
            return new_data, False  # Return original data and False
    
        return data, False  # Return original data and False

# Test function
def test_readjson_bool_series():
    result = pd.read_json("[true, true, false]", typ="series")
    expected = pd.Series([True, True, False])
    assert result.equals(expected)
