import numpy as np
from pandas import to_datetime, isna

class Parser():

    def _try_convert_to_date(self, data, min_stamp, date_unit):
        if not len(data):
            return data, False
        
        new_data = data
        if new_data.dtype == "object":
            try:
                new_data = new_data.astype("int64")
            except (TypeError, ValueError, OverflowError):
                pass

        if issubclass(new_data.dtype.type, np.number):
            in_range = (
                isna(new_data._values)
                | (new_data > min_stamp)
                | (new_data._values == np.iinfo(new_data.dtype).max)
            )
            if not in_range.all():
                return data, False

        for unit in (date_unit,) if date_unit else self._STAMP_UNITS:
            try:
                new_data = to_datetime(new_data, errors="raise", unit=unit)
                return new_data, True
            except (ValueError, OverflowError):
                continue
        
        return data, False
