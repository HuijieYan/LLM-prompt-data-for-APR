import pandas as pd

        if not len(data):  # no conversion on empty
            return data, False

        if new_data.dtype == "object":
            try:
                new_data = data.astype("int64")
            except (TypeError, ValueError, OverflowError):
                pass

            return new_data, False  # return original data if not a number
