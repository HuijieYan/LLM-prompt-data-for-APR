import numpy as np
from pandas import to_datetime

def _try_convert_to_date(self, data):
    if len(data) == 0:  # Fix the condition to check for empty data
        return data, False

    new_data = data.copy()  # Use copy to create a new array for manipulation
    if new_data.dtype == "object":
        try:
            new_data = new_data.astype("int64")  # Update "data" to "new_data"
        except (TypeError, ValueError, OverflowError):
            pass

    if issubclass(new_data.dtype.type, np.number) and "min_stamp" in self.__dict__:
        in_range = (
            new_data._values < self.min_stamp
        )
        if not in_range.all():
            return data, False

    date_units = (self.date_unit,) if self.date_unit else (None,)  # Add None as fallback in case self.date_unit is not defined
    for date_unit in date_units:
        try:
            new_data = to_datetime(new_data, errors="coerce", unit=date_unit)  # Correct errors parameter and use "coerce" to convert boolean values to NaT
        except (ValueError, OverflowError):
            continue
        return new_data, True

    return data, False
