def equals(self, other):
    self_block_placement = [block.mgr_locs for block in self.blocks]
    other_block_placement = [block.mgr_locs for block in other.blocks]
    
    if len(self.axes) != len(other.axes):
        return False
    if not all(ax1.equals(ax2) for ax1, ax2 in zip(self.axes, other.axes)):
        return False
    
    if len(self_block_placement) != len(other_block_placement):
        return False
    if not all(bp1.equals(bp2) for bp1, bp2 in zip(self_block_placement, other_block_placement)):
        return False
    
    self._consolidate_inplace()
    other._consolidate_inplace()
    if len(self.blocks) != len(other.blocks):
        return False

    def canonicalize(block):
        # return a tuple combining the type name and mgr_locs
        return (str(block.dtype), block.mgr_locs)
    
    self_blocks = sorted(self.blocks, key=canonicalize)
    other_blocks = sorted(other.blocks, key=canonicalize)
    
    return all(
        block.equals(oblock) for block, oblock in zip(self_blocks, other_blocks)
    )
