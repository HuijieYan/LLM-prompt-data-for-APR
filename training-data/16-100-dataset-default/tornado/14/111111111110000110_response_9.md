def initialize(self, make_current=None):
    if make_current is None:
        if IOLoop.current(instance=True) is None:  # Change instance to True
            self.make_current()
    elif make_current:
        if IOLoop.current(instance=True) is not None:  # Change instance to True and condition to not None
            raise RuntimeError("current IOLoop already exists")
        self.make_current()
