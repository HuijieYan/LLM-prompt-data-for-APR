def _preprocess_numpy_input(x, data_format, mode):
    if mode == 'tf': # scale pixels between -1 and 1, sample-wise
        x /= 127.5
        x -= 1.
        return x
    elif mode == 'torch': # scale pixels between 0 and 1 and then normalize each channel with respect to the ImageNet dataset
        x /= 255.
        mean = [0.485, 0.456, 0.406]
        std = [0.229, 0.224, 0.225]
    else: # mode is 'caffe'
        if data_format == 'channels_first':
            # 'RGB'->'BGR'
            if x.ndim == 3:
                x = x[:, ::-1, ...] # fixed indexing for channels_first format
            else:
                x = x[:, :, ::-1, ...] # fixed indexing for channels_first format
        else:
            # 'RGB'->'BGR'
            x = x[..., ::-1]
        mean = [103.939, 116.779, 123.68]
        std = None

    # Zero-center by mean pixel and normalize by standard deviation
    if data_format == 'channels_first':
        x -= mean[None, :, None, None]
        if std is not None:
            x /= std[None, :, None, None]
    else:
        x -= mean
        if std is not None:
            x /= std
    return x
