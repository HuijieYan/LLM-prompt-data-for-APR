def _preprocess_numpy_input(x, data_format, mode):
    if x.dtype != 'float32':
        x = x.astype('float32')
    
    if mode == 'tf':
        x /= 127.5
        x -= 1.
        return x

    if mode == 'torch':
        x /= 255.
        mean = [0.485, 0.456, 0.406]
        std = [0.229, 0.224, 0.225]
    else:
        if data_format == 'channels_first':
            # 'RGB'->'BGR'
            if x.ndim == 3:
                x = x[:, ::-1, ...]
            else:
                x = x[..., ::-1]
            mean = [103.939, 116.779, 123.68]
            std = None
        else:
            # 'RGB'->'BGR'
            x = x[..., ::-1]
            mean = [103.939, 116.779, 123.68]
            std = None

    # Zero-center by mean pixel
    x -= mean
    if std is not None:
        x /= std
    return x
