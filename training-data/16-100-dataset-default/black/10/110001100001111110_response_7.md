def _partially_consume_prefix(self, prefix, column):
    lines = []
    current_line = ""
    current_column = 0
    wait_for_nl = False
    for char in prefix:
        current_line += char
        if wait_for_nl:
            if char == '\n':
                if current_line.strip() and current_column < column:
                    res = ''.join(lines)
                    return res, prefix[len(res):]

                lines.append(current_line)
                current_line = ""
                current_column = 0
                wait_for_nl = False
                continue  # Add this line to skip the remaining code in the loop
        if char == ' ':
            current_column += 1
        elif char == '\t':
            current_column += 8 - (current_column % 8)  # Adjust the column for tab characters
        elif char == '\n':
            current_column = 0
            wait_for_nl = True  # Set wait_for_nl after resetting the column
        else:
            # indent is finished
            wait_for_nl = False  # Set wait_for_nl to False when indent is finished
    return ''.join(lines), current_line
