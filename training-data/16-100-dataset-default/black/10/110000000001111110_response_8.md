def _partially_consume_prefix(self, prefix, column):
    lines = []
    current_line = ""
    current_column = 0
    wait_for_nl = False
    for char in prefix:
        current_line += char
        if wait_for_nl:
            if char == '\n':
                if current_line.strip() and current_column < column:
                    res = ''.join(lines)
                    return res, prefix[len(res):]

                lines.append(current_line)
                current_line = ""
                current_column = 0
                wait_for_nl = False
        elif char == ' ':
            current_column += 1
        elif char == '\t':
            current_column += 8  # Fixed: increase current_column by 8 instead of 4
        elif char == '\n':
            lines.append(current_line)  # Fixed: Move this line to handle all cases of '\n'
            current_line = ""
            current_column = 0
            wait_for_nl = False
        else:
            wait_for_nl = True
    res = ''.join(lines)  # Fixed: Move the join outside the loop to include the last line
    return res, prefix[len(res):] if len(prefix) > len(res) else ""
