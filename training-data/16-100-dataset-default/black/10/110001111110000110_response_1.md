def _partially_consume_prefix(self, prefix, column):
    lines = []
    current_line = ""
    current_column = 0
    wait_for_nl = False
    for char in prefix:
        current_line += char
        if wait_for_nl:
            if char == '\n':
                if current_line.strip() and current_column < column:
                    res = ''.join(lines)
                    return res, prefix[len(res):]

                lines.append(current_line)
                current_line = ""
                current_column = 0
                wait_for_nl = False
                continue  # Add continue statement to skip the rest of the loop iteration
        elif char == ' ':
            current_column += 1
        elif char == '\t':
            current_column += 4
        elif char == '\n':
            lines.append(current_line)  # Move this line here to handle unexpected empty line
            current_line = ""  # Reset current_line for new line
            current_column = 0  # Reset current_column for new line
            continue  # Add continue statement to skip the rest of the loop iteration
        else:
            wait_for_nl = True
    lines.append(current_line)  # Append remaining line outside the loop
    return ''.join(lines), current_line
