def _partially_consume_prefix(self, prefix, column):
    lines = []
    current_line = ""
    current_column = 0
    wait_for_nl = False
    for char in prefix:
        current_line += char
        if wait_for_nl:
            if char == '\n':
                if current_line.strip() and current_column < column:
                    res = ''.join(lines) + current_line  # Changed here to include the current_line
                    return res, prefix[len(res):]
                lines.append(current_line)
                current_line = ""
                current_column = 0
                wait_for_nl = False
        elif char == ' ':
            current_column += 1
        elif char == '\t':
            current_column += 4
        elif char == '\n':
            lines.append(current_line)  # Moved this inside the 'if' block
            current_line = ""  # Reset current_line here instead of setting it to ''
            current_column = 0
            wait_for_nl = False
        else:
            wait_for_nl = True
    return ''.join(lines), current_line  # Changed here to return current_line instead of joining it with lines
