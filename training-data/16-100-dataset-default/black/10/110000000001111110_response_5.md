def _partially_consume_prefix(self, prefix, column):
    lines = []
    current_line = ""
    current_column = 0
    wait_for_nl = False
    for char in prefix:
        current_line += char
        if wait_for_nl:
            if char == '\n':
                if current_line.strip() and current_column < column:
                    res = ''.join(lines)
                    return res, prefix[len(res):]

                lines.append(current_line)
                current_line = ""
                current_column = 0
                wait_for_nl = False
                continue
        elif char == ' ':
            current_column += 1
        elif char == '\t':
            current_column += 8  # Change from 4 to 8 to match the standard tab size
        elif char == '\n':
            if current_line.strip():  # Check for unexpected empty line after dedent
                if current_column < column:  # Check if the column position is less than the required
                    res = ''.join(lines)
                    return res, prefix[len(res):]
            current_column = 0
            wait_for_nl = False
        else:
            # indent is finished
            wait_for_nl = True
    return ''.join(lines), current_line
