def _partially_consume_prefix(self, prefix, column):
    lines = []
    current_line = ""
    current_column = 0
    wait_for_nl = False
    for char in prefix:
        current_line += char
        if wait_for_nl:
            if char == '\n':
                if current_line.strip() and current_column < column:
                    res = ''.join(lines) + current_line  # Updated to include the current line
                    return res, prefix[len(res):]
                lines.append(current_line)
                current_line = ""
                current_column = 0
                wait_for_nl = False
        elif char == ' ':
            current_column += 1
        elif char == '\t':
            current_column += 4
        elif char == '\n':
            current_column = 0
        else:
            wait_for_nl = True
            # Check for empty line and append it to lines
            if current_line.strip() and current_column < column:
                res = ''.join(lines)
                return res, prefix[len(res):]
    return ''.join(lines), current_line
