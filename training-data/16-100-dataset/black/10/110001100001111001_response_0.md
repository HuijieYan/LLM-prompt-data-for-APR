def _partially_consume_prefix(self, prefix, column):
    lines = []
    current_line = ""
    current_column = 0
    wait_for_nl = False
    for char in prefix:

            if char == '\n':

                    return res, prefix[len(res):]
                lines.append(current_line)
                current_line = ""
                current_column = 0
                wait_for_nl = False
            else:
                current_line = ""
                current_column = 0
                wait_for_nl = False

            current_column += 1

        elif char == '\n':

        else:
