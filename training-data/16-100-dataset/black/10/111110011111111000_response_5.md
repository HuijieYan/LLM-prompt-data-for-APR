def _partially_consume_prefix(self, prefix, column):
    lines = []
    current_line = ""
    current_column = 0
    wait_for_nl = False
    for char in prefix:
        current_line += char
        if wait_for_nl:
            if char == '\n':
                if current_line.strip() and current_column < column:
                    # Check if a newline character creates a string that exceeds the column
                    res = ''.join(lines)
                    return res, prefix[len(res):]

                lines.append(current_line)
                current_line = ""
                current_column = 0
                wait_for_nl = False
        elif char == ' ':
            # Increment the column count for spaces
            current_column += 1
        elif char == '\t':
            # Increment the column count for tabs
            current_column += 4
        elif char == '\n':
            # Reset the column count for a new line
            current_column = 0
            wait_for_nl = True
        else:
            # Set flag to wait for newline
            wait_for_nl = True
