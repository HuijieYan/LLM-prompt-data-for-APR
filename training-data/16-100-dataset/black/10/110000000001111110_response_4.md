def _partially_consume_prefix(self, prefix, column):
    lines = []
    current_line = ""
    current_column = 0
    wait_for_nl = False
    for char in prefix:
        current_line += char
        if wait_for_nl:
            if char == '\n':
                if current_line.strip() and current_column <= column:  # changed '<' to '<='
                    res = ''.join(lines)
                    return res, prefix[len(res):]

                lines.append(current_line)
                current_line = ""
                current_column = 0
                wait_for_nl = False
        elif char == ' ':
            current_column += 1
        elif char == '\t':
            current_column += 4
        elif char == '\n':
            lines.append(current_line)  # moved this line to include the current line
            current_line = ""  # moved this line to include the current line
            current_column = 0
        else:
            current_column += 1  # changed current_column += 1 to include all characters other than space and tab
            wait_for_nl = True
    return ''.join(lines), current_line  # moved this line outside the loop
