def _partially_consume_prefix(self, prefix, column):
    lines = []
    current_line = ""
    current_column = 0
    wait_for_nl = False
    for char in prefix:
        if wait_for_nl:
            if char == '\n':
                if current_line.strip() and current_column < column:
                    res = ''.join(lines)
                    return res, prefix[len(res):]
                lines.append(current_line)
                current_line = ""
                current_column = 0
                wait_for_nl = False
        if char == ' ':
            current_column += 1
            current_line += char
        elif char == '\t':
            current_column += 4
            current_line += char
        elif char == '\n':
            current_column = 0
            lines.append(current_line + char)
            current_line = ""
            wait_for_nl = False
        else:
            wait_for_nl = True
            current_line += char
    return ''.join(lines), current_line
