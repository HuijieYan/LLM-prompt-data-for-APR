    lines = []
    current_line = ""
    current_column = 0
    wait_for_nl = False

                current_line = ""
                current_column = 0
                wait_for_nl = False

            current_column += 1
        elif char == '\t':
            current_column += 4
        elif char == '\n':
            # unexpected empty line
            current_column = 0
            current_line = ""  # reset current_line to an empty string
        else:
            # indent is finished
            wait_for_nl = True
