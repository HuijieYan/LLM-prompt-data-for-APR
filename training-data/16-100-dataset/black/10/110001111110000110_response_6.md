def _partially_consume_prefix(self, prefix, column):
    lines = []
    current_line = ""
    current_column = 0
    wait_for_nl = False
    for char in prefix:
        current_line += char
        if wait_for_nl:
            if char == '\n':
                if current_line.strip() and current_column < column:
                    res = ''.join(lines)
                    return res, prefix[len(res):]

                lines.append(current_line)
                current_line = ""
                current_column = 0
                wait_for_nl = False
        elif char == ' ':
            current_column += 1
        elif char == '\t':
            current_column += 4
        elif char == '\n':
            # unexpected empty line
            current_column = 0
            lines.append(current_line)  # Move this line here to handle unexpected empty lines correctly
            current_line = ""  # Reset current_line after adding it to lines
        else:
            # indent is finished
            wait_for_nl = True
    lines.append(current_line)  # Add the remaining current_line to lines after the loop ends
    return ''.join(lines), current_line
