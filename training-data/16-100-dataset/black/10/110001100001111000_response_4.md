def _partially_consume_prefix(self, prefix, column):
    lines = []
    current_line = ""
    current_column = 0
    wait_for_nl = False
    for char in prefix:
        if wait_for_nl:
            if char == '\n':
                if current_line.strip() and current_column < column:
                    res = ''.join(lines)
                    return res, prefix[len(res):]
                else:
                    lines.append(current_line)
                    current_line = ""
                    current_column = 0
                    wait_for_nl = False
        elif char in [' ', '\t']:
            current_column += 1 if char == ' ' else 4
        elif char == '\n':
            current_column = 0
        else:
            wait_for_nl = True
        current_line += char
    return ''.join(lines), current_line
