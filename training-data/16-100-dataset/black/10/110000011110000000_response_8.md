def _partially_consume_prefix(self, prefix, column):
    lines = []
    current_column = 0
    for char in prefix:
        if char == '\n':
            current_column = 0
        else:
            if char == ' ':
                current_column += 1
            elif char == '\t':
                current_column += 4
            lines.append(char)
            if current_column >= column:
                break
    return ''.join(lines), prefix[len(lines):]
