# The declaration of the class containing the fixed function
class Driver(object):
  def _partially_consume_prefix(self, prefix, column):
      lines = []
      current_line = ""
      current_column = 0
      wait_for_nl = False
      tabs_count = 0
      for char in prefix:
          if char == '\t':
              tabs_count += 1
              current_column += 4
          else:
              current_line += char
              if wait_for_nl:
                  if char == '\n':
                      if current_line.strip() and current_column < column:
                          res = ''.join(lines)
                          return res, prefix[len(res):]

                      lines.append(current_line)
                      current_line = ""
                      current_column = 0
                      wait_for_nl = False
              elif char == ' ':
                  current_column += 1
              elif char == '\n':
                  # unexpected empty line
                  current_column = 0
                  if tabs_count > 0:
                      current_column += 4 * tabs_count
                      tabs_count = 0
              else:
                  # indent is finished
                  wait_for_nl = True
      return ''.join(lines), current_line
