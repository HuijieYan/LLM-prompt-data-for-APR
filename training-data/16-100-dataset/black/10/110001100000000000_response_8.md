def _partially_consume_prefix(self, prefix, column):
    lines = []
    current_line = ""
    current_column = 0
    wait_for_nl = False
    for char in prefix:
        current_line += char
        if wait_for_nl:
            if char == '\n':
                if current_line.strip() and current_column < column:
                    res = ''.join(lines)
                    return res, prefix[len(res):]

                lines.append(current_line)
                current_line = ""
                current_column = 0
                wait_for_nl = False
        elif char == ' ':
            current_column += 1
        elif char == '\t':
            current_column += 8  # changed from 4 to 8 to handle tab as 8 spaces
        elif char == '\n':
            lines.append(current_line)  # moved this line here to store the current line in lines array
            current_line = ""  # reset current_line after storing it in lines array
            current_column = 0
            wait_for_nl = False
        else:
            # indent is finished
            wait_for_nl = True
    return ''.join(lines), current_line
