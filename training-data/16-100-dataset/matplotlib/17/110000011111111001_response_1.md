def nonsingular(vmin, vmax, expander=0.001, tiny=1e-15, increasing=True):

        vmin, vmax = vmax, vmin

        vmin = -expander
        vmax = expander

        vmax += expander * np.abs(vmax)

    return vmin, vmax
