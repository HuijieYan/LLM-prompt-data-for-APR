```python
import pandas as pd

def pivot(data: pd.DataFrame, index=None, columns=None, values=None) -> pd.DataFrame:
    if values is None:
        pivot_table = data.pivot(index=index, columns=columns)
    else:
        pivot_table = data.pivot(index=index, columns=columns, values=values)

    return pivot_table
```
