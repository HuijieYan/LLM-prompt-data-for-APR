import pandas as pd

def pivot(data: "DataFrame", index=None, columns=None, values=None) -> "DataFrame":
    if values is None: 
        pivot_table = pd.pivot_table(data, index=index, columns=columns)
    else: 
        pivot_table = pd.pivot_table(data, index=index, columns=columns, values=values)

    return pivot_table
