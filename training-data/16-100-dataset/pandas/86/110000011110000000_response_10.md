import pandas as pd

def Substitution(value: str) -> str:
    return value

def Appender(value: str, indents: int) -> str:
    return value

def is_list_like(values: Union[list, tuple]) -> bool:
    return isinstance(values, (list, tuple))

def pivot(data: "DataFrame", index=None, columns=None, values=None) -> "DataFrame":
    indexed = None
    if values is None:
        cols = [columns] if index is None else [index, columns]
        append = index is None
        indexed = data.set_index(cols, append=append)
    else:
        if columns is None:
            raise TypeError("pivot() missing 1 required argument: 'columns'")
        if index is None:
            index = data.index
        else:
            index = data[index]
        index = pd.MultiIndex.from_arrays([index, data[columns]])

        if is_list_like(values) and not isinstance(values, tuple):
            indexed = pd.DataFrame(
                data[values].values, index=index, columns=values
            )
        else:
            indexed = pd.DataFrame(data[values].values, index=index)
    return indexed.unstack(columns)

def test_pivot_columns_none_raise_error(self):
    # GH 30924
    df = pd.DataFrame(
        {"col1": ["a", "b", "c"], "col2": [1, 2, 3], "col3": [1, 2, 3]}
    )
    msg = r"pivot\(\) missing 1 required argument: 'columns'"
    with pytest.raises(TypeError, match=msg):
        df.pivot(index="col1", values="col3")
