def _cython_agg_blocks(
    self, how: str, alt=None, numeric_only: bool = True, min_count: int = -1
) -> "Tuple[List[Block], Index]":
    data: BlockManager = self._get_data_to_aggregate()
    
    if numeric_only:
        data = data.get_numeric_data(copy=False)
    
    agg_blocks: List[Block] = []
    new_items: List[np.ndarray] = []
    deleted_items: List[np.ndarray] = []
    
    split_items: List[np.ndarray] = []
    split_frames: List[DataFrame] = []
    
    no_result = object()
    for block, locs in zip(data.blocks, data._blknos):
        result = no_result
        try:
            result, _ = self.grouper.aggregate(
                block.values, how, axis=1, min_count=min_count
            )
        except NotImplementedError:
            if alt is None:
                assert how == "ohlc"
                deleted_items.append(locs)
                continue
            
            obj = self.obj._mgr.getitem_array(data.items.get_indexer_for(locs))
            if obj.shape[1] == 1:
                obj = obj.iloc[:, 0]
            
            s = get_groupby(obj, self.grouper)
            try:
                result = s.aggregate(lambda x: alt(x, axis=self.axis))
            except TypeError:
                deleted_items.append(locs)
                continue
            else:
                result = cast(DataFrame, result)
                
                split_blocks = result._split_op_result(locs)
                if split_blocks is not None:
                    split_items.append(locs)
                    split_frames.extend(split_blocks)
                    continue
                
                result = result.values
    
        if isinstance(result, DataFrame):
            # We may have an exception in trying to aggregate
            deleted_items.append(locs)
            continue

        if result is not no_result:
            
            if isinstance(result, ABCDataFrame):
                result = cast(DataFrame, result)

            agg_blocks.append(block.make_block(result))
    
    if len(agg_blocks) == 0 and len(split_frames) == 0:
        raise DataError("No numeric types to aggregate")

    if len(split_items) > 0:
        new_items = data.items.take(np.sort(np.concatenate(new_items)))
    
    if len(deleted_items) > 0:
        indexer = np.arange(len(data))
        indexer = np.delete(indexer, np.hstack(deleted_items)).argsort()
        agg_blocks = [b for b in agg_blocks if len(b.mgr_locs) > 0]
        new_items = data.items.take(indexer)
    
    for loc, blk in zip(new_items, agg_blocks):
        loc.mgr_locs = blk.mgr_locs

    return agg_blocks, new_items
