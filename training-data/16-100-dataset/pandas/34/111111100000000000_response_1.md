from pandas.tseries.frequencies import to_offset
from pandas.tseries.common import (Timestamp, lib, maybe_convert_dtype)

class TimeGrouper(Grouper):
   
    def _get_timestamp_range_edges(self, axis):
        start = axis.min()
        end = axis.max()
        _offset = to_offset(self.freq)
        return start, end, _offset


    def _get_time_bins(self, axis):
        if not isinstance(axis, DatetimeIndex):
            raise TypeError(
                "axis must be a DatetimeIndex, but got "
                f"an instance of {type(axis).__name__}"
            )
    
        if len(axis) == 0:
            binner = labels = DatetimeIndex(data=[], freq=self.freq, name=axis.name)
            return binner, [], labels
    
        first, last, offset = self._get_timestamp_range_edges(axis)
        
        binner = labels = date_range(
            start=first,
            end=last,
            periods=offset,
            tz=axis.tz,
            name=axis.name,
            closed=self.closed,
            ambiguous="infer",
            nonexistent="shift_forward",
        )
    
        ax_values = axis.asi8
        binner, bin_edges = self._adjust_bin_edges(binner, ax_values)
    
        bins = lib.generate_bins_dt64(
            ax_values, bin_edges, self.closed, hasnans=maybe_convert_dtype(axis.hasnans)
        )
    
        if self.closed == "right":
            labels = binner
            if self.label == "right":
                labels = labels[1:]
        elif self.label == "right":
            labels = labels[1:]
    
        if axis.hasnans:
            binner = binner.insert(0, NaT)
            labels = labels.insert(0, NaT)
    
        if len(bins) < len(labels):
            labels = labels[: len(bins)]
    
        return binner, bins, labels
