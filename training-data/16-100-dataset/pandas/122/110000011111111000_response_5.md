def equals(self, other):
    self_axes, other_axes = self.axes, other.axes
    if len(self_axes) != len(other_axes):
        return False
    if not all(ax1.equals(ax2) for ax1, ax2 in zip(self_axes, other_axes)):
        return False

    # Add a condition to handle if the blocks are of different types
    if len(self.blocks) != len(other.blocks) or type(self.blocks) != type(other.blocks):
        return False

    # Add a condition to check if the blocks are equal
    if any(block.shape != other.blocks[i].shape or not block.equals(other.blocks[i]) for i, block in enumerate(self.blocks)):
        return False

    return True
