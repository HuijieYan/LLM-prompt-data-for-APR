def equals(self, other):
    self_axes, other_axes = self._get_axis_list(), other._get_axis_list()
    if len(self_axes) != len(other_axes):
        return False
    if not all(ax1.equals(ax2) for ax1, ax2 in zip(self_axes, other_axes)):
        return False
    self._consolidate_inplace()
    other._consolidate_inplace()
    if len(self._mgr.blocks) != len(other._mgr.blocks):
        return False

    # canonicalize block order, using a tuple combining the type
    # name and then mgr_locs because there might be unconsolidated
    # blocks (say, Categorical) which can only be distinguished by
    # the iteration order
    def canonicalize(block):
        return (block._values, block.mgr_locs)

    self_blocks = sorted(self._mgr.blocks, key=canonicalize)
    other_blocks = sorted(other._mgr.blocks, key=canonicalize)
    return all(
        block.equals(oblock) for block, oblock in zip(self_blocks, other_blocks)
    )
