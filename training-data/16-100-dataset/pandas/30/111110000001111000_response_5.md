import numpy as np
from pandas import isna, iNaT, to_datetime, RangeIndex, Series, DataFrame

class Parser():

    def __init__(self, date_unit):
        self.date_unit = date_unit
        self.min_stamp = 31536000
        self._STAMP_UNITS = ('s', 'ms', 'us', 'ns')

    def _try_convert_to_date(self, data):
        if not len(data):
            return data, False
        
        new_data = data
        if isinstance(data, Series) or isinstance(data, DataFrame):
            try:
                new_data = data.astype("int64")
            except (TypeError, ValueError, OverflowError):
                pass
        else:
            new_data = np.array(data, dtype='int64')

        if np.issubdtype(new_data.dtype, np.number):
            in_range = (
                isna(new_data)
                | (new_data > self.min_stamp)
                | (new_data == iNaT)
            )
            if not in_range.all():
                return data, False

        date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
        for date_unit in date_units:
            try:
                new_data = to_datetime(new_data, errors="raise", unit=date_unit)
            except (ValueError, OverflowError):
                continue
            return new_data, True
        return data, False
