import numpy as np
from pandas.core.dtypes.common import is_numeric_dtype
from pandas.core.nanops import isna
from pandas.io.json import iNaT
from pandas import to_datetime

class Parser():

    def __init__(self, date_unit=None):
        self.date_unit = date_unit
        self._STAMP_UNITS = ('s', 'ms', 'us', 'ns')
        self.min_stamp = 31536000

    def _try_convert_to_date(self, data):
        if not len(data):
            return data, False

        new_data = data
        if new_data.dtype == "object":
            try:
                new_data = new_data.astype("int64")
            except (TypeError, ValueError, OverflowError):
                pass

        if is_numeric_dtype(new_data):
            in_range = (
                isna(new_data) | (new_data > self.min_stamp) | (new_data == iNaT)
            )
            if not in_range.all():
                return new_data, False

        date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
        for date_unit in date_units:
            try:
                new_data = to_datetime(new_data, errors="raise", unit=date_unit)
            except (ValueError, OverflowError):
                continue
            return new_data, True
        return data, False
