from pandas._libs.missing import iNaT
from pandas._libs.tslibs.nattype import iNaT as NaT
from pandas._typing import Dtype, DateType
from pandas.api.types import isna
from pandas.tseries.common import to_datetime

class Parser():
    def __init__(self, date_unit: DateType = "ns", min_stamp: int = 31536000):
        self.date_unit = date_unit
        self.min_stamp = min_stamp
        self._STAMP_UNITS = ('s', 'ms', 'us', 'ns')

    def _try_convert_to_date(self, data):
        if not len(data):
            return data, False

        new_data = data
        if new_data.dtype == "object":
            try:
                new_data = data.astype("int64")
            except (TypeError, ValueError, OverflowError):
                pass

        if issubclass(new_data.dtype.type, np.number) or issubclass(new_data.dtype.type, bool):
            in_range = (
                isna(new_data._values)
                | (new_data > self.min_stamp)
                | (new_data._values == iNaT)
            )
            if not in_range.all():
                return data, False

        date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
        for date_unit in date_units:
            try:
                new_data = to_datetime(new_data, errors="raise", unit=date_unit)
            except (ValueError, OverflowError):
                continue
            return new_data, True
        return data, False
