import pandas as pd
import numpy as np
from pandas.api.types import is_datetime64_any_dtype as is_datetime
from pandas.errors import PerformanceWarning

class CustomJSONReader(pd.io.json._json.JSONReader):
    
    def __init__(self, filepath_or_buffer, typ='frame', convert_axes=True, convert_dates=True, keep_default_dates=True, numpy=False, precise_float=False, date_unit=None, encoding=None, lines=False, chunksize=None):
        super().__init__(filepath_or_buffer, typ=typ, convert_axes=convert_axes, convert_dates=convert_dates, keep_default_dates=keep_default_dates, numpy=numpy, precise_float=precise_float, date_unit=date_unit, encoding=encoding, lines=lines, chunksize=chunksize)
        if self.typ == 'series':
            self.date_unit = date_unit
            self.convert_dates = convert_dates
            self.min_stamp = np.datetime64('NaT').astype('datetime64[ns]')
        else:
            self.date_unit = None
            self.convert_dates = False
            self.min_stamp = None
    
    # function to try to parse a ndarray like into a date column
    def _try_convert_to_date(self, data):
        """
        Try to coerce object in epoch/iso formats and integer/float in epoch
        formats. Return a boolean if parsing was successful.
        """
        # no conversion on empty
        if not len(data):
            return data, False

        new_data = data
        if new_data.dtype == "object":
            try:
                new_data = data.astype("int64")
            except (TypeError, ValueError, OverflowError):
                pass

        # ignore numbers that are out of range
        if is_datetime(new_data.dtype):
            in_range = (
                pd.isna(new_data)
                | (new_data > self.min_stamp)
                | (new_data == np.datetime64('NaT'))
            )
            
            if not in_range.all():
                return data, False

        date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
        for date_unit in date_units:
            try:
                new_data = pd.to_datetime(new_data, errors="raise", unit=date_unit)
            except (ValueError, OverflowError):
                continue
            return new_data, True
        return data, False

pd.read_json('[true, true, false]', typ="frame")
pd.read_json('[true, true, false]', typ="series", cls=CustomJSONReader)
