```python
import numpy as np
from pandas.core.nanops import nanops
from pandas.core.dtypes.cast import maybe_cast_to_datetime
from pandas._libs.tslibs import iNaT
from pandas._typing import DateDtype
from pandas._libs.tslibs import Timestamp, NaT
from pandas.util._decorators import Appender

@Appender(
    "Try to coerce object in epoch/iso formats and integer/float in epoch "
    "formats. Return a boolean if parsing was successful."
)
def _try_convert_to_date(self, data):
    """
    Try to parse a Series like into a date column.

    Parameters
    ----------
    data : Series
        The Series being parsed.

    Returns
    -------
    data : Series
        The converted Series if successful, original Series otherwise.
    success : bool
        True if successful.
    """
    # no conversion on empty
    if not len(data):
        return data, False

    dtype = data.dtype
    if dtype is not None:
        if dtype == "object":
            # try to coerce to int64, if that fails, fallback to object
            try:
                data = data.astype(np.int64)
            except (TypeError, ValueError, OverflowError):
                pass
        elif issubclass(dtype.type, np.number):
            # ignore numbers that are out of range
            in_range = nanops.isfinite(data)
            if not in_range.all():
                return data, False
        else:
            pass  # we have a unit. So, we ignore

    # tz-aware is okay here
    parsed, _ = maybe_cast_to_datetime(data, errors="ignore")

    if parsed is not NotImplemented:
        return parsed, True

    return data, False
```
