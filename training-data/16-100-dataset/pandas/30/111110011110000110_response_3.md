import pandas as pd
import numpy as np
from pandas.api.types import is_list_like, is_datetime64_any_dtype
from pandas.core.dtypes.dtypes import DatetimeTZDtype
from pandas.core.missing import iNaT, isna
from pandas.core.tools.datetimes import to_datetime
from numpy import nan as NaN

class Parser:

    def __init__(self, data, date_unit=None):
        self.data = data
        self.date_unit = date_unit
        self.min_stamp = 0

    def _try_convert_to_date(self, data):
        if not len(data):
            return data, False

        new_data = data
        if new_data.dtype == "object":
            try:
                new_data = data.astype("int64")
            except (TypeError, ValueError, OverflowError):
                pass

        if is_list_like(new_data) and not is_datetime64_any_dtype(new_data):
            new_data = np.array([to_datetime(data, errors='raise') for data in new_data])

        return new_data, True
