import pandas as pd
import numpy as np
from pandas._libs.missing import iNaT
from pandas.core.dtypes.common import is_datetime64_any_dtype as is_datetime64_any_dtype
from pandas.core.missing import isna
from pandas.core.tools.datetimes import _guess_datetime_format_for_array as _guess_datetime_format_for_array
from pandas.util._decorators import Appender

class Parser():

    def _try_convert_to_date(self, data):
        new_data = data
        if new_data.dtype == "object":
            # try to coerce strings to ints
            try:
                new_data = data.astype("int64")
            except (TypeError, ValueError, OverflowError):
                pass

        if is_datetime64_any_dtype(new_data):
            return new_data, True
        
        date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
        for date_unit in date_units:
            try:
                new_data = pd.to_datetime(new_data, errors="raise", unit=date_unit)
            except (ValueError, OverflowError):
                continue
            return new_data, True
        return data, False
