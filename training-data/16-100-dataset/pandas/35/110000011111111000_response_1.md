import weakref

def cache_readonly(func):
    @property
    def wrapper(self):
        if not hasattr(self, '_cached'):
            self._cached = func(self)
        return self._cached
    return wrapper

def _engine(self):
    # To avoid a reference cycle, pass a weakref of self to _engine_type.
