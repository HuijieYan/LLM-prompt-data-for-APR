import weakref

def cache_readonly(fn):
    cache = {}
    def wrapped(self):
        if fn.__name__ not in cache:
            result = fn(self)
            cache[fn.__name__] = weakref.ref(result)
        else:
            result = cache[fn.__name__]()
            if result is None:
                result = fn(self)
                cache[fn.__name__] = weakref.ref(result)
        return result
    return wrapped
