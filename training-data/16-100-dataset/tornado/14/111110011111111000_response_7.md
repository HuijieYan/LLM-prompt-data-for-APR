def initialize(self, make_current=False):
    # If make_current is True or None, we want to set the current IOLoop,
    # so we check if there is already a current instance and raise an error if it exists.
    # Otherwise, we set the current IOLoop.
    if make_current is None or make_current:
        if IOLoop.current(instance=False) is not None:
            raise RuntimeError("current IOLoop already exists")
        self.make_current()
