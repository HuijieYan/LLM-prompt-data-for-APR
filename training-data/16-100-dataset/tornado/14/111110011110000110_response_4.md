class IOLoop(Configurable):
    def __init__(self, **kwargs):
        # Initialize make_current with None if not provided
        self.make_current = kwargs.pop('make_current', None)
        super(IOLoop, self).__init__(**kwargs)

    def initialize(self):
        if self.make_current is None:
            if self.current(instance=False) is None:
                self.make_current()
        elif self.make_current:
            if self.current(instance=False) is not None:
                raise RuntimeError("current IOLoop already exists")
            self.make_current()
