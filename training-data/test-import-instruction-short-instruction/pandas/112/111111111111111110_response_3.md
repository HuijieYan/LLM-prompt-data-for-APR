def get_indexer(
    self,
    target: AnyArrayLike,
    method: Optional[str] = None,
    limit: Optional[int] = None,
    tolerance: Optional[Any] = None,
) -> np.ndarray:

    self._check_method(method)

    if self.is_overlapping() or np.array_equal(self.values, target):
        return np.arange(len(self), dtype="intp")
    else:
        return np.repeat(-1, len(target))
