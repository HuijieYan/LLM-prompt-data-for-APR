    if len(clocs) == 0:
        return data

    # NOTE: This doesn't deal with hierarchical columns yet
    
    index = data.index


    if not isinstance(clocs, (list, tuple)):
        clocs = [clocs]

    to_unstack = clocs

    colidxs = [index._get_level_number(name) for name in to_unstack]

    mask = np.zeros(index.nlevels, dtype=bool)
    mask[colidxs] = True

    mask = np.logical_not(mask)

    ridxs = np.where(mask)[0]
