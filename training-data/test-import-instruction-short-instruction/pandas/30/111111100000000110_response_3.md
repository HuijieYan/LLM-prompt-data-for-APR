        Try to parse a ndarray like into a date column.

        if is_scalar(data):
            return data, False

        if not len(data):
            return data, False

        new_data = data
        if new_data.dtype == "object":
            try:
                new_data = data.astype("datetime64[ns]")
            except (TypeError, ValueError, OverflowError):
                pass

        if is_list_like(new_data):
            new_data = np.array(new_data, dtype="datetime64[ns]")

        # dates parsed, do nothing
        if new_data.dtype.type == np.datetime64:
            return new_data, True

        # ignore numbers that are out of range
        if issubclass(new_data.dtype.type, np.number):
            in_range = (
                isna(new_data)
                | (new_data > self.min_stamp)
                | (new_data == iNaT)
            )
            if not in_range.all():
                return data, False

        try:
            new_data = to_datetime(new_data, errors="raise")
        except (ValueError, OverflowError, OutOfBoundsDatetime):
            pass
        return new_data, True
