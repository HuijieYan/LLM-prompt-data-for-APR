```python
import sys
from typing import Union
from pathlib import Path
import utils
import pycompat

output: Union[None, pycompat.PathLike, str, utils.WritableStream]

def get_write_function(output):
    if output is None:
        def write(s):
            stderr = sys.stderr
            stderr.write(s)
    elif isinstance(output, (pycompat.PathLike, str)):
        def write(s):
            with open(output, 'a') as output_file:
                output_file.write(s)
    else:
        assert isinstance(output, utils.WritableStream)
        def write(s):
            output.write(s)

    return write
```
