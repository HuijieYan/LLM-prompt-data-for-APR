        path=url.path or '/',
        query='?' + url.query if url.query else ''
    )

    headers = dict(self._orig.headers)

    if 'Host' not in headers:

    headers = ['%s: %s' % (name, value)
               for name, value in headers.items()]

    headers.insert(0, request_line)
    headers = '\r\n'.join(headers).strip()
