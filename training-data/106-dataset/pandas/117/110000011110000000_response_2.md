```python
import numpy as np
import pandas as pd
from pandas import Series, MultiIndex, DatetimeArray
import pandas._testing as tm

def _isna_ndarraylike_old(obj):
    """
    Detect missing values in ndarray-like objects
    """
    return np.isnan(obj)

def _isna_old(obj):
    """
    Detect missing values, treating None, NaN, INF, -INF as null.

    Parameters
    ----------
    arr: ndarray or object value

    Returns
    -------
    boolean ndarray or boolean
    """
    if pd.api.types.is_scalar(obj):
        return pd.api.types.isna(obj)
    # hack (for now) because MI registers as ndarray
    elif isinstance(obj, pd.MultiIndex):
        raise NotImplementedError("isna is not defined for MultiIndex")
    elif isinstance(obj, type):
        return False
    elif isinstance(obj, (Series, np.ndarray, pd.Index)):
        return _isna_ndarraylike_old(obj)
    elif isinstance(obj, pd.api.types.ABCGeneric):
        return obj._constructor(obj._data._isna(func=_isna_old))
    elif isinstance(obj, list):
        return _isna_ndarraylike_old(np.asarray(obj, dtype=object))
    elif hasattr(obj, "__array__"):
        return _isna_ndarraylike_old(np.asarray(obj))
    else:
        return obj is None
```
