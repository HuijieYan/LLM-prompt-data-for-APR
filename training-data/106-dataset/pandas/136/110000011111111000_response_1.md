def _get_merge_keys(self):
    # note this function has side effects
    (left_join_keys, right_join_keys, join_names) = super()._get_merge_keys()

    # validate index types are the same
    for i, (lk, rk) in enumerate(zip(left_join_keys, right_join_keys)):
        if not is_dtype_equal(lk.dtype, rk.dtype):
            # check if lk and rk are categorical data
            if is_categorical_dtype(lk.dtype) and is_categorical_dtype(rk.dtype):
                msg = (
                    f"incompatible merge keys [{i}] {lk.dtype} and "
                    f"{rk.dtype}, both sides category, but not equal ones"
                )
            else:
                msg = (
                    f"incompatible merge keys [{i}] {lk.dtype} and "
                    f"{rk.dtype}, must be the same type"
                )
            raise MergeError(msg)

    # validate tolerance
    if self.tolerance is not None:
        if self.left_index:
            lt = self.left.index
        else:
            lt = left_join_keys[-1]

        msg = (
            f"incompatible tolerance {type(self.tolerance)}, must be compat "
            f"with type {lt.dtype}"
        )

        if is_datetimelike(lt):
            if not isinstance(self.tolerance, datetime.timedelta):
                raise MergeError(msg)
            if self.tolerance < Timedelta(0):
                raise MergeError("tolerance must be positive")

        elif is_integer(lt):
            if not isinstance(self.tolerance, int) or is_integer(self.tolerance):
                raise MergeError(msg)
            if self.tolerance < 0:
                raise MergeError("tolerance must be positive")

        elif is_numeric(lt):
            if not is_number(self.tolerance):
                raise MergeError(msg)
            if self.tolerance < 0:
                raise MergeError("tolerance must be positive")

        else:
            raise MergeError("key must be integer, timestamp or float")

    # validate allow_exact_matches
    if not isinstance(self.allow_exact_matches, bool):
        msg = f"allow_exact_matches must be boolean, passed {self.allow_exact_matches}"
        raise MergeError(msg)

    return left_join_keys, right_join_keys, join_names
