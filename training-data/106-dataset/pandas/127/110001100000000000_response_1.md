# updated the pct_change function and also added the missing imports for np and isna
@Appender(_shared_docs["pct_change"] % _shared_doc_kwargs)
def pct_change(self, periods=1, fill_method="pad", limit=None, freq=None, **kwargs):
    import numpy as np
    from pandas.core.dtypes.common import isna
    axis = self._get_axis_number(kwargs.pop("axis", self._stat_axis_name))
    if fill_method is None:
        data = self
    else:
        data = self.fillna(value=fill_method, limit=limit, axis=axis)
    
    rs = data.div(data.shift(periods=periods, freq=freq, axis=axis, fill_value=None)) - 1
    rs = rs.reindex_like(data)
    if freq is None:
        mask = isna(data.values)
        np.putmask(rs.values, mask, np.nan)
    return rs
