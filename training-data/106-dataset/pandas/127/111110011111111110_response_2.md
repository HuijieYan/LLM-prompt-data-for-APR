# Fixed function
@Appender(_shared_docs["pct_change"] % _shared_doc_kwargs)
def pct_change(self, periods=1, fill_method="pad", limit=None, freq=None, **kwargs):
    axis = self._get_axis_number(kwargs.pop("axis", self._stat_axis_name))
    if fill_method is None:
        data = self
    else:
        data = self.fillna(value=fill_method, method=limit, axis=axis)
  
    rs = data.div(data.shift(periods=periods, freq=freq, axis=axis, fill_value=None, **kwargs)) - 1
    rs = rs.reindex(data.index, method=None, copy=True, limit=None, tolerance=None)
    if freq is None:
        mask = rs.isna()
        rs.mask(mask, inplace=True)
    return rs
