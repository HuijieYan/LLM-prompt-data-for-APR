def run(cmdline_args=None, main_task_cls=None, worker_scheduler_factory=None, use_dynamic_argparse=False, local_scheduler=False):
        """
        Please dont use. Instead use `luigi` binary.

        Run from cmdline using argparse.

        :param cmdline_args:
        :param main_task_cls:
        :param worker_scheduler_factory:
        :param use_dynamic_argparse:
        :param local_scheduler:
        """
        # Instantiate the interface based on use_dynamic_argparse
        if use_dynamic_argparse:
            interface = DynamicArgParseInterface()
        else:
            interface = ArgParseInterface()
        
        # Check if main_task_cls is provided, then insert it into cmdline_args
        if main_task_cls:
            cmdline_args.insert(0, main_task_cls.task_family)
        
        # Check if local_scheduler is True, then insert '--local-scheduler' into cmdline_args
        if local_scheduler:
            cmdline_args.insert(0, '--local-scheduler')
        
        # Parse the cmdline_args using the interface
        tasks = interface.parse(cmdline_args)
        
        # Return the result of running the tasks using the interface and worker_scheduler_factory
        return interface.run(tasks, worker_scheduler_factory)
