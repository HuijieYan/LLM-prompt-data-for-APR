    def run(cmdline_args=None, main_task_cls=None,
            worker_scheduler_factory=None, use_dynamic_argparse=False, local_scheduler=False):
        """
        Please don't use. Instead use the `luigi` binary.

        Run from the command line using argparse.

        :param cmdline_args: Optional. List of command line arguments.
        :param main_task_cls: Optional. The main task class to run.
        :param worker_scheduler_factory: Optional.
        :param use_dynamic_argparse: Optional. True if dynamic argparse is used, False otherwise.
        :param local_scheduler: Optional. True if using a local scheduler, False otherwise.
        """
        if use_dynamic_argparse:
            interface = DynamicArgParseInterface()
        else:
            interface = ArgParseInterface()
        if main_task_cls:
            if cmdline_args is None:
                cmdline_args = []
            cmdline_args.insert(0, main_task_cls.task_family)
        if local_scheduler:
            if cmdline_args is None:
                cmdline_args = []
            cmdline_args.insert(0, '--local-scheduler')
        tasks = interface.parse(cmdline_args)
        return interface.run(tasks, worker_scheduler_factory)
