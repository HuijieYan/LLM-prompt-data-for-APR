from typing import Any, Set
from pydantic import Field, Response, jsonable_encoder
from pydantic.errors import ErrorWrapper, ValidationError
from your_module import Field, Response

def serialize_response(
    response: Response,
    field: Field = None,
    include: Set[str] = None,
    exclude: Set[str] = None,
    by_alias: bool = True,
    skip_defaults: bool = False,
) -> Any:
    if field:
        errors = []
        value, errors_ = field.validate(response, {}, loc=("response",))
        if isinstance(errors_, ErrorWrapper):
            errors.append(errors_)
        elif isinstance(errors_, list):
            errors.extend(errors_)
        if errors:
            raise ValidationError(errors)
        return jsonable_encoder(
            value,
            include=include,
            exclude=exclude,
            by_alias=by_alias,
            skip_defaults=skip_defaults,
        )
    else:
        return jsonable_encoder(response, skip_defaults=skip_defaults)
