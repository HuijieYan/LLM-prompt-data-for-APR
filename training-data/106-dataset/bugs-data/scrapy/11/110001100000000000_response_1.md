```python
import gzip
from io import BytesIO

def gunzip(data):
    """Gunzip the given data and return as much data as possible.

    This is resilient to CRC checksum errors.
    """
    f = gzip.GzipFile(fileobj=BytesIO(data), mode='rb')
    output = b''
    while True:
        chunk = f.read(8196)
        if chunk:
            output += chunk
        else:
            break
    return output
```
