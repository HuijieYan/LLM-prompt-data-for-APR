from io import BytesIO

def gunzip(data):
    """Gunzip the given data and return as much data as possible.

    This is resilient to CRC checksum errors.
    """
    try:
        f = gzip.GzipFile(fileobj=BytesIO(data))
        output = f.read()
        return output
    except FileNotFoundError as e:
        print(f"Error: {e}")
        return None
