from datetime import timedelta

class tqdm:
    @staticmethod
    def format_sizeof(n):
        return str(n)

    @staticmethod
    def format_interval(elapsed):
        return str(timedelta(seconds=elapsed))

    @staticmethod
    def format_meter(n, total, elapsed, ncols=None, prefix='',
                 ascii=False, unit='it', unit_scale=False, rate=None,
                 bar_format=None):
        
        percentage = (n / total) * 100 if total else 0
        remaining_str = tqdm.format_interval((total - n) / rate) if rate else '?' if total else '?'
        
        l_bar = (prefix if prefix else '') + '{0:3.0f}%|'.format(percentage)
        r_bar = '| {0}/{1} [{2}<{3}]'.format(n, total, elapsed, remaining_str)
        
        if ncols == 0:
            return l_bar[:-1] + r_bar[1:]
        
        if bar_format:
            bar_args = {
                'n': n,
                'total': total,
                'percentage': percentage,
                'rate': rate,
                'elapsed': tqdm.format_interval(elapsed),
                'remaining': remaining_str,
                'l_bar': l_bar,
                'r_bar': r_bar,
                'desc': prefix if prefix else ''
            }
            
            return bar_format.format(**bar_args)
        
        return l_bar + r_bar
