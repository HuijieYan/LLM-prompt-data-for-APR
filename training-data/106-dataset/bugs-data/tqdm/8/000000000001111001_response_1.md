@staticmethod
def format_meter(n, total, elapsed, ncols=None, prefix='',
                 ascii=False, unit='it', unit_scale=False, rate=None,
                 bar_format=None):

    # Calculate elapsed time string
    elapsed_str = "{:02d}:{:02d}".format(int(elapsed/60), int(elapsed % 60))  # format elapsed time

    # Calculate rate if not provided
    if rate is None and elapsed:
        rate = n / elapsed
    
    rate_fmt = '{0:5.2f}{1}/{2}'.format(rate, unit, 's' if rate < 1 else unit)  # format the rate
    
    # Calculate progress stats if total is known
    if total:
        frac = n / total
        percentage = frac * 100
        remaining_str = "{:02d}:{:02d}".format(int((total - n) / rate / 60), int((total - n) / rate % 60))
        
        # Format the left and right sides of the bar
        l_bar = (prefix if prefix else '') + '{0:3.0f}%|'.format(percentage)
        r_bar = '| {0}/{1} [{2}<{3}, {4}]'.format(n, total, elapsed_str, remaining_str, rate_fmt)

        if ncols == 0:
            return l_bar[:-1] + r_bar[1:]

        if bar_format:
            bar_args = {'n': n, 'n_fmt': n, 'total': total, 'total_fmt': total, 'percentage': percentage, 'rate': rate,
                        'rate_noinv': rate, 'rate_fmt': rate_fmt, 'elapsed': elapsed_str, 'remaining': remaining_str,
                        'l_bar': l_bar, 'r_bar': r_bar, 'desc': prefix if prefix else ''}
            return bar_format.format(**bar_args)

        # Format progress bar
        N_BARS = max(1, ncols - len(l_bar) - len(r_bar)) if ncols else 10
        bar_length = int(frac * N_BARS)
        
        if ascii:
            bar = ('#' * bar_length) + ('-' * (N_BARS - bar_length))
        else:
            bar = 'â–ˆ' * bar_length + ' ' * (N_BARS - bar_length)

        full_bar = l_bar + bar + r_bar
        return full_bar
    else:
        return (prefix if prefix else '') + '{0}{1} [{2}, {3}]'.format(n, unit, elapsed_str, rate_fmt)
