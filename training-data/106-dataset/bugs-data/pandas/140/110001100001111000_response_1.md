def _recast_datetimelike_result(result: DataFrame) -> DataFrame:
    """
    If we have date/time like in the original, then coerce dates
    as we are stacking can easily have object dtypes here.

    Parameters
    ----------
    result : DataFrame

    Returns
    -------
    DataFrame

    Notes
    -----
    - Assumes Groupby._selected_obj has ndim==2 and at least one
    datetimelike column
    """
    result_copy = result.copy()  # Make a copy of the input DataFrame
    result_copy_obj_cols = [
        idx for idx in range(len(result_copy.columns)) if pd.api.types.is_object_dtype(result_copy.dtypes[idx])
    ]  # Find the columns with object dtype

    # Loop through the object columns and convert the values
    for n in result_copy_obj_cols:
        converted = pd.to_numeric(result_copy.iloc[:, n], errors='coerce')  # Convert the values to numeric
        result_copy.iloc[:, n] = converted  # Assign the converted values back to the DataFrame

    return result_copy  # Return the modified DataFrame
