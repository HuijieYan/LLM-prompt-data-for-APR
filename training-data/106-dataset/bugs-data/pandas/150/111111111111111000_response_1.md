import numpy as np
from pandas.api.types import is_string_dtype, is_float_dtype, is_complex_dtype, is_datetimelike_v_numeric, needs_i8_conversion, is_dtype_equal
from pandas.core.algorithms import isin

def array_equivalent(left, right, strict_nan=False):
    left, right = np.asarray(left), np.asarray(right)
    
    # shape compat
    if left.shape != right.shape:
        return False

    if is_string_dtype(left) or is_string_dtype(right):
        return _string_array_equivalent(left, right, strict_nan=strict_nan)

    if is_float_dtype(left) or is_complex_dtype(left):
        return _float_array_equivalent(left, right, strict_nan=strict_nan)
    
    if is_datetimelike_v_numeric(left, right):
        return _datetimelike_array_equivalent(left, right, strict_nan=strict_nan)
    
    if needs_i8_conversion(left) and needs_i8_conversion(right):
        return _numeric_array_equivalent(left, right, strict_nan=strict_nan)

    if left.dtype.type is np.void or right.dtype.type is np.void:
        return _void_array_equivalent(left, right, strict_nan=strict_nan)

    return np.array_equal(left, right)

def _string_array_equivalent(left, right, strict_nan=False):
    if not strict_nan:
        return (left == right).all()

    # isna considers NaN and None to be equivalent.
    return isin(left, right)

def _float_array_equivalent(left, right, strict_nan=False):
    if not strict_nan:
        return ((left == right) |
                (left != left) & (right != right)).all()

    return ((left == right) |
            (left != left) & (right != right) |
            (np.isnan(left) & np.isnan(right))).all()

def _datetimelike_array_equivalent(left, right, strict_nan=False):
    if isinstance(left.dtype, np.datetime_data):
        if not isinstance(right.dtype, np.datetime_data):
            return False
    else:
        if not isinstance(left.dtype, type(right.dtype)):
            return False
    
    return ((left == right) |
            (left != left) & (right != right)).all()

def _numeric_array_equivalent(left, right, strict_nan=False):
    return ((left == right) |
            (left != left) & (right != right)).all()

def _void_array_equivalent(left, right, strict_nan=False):
    if left.dtype != right.dtype:
        return False

    return np.array_equal(left, right)
