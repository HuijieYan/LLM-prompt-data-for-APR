    def pct_change(self, periods=1, fill_method="pad", limit=None, freq=None, **kwargs):
        axis = self._get_axis_number(kwargs.get("axis", self._stat_axis_name))
        if fill_method is None:
            data = self
        else:
            data = self.fillna(method=fill_method, limit=limit)
    
        rs = data.div(data.shift(periods=periods, freq=freq, axis=axis)) - 1
        rs = rs.reindex_like(data, method=None)  # updated to specify method=None as reindex_like by default does not handle duplicates
        if freq is None:
            mask = rs.isna()  # updated to use isna instead of isna(com.values_from_object(data))
            rs[mask] = np.nan  # updated to use bracket notation instead of the putmask method
        return rs
