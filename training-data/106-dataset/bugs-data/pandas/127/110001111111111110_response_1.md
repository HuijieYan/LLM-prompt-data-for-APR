def pct_change(self, periods=1, fill_method="pad", limit=None, freq=None, **kwargs):
    axis = self._get_axis_number(kwargs.pop("axis", self._stat_axis_name))
    if fill_method is None:
        data = self
    else:
        data = self.fillna(value=fill_method, limit=limit, axis=axis)  # fixed fillna method call
    
    rs = data.div(data.shift(periods=periods, freq=freq, axis=axis, **kwargs)) - 1
    rs = rs.reindex_like(self)  # fixed reindex_like method call
    mask = rs.isna()  # fixed isna method call
    rs = rs.mask(mask, np.nan)  # fixed mask method call
    return rs
