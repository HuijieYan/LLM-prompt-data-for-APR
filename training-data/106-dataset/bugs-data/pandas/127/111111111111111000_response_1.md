def pct_change(self, periods=1, fill_method="pad", limit=None, freq=None, **kwargs):
    axis = self._get_axis_number(self)  # adjusted
    if fill_method is None:
        data = self
    else:
        data = self.fillna(value=fill_method, limit=limit, axis=axis)  # adjusted

    rs = data.div(data.shift(periods=periods, freq=freq, axis=axis)) - 1  # adjusted
    rs = rs.reindex(index=data.index)  # adjusted, changed to reindex with 'index'
    mask = rs.index.duplicated()  # adjusted
    rs[mask] = np.nan  # adjusted
    return rs
