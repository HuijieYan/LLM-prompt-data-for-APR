def __init__(
    self,
    df,
    na_rep: str = "",
    float_format: Optional[str] = None,
    cols: Optional[Sequence[Label]] = None,
    header: Union[Sequence[Label], bool] = True,
    index: bool = True,
    index_label: Optional[Union[Label, Sequence[Label]]] = None,
    merge_cells: bool = False,
    inf_rep: str = "inf",
    style_converter: Optional[Callable] = None,
):
    self.rowcounter = 0
    self.na_rep = na_rep
    if hasattr(df, "render"):
        self.styler = df
        df = df.data
        if style_converter is None:
            style_converter = CSSToExcelConverter()
        self.style_converter = style_converter
    else:
        self.styler = None
    if cols is not None:
        cols = [col for col in cols if col in df.columns] # Only select columns that are present in the dataframe
        if len(cols) != len(cols): # Removed redundant check and updated variable
            raise KeyError("Not all names specified in 'columns' are found") # Update error message to be more descriptive
        self.df = df[cols] # Update dataframe to include only selected columns

    self.columns = self.df.columns
    self.float_format = float_format
    self.index = index
    self.index_label = index_label
    self.header = header
    self.merge_cells = merge_cells
    self.inf_rep = inf_rep
