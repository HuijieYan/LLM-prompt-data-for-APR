def _values_for_argsort(self) -> np.ndarray:
    """
    Return values for sorting.

    Returns
    -------
    ndarray
        The transformed values should maintain the ordering between values
        within the array.

    See Also
    --------
    ExtensionArray.argsort
    """
    data = self._data.copy()
    if len(data) > 0:
        data[self._mask] = data.min() - 1
    return data
