def _cython_agg_blocks(
    self, how: str, alt=None, numeric_only: bool = True, min_count: int = -1
) -> "Tuple[List[Block], Index]":
    data: BlockManager = self.obj._data
    if numeric_only:
        data = data.convert_dtypes()
        data = data.select_dtypes(include=["int", "float"], allow_narrow=True)

    agg_blocks: List[Block] = []
    new_items: List[np.ndarray] = []

    if how == "ohlc":
        deleted_items: List[np.ndarray] = []
    else:
        deleted_items = None
        split_items: List[np.ndarray] = []

    for indexer in range(len(data.blocks)):
        result = pd.DataFrame()
        for locs in data.blocks[indexer].mgr_locs.as_array:
            obj = self.obj.copy().iloc[locs, :]
            if obj.shape[1] == 1:
                obj = obj.stack(dropna=False)

            s = obj.groupby(level=self.grouper.names[0])
            agg_result = s.aggregate(alt)
            if isinstance(agg_result, pd.DataFrame):
                split_items.append(locs)
                agg_result = agg_result.transpose().to_numpy()
            result = pd.concat([result, agg_result])

        result_values = result.values
        if result_values.size == 0:
            result_values = np.array([np.nan])

        agg_block = pd.blocks.make_block(
            values=result_values, placement=pd.block.mgr_locs,
        )

        new_items.append(pd.block.mgr_locs.as_array)
        agg_blocks.append(agg_block)

    return agg_blocks, agg_items
