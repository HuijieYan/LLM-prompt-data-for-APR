import pandas as pd

def pivot_table(data, values=None, index=None, columns=None, aggfunc='mean', fill_value=None, margins=False, dropna=True, margins_name='All', observed=False) -> 'DataFrame':
    if isinstance(columns, tuple):
        table = data.pivot_table(values=values, index=index, columns=list(columns), fill_value=fill_value, aggfunc=aggfunc, margins=margins, dropna=dropna, margins_name=margins_name, observed=observed)
    else:
        table = data.pivot_table(values=values, index=index, columns=columns, fill_value=fill_value, aggfunc=aggfunc, margins=margins, dropna=dropna, margins_name=margins_name, observed=observed)
    
    return table

# Testing the fixed function with the provided use cases
df = pd.DataFrame({'k': [1, 2, 3], 'v': [4, 5, 6]})
print(df.pivot_table(values='v', columns='k'))

df = pd.DataFrame({'k': [1, 2, 3], 'v': [4, 5, 6]})
print(df.pivot_table(values='v', index='k'))

df2 = pd.DataFrame({'k1': [1, 2, 3], 'k2': [1, 2, 3], 'v': [4, 5, 6]})
print(df2.pivot_table(values='v', index=('k1','k2')))
print(df2.pivot_table(values='v', columns=('k1','k2')))
