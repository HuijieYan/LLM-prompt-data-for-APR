# This includes the corrected version of the pivot_table function/method

from pandas.core.reshaping.pivot import pivot_table
from pandas.core.dtypes.common import is_list_like
from pandas.core.groupby.grouper import Grouper
from pandas.core.frame import DataFrame
from pandas.core.series import Series
from pandas.core.dtypes.generic import ABCDataFrame
from pandas.core.indexes.multi import MultiIndex
from pandas.core.reshape.concat import concat
from pandas.core.algorithms import maybe_downcast_to_dtype
from pandas.core.indexes.api import MultiIndex, cartesian_product

from typing import List

def pivot_table(
    data: DataFrame = None,
    values: Series = None,
    index: Series = None,
    columns: Series = None,
    aggfunc: str = "mean",
    fill_value = None,
    margins: bool = False,
    dropna: bool = True,
    margins_name: str = "All",
    observed: bool = False
) -> DataFrame:
    index_names = []
    for inx in index:
        index_names.append(inx)
    if columns is not None:
        for col in columns:
            index_names.append(col)

    values_passed = values is not None
    if values_passed:
        if is_list_like(values):
            values_multi = True
        else:
            values_multi = False
            values = [values]

        for value in values:
            if value not in data:
                raise KeyError(value)

        to_filter = []
        for element in index_names + values:
            if isinstance(element, Grouper):
                element = element.key
            if element in data.columns:
                to_filter.append(element)

        if len(to_filter) != len(data.columns):
            data = data[to_filter]

    else:
        values = list(data.columns)
        for key in index_names:
            try:
                values = values.drop(key)
            except (TypeError, ValueError, KeyError):
                pass
        values = list(values)

    grouped_data = data.groupby(index_names, observed=observed)
    aggregated_data = grouped_data.agg(aggfunc)
    if dropna and isinstance(aggregated_data, ABCDataFrame) and len(aggregated_data.columns):
        aggregated_data = aggregated_data.dropna()
        for value in values:
            if data[value].dtype == 'int' and aggregated_data[value].dtype != 'int':
                aggregated_data[value] = maybe_downcast_to_dtype(aggregated_data[value], data[value].dtype)

    table = aggregated_data
    if len(table.index.names) > 1:
        to_unstack = []
        for i in range(len(index_names), len(index_names + columns)):
            name = table.index.names[i]
            if name in index_names:
                to_unstack.append(i)
            else:
                to_unstack.append(name)
        table = table.unstack(to_unstack)

    if not dropna:
        if len(table.index.names) > 1:
            m = MultiIndex.from_arrays(cartesian_product(table.index.levels), names=table.index.names)
            table = table.reindex(m, axis=0)
        if len(table.columns.names) > 1:
            m = MultiIndex.from_arrays(cartesian_product(table.columns.levels), names=table.columns.names)
            table = table.reindex(m, axis=1)
    if isinstance(table, ABCDataFrame):
        table = table.sort_index(axis=1)
    if fill_value is not None:
        table = table.fillna(fill_value)

    if margins:
        if dropna:
            data = data.dropna()
        table = concat([table, data], keys=None, axis=0)

    if values_passed and not values_multi and not table.empty and (len(table.columns.names) > 1):
        table = table[values[0]]

    if len(index_names) == 0 and len(columns) > 0:
        table = table.T

    if isinstance(table, ABCDataFrame):
        table = table.dropna(axis=1)

    return table
