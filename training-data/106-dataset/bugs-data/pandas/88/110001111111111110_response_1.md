```python
def pivot_table(
        data,
        values=None,
        index=None,
        columns=None,
        aggfunc="mean",
        fill_value=None,
        margins=False,
        dropna=True,
        margins_name="All",
        observed=False,
    ) -> "DataFrame":
    
    # check if columns is a tuple, if not convert it to a tuple
    if not isinstance(columns, tuple):
        columns = (columns,)

    table = data.pivot_table(values=values, index=index, columns=columns, aggfunc=aggfunc, fill_value=fill_value, margins=margins, dropna=dropna, margins_name=margins_name, observed=observed)
    
    return table
```
