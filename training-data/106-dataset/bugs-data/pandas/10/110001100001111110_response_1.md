def putmask(
    self, mask, new, inplace: bool = False, axis: int = 0, transpose: bool = False,
) -> List["Block"]:
    """
    See Block.putmask.__doc__
    """
    inplace = validate_bool_kwarg(inplace, "inplace")

    mask = _extract_bool_array(mask)

    new_values = self.values._data if inplace else self.values._data.copy()

    if isinstance(new_values, pd.core.arrays.string.StringArray):
        mask = _safe_reshape(mask, new_values.shape)
        new_values[mask] = new
    elif isinstance(new_values, pd.core.arrays.integer.IntegerArray):
        mask = _safe_reshape(mask, new_values.shape)
        new_values[mask] = new
    elif isinstance(new_values, pd.core.arrays.boolean.BooleanArray):
        mask = _safe_reshape(mask, new_values.shape)
        new_values[mask] = new
    
    return [self.make_block(values=new_values)]
