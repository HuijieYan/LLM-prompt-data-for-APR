                result = True
            except (TypeError, ValueError):

    # don't coerce 0-len data
    if len(data) and (data.dtype == "float" or data.dtype == "object"):
        # coerce ints if we can
        try:
            new_data = data.astype("int64")
            if (new_data == data).all():
                data = new_data
                result = True
        except (TypeError, ValueError):
            pass

    # coerce ints to 64
    if data.dtype == "int":
        try:
            data = data.astype("int64")
            result = True
        except (TypeError, ValueError):
            pass
