def add(self, layer):
    if not isinstance(layer, Layer):
        raise TypeError('The added layer must be an instance of class Layer. Found: ' + str(layer))

    if not self._layers:
        if isinstance(layer, InputLayer):
            self.inputs = layer
            self.outputs = layer
        else:
            raise ValueError('The first layer in the model must be an instance of type InputLayer.')
    else:
        if not hasattr(self, 'inputs'):
            raise ValueError('The model must have an input layer before adding additional layers.')
        self.outputs = layer(self.inputs)

    self._layers.append(layer)
