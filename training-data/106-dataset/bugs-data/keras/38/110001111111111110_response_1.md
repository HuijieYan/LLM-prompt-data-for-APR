def build(self, input_shape):
    for cell in self.cells:
        # check if the cell is an instance of a Layer
        if isinstance(cell, Layer):
            cell.build(input_shape)
        
        # check if the cell has a state_size attribute
        if hasattr(cell, 'state_size') and hasattr(cell.state_size, '__len__'):
            output_dim = cell.state_size[0]
        else:
            output_dim = cell.state_size
        
        # update the input_shape
        input_shape = (input_shape[0], input_shape[1], output_dim)
    
    # set self.built to True after the loop completes
    self.built = True
