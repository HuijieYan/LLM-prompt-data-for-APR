The bug occurs because in the `setter` function, there is a recursive call to `setter` which should actually be calling the `getter` function. This recursive call causes an infinite loop and incorrect behavior.

To fix this bug, the `setter` function should call the `getter` function to get the oldmin and oldmax values, instead of calling itself.

def _make_getset_interval(method_name, lim_name, attr_name):
    """
    Helper to generate ``get_{data,view}_interval`` and
    ``set_{data,view}_interval`` implementations.
    """

    def getter(self):
        # docstring inherited.
        return getattr(getattr(self.axes, lim_name), attr_name)

    def setter(self, vmin, vmax, ignore=False):
        # docstring inherited.
        if ignore:
            setattr(getattr(self.axes, lim_name), attr_name, (vmin, vmax))
        else:
            oldmin, oldmax = getter(self)
            if oldmin < oldmax:
                setattr(getattr(self.axes, lim_name), attr_name, (min(vmin, vmax, oldmin), max(vmin, vmax, oldmax)))
            else:
                setattr(getattr(self.axes, lim_name), attr_name, (max(vmin, vmax, oldmax), min(vmin, vmax, oldmin)))
        self.stale = True

    getter.__name__ = f"get_{method_name}_interval"
    setter.__name__ = f"set_{method_name}_interval"

    return getter, setter
