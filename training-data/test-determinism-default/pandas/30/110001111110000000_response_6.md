import pandas as pd
import numpy as np
from pandas.core.dtypes.cast import astype_nansafe

from pandas._libs.tslibs import NaT, iNaT

class BuggyFunctionFixed:
    
    min_stamp = pd.Timestamp.min.replace(tzinfo=None)
    date_unit = None  # not specified in the original code

    def _try_convert_to_date(self, data):
        if not len(data):
            return data, False

        new_data = data
        if is_object_dtype(new_data):
            try:
                new_data = astype_nansafe(new_data, 'datetime64[ns]')
            except (TypeError, ValueError, OverflowError):
                pass

        if issubclass(new_data.dtype.type, np.number):
            in_range = (
                isna(new_data) | (new_data > self.min_stamp) | (new_data == iNaT)
            )
            if not in_range.all():
                return data, False

        date_units = (self.date_unit,) if self.date_unit else _STAMP_UNITS
        for date_unit in date_units:
            try:
                new_data = to_datetime(new_data, errors="raise", unit=date_unit)
            except (ValueError, OverflowError):
                continue
            return new_data, True
        return data, False

def test_readjson_bool_series():
    # GH31464
    result = BuggyFunctionFixed()._try_convert_to_date(pd.Series(['true', 'true', 'false']))
    expected = pd.Series([True, True, False])
    assert result.all() == expected.all()
