import numpy as np
from pandas.core.dtypes.common import is_datetime64_any_dtype as is_datetime_dtype
from pandas._libs.missing import iNaT
from pandas.core.dtypes.generic import is_bool_dtype
from pandas.api.types import is_bool

def _try_convert_to_date(self, data):
    if isinstance(data, pd.Series) and is_bool_dtype(data) and self.date_unit is None:
        return pd.Series(data, copy=True), True
    if self.date_unit is None:
        return data, False

    # no conversion on empty
    if not len(data):
        return data, False

    new_data = data
    if new_data.dtype == "object":
        try:
            new_data = data.astype("int64")
        except (TypeError, ValueError, OverflowError):
            pass

    # ignore numbers that are out of range
    if issubclass(new_data.dtype.type, np.number):
        in_range = (
            isna(new_data._values)
            | (new_data > self.min_stamp)
            | (new_data._values == iNaT)
        )
        if not in_range.all():
            return data, False

    date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
    for date_unit in date_units:
        try:
            if is_bool(new_data):
                new_data = new_data.astype('<M8[ns]')
            new_data = to_datetime(new_data, errors="raise", unit=date_unit)
        except (TypeError, ValueError, OverflowError):
            continue
        return new_data, True
    return data, False
