def _try_convert_to_date(self, data, errors: str = "raise") -> Tuple[Union["_ArrayType", Series], bool]:
    """
    Try to parse a ndarray like into a date column.

    Try to coerce object in epoch/iso formats and integer/float in epoch
    formats. Return a boolean if parsing was successful.
    """
    # no conversion on empty
    if not len(data):
        return data, False

    new_data = data
    data_is_data_or_discrete = (
        not is_datetime64_any_dtype(data)
        and is_datetime64_any_or_timedelta_dtype(data)
        and not is_datetime64_or_timedelta_dtype(data)
    )
    # ignore datetime
    if (errors == "coerce" and data_is_data_or_discrete) or errors == "ignore":
        if self.errors in ["ignore", "coerce"]:
            return data, False

    if new_data.dtype == "object":
        try:
            result = new_data.astype("int64")
            new_data = new_data[result == new_data]._values
        except (TypeError, ValueError, OverflowError):
            pass

    # ignore numbers that are out of range
    if issubclass(new_data.dtype.type, np.number):
        if (
            self.errors in [None, "coerce"]
            and (not isna(new_data._values))
            & ((new_data > iNaT) | (new_data._values == iNaT))
        ).any():
            return data, False

    date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
    for date_unit in date_units:
        try:
            new_data = to_datetime(new_data, errors=self.errors, unit=date_unit)
        except (ValueError, OverflowError, TypeError):
            continue
        return new_data, True
    return data, False
