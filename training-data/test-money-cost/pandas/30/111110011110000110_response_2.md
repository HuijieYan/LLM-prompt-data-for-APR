import numpy as np
import pandas as pd
from pandas.api.types import is_string_dtype

class Parser():

    def __init__(self, min_stamp, date_unit):
        self.min_stamp = min_stamp
        self.date_unit = date_unit
        self._STAMP_UNITS = ['ns', 'us', 'ms', 's', 'm', 'h']
        
    def _try_convert_to_date(self, data):
        if len(data) == 0:
            return data, False
    
        new_data = data
        if is_string_dtype(new_data):
            try:
                new_data = new_data.astype("int64")
            except (TypeError, ValueError, OverflowError):
                pass
    
        if issubclass(new_data.dtype.type, np.number):
            in_range = (
                np.isnan(new_data._values)
                | (new_data > self.min_stamp)
            )
            if not in_range.all():
                return data, False

        date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
        for date_unit in date_units:
            try:
                new_data = pd.to_datetime(new_data, errors="raise", unit=date_unit)
            except (ValueError, OverflowError):
                continue
            return new_data, True
        return data, False

def read_json(data, typ):
    return pd.read_json(data, typ=typ, convert_dates=False)

# Test
def test_readjson_bool_series():
    result = read_json("[true, true, false]", typ="series")
    expected = pd.Series([True, True, False])
    assert result.equals(expected)
