import pandas as pd
import numpy as np
from pandas.api.types import isna
from pandas.io.json import json_normalize

class Parser():
    min_stamp = pd.Timestamp.min
    iNaT = pd.Timestamp.iNaT
    date_unit = 'ms'
    _STAMP_UNITS = ['ms', 'us', 'ns']


    def _try_convert_to_date(self, data):
        if len(data) == 0 or data.dtype == "object":
            return data, False

        new_data = data
        try:
            new_data = pd.to_datetime(new_data, errors="coerce", format=None, unit=None, infer_datetime_format=False, origin='unix', cache=True)
        except (TypeError, ValueError, OverflowError):
            return data, False
        return new_data, True
