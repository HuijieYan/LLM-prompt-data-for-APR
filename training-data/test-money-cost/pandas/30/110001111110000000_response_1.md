import pandas as pd
import numpy as np
from pandas.io.json import read_json
from pandas.errors import iNaT, isna
from pandas.compat.numpy import dict_to_year_month_day

class JsonReader:
    def __init__(self, data):
        self.data = data

    def read(self):
        obj = self._get_object_parser(self.data)
        return obj

    def _get_object_parser(self, data, typ="series"):
        if typ == "series":
            return SeriesParser(data).parse()

class SeriesParser:
    def __init__(self, json):
        self.json = json

    def parse(self):
        self._try_convert_types()
        return self.json

    def _try_convert_types(self):
        if isinstance(self.json, pd.Series):
            return self.json

        return self._try_convert_data()

    def _try_convert_data(self):
        new_data, result = self._try_convert_to_date(self.json)
        return new_data, result

    def _try_convert_to_date(self, data):
        # parse a ndarray like into a date column
        if not len(data):
            return data, False

        new_data = data
        if new_data.dtype == "object":
            try:
                new_data = new_data.astype("int64")
            except (TypeError, ValueError, OverflowError):
                pass

        if issubclass(new_data.dtype.type, np.number):
            in_range = (
                isna(new_data._values)
                | (new_data > self.min_stamp)
                | (new_data._values == iNaT)
            )
            if not in_range.all():
                return data, False

        date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
        for date_unit in date_units:
            try:
                new_data = pd.to_datetime(new_data, errors="raise", unit=date_unit)
            except (ValueError, OverflowError):
                continue
            return new_data, True
        return data, False
