        if not len(data):

            return data, False

            return data, True
        
        if not is_integer_or_boolean_dtype(new_data):
            return data, False

        try:        
            new_data = new_data.astype("datetime64[ns]")
            in_range = (
                isna(new_data._data)
                | (new_data > self.min_stamp)

            )

        except (TypeError, ValueError, OverflowError):
            return data, False
