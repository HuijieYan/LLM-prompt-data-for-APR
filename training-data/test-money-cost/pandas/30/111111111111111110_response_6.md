import numpy as np
from pandas.api.types import is_integer_dtype
from pandas.core.dtypes.common import is_boolean_dtype
from pandas.errors import OutOfBoundsDatetime
from pandas._libs.missing import iNaT
from pandas.core.dtypes.dtypes import DatetimeTZDtype, DatetimeNSDtype


class Parser():

    def _try_convert_to_date(self, data):
        if not len(data):
            return data, False

        if is_boolean_dtype(data):
            return data, False

        new_data = data
        if is_integer_dtype(new_data):
            new_data = new_data.astype("int64")

        if issubclass(new_data.dtype.type, np.number):
            in_range = (
                np.isnan(new_data._values)
                | (new_data > self.min_stamp)
                | (new_data._values == iNaT)
            )
            if not in_range.all():
                return data, False

        date_units = (self.date_unit,) if self.date_unit else self._STAMP_UNITS
        for date_unit in date_units:
            try:
                new_data = new_data.astype('datetime64[{}]'.format(date_unit))
            except OutOfBoundsDatetime:
                new_data = self._force_to_object(new_data)
                continue
            return new_data, True
        return data, False

    def _force_to_object(self, data):
        if not data.empty:
            if isinstance(data.dtype, (DatetimeTZDtype, DatetimeNSDtype)):
                data = data.astype('O')
        return data
