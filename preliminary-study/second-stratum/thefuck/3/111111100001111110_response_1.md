class Fish(Generic):

    def info(self):
        """Returns the name and version of the current shell"""
        proc = Popen(['fish', '-c', 'echo $FISH_VERSION'],
                     stdout=PIPE, stderr=DEVNULL, encoding='utf-8')
        version = proc.communicate()[0].strip()
        return u'fish, version {}'.format(version)
