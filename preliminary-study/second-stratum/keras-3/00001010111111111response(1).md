        raise ValueError('Expected `model` argument to be a `Model` instance, got ', model)
    if model.built and input_tensors is None:

            input_tensors = [tf_utils.get_tensor_from_tensor_or_input(x) for x in model.inputs]
        except AttributeError:
            raise ValueError("When a model instance has not been compiled, "
