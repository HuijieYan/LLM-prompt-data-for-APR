```python
from pandas import DataFrame, MultiIndex
import pandas as pd

def pivot_table(
    data,
    values=None,
    index=None,
    columns=None,
    aggfunc="mean",
    fill_value=None,
    margins=False,
    dropna=True,
    margins_name="All",
    observed=False,
) -> "DataFrame":
    if not isinstance(columns, tuple):
        columns = (columns,)  # Wrap single column in a tuple

    result = pd.pivot_table(
        data=data,
        values=values,
        index=index,
        columns=columns,
        aggfunc=aggfunc,
        fill_value=fill_value,
        margins=margins,
        dropna=dropna,
        margins_name=margins_name,
        observed=observed
    )

    return result
```
