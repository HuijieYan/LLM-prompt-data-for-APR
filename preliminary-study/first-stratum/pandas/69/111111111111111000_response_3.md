def _convert_key(self, key, is_setter: bool = False):
    """
    Require the keys to be the same type as the index. (so we don't
    fall back)
    """
    # allow arbitrary setting
    if is_setter:
        return list(key)

    for ax, i in zip(self.obj.axes, key):
        if isinstance(ax, pd.Float64Index):
            if not isinstance(i, float):
                raise ValueError(
                    "At-based indexing on an integer index can only have "
                    "integer indexers"
                )
        else:
            if isinstance(i, float) and not isinstance(ax, pd.Float64Index):
                raise ValueError(
                    "At-based indexing on a non-integer index can only have "
                    "non-integer indexers"
                )
    return key
