def copy(self, cursor, f):
        
        logger.info("Inserting file: %s", f)
        colnames = ''
        if self.columns and len(self.columns) > 0:  # Fix the condition to check for NoneType
            colnames = ",".join([x[0] for x in self.columns])
            colnames = '({})'.format(colnames)
    
        cursor.execute("""
            COPY {table} {colnames}
            FROM '{source}'
            CREDENTIALS '{creds}'
            {options};
            """.format(
                table=self.table,
                colnames=colnames,
                source=f,
                creds=self._credentials(),

        )
