        x /= 127.5
        x -= 1.
        return x

        x /= 255.
        mean = [0.485, 0.456, 0.406]
        std = [0.229, 0.224, 0.225]
        x = (x - mean) / std
        return x
    else:
        if data_format == 'channels_first':
            # 'RGB'->'BGR'
            if x.ndim == 3:
                x = x[:, ::-1, ...]

        else:
            # 'RGB'->'BGR'
