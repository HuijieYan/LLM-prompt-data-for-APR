def _preprocess_numpy_input(x, data_format, mode):
    import numpy as np
    if mode == 'tf':
        x = np.divide(x, 127.5)
        x = np.subtract(x, 1.)
        return x

    if mode == 'torch':
        x = np.divide(x, 255.)
        mean = np.array([0.485, 0.456, 0.406])
        std = np.array([0.229, 0.224, 0.225])
    else:
        if data_format == 'channels_first':
            # 'RGB'->'BGR'
            if x.ndim == 3:
                x = x[:, ::-1, ...]
            else:
                x = x[:, :, ::-1, ...]
            mean = np.array([103.939, 116.779, 123.68])
            std = None
        else:
            # 'RGB'->'BGR'
            x = x[..., ::-1]
            mean = np.array([103.939, 116.779, 123.68])
            std = None

    # Zero-center by mean pixel
    if data_format == 'channels_first':
        if x.ndim == 3:
            x[0, :, :] -= mean[0]
            x[1, :, :] -= mean[1]
            x[2, :, :] -= mean[2]
            if std is not None:
                x[0, :, :] /= std[0]
                x[1, :, :] /= std[1]
                x[2, :, :] /= std[2]
        else:
            x[:, 0, :, :] -= mean[0]
            x[:, 1, :, :] -= mean[1]
            x[:, 2, :, :] -= mean[2]
            if std is not None:
                x[:, 0, :, :] /= std[0]
                x[:, 1, :, :] /= std[1]
                x[:, 2, :, :] /= std[2]
    else:
        x[..., 0] -= mean[0]
        x[..., 1] -= mean[1]
        x[..., 2] -= mean[2]
        if std is not None:
            x[..., 0] /= std[0]
            x[..., 1] /= std[1]
            x[..., 2] /= std[2]
    return x
