We need to fix the `_dict_arg` function to properly format the property and value pairs by wrapping them in quotes. We can do this by changing the line `command += [name, '"{0}={1}"'.format(prop, value)]` to `command += [name, '{0}="{1}"'.format(prop, value)]`

def _dict_arg(self, name, value):
    command = []
    if value and isinstance(value, dict):
        for prop, value in value.items():
            command += [name, '{0}="{1}"'.format(prop, value)]
    return command
