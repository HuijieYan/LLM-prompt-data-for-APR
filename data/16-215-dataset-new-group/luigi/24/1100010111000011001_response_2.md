def _dict_arg(self, name, value):
    command = []
    if value and isinstance(value, dict):
        for prop, val in value.items():  # change 'value' to 'val' to avoid variable shadowing
            command += [name, '"{0}={1}"'.format(prop, val)]  # change 'value' to 'val' to avoid variable shadowing
    return command
