def _dict_arg(self, name, value):
        command = []
        if value and isinstance(value, dict):
            for prop, val in value.items(): # Change the variable name from 'value' to 'val' to avoid conflicts
                command += [name, '"{0}={1}"'.format(prop, val)] # Change the variable name from 'value' to 'val' to use the correct value
        return command
