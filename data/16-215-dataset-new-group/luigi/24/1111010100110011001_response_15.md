```python
import luigi

class SparkSubmitTask(luigi.Task):

    def _dict_arg(self, name, value):
        command = []
        if value and isinstance(value, dict):
            for prop, val in value.items():  # FIX: Changed the variable name from value to val to avoid confusion
                command += [name, '"{0}={1}"'.format(prop, val)]  # FIX: Removed the unnecessary quotes around the value
        return command
```
