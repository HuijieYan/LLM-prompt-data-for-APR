def _dict_arg(self, name, value):
    command = []
    if value and isinstance(value, dict):
        for prop, val in value.items():  # renamed the inner variable to val to prevent overwriting the outer value variable
            command += [name, '--{0}={1}'.format(prop, val)]  # changed the format to '--{0}={1}' to match the expected format
    return command
