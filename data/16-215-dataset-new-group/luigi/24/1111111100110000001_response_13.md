# The import necessary for the fixed code
import luigi.contrib.spark
import luigi.task

# Renaming the variable to avoid confusion with function argument name
def _dict_arg(self, name, value_obj):
    command = []
    if value_obj and isinstance(value_obj, dict):
        for prop, value in value_obj.items():
            command += [name, "{0}={1}".format(prop, value)]
    return command
