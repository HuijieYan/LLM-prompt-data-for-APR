def convert_image(self, image, size=None):

        image = image.convert('RGB')
    elif image.mode != 'RGB':
        image = image.convert('RGBA')

    if size:
        thumbnail = image.copy()
        thumbnail.thumbnail(size, Image.ANTIALIAS)
    else:
        thumbnail = image

    buf = BytesIO()
    thumbnail.save(buf, 'JPEG')
    return thumbnail, buf
