        background = Image.new('RGBA', image.size, (255, 255, 255, 255))
        background = Image.alpha_composite(background, image)
        image = background.convert('RGB')
    elif image.mode != 'RGB':
        image = image.convert('RGB')

    if size:
        image = image.copy()
        image.thumbnail(size, Image.ANTIALIAS)

    buf = BytesIO()
