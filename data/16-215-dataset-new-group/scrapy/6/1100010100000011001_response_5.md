def convert_image(self, image, size=None):
    # Check if the image format is 'PNG' and the mode is 'RGBA'
    if image.format == 'PNG' and image.mode == 'RGBA':
        # Create a new RGBA image with a white background
        background = Image.new('RGBA', image.size, (255, 255, 255))
        # Paste the original image onto the new background
        background.paste(image, (0, 0), image)
        # Convert the image to RGB
        image = background.convert('RGB')
    # Check if the image mode is not RGB
    elif image.mode != 'RGB':
        # Convert the image to RGB
        image = image.convert('RGB')

    # If a size is provided, resize the image
    if size:
        image = image.copy()
        image.thumbnail(size, Image.ANTIALIAS)

    # Save the image to a buffer in JPEG format
    buf = BytesIO()
    image.save(buf, 'JPEG')
    # Return the image and the buffer
    return image, buf
