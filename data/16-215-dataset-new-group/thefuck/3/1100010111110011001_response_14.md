def info(self):
    """Returns the name and version of the current shell"""
    # Updated the command to retrieve the shell version
    proc = Popen(['fish', '--version'], stdout=PIPE, stderr=DEVNULL)
    version = proc.stdout.read().decode('utf-8').strip().split(', ')[1]  # Extracts the version from the output
    return u'Fish Shell {}'.format(version)
