from thefuck.types import Command
from thefuck.specific.git import git_support

@git_support
def match(command: Command) -> bool:
    if len(command.script.split()) > 1:
        return (command.script.split()[1] == 'stash'
                and 'usage:' in command.stderr)
    else:
        return False
