@git_support
def match(command):
    git_commands = ['add', 'bisect', 'branch', 'checkout', 'clone', 'commit', 'diff', 'fetch', 'grep', 
                    'init', 'log', 'merge', 'mv', 'pull', 'push', 'rebase', 'reset', 'rm', 'show', 
                    'status', 'tag']
    if command.script.split()[0] == 'git' and 'usage:' in command.stderr and not any(cmd in command.script.split()[1] for cmd in git_commands):
        return True
    else:
        return False
