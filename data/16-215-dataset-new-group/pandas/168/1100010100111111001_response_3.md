def _get_grouper(
    obj,
    key=None,
    axis=0,
    level=None,
    sort=True,
    observed=False,
    mutated=False,
    validate=True,
):
    """
    create and return a BaseGrouper, which is an internal
    mapping of how to create the grouper indexers.
    ...
    """
    group_axis = obj._get_axis(axis)
    
    # validate that the passed single level is compatible with the passed
    # axis of the object
    if level is not None:
        if isinstance(group_axis, MultiIndex):
            # Get the level values from group_axis
            level = group_axis.get_level_values(level)
            key = group_axis
        else:
            raise ValueError("level > 0 or level < -1 only valid with MultiIndex")

    # make sure key is a list
    if key is None or isinstance(key, str):
        key = [key]

    # create the internals grouper
    grouper = BaseGrouper(group_axis, [Grouping(key, obj=obj, name=None, level=level, sort=sort, observed=observed)], sort=sort, mutated=mutated)
    return grouper, None, obj
