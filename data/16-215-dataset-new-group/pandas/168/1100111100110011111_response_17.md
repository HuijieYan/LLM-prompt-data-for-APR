def _get_grouper(
    obj,
    key=None,
    axis=0,
    level=None,
    sort=True,
    observed=False,
    mutated=False,
    validate=True,
):

    if isinstance(key, (list, tuple)) and len(key) == 1:
        key = key[0]

    if isinstance(key, str):
        if key not in obj:
            raise KeyError(key)
