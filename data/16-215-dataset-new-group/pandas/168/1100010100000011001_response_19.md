def _get_grouper(
    obj,
    key=None,
    axis=0,
    level=None,
    sort=True,
    observed=False,
    mutated=False,
    validate=True,
):
    group_axis = obj._get_axis(axis)

    if level is not None:
        if not isinstance(group_axis, MultiIndex):
            if isinstance(level, str) and obj.index.name != level:
                raise ValueError(
                    "level name {} is not the name of the index".format(level)
                )
            elif level > 0 or level < -1:
                raise ValueError("level > 0 or level < -1 only valid with MultiIndex")

    keys = []
    levels = []
    groupings = []
    exclusions = []

    def add_grouping(gpr, level=None):
        if isinstance(gpr, Grouping) and gpr.key is not None:
            exclusions.append(gpr.key)
        else:
            if is_categorical_dtype(gpr) and len(gpr) != obj.shape[axis]:
                raise ValueError(
                    (
                        "Length of grouper ({len_gpr}) and axis ({len_axis})"
                        " must be same length".format(
                            len_gpr=len(gpr), len_axis=obj.shape[axis]
                        )
                    )
                )
            ping = Grouping(
                group_axis,
                gpr,
                obj=obj,
                level=level,
                sort=sort,
                observed=observed
            )
            groupings.append(ping)

    for i, (gpr, level) in enumerate(zip(keys, levels)):
        add_grouping(gpr, level)

    if len(groupings) == 0 and len(obj):
        raise ValueError("No group keys passed!")
    elif len(groupings) == 0:
        group_index = Index([], dtype="int")
        group_array = np.array([], dtype=np.intp)
        groupings.append(Grouping(group_index, group_array))

    grouper = BaseGrouper(group_axis, groupings, sort=sort, mutated=mutated)
    return grouper, exclusions, obj
