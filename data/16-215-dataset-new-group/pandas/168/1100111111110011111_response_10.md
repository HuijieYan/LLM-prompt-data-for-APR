def _get_grouper(
    obj,
    key=None,
    axis=0,
    level=None,
    sort=True,
    observed=False,
    mutated=False,
    validate=True,
):
    """
    create and return a BaseGrouper, which is an internal
    mapping of how to create the grouper indexers.
    This may be composed of multiple Grouping objects, indicating
    multiple groupers

    Groupers are ultimately index mappings. They can originate as:
    index mappings, keys to columns, functions, or Groupers

    Groupers enable local references to axis,level,sort, while
    the passed in axis, level, and sort are 'global'.

    This routine tries to figure out what the passing in references
    are and then creates a Grouping for each one, combined into
    a BaseGrouper.

    If observed & we have a categorical grouper, only show the observed
    values

    If validate, then check for key/level overlaps

    """
    group_axis = obj._get_axis(axis)

    # validate that the passed single level is compatible with the passed
    # axis of the object
    if level is not None:
        if isinstance(group_axis, MultiIndex):
            if is_list_like(level) and len(level) == 1:
                level = level[0]

            if key is None and is_scalar(level):
                # Get the level values from group_axis
                key = group_axis.get_level_values(level)
                level = None
        else:
            if is_list_like(level):
                nlevels = len(level)
                if nlevels == 1:
                    level = level[0]
                elif nlevels == 0:
                    raise ValueError("No group keys passed!")
                else:
                    raise ValueError("multiple levels only valid with MultiIndex")

            if isinstance(level, str):
                if obj.index.name != level:
                    raise ValueError(
                        "level name {} is not the name of the index".format(level)
                    )
            elif level > 0 or level < -1:
                raise ValueError("level > 0 or level < -1 only valid with MultiIndex")

            level = None
            key = group_axis

    if isinstance(key, Grouper):
        if key.key is None:
            return key._get_grouper(obj, validate=False)[1:3] + ([])
        return key._get_grouper(obj, validate=False)[1:]
    elif isinstance(key, BaseGrouper):
        return (key, [], obj)
    else:
        if isinstance(key, tuple):
            msg = (
                "Interpreting tuple 'by' as a list of keys, rather than "
                "a single key. Use 'by=[...]' instead of 'by=(...)'. In "
                "the future, a tuple will always mean a single key."
            )
            warnings.warn(msg, FutureWarning, stacklevel=5)
            key = list(key)
        keys = [key] if not isinstance(key, list) else key
        match_axis_length = len(keys) == len(group_axis)

        match_values = []
        match_axis = (
            obj.columns if axis else obj.index if level is None else group_axis
        )
        for k in keys:
            match = (
                False
                if callable(k) or isinstance(k, dict)
                else isinstance(k, (list, tuple, Series, Index, np.ndarray))
                or (len(keys) > 1 and isinstance(k, str) and len(keys) == len(group_axis))
            )
            if not match and len(keys) == len(group_axis):
                match = k in match_axis.names
            match_values.append(match)
        if all(match_values):
            return (BaseGrouper(group_axis, [], sort=sort, mutated=mutated), [], obj)
        groups, exclusions = [], []
        for i, key in enumerate(keys):
            try:
                level = level[i] if isinstance(level, (tuple, list)) else level
                selection = Grouping(
                    group_axis,
                    key,
                    name=group_axis.names[i] if isinstance(key, str) else None,
                    level=level,
                    sort=sort,
                    in_axis=key in group_axis,
                    obj=obj,
                )
                groups.extend([selection])
                if key not in obj or group_axis.is_level_reference(key):
                    exclusions.extend([key])
                else:
                    in_axis = obj._data.items.get_loc(key) if not _is_label_like(key) else True
                    if validate and not in_axis:
                        obj._check_label_or_level_ambiguity(key)
            except (ValueError, KeyError):
                exclusions.extend([key])
        if not groups:
            raise ValueError("No group keys passed!")
        return (
            BaseGrouper(group_axis, groups, sort=sort, mutated=mutated),
            exclusions,
            obj,
        )
