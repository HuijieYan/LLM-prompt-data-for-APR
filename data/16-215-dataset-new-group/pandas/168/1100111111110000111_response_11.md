def _get_grouper(
    obj,
    key=None,
    axis=0,
    level=None,
    sort=True,
    observed=False,
    mutated=False,
    validate=True,
):
    """
    create and return a BaseGrouper, which is an internal
    mapping of how to create the grouper indexers.
    This may be composed of multiple Grouping objects, indicating
    multiple groupers

    Groupers are ultimately index mappings. They can originate as:
    index mappings, keys to columns, functions, or Groupers

    Groupers enable local references to axis,level,sort, while
    the passed in axis, level, and sort are 'global'.

    This routine tries to figure out what the passing in references
    are and then creates a Grouping for each one, combined into
    a BaseGrouper.

    If observed & we have a categorical grouper, only show the observed
    values

    If validate, then check for key/level overlaps

    """
    group_axis = obj._get_axis(axis)

    # validate that the passed single level is compatible with the passed
    # axis of the object
    if level is not None:
        if is_list_like(level):
            nlevels = len(level)
            if nlevels == 1:
                level = level[0]
            elif nlevels == 0:
                raise ValueError("No group keys passed!")
            else:
                raise ValueError("multiple levels only valid with MultiIndex")
        if isinstance(level, str):
            if obj.index.name != level and group_axis.name != level:
                raise ValueError(
                    "level name {} is not the name of the index or columns"
                    .format(level)
                )
        elif level > 0 or level < -1:
            raise ValueError("level > 0 or level < -1 only valid with MultiIndex")

    keys = [key] if is_scalar(key) else key

    groupings = []
    exclusions = []

    for i, (group_key, grp_level) in enumerate(zip(keys, [level] * len(keys))):    
        if isinstance(group_key, Grouper) and group_key.key is not None:
            # Add key to exclusions
            exclusions.append(group_key.key)
            in_axis, group_key, grp_level, level_type = False, group_key.key, None, "Index"
        else:
            if axis == 1 and is_label_like(group_key) and is_list_like(group_key):
                level_type = "Columns"
            else:
                level_type = "Index"
            
            if level_type == "Columns":
                if group_key in obj:
                    if validate:
                        obj._check_label_or_level_ambiguity(group_key, axis=1)
                    in_axis, name = True, group_key
                    if isinstance(obj, DataFrame) and group_key in obj.columns:
                        group_key = obj[group_key]
                    exclusions.append(name)
                elif obj._is_level_reference(group_key):
                    in_axis, name, grp_level, group_key = False, None, group_key, None
                else:
                    raise KeyError(group_key)
            else:  # Index level/reference
                if group_key in obj:
                    in_axis, name = True, group_key
                    if group_key in group_axis:
                        group_key = group_axis.get_level_values(group_key)
                    exclusions.append(group_key)
                else:
                    raise KeyError(group_key)

        # Create the Grouping
        ping = (
            Grouping(
                group_axis,
                group_key,
                obj=obj,
                name=name,
                level=grp_level,
                sort=sort,
                observed=observed,
                in_axis=in_axis,
            )
        )
        groupings.append(ping)

    if not groupings and len(obj):
        raise ValueError("No group keys passed!")

    grouper = BaseGrouper(group_axis, groupings, sort=sort, mutated=mutated)
    return grouper, exclusions, obj
