def _get_grouper(
    obj,
    key=None,
    axis=0,
    level=None,
    sort=True,
    observed=False,
    mutated=False,
    validate=True,
):
    group_axis = obj.groupby(obj.columns, axis=axis).grouper.groupings
    if "Grouping" in str(type(group_axis)):
        return group_axis, [], obj
    else:
        raise ValueError("No group keys passed!")
