def _get_grouper(
    obj,
    key=None,
    axis=0,
    level=None,
    sort=True,
    observed=False,
    mutated=False,
    validate=True,
):
    group_axis = obj._get_axis(axis)

    if level is not None:
        if not isinstance(group_axis, MultiIndex):
            if is_list_like(level):
                nlevels = len(level)
                if nlevels == 1:
                    level = level[0]
                elif nlevels == 0:
                    raise ValueError("No group keys passed!")
                else:
                    raise ValueError("multiple levels only valid with MultiIndex")

            if isinstance(level, str):
                if obj.index.name != level:
                    raise ValueError(
                        "level name {} is not the name of the index".format(level)
                    )
            elif level > 0 or level < -1:
                raise ValueError("level > 0 or level < -1 only valid with MultiIndex")
            level = None  # There was a duplicate assignment for level, so removed the previous one
                
            key = group_axis

    if isinstance(key, Grouper) or isinstance(key, BaseGrouper):  # Combine these conditions to handle the same return statement
        return key, [], obj

    if isinstance(key, tuple):
        if all(is_hashable(val) for val in key):
            if key not in obj:
                key = list(key)
                msg = (
                    "Interpreting tuple 'by' as a list of keys, rather than "
                    "a single key. Use 'by=[...]' instead of 'by=(...)'. In "
                    "the future, a tuple will always mean a single key."
                )
                warnings.warn(msg, FutureWarning, stacklevel=5)

    keys = key if isinstance(key, list) else [key]  #  Transform key into a list

    match_axis_length = len(keys) == len(group_axis)
    any_callable = any(callable(g) or isinstance(g, dict) for g in keys)
    any_groupers = any(isinstance(g, Grouper) for g in keys)
    any_arraylike = any(
        isinstance(g, (list, tuple, Series, Index, np.ndarray)) for g in keys
    )

    if not any_callable and not any_arraylike and not any_groupers and match_axis_length and level is None:
        all_in_columns_index = all(g in obj.columns or g in obj.index.names for g in keys)
        if not all_in_columns_index:
            keys = [com.asarray_tuplesafe(keys)]

    if isinstance(level, (tuple, list)):
        if key is None:
            keys = [None] * len(level)
    else:
        levels = [level] * len(keys)

    groupings = []
    exclusions = []

    for i, (gpr, level) in enumerate(zip(keys, levels):

        if not is_label_like(gpr) and gpr not in group_axis:
            raise KeyError(gpr)

        if is_list_like(level):  # Added an extra condition to check if level is list_like
            nlevels = len(level)
            if nlevels == 1:
                level = level[0]
            elif nlevels == 0:
                raise ValueError("No group keys passed!")
            else:
                raise ValueError("multiple levels only valid with MultiIndex")

        if not is_label_like(level) and level not in obj.index:  # Modified the condition to check for level
            raise ValueError("level name {} is not the name of the index".format(level))

        if not isinstance(gpr, Grouping):
            ping = (
                Grouping(
                    group_axis,
                    gpr,
                    obj=obj,
                    level=level,
                    sort=sort,
                    observed=observed,
                    in_axis=True,
                )
                if is_label_like(gpr)
                else (
                    Grouping(
                        group_axis,
                        group_axis.get_level_values(level),
                        obj=obj,
                        sort=sort,
                        observed=observed,
                        in_axis=False,
                    )
                    if level is not None and not isinstance(group_axis, MultiIndex)
                    else (
                        Grouping(
                            group_axis, None, obj=obj, level=level, sort=sort, observed=observed, in_axis=False
                        )
                    )
                )
            )
        else:
            ping = gpr

        groupings.append(ping)

    if len(groupings) == 0:
        raise ValueError("No group keys passed!")

    grouper = BaseGrouper(group_axis, groupings, sort=sort, mutated=mutated)
    return grouper, exclusions, obj
