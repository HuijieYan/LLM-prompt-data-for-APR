def _get_grouper(
    obj,
    key=None,
    axis=0,
    level=None,
    sort=True,
    observed=False,
    mutated=False,
    validate=True,
):
    # Grouping along axis 0 (index)
    if axis == 0:
        return obj._get_axis(axis), [], obj
    # Grouping along axis 1 (columns)
    elif axis == 1:
        if level is not None:
            raise ValueError(
                "Level parameter is not supported for grouping along columns"
            )

        if key in obj.columns:
            # Grouping by column name
            gpr = obj[key]
        else:
            # Grouping by an iterable of keys
            gpr = [obj[key] for key in key]

        # Creating the Grouping object
        ping = Grouping(
            obj._get_axis(axis),
            gpr,
            obj=obj,
            name=key,
            level=level,
            sort=sort,
            observed=observed,
            in_axis=True,
        )

        # return the BaseGrouper
        return BaseGrouper(obj._get_axis(axis), [ping], sort=sort, mutated=mutated), [], obj
    else:
        raise ValueError("Invalid value for 'axis'. Should be 0 or 1.")
