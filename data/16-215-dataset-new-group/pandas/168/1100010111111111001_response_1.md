def _get_grouper(
    obj,
    key=None,
    axis=0,
    level=None,
    sort=True,
    observed=False,
    mutated=False,
    validate=True,
):
    """
    create and return a BaseGrouper, which is an internal
    mapping of how to create the grouper indexers.
    This may be composed of multiple Grouping objects, indicating
    multiple groupers

    Groupers are ultimately index mappings. They can originate as:
    index mappings, keys to columns, functions, or Groupers

    Groupers enable local references to axis,level,sort, while
    the passed in axis, level, and sort are 'global'.

    This routine tries to figure out what the passing in references
    are and then creates a Grouping for each one, combined into
    a BaseGrouper.

    If observed & we have a categorical grouper, only show the observed
    values

    If validate, then check for key/level overlaps

    """
    group_axis = obj._get_axis(axis)

    level, key = validate_group_level_key(obj, level, key, group_axis)
    
    groupings = []
    exclusions = []

    for k, l in zip(key, level):
        is_multi_level = isinstance(obj.index, MultiIndex)
        if is_multi_level and (not is_list_like(k) or len(k) > 1):
            raise ValueError("Level type is a MultiIndex, only a one-element list can be used as key.")
        
        if is_multi_level and is_scalar(k):
            k = [k]

        # Check if the key is in index or index name, and get the group values accordingly
        if k in obj.index or (obj.index.name == k and len(k) == 1):
            key = obj.index.get_level_values(k)
        else:
            raise KeyError(f"Key '{k}' not found in index or index name")

        # Create the Grouping and add it to the groupings list
        groupings.append(
            Grouping(
                group_axis,
                key,
                obj=obj,
                name=k,
                level=l,
                sort=sort,
                observed=observed,
                in_axis=is_in_axis(k, obj),
            )
        )

    if len(groupings) == 0 and len(obj):
        raise ValueError("No group keys passed!")

    # create the internals grouper
    grouper = BaseGrouper(group_axis, groupings, sort=sort, mutated=mutated)
    return grouper, exclusions, obj
