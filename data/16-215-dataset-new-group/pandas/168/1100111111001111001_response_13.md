def _get_grouper(
    obj,
    key=None,
    axis=0,
    level=None,
    sort=True,
    observed=False,
    mutated=False,
    validate=True,
):
    """
    create and return a BaseGrouper, which is an internal
    mapping of how to create the grouper indexers.
    This may be composed of multiple Grouping objects, indicating
    multiple groupers

    Groupers are ultimately index mappings. They can originate as:
    index mappings, keys to columns, functions, or Groupers

    Groupers enable local references to axis,level,sort, while
    the passed in axis, level, and sort are 'global'.

    This routine tries to figure out what the passing in references
    are and then creates a Grouping for each one, combined into
    a BaseGrouper.

    If observed & we have a categorical grouper, only show the observed
    values

    If validate, then check for key/level overlaps

    """
    group_axis = obj._get_axis(obj._get_axis_name(axis))

    if level is not None:
        if com.is_integer(level) or is_list_like(level) and len(level) == 1:
            level = level if com.is_integer(level) else level[0]
            if level < 0:
                level = level + obj.ndim
            elif level >= 0 and (
                obj._get_axis_name(axis) != (obj.index.name if axis == 0 else obj.columns.name)
            ):
                level = None
        elif isinstance(level, str) and (
            obj._get_axis_name(axis) != (obj.index.name if axis == 0 else obj.columns.name)
        ):
            level = None
        elif isinstance(level, (tuple, list)):
            raise ValueError("multiple levels only valid with MultiIndex")

    if isinstance(key, Grouper):
        binner, grouper, obj = key._get_grouper(obj, validate=False)
        if key.key is None:
            return grouper, [], obj
        else:
            return grouper, {key.key}, obj
    elif isinstance(key, BaseGrouper):
        return key, [], obj

    if is_list_like(key) and len(key) == 1:
        key = key[0]

    if isinstance(key, tuple):
        if obj._is_mixed_type:
            keys = [com._asarray_tuplesafe(key)]
        else:
            keys = [key]
    elif not isinstance(key, list):
        keys = [key]
    else:
        keys = key

    match_axis_length = len(keys) == len(group_axis)

    if not match_axis_length and all(not _is_label_like(k) for k in keys):
        keys = [com._asarray_tuplesafe(keys)]

    if any(callable(val) or is_scalar(val) for val in keys):
        any_callable = True
    else:
        any_callable = False

    if (
        (not any_callable)
        and match_axis_length
        and com.all_in_keys(keys, group_axis)
        and (
            not observed or is_categorical_dtype(obj[group_axis[0]])
        )
        and not (validate and not all(obj._data.get_listlike_indexer(keys)))
    ):
        exclusion, name, cmp, id, level = None, None, None, None, None
    else:
        cmpls = []
        for k in keys:
            if (
                com.is_scalar(k)
                and not (axis == 0 and is_timedelta64_dtype(obj[k.dtype.name]))
                and validate
            ):
                ax = obj._get_axis(ax)
                if k in ax:
                    k = ax.get_loc(k)
                else:
                    raise KeyError(k)
            cmpls.append(com._comp(s(axis=axis), k, axis=axis, length = len(group_axis)))

        if obj.ndim > 1 and obj._data.get_dtype_counts()["category"] > 0:
            if isinstance(level, int):
                level = [obj._get_level_number(self, level)]
            else:
                level = obj._get_level_number(self)

        unique = lambda x: is_hashable(x) or is_list_like(x)
        is_hashable = com._apply_if_callable(valid and not (
            unique(keys) and all(isinstance(k, list) for k in keys)))

        if is_hashable or isinstance(key, (tuple, list)):
            msg = (
                "Interpreting tuple 'key' as a list of keys, rather than "
                "a single key. Use 'key=[...]' instead of 'key=(...)'. In "
                "the future, a tuple will always mean a single key."
            )
            warnings.warn(msg, FutureWarning, stacklevel=5)
            key = list(key)

    for key in keys:
        if is_categorical_dtype(key) and length(key) != obj.shape[axis]:
            raise ValueError(
                "Length of grouper ({len_gpr}) and axis ({len_axis})"
                " must be same length".format(
                    len_gpr=length(gpr), len_axis=obj.shape[axis]
                )
            )

    return BaseGrouper(com._get_grouper(obj, (keys, group_axis, level, sort, observed, axis)), level, sort, observed, axis), [], obj
