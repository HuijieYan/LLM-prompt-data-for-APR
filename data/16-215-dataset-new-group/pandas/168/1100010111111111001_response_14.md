def _get_grouper(
    obj,
    key=None,
    axis=0,
    level=None,
    sort=True,
    observed=False,
    mutated=False,
    validate=True,
):
    group_axis = obj._get_axis(axis)
    groupings = []
    exclusions = []

    if is_list_like(key) and len(key) == 1:
        key = key[0]

    # Removed the code handling tuple key, and restructured the code regarding the key and level conditions while keeping the MultiIndex instance check as is.
    if isinstance(level, (tuple, list)):
        if key is None:
            keys = [None] * len(level)
        levels = level
    else:
        keys = [key]
        levels = [level] * len(keys)

    for i, (gpr, level) in enumerate(zip(keys, levels)):
        if is_categorical_dtype(gpr) and len(gpr) != obj.shape[axis]:
            raise ValueError(
                (
                    "Length of grouper ({len_gpr}) and axis ({len_axis})"
                    " must be same length".format(
                        len_gpr=len(gpr), len_axis=obj.shape[axis]
                    )
                )
            )

        ping = (
            Grouping(
                group_axis,
                gpr,
                obj=obj,
                name=gpr.name if isinstance(gpr, Grouper) else None,
                level=level,
                sort=sort,
                observed=observed,
            )
            if not isinstance(gpr, Grouping)
            else gpr
        )
        groupings.append(ping)

    if not groupings:
        raise ValueError("No group keys passed!")

    grouper = BaseGrouper(group_axis, groupings, sort=sort, mutated=mutated)
    return grouper, exclusions, obj
