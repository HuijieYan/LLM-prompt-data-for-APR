def _get_grouper(
    obj,
    key=None,
    axis=0,
    level=None,
    sort=True,
    observed=False,
    mutated=False,
    validate=True,
):
    """
    create and return a BaseGrouper, which is an internal
    mapping of how to create the grouper indexers.
    This may be composed of multiple Grouping objects, indicating
    multiple groupers

    Groupers are ultimately index mappings. They can originate as:
    index mappings, keys to columns, functions, or Groupers

    Groupers enable local references to axis,level,sort, while
    the passed in axis, level, and sort are 'global'.

    This routine tries to figure out what the passing in references
    are and then creates a Grouping for each one, combined into
    a BaseGrouper.

    If observed & we have a categorical grouper, only show the observed
    values

    If validate, then check for key/level overlaps

    """
    import pandas as pd
    group_axis = obj._get_axis(axis)
    
    # Handle the case when key is a single string
    if isinstance(key, str):
        key = [key]

    if level is not None and not isinstance(group_axis, pd.MultiIndex):
        group_axis = group_axis.to_series()

    groupings = []
    exclusions = []
    
    for gpr, lev in zip(key or [], level or []):
        try:
            if gpr in obj:
                gpr = obj[gpr]
            elif gpr in group_axis:
                gpr = group_axis[gpr]

            if lev is not None and lev > 0:
                raise ValueError(f"level > 0 or level < -1 only valid with MultiIndex")
            elif lev is not None and isinstance(group_axis, pd.MultiIndex):
                gpr = gpr.droplevel(level=lev) if gpr.ndim == 2 else gpr
        except KeyError:
            raise KeyError(f"{gpr}")
        
        groupings.append(
            pd.core.groupby.grouper.Grouping(
                group_axis, gpr, obj=obj, level=lev, sort=sort, observed=observed, in_axis=True
            )
        )

    if not groupings:
        raise ValueError("No group keys passed!")

    # create the internals grouper
    grouper = pd.core.groupby.ops.BaseGrouper(group_axis, groupings, sort=sort, mutated=mutated)
    return grouper, exclusions, obj
