def _get_grouper(
    obj,
    key=None,
    axis=0,
    level=None,
    sort=True,
    observed=False,
    mutated=False,
    validate=True,
):
    """
    create and return a BaseGrouper, which is an internal
    mapping of how to create the grouper indexers.
    This may be composed of multiple Grouping objects, indicating
    multiple groupers

    Groupers are ultimately index mappings. They can originate as:
    index mappings, keys to columns, functions, or Groupers

    Groupers enable local references to axis,level,sort, while
    the passed in axis, level, and sort are 'global'.

    This routine tries to figure out what the passing in references
    are and then creates a Grouping for each one, combined into
    a BaseGrouper.

    If observed & we have a categorical grouper, only show the observed
    values

    If validate, then check for key/level overlaps

    """
    group_axis = obj._get_axis(axis)
    group_axis_name = obj.index.name

    if level is not None:
        if isinstance(group_axis, MultiIndex):
            # Check for MultiIndex and handle accordingly
            if is_list_like(level) and len(level) == 1:
                level = level[0]
            if key is None and is_scalar(level):
                key = group_axis.get_level_values(level)
                level = None
        else:
            # Handle index and level for non-MultiIndex
            if is_list_like(level):
                nlevels = len(level)
                if nlevels == 1:
                    level = level[0]
                elif nlevels == 0:
                    raise ValueError("No group keys passed!")
                else:
                    raise ValueError("multiple levels only valid with MultiIndex")

            if isinstance(level, str):
                if group_axis_name != level:
                    raise ValueError(f"level name {level} is not the name of the index")
            elif level != 0 and level != -1:
                raise ValueError("level 0 or -1 only valid with MultiIndex")
            level = None
            key = group_axis

    if isinstance(key, BaseGrouper):
        return key, [], obj

    if isinstance(key, tuple):
        if all(is_hashable(element) for element in key):
            if key not in obj and set(key).issubset(obj):
                warnings.warn(
                    (
                        "Interpreting tuple 'by' as a list of keys, rather than "
                        "a single key. Use 'by=[...]' instead of 'by=(...)'. In "
                        "the future, a tuple will always mean a single key."
                    ),
                    FutureWarning,
                    stacklevel=5,
                )
                key = list(key)
            else:
                key = [key]

    keys = [key] if not isinstance(key, list) else key
    match_axis_length = len(keys) == len(group_axis)

    # Create Grouping objects for each key
    groupings = []
    exclusions = []
    for i, key in enumerate(keys):
        gpr = key if isinstance(key, str) else None

        if is_categorical_dtype(gpr) and len(gpr) != len(group_axis):
            raise ValueError(
                (
                    "Length of grouper ({len_gpr}) and axis ({len_axis})"
                    " must be same length".format(
                        len_gpr=len(gpr), len_axis=len(group_axis)
                    )
                )
            )

        ping = Grouping(
            group_axis,
            gpr,
            obj=obj,
            name=None if isinstance(key, str) else gpr,
            level=level,
            sort=sort,
            observed=observed,
            in_axis=True if isinstance(key, str) else False,
        )
        groupings.append(ping)

    if len(groupings) == 0 and len(obj) > 0:
        raise ValueError("No group keys passed!")
    elif len(groupings) == 0:
        groupings.append(Grouping(Index([], dtype='int'), np.array([], dtype=np.intp)))

    grouper = BaseGrouper(group_axis, groupings, sort=sort, mutated=mutated)
    return grouper, exclusions, obj
