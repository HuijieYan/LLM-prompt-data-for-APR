def _get_grouper(
    obj,
    key=None,
    axis=0,
    level=None,
    sort=True,
    observed=False,
    mutated=False,
    validate=True,
):
    group_axis = obj._get_axis(axis)

    if key is None or key == obj.columns:
        return BaseGrouper(group_axis, [], sort=sort, mutated=mutated), [], obj
    else:
        if key == obj.index:
            key = [group_axis.get_loc(k) for k in key]

        if not isinstance(key, (list, tuple)):
            key = [key]

        if level is not None:
            if not isinstance(level, list):
                level = [level] * len(key)

        groupings = []
        exclusions = []

        for k, l in zip(key, level or [None] * len(key)):
            ping = Grouping(
                group_axis,
                k,
                obj=obj,
                name=k,
                level=l,
                sort=sort,
                observed=observed
            )
            groupings.append(ping)

        return BaseGrouper(group_axis, groupings, sort=sort, mutated=mutated), exclusions, obj
