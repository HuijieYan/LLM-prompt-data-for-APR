def apply(self, other):
    if isinstance(other, datetime):
        # used for detecting edge condition
        nanosecond = getattr(other, "nanosecond", 0)

        # adjust other to reduce cases to handle
        other = datetime(
            other.year,
            other.month,
            other.day,
            other.hour,
            other.minute,
            other.second,
            other.microsecond,
        )
        n = self.n

        # get total business hours by sec in one business day
        businesshours = sum(
            self._get_business_hours_by_sec(st, en) for st, en in zip(self.start, self.end)
        )

        bd, r = divmod(abs(n * 60), businesshours // 60)
        if n < 0:
            bd, r = -bd, -r

        if bd != 0:
            # adjust by business days first
            skip_bd = BusinessDay(n=bd)
            prev_open = self._prev_opening_time(other)

            # midnight business hour may not be on BusinessDay
            if not self.next_bday._is_on_offset(other):
                remain = other - prev_open
                other = prev_open + skip_bd + remain
            else:
                other = other + skip_bd
        
        bhour_remain = timedelta(minutes=r)
        end_st = shift_month(other) if n < 0 else other

        while bhour_remain > timedelta(0):
            bhour = (self._next_opening_time(other) - end_st) if n < 0 else (end_st - other)

            if bhour_remain < bhour or (
                bhour_remain == bhour and nanosecond != 0
            ):
                # finish adjusting if possible
                other = other + bhour_remain if n >= 0 else end_st - bhour_remain
                bhour_remain = timedelta(0)
            else:
                # go to next business time interval
                bhour_remain -= bhour
                other = (
                    self._get_closing_time(self._next_opening_time(end_st))
                    if n < 0
                    else self._next_opening_time(other) + bhour
                )

    return other
