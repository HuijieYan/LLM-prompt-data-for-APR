from pandas.core.indexers import convert_to_index_sliceable, convert_to_index_labels

def _get_with(self, key):
    # other: fancy integer or otherwise
    if isinstance(key, slice):
        slobj = convert_to_index_sliceable(self.index, key)
        return self._slice(slobj)
    elif isinstance(key, (list, np.ndarray, ExtensionArray, Series)):
        key = convert_to_index_labels(key, self.index)
    elif isinstance(key, ABCDataFrame):
        raise TypeError(
            "Indexing a Series with DataFrame is not "
            "supported, use the appropriate DataFrame column"
        )
    elif isinstance(key, tuple):
        return self._get_values_tuple(key)

    if not is_list_like(key):
        return self.loc[key]

    if not self.index.isin(key).all():
        raise KeyError(f"None of {key} are in the index")

    if is_integer(key):
        if not self.index._should_fallback_to_positional():
            return self.loc[key]
        else:
            return self.iloc[key]

    # handle the dup indexing case
    return self.loc[key].drop_duplicates()
