def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data

    index = data.index
    clocs = [index._get_level_number(i) for i in clocs]
  
    # new_columns, new_levels, and new_codes calculation is corrected
    rlocs = [i for i in range(index.nlevels) if i not in clocs]

    if rlocs:
        idx_rlocs = [index.get_level_values(i) for i in rlocs]
        idx_clocs = [index.get_level_values(i) for i in clocs]
        dummy_index = pd.MultiIndex.from_tuples(
            zip(*[x.ravel() for x in np.meshgrid(*idx_rlocs, indexing='ij')] + idx_clocs),
            names=index.names[rlocs] + index.names[clocs]
        )
        unstacked = pd.concat({level: data}, axis=1, names=index.names[rlocs]).unstack(level=clocs)

        new_columns = unstacked.columns
    else:
        # If there are no rlocs, the index is the same
        idx_clocs = [index.get_level_values(i) for i in clocs]

        dummy_index = pd.MultiIndex.from_tuples(
            zip(*idx_clocs), names=index.names[clocs]
        )

        unstacked = data.unstack(clocs)
        new_columns = unstacked.columns

    return unstacked, dummy_index, new_columns
