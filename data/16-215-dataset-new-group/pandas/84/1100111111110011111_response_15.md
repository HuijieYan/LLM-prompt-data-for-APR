import pandas as pd
from pandas.core.indexes.api import Index, MultiIndex
from pandas.core.series import Series
from pandas.core.sorting import compress_group_index, decons_obs_group_ids, get_compressed_ids, get_group_index
import numpy as np


def _unstack_multiple(data, clocs, fill_value=None):
    original_data = data.copy()

    if len(clocs) == 0:
        return data

    level_values = [[(-1, i, j), None] for i, j in zip(data.index.names, clocs)]

    # NOTE: This doesn't deal with hierarchical columns yet

    index = data.index

    rlocs = [-1]  # setting for a single level

    clevels = [index.levels[i] for i in clocs]
    ccodes = [index.codes[i] for i in clocs]
    cnames = [index.names[i] for i in clocs]
    rlevels = index.levels[rlocs[0]]
    rcodes = index.codes[rlocs[0]]
    rnames = index.names[rlocs[0]]

    # New level to unstack by
    shape = [len(x) for x in clevels]
    group_index = get_group_index(ccodes, shape, sort=False, xnull=False)

    comp_ids, obs_ids = compress_group_index(group_index)
    recons_codes = decons_obs_group_ids(comp_ids, obs_ids, shape, ccodes, xnull=False)

    # The <if> and <else> blocks are modified to ensure the correct unstacking using the restructured multi-level index
    if isinstance(data, Series):
        dummy_index = data.index.take(recons_codes[0])
        dummy_index = dummy_index.set_names(level_values[0][1], level_values[0][0])
        dummy = data.copy()
        dummy.index = dummy_index

        unstacked = dummy.unstack(fill_value=fill_value)
        new_levels = [rlevels] + clevels
        new_names = [rnames] + cnames
        new_codes = recons_codes
    else:
        dummy_index = data.index.take(recons_codes[-1])

        dummy_index = dummy_index.set_names(level_values[-1][1], level_values[-1][0])
        dummy = data.copy()
        dummy.index = dummy_index

        unstacked = dummy.unstack(fill_value=fill_value)
        new_levels = clevels
        new_names = cnames
        new_codes = recons_codes[1:]

    new_columns = MultiIndex(levels=new_levels, codes=new_codes, names=new_names, verify_integrity=False)
    unstacked = unstacked.reorder_levels(rlocs[0], axis=1)

    unstacked.columns = new_columns
    return unstacked
