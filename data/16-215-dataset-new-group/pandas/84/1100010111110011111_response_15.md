def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data

    # Get the index of the data
    index = data.index
    
    # Extract the positions of the levels from the data index
    clocs_idx = [index._get_level_number(i) for i in clocs]

    rlocs = [i for i in range(index.nlevels) if i not in clocs_idx]

    shape = [len(index.levels[i]) for i in clocs_idx]
    group_index = get_group_index(index.codes, shape, sort=False, xnull=False)

    # Split the group index into compressed and observed ids
    comp_ids, obs_ids = compress_group_index(group_index, sort=False)
    recons_codes = decons_obs_group_ids(comp_ids, obs_ids, shape, index.codes, xnull=False)

    if rlocs == []:
        # Create a dummy index which contains only observed ids
        dummy_index = Index(obs_ids, name="__placeholder__")
    else:
        dummy_index = MultiIndex(
            levels=rlevels + [obs_ids],
            codes=rcodes + [comp_ids],
            names=rnames + ["__placeholder__"],
            verify_integrity=False,
        )

    if isinstance(data, Series):
        # If the input data is a Series object, create a copy and change the index
        dummy = data.copy()
        dummy.index = dummy_index
        # Unstack using the placeholder level with the specified fill value
        unstacked = dummy.unstack("__placeholder__", fill_value=fill_value)
        new_levels = index.levels
        new_names = [data.name] + to_list(index.names)
        new_codes = recons_codes
    else:
        # If the input data is a DataFrame object
        if not is_list_like(clocs, allow_generators=True):
            clocs = [clocs]

        # Cast the columns to a MultiIndex
        result = data.copy()
        for i in range(len(clocs)):
            result = result.unstack(clocs[i], fill_value=fill_value)
            clocs = [v if i < v else v - 1 for v in clocs]

        return result

    new_columns = MultiIndex(
        levels=new_levels, codes=new_codes, names=new_names, verify_integrity=False
    )

    if isinstance(unstacked, Series):
        unstacked.index = new_columns
    else:
        unstacked.columns = new_columns

    return unstacked
