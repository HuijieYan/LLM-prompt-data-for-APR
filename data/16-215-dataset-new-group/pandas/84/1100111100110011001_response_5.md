def _unstack_multiple(data, clocs, fill_value=None):
    if len(clocs) == 0:
        return data
    
    # Retrieve the index
    index = data.index

    # Extract level numbers from clocs
    cloc_numbers = [index._get_level_number(i) if isinstance(i, str) else i for i in clocs]

    # Get remaining level numbers
    rloc_numbers = [i for i in range(index.nlevels) if i not in cloc_numbers]
    
    # Extract levels, codes, and names
    clevels = [index.levels[i] for i in cloc_numbers]
    ccodes = [index.codes[i] for i in cloc_numbers]
    cnames = [index.names[i] for i in cloc_numbers]
    rlevels = [index.levels[i] for i in rloc_numbers]
    rcodes = [index.codes[i] for i in rloc_numbers]
    rnames = [index.names[i] for i in rloc_numbers]
    
    # Get the shape
    shape = [len(x) for x in clevels]
    
    # Compress the group index
    group_index = get_group_index(ccodes, shape, sort=False, xnull=False)
    comp_ids, obs_ids = compress_group_index(group_index, sort=False)
    recons_codes = decons_obs_group_ids(comp_ids, obs_ids, shape, ccodes, xnull=False)
    
    # Check if there are no remaining levels
    if not rloc_numbers:
        dummy_index = Index(obs_ids, name="__placeholder__")
        unstacked = data.reindex(dummy_index, fill_value=fill_value).to_frame().unstack("__placeholder__")
        unstacked.index = MultiIndex.from_arrays(unstacked.index.values, names=cnames+['__placeholder__'])
        unstacked.columns = unstacked.columns.droplevel()
        return unstacked
    else:
        dummy_index = MultiIndex(
            levels=rlevels + [obs_ids],
            codes=rcodes + [comp_ids],
            names=rnames + ["__placeholder__"],
            verify_integrity=False,
        )
        dummy = data.reindex(dummy_index, fill_value=fill_value)
        unstacked = dummy.to_frame().unstack('__placeholder__')
        
        # Construct the new level order and names
        if isinstance(data, Series):
            new_levels_order = [data.index, *clevels]
            new_names = data.index.names + cnames
        else:
            new_levels_order = [unstacked.columns.levels[0], *clevels]
            new_names = [data.columns.names, *cnames]

        # Create the new MultiIndex from the new level order and names
        new_multiindex = MultiIndex.from_arrays(new_levels_order, names=new_names)
        
        # Update the index or columns based on whether data is a Series or DataFrame
        if isinstance(unstacked, Series):
            unstacked.index = new_multiindex
        else:
            unstacked.columns = new_multiindex
        
        return unstacked
