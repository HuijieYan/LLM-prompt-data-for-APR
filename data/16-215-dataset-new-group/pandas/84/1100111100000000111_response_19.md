def _unstack_multiple(data, clocs, fill_value=None):

    level_numbers = []
    for c in clocs:
        if isinstance(c, (str, tuple)):
            level = c if isinstance(c, str) else tuple(c)
            level_num = index._get_level_number(level)
        else:

        level_numbers.append(level_num)

    if not rlocs:
        unstacked = data.unstack(level_numbers[0], fill_value=fill_value)

        unstacked = data.unstack(level_numbers, fill_value=fill_value)

    return unstacked
