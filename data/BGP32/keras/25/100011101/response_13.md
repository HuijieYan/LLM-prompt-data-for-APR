def _preprocess_numpy_input(x, data_format, mode):

    # Arguments

        data_format: Data format of the image array.

            - tf: will scale pixels between -1 and 1,

            - torch: will scale pixels between 0 and 1 and then

        Preprocessed Numpy array.
    """
