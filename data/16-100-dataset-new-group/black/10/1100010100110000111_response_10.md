def _partially_consume_prefix(self, prefix, column):
    lines = []
    current_line = ""
    current_column = 0
    wait_for_nl = False
    for char in prefix:
        current_line += char
        if wait_for_nl:
            if char == '\n':
                if current_column < column:  # Check if the current column is less than the specified column
                    res = ''.join(lines)
                    return res, prefix[len(res):]

                lines.append(current_line)
                current_line = ""
                current_column = 0
                wait_for_nl = False
        elif char == ' ':
            current_column += 1
        elif char == '\t':
            current_column += (4 - (current_column % 4))  # Adjust the column based on the current indentation level
        elif char == '\n':
            # unexpected empty line
            current_column = 0
            wait_for_nl = False  # Reset the wait_for_nl flag
        else:
            # indent is finished
            wait_for_nl = True
    return ''.join(lines), current_line
