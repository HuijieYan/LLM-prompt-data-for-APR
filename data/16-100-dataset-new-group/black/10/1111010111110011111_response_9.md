        for char in prefix:

                    if current_line.strip() and current_column < column:
                        res = ''.join(lines)
                        return res, prefix[len(res):]
    
                    lines.append(current_line)
                    current_line = ""
                    current_column = 0
                    wait_for_nl = False
            elif char == ' ':
                current_column += 1
            elif char == '\t':
                current_column += 4
            elif char == '\n':
                if wait_for_nl:  # only reset if waiting for newline
                    current_column = 0
                    wait_for_nl = False
            else:
                # indent is finished
                wait_for_nl = True
        return ''.join(lines), current_line
