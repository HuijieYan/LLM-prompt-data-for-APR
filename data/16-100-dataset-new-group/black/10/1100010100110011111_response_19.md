def _partially_consume_prefix(self, prefix, column):
    lines = []
    current_line = ""
    current_column = 0
    wait_for_nl = False
    for char in prefix:
        current_line += char
        if wait_for_nl:
            if char == '\n':
                if current_line.strip() and current_column < column:
                    res = ''.join(lines)
                    return res, prefix[len(res):]

                lines.append(current_line)
                current_line = ""
                current_column = 0
                wait_for_nl = False
        elif char == ' ':
            current_column += 1
            if current_column >= column:
                wait_for_nl = True
        elif char == '\t':
            current_column += 4
            if current_column >= column:
                wait_for_nl = True
        elif char == '\n':
            if current_column < column:
                lines.append(current_line)
                current_line = ""
            current_column = 0
        else:
            # indent is finished
            wait_for_nl = True
    if wait_for_nl and current_line.strip() and current_column < column:
        res = ''.join(lines)
        return res, prefix[len(res):]
    return ''.join(lines), current_line
