    lines = []
    current_line = ""
    current_column = 0
    wait_for_nl = False
    for char in prefix:
        if wait_for_nl:
            if char == '\n':
                if current_column < column:
                    res = ''.join(lines)
                    return res, prefix[len(res):]

                lines.append(current_line)
                current_line = ""
                current_column = 0
                wait_for_nl = False
        elif char == ' ':
            current_column += 1
        elif char == '\t':
            current_column += 4  # Assume each tab is equivalent to 4 spaces
        elif char == '\n':
            current_column = 0
        else:
            wait_for_nl = True
        
        current_line += char
        
    return ''.join(lines), current_line
