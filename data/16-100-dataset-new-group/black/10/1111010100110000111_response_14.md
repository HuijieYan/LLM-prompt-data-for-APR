                lines.append(current_line)

                wait_for_nl = False
        elif char in (' ', '\t'):

                current_column += 1
            elif char == '\t':
                current_column = (current_column // 4 + 1) * 4  # Adjust column based on tab width
        elif char == '\n':
            # Unexpected empty line
            current_column = 0
        else:
            # Indent is finished
            wait_for_nl = True
