def _partially_consume_prefix(self, prefix, column):
    lines = []
    current_line = ""
    current_column = 0
    wait_for_nl = False
    for char in prefix:
        current_line += char
        if wait_for_nl:
            if char == '\n':
                if current_line.strip() and current_column <= column:  # Modified condition to include the column value as well
                    res = ''.join(lines)
                    return res, prefix[len(res):]

                lines.append(current_line)
                current_line = ""
                current_column = 0
                wait_for_nl = False
        elif char == ' ':
            current_column += 1
        elif char == '\t':
            # Modified the increment value to handle the tab character
            current_column += 4 - (current_column % 4)
        elif char == '\n':
            # Reset current_column to 0 when encountering a new line
            current_column = 0
        else:
            # Set wait_for_nl to True when encountering any other character
            wait_for_nl = True
    return ''.join(lines), current_line
