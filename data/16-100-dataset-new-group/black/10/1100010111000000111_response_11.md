def _partially_consume_prefix(self, prefix, column):
    lines = []
    current_line = ""
    current_column = 0
    wait_for_nl = False
    for char in prefix:
        current_line += char
        if wait_for_nl:
            if char == '\n':
                if current_line.strip() and current_column <= column:  # Include lines at the same column
                    res = ''.join(lines)
                    return res, prefix[len(res):]
                lines.append(current_line)
                current_line = ""
                current_column = 0
                wait_for_nl = False
        elif char == ' ':
            current_column += 1
        elif char == '\t':
            current_column = (current_column // 4 + 1) * 4  # Handle tab indentation
        elif char == '\n':
            # unexpected empty line
            current_column = 0
        else:
            # indent is finished
            wait_for_nl = True
    return ''.join(lines), current_line
