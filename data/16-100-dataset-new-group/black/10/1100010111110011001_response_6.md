def _partially_consume_prefix(self, prefix, column):
    lines = []
    current_line = ""
    current_column = 0
    wait_for_nl = False
    for char in prefix:
        if wait_for_nl:
            if char == '\n':
                lines.append(current_line)
                current_line = ""
                current_column = 0
                wait_for_nl = False
                continue
        if char == ' ':
            current_column += 1
        elif char == '\t':
            current_column += 4
        elif char == '\n':
            lines.append(current_line)
            current_line = ""
            current_column = 0
            wait_for_nl = False
        else:
            wait_for_nl = True
        current_line += char
    if current_line.strip() and current_column < column:
        res = ''.join(lines)
        return res, prefix[len(res):]
    return ''.join(lines), current_line
