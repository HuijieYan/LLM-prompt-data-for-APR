class Driver(object):
    def _partially_consume_prefix(self, prefix, column):
        lines = []
        current_line = ""
        current_column = 0
        wait_for_nl = False
        inside_ws = False # track if we are inside whitespace
        for char in prefix:
            current_line += char
            if wait_for_nl:
                if char == '\n':
                    if current_line.strip() and current_column < column:
                        res = ''.join(lines)
                        return res, prefix[len(res):]
    
                    lines.append(current_line)
                    current_line = ""
                    current_column = 0
                    wait_for_nl = False
            elif char == ' ':
                current_column += 1
                inside_ws = True
            elif char == '\t':
                current_column += 4
                inside_ws = True
            elif char == '\n':
                if inside_ws:
                    lines.append(current_line)
                    current_line = ""
                    current_column = 0
                    inside_ws = False
                else:
                    wait_for_nl = True
                    inside_ws = False
            else:
                # indent is finished
                wait_for_nl = True
                inside_ws = False
        return ''.join(lines), current_line
