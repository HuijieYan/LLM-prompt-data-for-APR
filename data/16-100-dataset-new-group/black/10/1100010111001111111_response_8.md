    current_column = 0
    wait_for_nl = False

                current_column = 0
                wait_for_nl = False
        elif char == ' ':
            current_column += 1
        elif char == '\t':
            current_column += 8 - (current_column % 8)  # Adjust for tab width
        elif char == '\n':
            current_column = 0
        else:
            # indent is finished
            wait_for_nl = True
            if char == '#':  # Adjust comment indentation
                current_column = column - 1

    return ''.join(lines), current_line
