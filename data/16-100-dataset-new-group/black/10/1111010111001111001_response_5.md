class Driver(object):

    def _partially_consume_prefix(self, prefix, column):
        lines = []
        current_line = ""
        current_column = 0
        wait_for_nl = False
        for char in prefix:
            current_line += char
            if wait_for_nl:
                if char == '\n':
                    if current_line.strip() and current_column <= column:  # Fix: include lines with column <=
                        res = ''.join(lines)
                        return res, prefix[len(res):]

                    lines.append(current_line)
                    current_line = ""
                    current_column = 0
                    wait_for_nl = False
            elif char == ' ':
                current_column += 1
            elif char == '\t':
                current_column += 4
            elif char == '\n':
                wait_for_nl = True  # Fix: set wait_for_nl to True for newline character
                current_column = 0
            else:
                # indent is finished
                wait_for_nl = False   # Fix: set wait_for_nl to False when indent is finished
        return ''.join(lines), current_line
