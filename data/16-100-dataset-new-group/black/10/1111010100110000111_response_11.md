```python
class Driver(object):

    def _partially_consume_prefix(self, prefix, column):
        lines = []
        current_line = ""
        current_column = 0
        wait_for_nl = False
        tab_size = 4
        for char in prefix:
            current_line += char
            if wait_for_nl and char == '\n':
                if current_column < column and current_line.strip():
                    res = ''.join(lines)
                    return res, prefix[len(res):]
                lines.append(current_line)
                current_line = ""
                current_column = 0
                wait_for_nl = False
            elif char == ' ':
                current_column += 1
            elif char == '\t':
                current_column += tab_size - (current_column % tab_size)
            elif char == '\n':
                current_column = 0
            else:
                wait_for_nl = True
        return ''.join(lines), current_line
```
