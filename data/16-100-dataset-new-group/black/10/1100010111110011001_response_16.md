def _partially_consume_prefix(self, prefix, column):
    lines = []
    current_line = ""
    current_column = 0
    wait_for_nl = False
    for char in prefix:
        current_line += char
        if wait_for_nl:
            if char == '\n':
                if current_line.strip() and current_column <= column:  # changed < to <=
                    res = ''.join(lines)
                    return res, prefix[len(res):]

                lines.append(current_line)
                current_line = ""
                current_column = 0
                wait_for_nl = False
        elif char in [' ', '\t']:  # changed if-elif condition
            if char == ' ':
                current_column += 1
            elif char == '\t':
                current_column += 4
        elif char == '\n':
            # unexpected empty line
            current_column = 0
            current_line = ""  # reset current_line
        else:
            # indent is finished
            wait_for_nl = True
    return ''.join(lines), current_line
