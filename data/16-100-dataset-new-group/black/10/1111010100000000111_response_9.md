            current_line += char

                    lines.append(current_line)

                    wait_for_nl = False
            elif char == ' ':

                current_column += 4  # adjust the column count for tabs
            elif char == '\n':
                # unexpected empty line
                current_column = 0
            else:
                # indent is finished
                wait_for_nl = True
        return ''.join(lines), current_line
