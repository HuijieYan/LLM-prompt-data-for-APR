def _partially_consume_prefix(self, prefix, column):

    current_line = ""
    current_column = 0
    wait_for_nl = False

            if char == '\n':

                    res = ''.join(lines)
                    return res, prefix[len(res):]

                lines.append(current_line)
                current_line = ""
                current_column = 0
                wait_for_nl = False
        elif char == ' ':
            current_column += 1
        elif char == '\t':
            current_column += 4
        elif char == '\n':
            if current_line.strip() and current_column < column:
                res = ''.join(lines)
                return res, prefix[len(res):]
            current_column = 0
            lines.append(current_line)
            current_line = ""
        else:
            # indent is finished
            wait_for_nl = True
    return ''.join(lines), current_line
