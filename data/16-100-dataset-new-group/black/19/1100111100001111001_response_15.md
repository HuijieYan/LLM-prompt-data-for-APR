def _maybe_empty_lines(self, current_line: Line) -> Tuple[int, int]:
    max_allowed = 1
    depth = current_line.depth
    first_leaf = current_line.leaves[0] if current_line.leaves else None
    before = min(len(first_leaf.prefix.split("\n")), max_allowed) if first_leaf else 0
    self.previous_defs_count = self.previous_defs_count if current_line.is_flow_control else 0
    if current_line.is_decorator or current_line.is_def or current_line.is_class:
        if not current_line.is_decorator:
            self.previous_defs.append(depth)
        if self.previous_line is None or self.previous_line.is_decorator:
            return 0, 0
        newlines = 2
        if current_line.depth:
            newlines -= 1
        return newlines, 0

    if current_line.is_flow_control:
        return before, self.previous_defs_count

    if self.previous_line and self.previous_line.is_import and not current_line.is_import and depth == self.previous_line.depth:
        return max(before, 1), 0

    if self.previous_line and self.previous_line.is_yield and (not current_line.is_yield or depth != self.previous_line.depth):
        return max(before, 1), 0

    return before, 0
