def _maybe_empty_lines(self, current_line: Line) -> Tuple[int, int]:
    max_allowed = 1
    if current_line.depth == 0:
        max_allowed = 2
    if current_line.leaves:
        # Consume the first leaf's extra newlines.
        first_leaf = current_line.leaves[0]
        before = first_leaf.prefix.count("\n")
        before = min(before, max_allowed)
        first_leaf.prefix = ""
    else:
        before = 0
    # Check if the previous line is not None and is a decorator.
    previous_line_is_decorator = self.previous_line and self.previous_line.is_decorator
    if previous_line_is_decorator:
        # Don't insert empty lines before and between decorators.
        return 0, 0
    depth = current_line.depth
    if (
        self.previous_line
        and self.previous_line.is_import
        and not current_line.is_import
        and depth == self.previous_line.depth
    ):
        return (before or 1), 0
    if (
        self.previous_line
        and self.previous_line.is_yield
        and (not current_line.is_yield or depth != self.previous_line.depth)
    ):
        return (before or 1), 0
    # Change the condition to check if the current line is not a decorator, def or class.
    if not current_line.is_decorator and not current_line.is_def and not current_line.is_class:
        if depth == 0:
            return 0, 0
        newlines = 2
        return newlines, 0
    if current_line.is_flow_control:
        return before, 1
    if not current_line.is_decorator:
        # Append the current line's depth to self.previous_defs whenever it's not a decorator.
        self.previous_defs.append(depth)
    return before, 0
