def _preprocess_numpy_input(x, data_format, mode):
    if mode == 'tf':
        x /= 127.5
        x -= 1.
        return x

    if mode == 'torch':
        x /= 255.
        mean = [0.485, 0.456, 0.406]
        std = [0.229, 0.224, 0.225]
        if data_format == 'channels_first':
            x = x - mean[:, None, None]
            x = x / std[:, None, None]
        else:
            x = x - mean
            x = x / std
    else:
        if data_format == 'channels_first':
            # 'RGB'->'BGR'
            if x.ndim == 4:
                x[:, 0, :, :] -= 123.68
                x[:, 1, :, :] -= 116.779
                x[:, 2, :, :] -= 103.939
            else:
                x[0, :, :] -= 123.68
                x[1, :, :] -= 116.779
                x[2, :, :] -= 103.939
        else:
            # 'RGB'->'BGR'
            x[..., 0] -= 123.68
            x[..., 1] -= 116.779
            x[..., 2] -= 103.939

    return x
