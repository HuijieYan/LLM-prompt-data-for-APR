def _preprocess_numpy_input(x, data_format, mode):
    if mode == 'tf':
        x /= 127.5
        x -= 1.
        return x
    if mode == 'torch':
        x = x / 255.
        mean = [0.485, 0.456, 0.406]
        std = [0.229, 0.224, 0.225]
        x = (x - mean) / std
    else:
        if data_format == 'channels_first':
            if x.ndim == 3:
                x[:, 0, :, :] -= 103.939
                x[:, 1, :, :] -= 116.779
                x[:, 2, :, :] -= 123.68
            else:
                x[..., 0] -= 103.939
                x[..., 1] -= 116.779
                x[..., 2] -= 123.68
        else:
            x[..., 0] -= 103.939
            x[..., 1] -= 116.779
            x[..., 2] -= 123.68
    return x
