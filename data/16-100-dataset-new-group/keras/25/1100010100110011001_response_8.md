def _preprocess_numpy_input(x, data_format, mode):
    """Preprocesses a Numpy array encoding a batch of images.
    
    # Arguments
    x: Input array, 3D or 4D
    data_format: Data format of the image array
    mode: One of "caffe", "tf" or "torch"
    - caffe: convert the images from RGB to BGR,
    then zero-center each color channel with respect to the ImageNet dataset, without scaling.
    - tf: scale pixels between -1 and 1, sample-wise.
    - torch: scale pixels between 0 and 1, then normalize each channel with respect to the ImageNet dataset.
    
    # Returns
    Preprocessed Numpy array"""
    
    if mode == 'tf':
        x /= 127.5
        x -= 1.
        return x
    
    if mode == 'torch':
        x = x.astype('float32')  # converting x to float type
        x /= 255.
        mean = [0.485, 0.456, 0.406]
        std = [0.229, 0.224, 0.225]
    else:
        if data_format == 'channels_first':
            # 'RGB'->'BGR'
            if x.ndim == 3:
                x = x[:, ::-1, ...]  # reversing the order of channels
            else:
                x = x[:, ::-1, ...]  # reversing the order of channels
        else:
            # 'RGB'->'BGR'
            x = x[..., ::-1]
        mean = [103.939, 116.779, 123.68]
        std = None
    
    if std is not None:  # check if standard deviation is not None
        x -= mean
        x /= std
    else:
        x -= mean
    
    return x
