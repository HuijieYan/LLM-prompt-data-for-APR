            - torch: will scale pixels between 0 and 1 and then

        x /= 127.5
        x -= 1.
        return x

        x /= 255.0

        std = [0.229, 0.224, 0.225]

            if x.ndim == 3:

            else:
                x = x[:, ::-1, ...]
        else:

            x = x[..., ::-1]
            mean = [103.939, 116.779, 123.68]
            std = None

    if mode == 'torch' and data_format == 'channels_last':
