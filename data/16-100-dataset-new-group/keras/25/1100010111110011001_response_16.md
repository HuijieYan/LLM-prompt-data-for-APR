import numpy as np

def _preprocess_numpy_input(x, data_format, mode):
    if mode == 'tf':
        x /= 127.5
        x -= 1.
        return x

    if mode == 'torch':
        x = x.astype('float32')  # Convert to float32 to avoid ufunc type error
        x /= 255.
        mean = [0.485, 0.456, 0.406]
        std = [0.229, 0.224, 0.225]
    else:
        if data_format == 'channels_first':
            if x.ndim == 3:
                x = x[:, ::-1, ...]  # Corrected to properly convert 'RGB' to 'BGR'
            else:
                x = x[:, :, ::-1, ...]  # Corrected to properly convert 'RGB' to 'BGR'
        else:
            x = x[..., ::-1]  # Corrected to properly convert 'RGB' to 'BGR'
            mean = [103.939, 116.779, 123.68]
            std = None

    if mode == 'torch' and x.dtype == 'float32':  # Only perform zero-centering if input is in float32
        if data_format == 'channels_first':
            x[0] -= mean[0]
            x[1] -= mean[1]
            x[2] -= mean[2]
            if std is not None:
                x[0] /= std[0]
                x[1] /= std[1]
                x[2] /= std[2]
        else:
            x[..., 0] -= mean[0]
            x[..., 1] -= mean[1]
            x[..., 2] -= mean[2]
            if std is not None:
                x[..., 0] /= std[0]
                x[..., 1] /= std[1]
                x[..., 2] /= std[2]
    return x
