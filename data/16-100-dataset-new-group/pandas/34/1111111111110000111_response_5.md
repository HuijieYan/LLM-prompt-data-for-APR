        DatetimeIndex(start=first, end=last, freq=self.freq, name=ax.name, tz=ax.tz)
        .shift(0, freq=self.freq)
        .asfreq(self.freq, method="pad", ambiguous="infer", nonexistent="shift_forward")
    )

    ax_values = ax.asi8

    elif self.label == "right":
        labels = labels[1:]
