        raise TypeError(

            f"an instance of {type(ax).__name__}"
        )

    if len(ax) == 0:
        binner = labels = DatetimeIndex(data=[], freq=self.freq, name=ax.name)

    first, last = _get_timestamp_range_edges(

            ambiguous="infer",
            nonexistent="shift_forward",
        )
        start_end_shifted = start_end.shift(1, freq="H")

        bin_edges[-1] = start_end_shifted[-1].to_pydatetime().astimezone(ax.tz).timestamp() * 1e9
