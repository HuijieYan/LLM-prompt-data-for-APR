    def equals(self, other):
        # Check if the number of axes in both in BlockManagers are the same
        if len(self.axes) != len(other.axes):
            return False
        
        # Check if all the axes in both BlockManagers are equal
        for ax1, ax2 in zip(self.axes, other.axes):
            if not ax1.equals(ax2):
                return False
        
        # Consolidate both BlockManagers in place
        self._consolidate_inplace()
        other._consolidate_inplace()
        
        # Check if the number of blocks in both BlockManagers are the same
        if len(self.blocks) != len(other.blocks):
            return False
        
        # Sort the blocks in both BlockManagers using the type name and the manager locations
        def canonicalize(blocks):
            return (blocks.dtype.name, blocks.mgr_locs.as_array.tolist())
        
        self_blocks = sorted(self.blocks, key=canonicalize)
        other_blocks = sorted(other.blocks, key=canonicalize)
        
        # Check if all the blocks in both sorted BlockManagers are equal
        return all(block.equals(oblock) for block, oblock in zip(self_blocks, other_blocks)
