```python
def equals(self, other):
    self_axes, other_axes = self.axes, other.axes
    if self_axes != other_axes:  # Compare the axes directly
        return False
    
    self._consolidate_inplace()  # Consolidate the blocks in place
    other._consolidate_inplace()  # Consolidate the blocks in place
    
    self_blocks = sorted(self.blocks, key=lambda block: (block.dtype.name, block.mgr_locs.as_array.tolist()))
    other_blocks = sorted(other.blocks, key=lambda block: (block.dtype.name, block.mgr_locs.as_array.tolist()))
    if len(self_blocks) != len(other_blocks):
        return False
    
    return all(
        block.equals(oblock) for block, oblock in zip(self_blocks, other_blocks)
    )
```
