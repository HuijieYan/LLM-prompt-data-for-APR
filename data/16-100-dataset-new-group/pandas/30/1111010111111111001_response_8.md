def _try_convert_to_date(self, data):
    """
    Try to parse an array into a date column.

    Try to coerce an array with object data type into int64 data type to handle epoch/iso formats
    as well as integer/float data type in epoch formats. 
    Return a boolean if parsing was successful.
    """
    # no conversion on empty
    if not len(data):
        return data, False

    new_data = data
    if new_data.dtype == "object":
        try:
            new_data = new_data.astype("int64")
        except (TypeError, ValueError, OverflowError):
            pass

    # ignore numbers that are out of range
    if issubclass(new_data.dtype.type, np.number):
        in_range = (
            isna(new_data._values)
            | (new_data > self.min_stamp)
        )
        if not in_range.all():
            return data, False

    # convert to date
    try:
        new_data = to_datetime(new_data, errors="raise", unit=self.date_unit)
        return new_data, True
    except (ValueError, OverflowError):
        return data, False
