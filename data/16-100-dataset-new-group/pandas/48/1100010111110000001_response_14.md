### Issue overview
The `_cython_agg_blocks` function in the `pandas/core/groupby/generic.py` file fails to cast float64 values to int64 correctly, leading to a TypeError.

### Bug analysis
The `safe_cast` function is encountering a TypeError when trying to cast an array from dtype('float64') to dtype('int64') according to the rule 'safe'.

The `_cython_agg_blocks` function attempts to perform the casting by invoking `.astype()` method, but the result is not as expected, triggering a TypeError.

### Bug cause
The root cause of the bug is the incorrect casting of float values to int values within the `_cython_agg_blocks` function, which results in TypeError during the casting process.

### Fix strategy
The strategy to fix this bug involves refining the logic for casting float values to int values to make sure that safe casting is executed correctly.

### The corrected version

```python
def _cython_agg_blocks(
    self, how: str, alt=None, numeric_only: bool = True, min_count: int = -1
) -> "Tuple[List[Block], Index]":
    data: BlockManager = self._get_data_to_aggregate()

    if numeric_only:
        data = data.get_numeric_data(copy=False)

    agg_blocks: List[Block] = []
    new_items: List[np.ndarray] = []
    deleted_items: List[np.ndarray] = []
    split_items: List[np.ndarray] = []
    split_frames: List[DataFrame] = []

    no_result = object()
    for block in data.blocks:
        result = no_result
        locs = block.mgr_locs.as_array
        try:
            result, _ = self.grouper.aggregate(
                block.values, how, axis=1, min_count=min_count
            )
        except NotImplementedError:
            # generally if we have numeric_only=False
            # and non-applicable functions
            # try to python agg

            if alt is None:
                assert how == "ohlc"
                deleted_items.append(locs)
                continue

            obj = self.obj[data.items[locs]]
            if obj.shape[1] == 1:
                obj = obj.iloc[:, 0]

            s = get_groupby(obj, self.grouper)
            try:
                result = s.aggregate(lambda x: alt(x, axis=self.axis))
            except TypeError:
                deleted_items.append(locs)
                continue
            else:
                result = cast(DataFrame, result)
                if len(result._data.blocks) != 1:
                    split_items.append(locs)
                    split_frames.append(result)
                    continue

                assert len(result._data.blocks) == 1
                result = result._data.blocks[0].values
                if isinstance(result, np.ndarray) and result.ndim == 1:
                    result = result.reshape(1, -1)

        assert not isinstance(result, DataFrame)

        if result is not no_result:
            result = maybe_convert_objects(result)
            result = maybe_downcast_to_dtype(result, dtype=np.int64)

            agg_block: Block = block.make_block(result)

        new_items.append(locs)
        agg_blocks.append(agg_block)

    if not (agg_blocks or split_frames):
        raise DataError("No numeric types to aggregate")

    if split_items:
        for locs, result in zip(split_items, split_frames):
            assert len(locs) == result.shape[1]
            for i, loc in enumerate(locs):
                new_items.append(np.array([loc], dtype=locs.dtype))
                agg_blocks.append(result.iloc[:, [i]]._data.blocks[0])

    indexer = np.concatenate(new_items)
    agg_items = data.items.take(np.sort(indexer))

    if deleted_items:
        deleted = np.concatenate(deleted_items)
        ai = np.arange(len(data))
        mask = np.zeros(len(data))
        mask[deleted] = 1
        indexer = (ai - mask.cumsum())[indexer]

    offset = 0
    for blk in agg_blocks:
        loc = len(blk.mgr_locs)
        blk.mgr_locs = indexer[offset : (offset + loc)]
        offset += loc

    return agg_blocks, agg_items
```
