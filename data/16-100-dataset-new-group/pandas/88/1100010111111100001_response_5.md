    columns=None,
    aggfunc="mean",
    fill_value=None,
    margins=False,
    dropna=True,

) -> "DataFrame":

    columns = _convert_by(columns)

    if isinstance(aggfunc, list):

        keys = index + columns
        for func in aggfunc:
            grouped = data.groupby(keys, observed=observed)
            agged = grouped.agg(func)
            pieces.append(agged)

        table = concat(pieces, keys=aggfunc, axis=1)
    else:
        keys = index + columns
        grouped = data.groupby(keys, observed=observed)
