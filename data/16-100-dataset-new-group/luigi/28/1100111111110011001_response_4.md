class HiveCommandClient(HiveClient):
    def table_exists(self, table, database='default', partition=None):
        if partition is None:
            stdout = run_hive_cmd('use {0}; show tables in {1} like "{2}";'.format(database, database, table))

            return stdout and table in stdout
        else:
            partition_filter = " and ".join(["{}='{}'".format(k, v) for k, v in partition.items()])
            stdout = run_hive_cmd("use {}; show partitions {}.{} where {}".format(database, database, table, partition_filter))

            return bool(stdout)
