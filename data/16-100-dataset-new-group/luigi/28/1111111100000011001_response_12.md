def table_exists(self, table, database='default', partition=None):
    if partition is None:
        stdout = run_hive_cmd('use {0}; show tables like "{1}";'.format(database, table))

        return stdout and table in stdout
    else:
        if isinstance(partition, str):  # Check if partition is a string
            partition_filter = "partition ({0})".format(partition)
        else:
            partition_filter = partition
        
        stdout = run_hive_cmd("use {0}; show partitions {1} {2};".format(database, table, partition_filter))

        return bool(stdout)  # Convert stdout to boolean
