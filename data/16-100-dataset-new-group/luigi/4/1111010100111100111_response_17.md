    """
    Defines copying from s3 into redshift.

    logger.info("Inserting file: %s", f)
    colnames = ''

     COPY {table} {colnames} from '{source}'
     CREDENTIALS '{creds}'
     {options}

        colnames=colnames,
        source=f,
        creds=self._credentials(),
