    Defines copying from s3 into redshift.

    colnames = ''
    if self.columns is not None and len(self.columns) > 0:  # Explicitly check if self.columns is not None

     COPY {table} {colnames} from '{source}'
     CREDENTIALS '{creds}'

        source=f,
